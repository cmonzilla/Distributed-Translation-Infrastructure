125  void* allocate(size_type num) {
126  const size_type remains = getAvailableBytes();
127  void* ptr = NULL;
128 
129  //Check if there is more space needed!
130  if (remains > num) {
131  //Allocate the requested memory in the buffer
132  ptr = static_cast<void*> (static_cast<char*> (_pBuffer) + _allocBytes);
133  _allocBytes += num;
134  } else {
135  //WARNING: If we need more space we allocate it additionally in a not so efficient way.
136  LOG_DEBUG3 << "Allocating additional: " << num << " bytes!" << END_LOG;
137  ptr = static_cast<void*> (new TStorageData[num]);
138  _memoryBuffers.push_back(ptr);
