260  template<bool is_back_off>
261  inline bool get_query_context_Id(const TModelLevel ctxLen, TLongId & ctxId) {
262  const TModelLevel mGramEndIdx = (is_back_off ? (N - 2) : (N - 1));
263  const TModelLevel eIdx = mGramEndIdx;
264  const TModelLevel bIdx = mGramEndIdx - ctxLen;
265  TModelLevel idx = bIdx;
266 
267  LOG_DEBUG1 << "Computing ctxId for context length: " << SSTR(ctxLen)
268  << " for a " << (is_back_off ? "back-off" : "probability")
269  << " computation" << END_LOG;
270 
271  //Compute the first words' hash
272  ctxId = ATrie<N>::m_tmp_word_ids[idx];
273  LOG_DEBUG1 << "First word @ idx: " << SSTR(idx) << " has wordId: " << SSTR(ctxId) << END_LOG;
274  idx++;
275 
276  //Since the first word defines the second word context, and
277  //if this word is unknown then there is definitely no data
278  //for this N-gram in the trie ... so we through!
279  if (ctxId < AWordIndex::MIN_KNOWN_WORD_ID) {
280  LOG_DEBUG1 << "The first " << SSTR(ctxLen + 1) << " wordId == "
281  << SSTR(ctxId) << " i.e. <unk>, need to back-off!" << END_LOG;
282  return false;
283  } else {
284  //Compute the subsequent context ids
285  for (; idx < eIdx;) {
286  LOG_DEBUG1 << "Start searching ctxId for mGramWordIds[" << SSTR(idx) << "]: "
287  << SSTR(ATrie<N>::m_tmp_word_ids[idx]) << " prevCtxId: " << SSTR(ctxId) << END_LOG;
288  if (m_get_ctx_id_func(ATrie<N>::m_tmp_word_ids[idx], ctxId, (idx - bIdx) + 1)) {
289  LOG_DEBUG1 << "getContextId(" << SSTR(ATrie<N>::m_tmp_word_ids[idx])
290  << ", prevCtxId) = " << SSTR(ctxId) << END_LOG;
291  idx++;
292  } else {
293  //The next context id could not be retrieved
294  return false;
295  }
296  }
297 
298  LOG_DEBUG1 << "Resulting context hash for context length " << SSTR(ctxLen)
299  << " of a " << (is_back_off ? "back-off" : "probability")
300  << " computation is: " << SSTR(ctxId) << END_LOG;
301 
302  return true;
303  }

315  template<DebugLevelsEnum logLevel>
316  inline bool getContextId(const T_M_Gram & gram, TLongId &ctxId) {
317  //Try to retrieve the context from the cache, if not present then compute it
318  if (getCachedContextId(gram, ctxId)) {
319  //Get the start context value for the first token
320  TShortId wordId = ATrie<N>::get_word_index()->get_word_id(gram.tokens[0]);
321 
322  //There is no id cached for this M-gram context - find it
323  if (wordId != AWordIndex::UNKNOWN_WORD_ID) {
324  //The first word id is the first context id
325  ctxId = wordId;
326  LOGGER(logLevel) << "ctxId = getId('" << gram.tokens[0].str()
327  << "') = " << SSTR(ctxId) << END_LOG;
328 
329  //Iterate and compute the hash:
330  for (int i = 1; i < (gram.level - 1); i++) {
331  wordId = ATrie<N>::get_word_index()->get_word_id(gram.tokens[i]);
332  if (wordId != AWordIndex::UNKNOWN_WORD_ID) {
333  LOGGER(logLevel) << "wordId = getId('" << gram.tokens[i].str()
334  << "') = " << SSTR(wordId) << END_LOG;
335  if (m_get_ctx_id_func(wordId, ctxId, i + 1)) {
336  LOGGER(logLevel) << "ctxId = computeCtxId( "
337  << "wordId, ctxId ) = " << SSTR(ctxId) << END_LOG;
338  } else {
339  //The next context id could not be computed
340  LOGGER(logLevel) << "The next context for wordId: "
341  << SSTR(wordId) << " and ctxId: "
342  << SSTR(ctxId) << "on level: " << SSTR((i + 1))
343  << "could not be computed!" << END_LOG;
344  return false;
345  }
346  } else {
347  //The next word Id was not found, it is
348  //unknown, so we can stop searching
349  LOGGER(logLevel) << "The wordId for '" << gram.tokens[i].str()
350  << "' could not be found!" << END_LOG;
351  return false;
352  }
353  }
354 
355  //Cache the newly computed context id for the given n-gram context
356  setCacheContextId(gram, ctxId);
357 
358  //The context Id was found in the Trie
359  LOGGER(logLevel) << "The ctxId could be computed, "
360  << "it's value is: " << SSTR(ctxId) << END_LOG;
361  return true;
362  } else {
363  //The context id could not be computed as
364  //the first N-gram's word is already unknown
365  LOGGER(logLevel) << "The wordId for '" << gram.tokens[0].str()
366  << "' could not be found!" << END_LOG;
367  return false;
368  }
369  } else {
370  //The context Id was found in the cache
371  LOGGER(logLevel) << "The ctxId was found in cache, "
372  << "it's value is: " << SSTR(ctxId) << END_LOG;
373  return true;
374  }

383  inline bool getCachedContextId(const T_M_Gram &mGram, TLongId & result) {
384  if (m_chached_ctx == mGram.context) {
385  result = m_chached_ctx_id;

388  << ", cached ctxId: " << SSTR(m_chached_ctx_id) << END_LOG;
389  return false;
