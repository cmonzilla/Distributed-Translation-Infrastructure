89  inline uint64_t sub_hash(const TModelLevel begin_idx, const TModelLevel end_idx) const {
90  LOG_DEBUG3 << "Hashing tokens begin_idx: " << begin_idx << ", end_idx: " << end_idx << END_LOG;
91 
92  //Compute the length of the gram tokens in memory, including spaces between
93  const char * beginFirstPtr = tokens[begin_idx].getBeginCStr();
94  const TextPieceReader & last = tokens[end_idx];
95  const char * beginLastPtr = last.getBeginCStr();
96  const size_t totalLen = (beginLastPtr - beginFirstPtr) + last.getLen();
97  LOG_DEBUG3 << "Hashing tokens length: " << totalLen << END_LOG;
98 
99  //If the sanity check is on then test that the memory is continuous
100  //Compute the same length but with a longer iterative algorithms
101  if (DO_SANITY_CHECKS) {
102  //Compute the exact length
103  size_t exactTotalLen = (end_idx - begin_idx); //The number of spaces in between tokens
104  for (TModelLevel idx = begin_idx; idx <= end_idx; idx++) {
105  exactTotalLen += tokens[idx].getLen();
106  }
107  //Check that the exact and fast computed lengths are the same
108  if (exactTotalLen != totalLen) {
109  stringstream msg;
110  msg << "The memory allocation for M-gram tokens is not continuous: totalLen (" <<
111  SSTR(totalLen) << ") != exactTotalLen (" << SSTR(exactTotalLen) << ")";
112  throw Exception(msg.str());
113  }
114  }
115 
116  //Compute the hash using the gram tokens with spaces with them
117  return computeHash(beginFirstPtr, totalLen);
118  }
