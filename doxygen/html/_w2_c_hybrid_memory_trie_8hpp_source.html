110  virtual TProbBackOffEntry& make_M_GramDataRef(const TModelLevel level, const TShortId wordId, const TLongId ctxId) {
111  const TModelLevel idx = (level - ALayeredTrie<N>::MGRAM_IDX_OFFSET);
112 
113  //Get the word mapping first
114  StorageContainer*& ctx_mapping = m_mgram_mapping[idx][wordId];
115 
116  //If the mappings is not there yet for the contexts then create it
117  if (ctx_mapping == NULL) {
118  ctx_mapping = m_storage_factory->create(level);
119  LOG_DEBUG3 << "A new ACtxToPBStorage container is allocated for level " << SSTR(level) << END_LOG;
120  }
121 
122  //Add the new element to the context mapping
123  TShortId & nextCtxId = ctx_mapping->operator[](ctxId);
124  nextCtxId = next_ctx_id[idx]++;
125 
126  //Return the reference to it
127  return m_mgram_data[level - 1][nextCtxId];

170  virtual bool get_N_GramProb(const TShortId wordId, const TLongId ctxId,
171  TLogProbBackOff & prob) {
172  //Try to find the word mapping first
173  StorageContainer*& ctx_mapping = m_mgram_mapping[ALayeredTrie<N>::N_GRAM_IDX_IN_M_N_ARR][wordId];
174 
175  //If the mapping is present the search further, otherwise return false
176  if (ctx_mapping != NULL) {
177  typename StorageContainer::const_iterator result = ctx_mapping->find(ctxId);
178  if (result == ctx_mapping->end()) {
179  //The data could not be found
180  return false;
181  } else {
182  //The data could be found
183  LOG_DEBUG1 << "Found the probability value: " << SSTR((TLogProbBackOff) result->second)
184  << ", wordId: " << SSTR(wordId) << ", ctxId: " << SSTR(ctxId) << END_LOG;
185  prob = (TLogProbBackOff &) result->second;
186  return true;
187  }
188  } else {
189  LOG_DEBUG1 << "There are no elements @ level: " << SSTR(N) << " for wordId: " << SSTR(wordId) << "!" << END_LOG;
190  return false;
191  }

202  //M-Gram data for 1 <= M < N. This is a 2D array storing
203  //For each M-Gram level M an array of prob-back_off values
204  // m_mgram_data[M][0] - probability/back-off pair for the given M-gram
205  // m_mgram_data[M][1] --//--
206  // ...
207  // m_mgram_data[M][#M-Grams - 1] --//--
208  // m_mgram_data[M][#M-Grams] --//--
209  TProbBackOffEntry * m_mgram_data[N - 1];
210 
211  //M-Gram data for 1 < M <= N. This is a 2D array storing
212  //For each M-Gram level M an array of #words elements of
213  //pointers to C template parameter type:
214  // m_mgram_mapping[M][0] -> NULL (if there is no M-gram ending with word 0 in the level)
215  // m_mgram_mappin[M]g[1] -> C instance
216  // ...
217  // m_mgram_mapping[M][#words - 1] -> NULL
218  // m_mgram_mapping[M][#words] -> C instance
219  //
220  //For all 1 < M < N instances of C contain mappings from the
221  //context index to the index in the m_mgram_data[M] array -
222  //the array storing probability/back-off values.
223  //For M = N the pair value stored in the instance of C is the
224  //probability itself! Note that that the probabilities are
225  //stored as floats - 4 bytes and m_mgram_data[M] array is also a
226  //4 byte integer, so we minimize memory usage by storing float
227  //probability in place of the index.
228  StorageContainer** m_mgram_mapping[N - 1];

246  inline bool getContextId(const TShortId wordId, TLongId & ctxId, const TModelLevel level) {
247  LOG_DEBUG3 << "Retrieving context level: " << level << ", wordId: "
248  << wordId << ", ctxId: " << ctxId << END_LOG;
249  //Retrieve the context data for the given word
250  StorageContainer* ctx_mapping = m_mgram_mapping[level - ALayeredTrie<N>::MGRAM_IDX_OFFSET][wordId];
251 
252  //Check that the context data is available
253  if (ctx_mapping != NULL) {
254  typename StorageContainer::const_iterator result = ctx_mapping->find(ctxId);
255  if (result == ctx_mapping->end()) {
256  LOG_DEBUG2 << "Can not find ctxId: " << SSTR(ctxId) << " for level: "
257  << SSTR(level) << ", wordId: " << SSTR(wordId) << END_LOG;
258  return false;
259  } else {
260  LOG_DEBUG2 << "Found next ctxId: " << SSTR(result->second)
261  << " for level: " << SSTR(level) << ", wordId: "
262  << SSTR(wordId) << ", ctxId: " << SSTR(ctxId) << END_LOG;
263 
264  ctxId = result->second;
265  return true;
266  }
267  } else {
268  LOG_DEBUG2 << "No context data for: " << SSTR(level)
269  << ", wordId: " << SSTR(wordId) << END_LOG;
270  return false;
271  }
272  }
