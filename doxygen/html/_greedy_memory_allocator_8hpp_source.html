59  template<typename TContaner, typename TAllocator >
60  void allocate_container(TContaner ** ppContainer, TAllocator ** ppAllocator,
61  const size_t numEntries, const string ctName, const float factor) {
62  //Compute the number of bytes needed to store these words
63  const size_t factoredNumElems = numEntries * factor;
64 
65  LOG_DEBUG << "Computing the required storage for " << ctName
66  << ", numEntries=" << numEntries << ", factor=" << factor
67  << ", result=" << factoredNumElems << END_LOG;
68 
69  //Allocate the allocator
70  LOG_DEBUG << "Allocating the " << ctName << " allocator for " << factoredNumElems << " elements!" << END_LOG;
71  *ppAllocator = new TAllocator(factoredNumElems);
72 
73  //Allocate the map with the given allocator
74  LOG_DEBUG << "Allocating the " << ctName << " container with the created allocator!" << END_LOG;
75  *ppContainer = new TContaner(**ppAllocator);
76  }

84  template<typename TContaner, typename TAllocator >
85  void reserve_mem_unordered_map(TContaner ** ppContainer, TAllocator ** ppAllocator,
86  const size_t numEntries, const string ctName, const float factor = UNORDERED_MAP_MEMORY_FACTOR) {
87  //Call the generic allocation function with the appropriate factor
88  allocate_container<TContaner, TAllocator>(ppContainer, ppAllocator, numEntries, ctName, factor);
89 
90  //Reserve some memory for the buckets!
91  LOG_DEBUG << "Reserving " << numEntries << " buckets for the " << ctName << "!" << END_LOG;
92  (*ppContainer)->reserve(numEntries);
93  }

103  //First deallocate the memory of the map
104  if (*ppContainer != NULL) {
105  delete *ppContainer;
