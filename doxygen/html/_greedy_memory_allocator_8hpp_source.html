63  template<typename TContaner, typename TAllocator>
64  void allocate_container(TContaner ** ppContainer, TAllocator ** ppAllocator,
65  const size_t numEntries, const string ctName,
66  const float factor = UNORDERED_MAP_MEMORY_FACTOR) {
67  //Compute the number of bytes needed to store these words
68  const size_t factoredNumElems = numEntries * factor;
69 
70  LOG_DEBUG4 << "Computing the required storage for " << ctName
71  << ", numEntries=" << numEntries << ", factor=" << factor
72  << ", result=" << factoredNumElems << END_LOG;
73 
74  //Allocate the allocator
75  LOG_DEBUG4 << "Allocating the " << ctName << " allocator for " << factoredNumElems << " elements!" << END_LOG;
76  *ppAllocator = new TAllocator(factoredNumElems);
77 
78  //Allocate the map with the given allocator
79  LOG_DEBUG4 << "Allocating the " << ctName << " container with the created allocator!" << END_LOG;
80  *ppContainer = new TContaner(**ppAllocator);
81  }

93  template<typename TContaner, typename TAllocator>
94  void reserve_mem_unordered_map(TContaner ** ppContainer, TAllocator ** ppAllocator,
95  const size_t numEntries, const string ctName,
96  const float factor = UNORDERED_MAP_MEMORY_FACTOR) {
97  //Call the generic allocation function with the appropriate factor
98  allocate_container<TContaner, TAllocator>(ppContainer, ppAllocator, numEntries, ctName, factor);
99 
100  //Reserve some memory for the buckets!
