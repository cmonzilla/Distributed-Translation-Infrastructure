229  inline void get_bucket_id(const T_M_Gram &gram, TShortId & bucket_idx) {
230  //Compute the hash value for the given M-gram, it must
231  //be the M-Gram id in the M-Gram data storage
232  const uint64_t gram_hash = gram.hash();
233  LOG_DEBUG1 << "The " << gram.level << "-gram: " << tokensToString(gram)
234  << " hash is " << gram_hash << END_LOG;
235 
236  bucket_idx = get_bucket_id(gram_hash, gram.level);
237 
238  LOG_DEBUG1 << "Getting bucket for " << tokensToString(gram) << " bucket_idx: " << SSTR(bucket_idx) << END_LOG;
239 
240  //If the sanity check is on then check on that the id is within the range
241  if (DO_SANITY_CHECKS && ((bucket_idx < 0) || (bucket_idx >= num_buckets[gram.level - 1]))) {
242  stringstream msg;
243  msg << "The " << SSTR(gram.level) << "-gram: " << tokensToString<N>(gram)
244  << " was given an incorrect hash: " << SSTR(bucket_idx)
245  << ", must be within [0, " << SSTR(num_buckets[gram.level - 1]) << "]";
246  throw Exception(msg.str());
247  }

256  template<typename BUCKET_TYPE>
257  void post_M_N_Grams(BUCKET_TYPE * buckets, const TModelLevel level) {
258  //Iterate through all buckets and shrink/sort sub arrays
259  for (TShortId bucket_idx = 0; bucket_idx < num_buckets[level - 1]; ++bucket_idx) {
260  //First get the sub-array reference.
261  BUCKET_TYPE & ref = buckets[bucket_idx];
262 
263  LOG_DEBUG1 << "Shrinking the " << SSTR(level) << "-gram level bucket idx: " << SSTR(bucket_idx) << " ..." << END_LOG;
264  //Reduce capacity if there is unused memory
265  ref.shrink();
266  LOG_DEBUG1 << "Shrinking the " << SSTR(level) << "-gram level bucket idx: " << SSTR(bucket_idx) << " is done" << END_LOG;
267 
268  LOG_DEBUG1 << "Sorting the " << SSTR(level) << "-gram level bucket idx: " << SSTR(bucket_idx) << " ..." << END_LOG;
269  //Order the N-gram array as it is unordered and we will binary search it later!
270  ref.sort([&] (const typename BUCKET_TYPE::TElemType & first, const typename BUCKET_TYPE::TElemType & second) -> bool {
271  LOG_DEBUG1 << "Comparing " << SSTR((void*) first.id) << " with " << SSTR((void*) second.id) << END_LOG;
272  //Update the progress bar status
273  Logger::updateProgressBar();
274  //Return the result
275  return Byte_M_Gram_Id::is_less_m_grams_id(first.id, second.id, level);
276  });
277  LOG_DEBUG1 << "Sorting the " << SSTR(level) << "-gram level bucket idx: " << SSTR(bucket_idx) << " is done" << END_LOG;
278  }
