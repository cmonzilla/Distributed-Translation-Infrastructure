269  virtual void post_m_grams(const TModelLevel level) {
270  //Call the base class method first
271  if (BASE::is_post_grams(level)) {
272  BASE::post_m_grams(level);
273  }
274 
275  //Compute the m-gram index
276  const TModelLevel mgram_idx = (level - BASE::MGRAM_IDX_OFFSET);
277 
278  LOG_DEBUG2 << "Running post actions on " << level << "-grams, m-gram array index: " << mgram_idx << END_LOG;
279 
280  //Sort the entries per context with respect to the word index
281  //the order could be arbitrary if a non-basic word index is used!
282  const size_t num_prev_ctx = (level == M_GRAM_LEVEL_2) ? m_one_gram_arr_size : m_M_N_gram_num_ctx_ids[mgram_idx - 1];
283  LOG_DEBUG2 << "Number of previous contexts: " << num_prev_ctx << END_LOG;
284  for (size_t ctxId = 0; ctxId < num_prev_ctx; ++ctxId) {
285  const TSubArrReference & info = m_M_gram_ctx_2_data[mgram_idx][ctxId];
286  if (info.beginIdx != BASE::UNDEFINED_ARR_IDX) {
287  LOG_DEBUG3 << "Sorting for context id: " << ctxId << ", info.beginIdx: "
288  << info.beginIdx << ", info.endIdx: " << info.endIdx << END_LOG;
289  my_sort<TWordIdPBEntry>(&m_M_gram_data[mgram_idx][info.beginIdx], (info.endIdx - info.beginIdx) + 1);
290  }
291  }
292  }

294  virtual void post_n_grams() {
295  //Call the base class method first

300  LOG_DEBUG2 << "Sorting the N-gram's data: ptr: " << m_N_gram_data

303  //Order the N-gram array as it is unordered and we will binary search it later!
