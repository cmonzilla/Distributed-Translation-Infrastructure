218  virtual TProbBackOffEntry& make_M_GramDataRef(const TModelLevel level, const TShortId wordId, const TLongId ctxId) {
219  //Compute the m-gram index
220  const TModelLevel mgram_idx = level - ALayeredTrie<N>::MGRAM_IDX_OFFSET;
221 
222  LOG_DEBUG2 << "Adding " << SSTR(level) << "-gram with ctxId: "
223  << SSTR(ctxId) << ", wordId: " << SSTR(wordId) << END_LOG;
224 
225  //First get the sub-array reference.
226  TSubArrReference & ref = m_M_gram_ctx_2_data[mgram_idx][ctxId];
227 
228  //Check that the array is continuous in indexes, so that we add
229  //context after context and not switching between different contexts!
230  if (DO_SANITY_CHECKS && (ref.endIdx != ALayeredTrie<N>::UNDEFINED_ARR_IDX)
231  && (ref.endIdx + 1 != m_M_N_gram_next_ctx_id[mgram_idx])) {
232  stringstream msg;
233  msg << "The " << SSTR(level) << " -gram ctxId: " << SSTR(ctxId)
234  << " array is not ordered ref.endIdx = " << SSTR(ref.endIdx)
235  << ", next ref.endIdx = " << SSTR(m_M_N_gram_next_ctx_id[mgram_idx] + 1) << "!";
236  throw Exception(msg.str());
237  }
238 
239  //Get the new index and increment - this will be the new end index
240  ref.endIdx = m_M_N_gram_next_ctx_id[mgram_idx]++;
241 
242  //Check if we exceeded the maximum allowed number of M-grams
243  if (DO_SANITY_CHECKS && (ref.endIdx >= m_M_N_gram_num_ctx_ids[mgram_idx])) {
244  stringstream msg;
245  msg << "The maximum allowed number of " << SSTR(level) << "-grams: "
246  << SSTR(m_M_N_gram_num_ctx_ids[mgram_idx]) << " is exceeded )!";
247  throw Exception(msg.str());
248  }
249 
250  //Check if there are yet no elements for this context
251  if (ref.beginIdx == ALayeredTrie<N>::UNDEFINED_ARR_IDX) {
252  //There was no elements put into this context, the begin index is then equal to the end index
253  ref.beginIdx = ref.endIdx;
254  }
255 
256  //Store the word id
257  m_M_gram_data[mgram_idx][ref.endIdx].id = wordId;
258 
259  //Return the reference to the newly allocated element
260  return m_M_gram_data[mgram_idx][ref.endIdx].data;
261  };

358  virtual void post_m_grams(const TModelLevel level) {
359  //Call the base class method first
360  ATrie<N>::post_m_grams(level);
361 
362  //Compute the m-gram index
363  const TModelLevel mgram_idx = (level - ALayeredTrie<N>::MGRAM_IDX_OFFSET);
364 
365  LOG_DEBUG2 << "Running post actions on " << level << "-grams, m-gram array index: " << mgram_idx << END_LOG;
366 
367  //Sort the entries per context with respect to the word index
368  //the order could be arbitrary if a non-basic word index is used!
369  const size_t num_prev_ctx = (level == M_GRAM_LEVEL_2) ? m_one_gram_arr_size : m_M_N_gram_num_ctx_ids[mgram_idx - 1];
370  LOG_DEBUG2 << "Number of previous contexts: " << num_prev_ctx << END_LOG;
371  for (size_t ctxId = 0; ctxId < num_prev_ctx; ++ctxId) {
372  const TSubArrReference & info = m_M_gram_ctx_2_data[mgram_idx][ctxId];
373  if (info.beginIdx != ALayeredTrie<N>::UNDEFINED_ARR_IDX) {
374  LOG_DEBUG3 << "Sorting for context id: " << ctxId << ", info.beginIdx: "
375  << info.beginIdx << ", info.endIdx: " << info.endIdx << END_LOG;
376  my_sort<TWordIdPBEntry>(&m_M_gram_data[mgram_idx][info.beginIdx], (info.endIdx - info.beginIdx) + 1);
377  }
378  }

429  inline bool getContextId(const TShortId wordId, TLongId & ctxId, const TModelLevel level) {
430  //Compute the m-gram index
431  const TModelLevel mgram_idx = level - ALayeredTrie<N>::MGRAM_IDX_OFFSET;
432 
433  if (DO_SANITY_CHECKS && ((level == N) || (mgram_idx < 0))) {
434  stringstream msg;
435  msg << "Unsupported level id: " << level;
436  throw Exception(msg.str());
437  }
438 
439  LOG_DEBUG2 << "Searching for the next ctxId of " << SSTR(level)
440  << "-gram with wordId: " << SSTR(wordId) << ", ctxId: "
441  << SSTR(ctxId) << END_LOG;
442 
443  //First get the sub-array reference. Note that, even if it is the 2-Gram
444  //case and the previous word is unknown (ctxId == 0) we still can use
445  //the ctxId to get the data entry. The reason is that we allocated memory
446  //for it but being for an unknown word context it should have no data!
447  TSubArrReference & ref = m_M_gram_ctx_2_data[mgram_idx][ctxId];
448 
449  LOG_DEBUG2 << "Got context mapping for ctxId: " << SSTR(ctxId)
450  << ", with beginIdx: " << SSTR(ref.beginIdx) << ", endIdx: "
451  << SSTR(ref.endIdx) << END_LOG;
452 
453  //Check that there is data for the given context available
454  if (ref.beginIdx != ALayeredTrie<N>::UNDEFINED_ARR_IDX) {
455  TShortId nextCtxId = ALayeredTrie<N>::UNDEFINED_ARR_IDX;
456  //The data is available search for the word index in the array
457  //WARNING: The linear search here is much slower!!!
458  if (my_bsearch_id<TWordIdPBEntry>(m_M_gram_data[mgram_idx], ref.beginIdx, ref.endIdx, wordId, nextCtxId)) {
459  LOG_DEBUG2 << "The next ctxId for wordId: " << SSTR(wordId) << ", ctxId: "
460  << SSTR(ctxId) << " is nextCtxId: " << SSTR(nextCtxId) << END_LOG;
461  ctxId = nextCtxId;
462  return true;
463  } else {
464  LOG_DEBUG2 << "Unable to find M-gram ctxId for level: "
465  << SSTR(level) << ", prev ctxId: " << SSTR(ctxId)
466  << ", wordId: " << SSTR(wordId) << ", is not in the available range: ["
467  << SSTR(m_M_gram_data[mgram_idx][ref.beginIdx].id) << " ... "
468  << SSTR(m_M_gram_data[mgram_idx][ref.endIdx].id) << "]" << END_LOG;
469  return false;
470  }
471  } else {
472  LOG_DEBUG2 << "Unable to find M-gram context id for level: "
473  << SSTR(level) << ", prev ctxId: " << SSTR(ctxId)
474  << ", nothing present in that context!" << END_LOG;
475  return false;
476  }
477  }
