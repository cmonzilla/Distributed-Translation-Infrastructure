147  virtual TProbBackOffEntry & make_M_GramDataRef(const TModelLevel level, const TShortId wordId, TLongId ctxId) {
148  //Store the N-tires from length 2 on and indexing starts
149  //with 0, therefore "level-2". Get/Create the mapping for this

151 
152  //Obtain the context key and then create a new mapping
153  const TLongId key = TShortId_TShortId_2_TLongId(ctxId, wordId);

155  //Get the next context id

157  TShortId nextCtxId = m_M_gram_next_ctx_id[idx]++;

159  //Store the context mapping inside the map

162  //Return the reference to the piece of memory
