138  inline void trim(std::string& str,
139  const std::string& whitespace = WHITESPACES) {
140  LOG_DEBUG3 << "Trimming the string '" << str << "', with white spaces " << END_LOG;
141  if (str != "") {
142  const size_t strBegin = str.find_first_not_of(whitespace);
143  LOG_DEBUG3 << "First not of whitespaces pos: " << strBegin << END_LOG;
144 
145  if (strBegin == std::string::npos) {
146  str = ""; // no content
147  } else {
148  const size_t strEnd = str.find_last_not_of(whitespace);
149  LOG_DEBUG3 << "Last not of whitespaces pos: " << strEnd << END_LOG;
150  const size_t strRange = strEnd - strBegin + 1;
151  LOG_DEBUG3 << "Need a substring: [" << strBegin << ", " << (strBegin + strRange - 1) << "]" << END_LOG;
152 
153  str = str.substr(strBegin, strRange);
154  }
155  }
156  LOG_DEBUG3 << "The trimmed result is '" << str << "'" << END_LOG;
157  }

166  inline void reduce(std::string& str,
167  const std::string& fill = " ",
168  const std::string& whitespace = WHITESPACES) {
169  LOG_DEBUG3 << "Reducing the string '" << str << "', with white spaces" << END_LOG;
170  if (str != "") {
171  // trim first
172  trim(str, whitespace);
173 
174  // replace sub ranges
175  size_t beginSpace = str.find_first_of(whitespace);
176  while (beginSpace != std::string::npos) {
177  const size_t endSpace = str.find_first_not_of(whitespace, beginSpace);
178  const size_t range = endSpace - beginSpace;
179 
180  str.replace(beginSpace, range, fill);
181 
182  const size_t newStart = beginSpace + fill.length();
183  beginSpace = str.find_first_of(whitespace, newStart);
184  }
185  }
186  LOG_DEBUG3 << "The reduced result is '" << str << "'" << END_LOG;
187  }
188  }
