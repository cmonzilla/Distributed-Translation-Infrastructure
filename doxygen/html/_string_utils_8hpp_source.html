173  inline void reduce(std::string& str,
174  const std::string& fill = " ",
175  const std::string& whitespace = WHITESPACES) {
176  LOG_DEBUG4 << "Reducing the string '" << str << "', with white spaces" << END_LOG;
177  if (str != "") {
178  // trim first
179  trim(str, whitespace);
180 
181  // replace sub ranges
182  size_t beginSpace = str.find_first_of(whitespace);
183  while (beginSpace != std::string::npos) {
184  const size_t endSpace = str.find_first_not_of(whitespace, beginSpace);
185  const size_t range = endSpace - beginSpace;
186 
187  str.replace(beginSpace, range, fill);
188 
189  const size_t newStart = beginSpace + fill.length();
190  beginSpace = str.find_first_of(whitespace, newStart);
191  }
192  }
193  LOG_DEBUG4 << "The reduced result is '" << str << "'" << END_LOG;
194  }

224  inline bool fast_stoT(T & r, const char *p) {
225  r = 0.0;
226  int c = 0; // counter to check how many numbers we got!
227 
228  // Get the sign!
229  bool neg = false;
230  if (*p == '-') {
231  neg = true;
232  ++p;
233  } else if (*p == '+') {
234  neg = false;
235  ++p;
236  }
237 
238  // Get the digits before decimal point
239  while (valid_digit(*p)) {
240  r = (r * 10.0) + (*p - '0');
241  ++p;
242  ++c;
243  }
244 
245  // Get the digits after decimal point
246  if (*p == '.') {
247  T f = 0.0;
248  T scale = 1.0;
249  ++p;
250  while (valid_digit(*p)) {
251  f = (f * 10.0) + (*p - '0');
252  ++p;
253  scale *= 10.0;
254  ++c;
255  }
256  r += f / scale;
257  }
258 
259  // FIRST CHECK:
260  if (c == 0) {
261  return false;
262  } // we got no dezimal places! this cannot be any number!
263 
264 
265  // Get the digits after the "e"/"E" (exponenet)
266  if (*p == 'e' || *p == 'E') {
267  unsigned int e = 0;
268 
269  bool negE = false;
270  ++p;
271  if (*p == '-') {
272  negE = true;
273  ++p;
274  } else if (*p == '+') {
275  negE = false;
276  ++p;
277  }
278  // Get exponent
279  c = 0;
280  while (valid_digit(*p)) {
281  e = (e * 10) + (*p - '0');
282  ++p;
283  ++c;
284  }
285  if (!neg && e > (uint) std::numeric_limits<T>::max_exponent10) {
286  e = (uint) std::numeric_limits<T>::max_exponent10;
287  } else if (e < (uint) std::numeric_limits<T>::min_exponent10) {
288  e = (uint) std::numeric_limits<T>::max_exponent10;
289  }
290  // SECOND CHECK:
291  if (c == 0) {
292  return false;
293  } // we got no exponent! this was not intended!!
294 
295  T scaleE = 1.0;
296  // Calculate scaling factor.
297 
298  while (e >= 50) {
299  scaleE *= 1E50;
300  e -= 50;
301  }
302  //while (e >= 8) { scaleE *= 1E8; e -= 8; }
303  while (e > 0) {
304  scaleE *= 10.0;
305  e -= 1;
306  }
307 
308  if (negE) {
309  r /= scaleE;
310  } else {
311  r *= scaleE;
312  }
313  }
314 
315  // Apply sign to number
316  if (neg) {
317  r = -r;
318  }
319 
320  return true;
321  }
322  }
