193  static inline TReferenceHashSize createContext( vector<TWordHashSize> & hashes, const TTrieSize cLevel ) throw(Exception){
194  const TTrieSize currMaxIdx = (cLevel - MINIMUM_CONTEXT_LEVEL);
195  //Define and default initialize the context value
196  TReferenceHashSize context = hashes.at(currMaxIdx);
197  LOG_DEBUG << "initializing context = " << context << END_LOG;
198 
199  if( currMaxIdx > 0 ) {
200  LOG_DEBUG << "There is more than one element to create context from!" << END_LOG;
201  //If there is more than one element we need to create a hash for then iterate
202  for(TTrieSize idx = currMaxIdx; idx > 0; idx-- ) {
203  LOG_DEBUG << "context( " << hashes.at(idx-1) << ", " << context << " )" << END_LOG;
204  context = createContext(hashes.at(idx-1), context);
205  LOG_DEBUG << " = " << context << END_LOG;
206  }
207  } else {
208  if( currMaxIdx < 0 ) {
209  throw Exception("Unable to compute context of level <=1. It does not exist!");
210  }
211  }
212 
213  return context;
214  }
