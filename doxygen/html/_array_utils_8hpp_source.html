295  LOG_DEBUG2 << "Searching between indexes " << l_idx << " and " << u_idx << END_LOG;
296 
297  if (key > array[u_idx]) {
298  LOG_DEBUG3 << "key (+1) value@" << (uint32_t) u_idx << END_LOG;
299  //The key is larger than the last id so it is not in the array
300  return false;
301  } else {
302  if (key == array[u_idx]) {
303  LOG_DEBUG3 << "key (0) value@" << (uint32_t) u_idx << END_LOG;
304  //The key is equal to the last id so we found it!
305  found_pos = u_idx;
306  return true;
307  } else {
308  //The key is potentially inside array and it is not the last element!
309  for (found_pos = l_idx; found_pos < u_idx; ++found_pos) {
310  if (key < array[found_pos]) {
311  LOG_DEBUG3 << "key (-1) value@" << (uint32_t) found_pos << END_LOG;
312  //We bypassed the place where the value could have been
313  return false;
314  } else {
315  if (key == array[found_pos]) {
316  LOG_DEBUG3 << "key (0) value@" << (uint32_t) found_pos << END_LOG;
317  //We found the key!
318  return true;
319  }
320  }
321  }
322  }
323  }
324  //The key is outside the array values, no reason to search further
325  return false;

356  size, sizeof (ARR_ELEM_TYPE), [] (const void * p_a, const void * p_b) -> int {
357  const KEY_TYPE & v_a = *(static_cast<const KEY_TYPE*> (p_a));
358  const KEY_TYPE & v_b = (const KEY_TYPE) *(static_cast<const ARR_ELEM_TYPE*> (p_b));

360  if (v_a < v_b) {
361  return -1;
362  } else {
363  if (v_a > v_b) {

365  } else {
366  return 0;
367  }
368  }
369  }));
370 
371  //The resulting index is the difference between
372  //pointers, if the elements was found
373  if (result != NULL) {
374  //The found position is with respect to the beginning of the entire array
375  mid_pos = (INDEX_TYPE) (result - array);
376  LOG_DEBUG4 << "The element " << key << " is found, array[ "
