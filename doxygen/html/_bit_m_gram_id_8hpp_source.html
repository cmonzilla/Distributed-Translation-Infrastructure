54  //The memory in bits needed to store different M-gram id types in
55  //the M-gram id byte arrays
56  //These are the minimum values, if used then we use fewest memory
57  //but the data is not byte aligned. Therefore some bit copying operations
58  //are not done efficiently. Perhaps it is worth trying to round these
59  //values up to full bytes, this can improve performance @ some memory costs.

61  //The number of bites needed to store a 2-gram id type
62  //Possible id types: 32^2 = 1,024
63  //The number of bits needed to store the type is log_2(1,024) = 10
64  static const uint8_t M_GRAM_2_ID_TYPE_LEN_BITS = 10;
65  //The number of bites needed to store a 3-gram id type
66  //Possible id types: 32^3 = 32,768
67  //The number of bits needed to store the type is log_2(32,768) = 15
68  static const uint8_t M_GRAM_3_ID_TYPE_LEN_BITS = 15;
69  //The number of bites needed to store a 4-gram id type
70  //Possible id types: 32^4 = 1,048,576
71  //The number of bits needed to store the type is log_2(1,048,576) = 20
72  static const uint8_t M_GRAM_4_ID_TYPE_LEN_BITS = 20;
73  //The number of bites needed to store a 5-gram id type
74  //Possible id types: 32^5 = 33,554,432
75  //The number of bits needed to store the type is log_2(33,554,432) = 25
76  static const uint8_t M_GRAM_5_ID_TYPE_LEN_BITS = 25;
77  //The number of bites needed to store a 6-gram id type
78  //Possible id types: 32^6 = 1,073,741,824
79  //The number of bits needed to store the type is log_2(1,073,741,824) = 30

92  //Stores the maximum number of bits up to and including M-grams
93  //of level 6. We use sizeof (TShortId) as each wordId is of type
94  //TShortId, and the maximum number of bits is thus defined by the
95  //number of word_ids in the M-gram and their max size in bytes.
96  static constexpr uint8_t M_GRAM_MAX_ID_LEN_BYTES[] = {
97  0, 0,
98  2 * sizeof (TShortId) + NUM_BYTES_4_BITS(M_GRAM_2_ID_TYPE_LEN_BITS), // 2 TShortId values for 2 word ids, plus the memory needed to store type
99  3 * sizeof (TShortId) + NUM_BYTES_4_BITS(M_GRAM_3_ID_TYPE_LEN_BITS), // 3 TShortId values for 3 word ids, plus the memory needed to store type
100  4 * sizeof (TShortId) + NUM_BYTES_4_BITS(M_GRAM_4_ID_TYPE_LEN_BITS), // 4 TShortId values for 4 word ids, plus the memory needed to store type
101  5 * sizeof (TShortId) + NUM_BYTES_4_BITS(M_GRAM_5_ID_TYPE_LEN_BITS), // 5 TShortId values for 5 word ids, plus the memory needed to store type
102  6 * sizeof (TShortId) + NUM_BYTES_4_BITS(M_GRAM_6_ID_TYPE_LEN_BITS) // 6 TShortId values for 6 word ids, plus the memory needed to store type
103  };
