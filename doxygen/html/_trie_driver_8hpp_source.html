142  void execute(TMGramQuery & query) {
143  LOG_DEBUG << "Starting to execute:" << tokensToString(query.m_gram) << END_LOG;
144 
145  //Make sure that the query is prepared for execution
146  query.prepare_query();
147 
148  //Compute the probability in the loop fashion, should be faster that recursion.
149  while ((query.result.prob == ZERO_PROB_WEIGHT) && (!DO_SANITY_CHECKS || (query.curr_level != 0))) {
150  //Try to compute the next probability with decreased level
151  cache_check_get_prob_weight(query);
152  //Decrease the level
153  query.curr_level--;
154  }
155 
156  //If the probability is log-zero or snaller then there is no
157  //need for a back-off as then we will only get smaller values.
158  if (query.result.prob > ZERO_LOG_PROB_WEIGHT) {
159  //If the curr_level is smaller than the original level then
160  //it means that we needed to back-off, add back-off weights
161  for (++query.curr_level; query.curr_level != query.m_gram.level; ++query.curr_level) {
162  //Get the back_off
163  cache_check_add_back_off_weight(query);
164  }
165  }
166 
167  LOG_DEBUG << "The computed log_" << LOG_PROB_WEIGHT_BASE
168  << " probability is: " << query.result.prob << END_LOG;
169  }

184  void cache_check_get_prob_weight(TMGramQuery & query) {
185  LOG_DEBUG << "cache_check_add_prob_value(" << query.curr_level
186  << ") = " << query.result.prob << END_LOG;
187 
188  //Try getting the probability value.
189  //1. If the level is one go on: we can get smth
190  //even if the 1-Gram consists of just an unknown word.
191  //2. If the context length is more then one and there is
192  //an unknown word in the gram then it makes no sense to do
193  //searching as there are no M-grams with <unk> in them
194  if ((query.curr_level == M_GRAM_LEVEL_1) ||
195  ((query.curr_level > M_GRAM_LEVEL_1)
196  && query.template has_no_unk_words<false>()
197  && m_trie.template is_bitmap_hash_cache<false>(query))) {
198  //Let's look further, may be we will find something!
199  LOG_DEBUG1 << "All pre-checks are passed, calling add_prob_value(level, prob)!" << END_LOG;
200  m_trie.get_prob_weight(query);
201  } else {
202  LOG_DEBUG << "Could try to get probs but it will not be "
203  << "successful due to the present unk words! "
204  << "Thus backing off right away!" << END_LOG;
205  }
206  }

214  void cache_check_add_back_off_weight(TMGramQuery & query) {
215  LOG_DEBUG << "cache_check_add_back_off_weight(" << query.curr_level
216  << ") = " << query.result.prob << END_LOG;
217 
218  //Try getting the back-off weight.
219  //1. If the context length is one go on: we can get smth
220  //even if the 1-Gram consists of just an unknown word.
221  //2. If the context length is more then one and there is
222  //an unknown word in the gram then it makes no sense to do
223  //searching as there are no M-grams with <unk> in them
224  if ((query.curr_level == M_GRAM_LEVEL_1) ||
225  ((query.curr_level > M_GRAM_LEVEL_1)
226  && query.template has_no_unk_words<true>()
227  && m_trie.template is_bitmap_hash_cache<true>(query))) {
228  //Let's look further, we definitely get some back-off weight or zero!
229  LOG_DEBUG1 << "All pre-checks are passed, calling add_back_off_weight(level, prob)!" << END_LOG;
230  m_trie.add_back_off_weight(query);
231  } else {
232  LOG_DEBUG << "Could try to back off but it will not be "
233  << "successful due to the present unk words! Thus "
234  << "the back-off weight is zero!" << END_LOG;
235  }
236  }
