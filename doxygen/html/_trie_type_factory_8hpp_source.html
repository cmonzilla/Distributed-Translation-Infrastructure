73  static inline AWordIndex * getWordIndex(const WordIndexTypesEnum word_index_type) {
74  const size_t memory_factor = __HashMapWordIndex::MEMORY_FACTOR;
75  switch (word_index_type) {
76  case WordIndexTypesEnum::BASIC_WORD_INDEX:
77  LOG_USAGE << "Using BASIC_WORD_INDEX to instantiate the trie." << END_LOG;
78  return new BasicWordIndex(memory_factor);
79  case WordIndexTypesEnum::COUNTING_WORD_INDEX:
80  LOG_USAGE << "Using COUNTING_WORD_INDEX to instantiate the trie." << END_LOG;
81  return new CountingWordIndex(memory_factor);
82  case WordIndexTypesEnum::OPTIMIZING_BASIC_WORD_INDEX:
83  LOG_USAGE << "Using OPTIMIZING_BASIC_WORD_INDEX to instantiate the trie." << END_LOG;
84  return new OptimizingWordIndex(new BasicWordIndex(memory_factor));
85  case WordIndexTypesEnum::OPTIMIZING_COUNTING_WORD_INDEX:
86  LOG_USAGE << "Using OPTIMIZING_COUNTING_WORD_INDEX to instantiate the trie." << END_LOG;
87  return new OptimizingWordIndex(new CountingWordIndex(memory_factor));
88  default:
89  stringstream msg;
90  msg << "Unrecognized word index type: " << word_index_type;
91  throw Exception(msg.str());
92  }
93  }
