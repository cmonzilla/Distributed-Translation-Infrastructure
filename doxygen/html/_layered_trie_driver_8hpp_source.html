210  template<bool is_back_off>
211  inline bool get_query_context_Id(const TMGramQuery & query, TLongId & ctx_id) {
212  const TModelLevel mgram_end_idx = (is_back_off ? (TrieType::max_level - 2) : (TrieType::max_level - 1));
213  const TModelLevel end_idx = mgram_end_idx;
214  const TModelLevel begin_idx = mgram_end_idx - (query.curr_level - 1);
215  TModelLevel idx = begin_idx;
216 
217  LOG_DEBUG1 << "Computing id of the " << SSTR(query.curr_level)
218  << "-gram " << (is_back_off ? "back-off" : "probability")
219  << " context" << END_LOG;
220 
221  //Compute the first words' hash
222  ctx_id = query.m_query_word_ids[idx];
223  LOG_DEBUG1 << "First word @ idx: " << SSTR(idx) << " has wordId: " << SSTR(ctx_id) << END_LOG;
224  idx++;
225 
226  //Compute the subsequent context ids
227  for (; idx < end_idx;) {
228  LOG_DEBUG1 << "Start searching ctx_id for m_query_word_ids[" << SSTR(idx) << "]: "
229  << SSTR(query.m_query_word_ids[idx]) << " prevCtxId: " << SSTR(ctx_id) << END_LOG;
230  if (m_trie.get_ctx_id(query.m_query_word_ids[idx], ctx_id, (idx - begin_idx) + 1)) {
231  LOG_DEBUG1 << "getContextId(" << SSTR(query.m_query_word_ids[idx])
232  << ", prevCtxId) = " << SSTR(ctx_id) << END_LOG;
233  idx++;
234  } else {
235  //The next context id could not be retrieved
236  return false;
237  }
238  }
239 
240  LOG_DEBUG1 << "Resulting id for the " << SSTR(query.curr_level)
241  << "-gram " << (is_back_off ? "back-off" : "probability")
242  << " context is: " << SSTR(ctx_id) << END_LOG;
243 
244  return true;
245  }

256  template<DebugLevelsEnum logLevel>
257  inline bool get_context_id(const T_M_Gram & gram, TLongId &ctxId) {
258  //Try to retrieve the context from the cache, if not present then compute it
259  if (getCachedContextId(gram, ctxId)) {
260  //Get the start context value for the first token
261  TShortId wordId = m_trie.get_word_index().get_word_id(gram.tokens[0]);
262 
263  //There is no id cached for this M-gram context - find it
264  if (wordId != WordIndexType::UNKNOWN_WORD_ID) {
265  //The first word id is the first context id
266  ctxId = wordId;
267  LOGGER(logLevel) << "ctxId = getId('" << gram.tokens[0].str()
268  << "') = " << SSTR(ctxId) << END_LOG;
269 
270  //Iterate and compute the hash:
271  for (int i = 1; i < (gram.level - 1); i++) {
272  wordId = m_trie.get_word_index().get_word_id(gram.tokens[i]);
273  if (wordId != WordIndexType::UNKNOWN_WORD_ID) {
274  LOGGER(logLevel) << "wordId = getId('" << gram.tokens[i].str()
275  << "') = " << SSTR(wordId) << END_LOG;
276  if (m_trie.get_ctx_id(wordId, ctxId, i + 1)) {
277  LOGGER(logLevel) << "ctxId = computeCtxId( "
278  << "wordId, ctxId ) = " << SSTR(ctxId) << END_LOG;
279  } else {
280  //The next context id could not be computed
281  LOGGER(logLevel) << "The next context for wordId: "
282  << SSTR(wordId) << " and ctxId: "
283  << SSTR(ctxId) << "on level: " << SSTR((i + 1))
284  << "could not be computed!" << END_LOG;
285  return false;
286  }
287  } else {
288  //The next word Id was not found, it is
289  //unknown, so we can stop searching
290  LOGGER(logLevel) << "The wordId for '" << gram.tokens[i].str()
291  << "' could not be found!" << END_LOG;
292  return false;
293  }
294  }
295 
296  //Cache the newly computed context id for the given n-gram context
297  setCacheContextId(gram, ctxId);
298 
299  //The context Id was found in the Trie
300  LOGGER(logLevel) << "The ctxId could be computed, "
301  << "it's value is: " << SSTR(ctxId) << END_LOG;
302  return true;
303  } else {
304  //The context id could not be computed as
305  //the first N-gram's word is already unknown
306  LOGGER(logLevel) << "The wordId for '" << gram.tokens[0].str()
307  << "' could not be found!" << END_LOG;
308  return false;
309  }
310  } else {
311  //The context Id was found in the cache
312  LOGGER(logLevel) << "The ctxId was found in cache, "
313  << "it's value is: " << SSTR(ctxId) << END_LOG;
314  return true;
315  }
316  }
