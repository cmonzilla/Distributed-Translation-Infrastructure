122  static void report_memory_usage(const char* action, TMemotyUsage msStart, TMemotyUsage msEnd, const bool isDoInfo) {
123  LOG_USAGE << "Action: \'" << action << "\' memory change:" << END_LOG;
124  LOG_DEBUG << "\tmemory before: vmsize=" << SSTR(msStart.vmsize) << " Kb, vmpeak="
125  << SSTR(msStart.vmpeak) << " Kb, vmrss=" << SSTR(msStart.vmrss)
126  << " Kb, vmhwm=" << SSTR(msStart.vmhwm) << " Kb" << END_LOG;
127  LOG_DEBUG << "memory after: vmsize=" << SSTR(msEnd.vmsize) << " Kb, vmpeak="
128  << SSTR(msEnd.vmpeak) << " Kb, vmrss=" << SSTR(msEnd.vmrss)
129  << " Kb, vmhwm=" << SSTR(msEnd.vmhwm) << " Kb" << END_LOG;
130 
131  int vmsize = ((msEnd.vmsize < msStart.vmsize) ? 0 : msEnd.vmsize - msStart.vmsize) / BYTES_ONE_MB;
132  int vmpeak = ((msEnd.vmpeak < msStart.vmpeak) ? 0 : msEnd.vmpeak - msStart.vmpeak) / BYTES_ONE_MB;
133  int vmrss = ((msEnd.vmrss < msStart.vmrss) ? 0 : msEnd.vmrss - msStart.vmrss) / BYTES_ONE_MB;
134  int vmhwm = ((msEnd.vmhwm < msStart.vmhwm) ? 0 : msEnd.vmhwm - msStart.vmhwm) / BYTES_ONE_MB;
135  LOG_USAGE << showpos << "vmsize=" << vmsize << " Mb, vmpeak=" << vmpeak
136  << " Mb, vmrss=" << vmrss << " Mb, vmhwm=" << vmhwm
137  << " Mb" << noshowpos << END_LOG;
138  if (isDoInfo) {
139  LOG_INFO << " vmsize - Virtual memory size; vmpeak - Peak virtual memory size" << END_LOG;
140  LOG_INFO << " Virtual memory size is how much virtual memory the process has in total (RAM+SWAP)" << END_LOG;
141  LOG_INFO << " vmrss - Resident set size; vmhwm - Peak resident set size" << END_LOG;
142  LOG_INFO << " Resident set size is how much memory this process currently has in main memory (RAM)" << END_LOG;
143  }
144  }

174  static string get_mgram_prob_str(const T_M_Gram & mgram) {
175  if (mgram.level == 1) {
176  return mgram.tokens[0].str().empty() ? "<empty>" : mgram.tokens[0].str();
177  } else {
178  if (mgram.level > 1) {
179  string result = mgram.tokens[mgram.level - 1].str() + " |";
180  for (TModelLevel idx = 0; idx < (mgram.level - 1); idx++) {

184  } else {

196  template<typename TrieType>
197  static void read_and_execute_queries(TrieType & trie, AFileReader &testFile) {
198  //Declare time variables for CPU times in seconds
199  double startTime = 0.0, endTime = 0.0;
200  //Will store the read line (word1 word2 word3 word4 word5)
201  TextPieceReader line;
202  //Will store the M-gram query and its internal state
203  MGramQuery < TrieType::max_level, typename TrieType::WordIndexType > query(trie.get_word_index());
204 
205  //Start the timer
206  startTime = StatisticsMonitor::getCPUTime();
207 
208  //Read the test file line by line
209  while (testFile.getLine(line)) {
210  LOG_DEBUG << "Got query line [ " << line.str() << " ]" << END_LOG;
211 
212  //Parse the line into an N-Gram
213  ARPAGramBuilder::gram_to_tokens(line, query.m_gram);
214 
215  //There can be an empty or "unreadable" line in the text file, just skip it ...
216  if (query.m_gram.level > 0) {
217 
218  //Query the Trie for the results
219  trie.execute(query);
220 
221  //Print the results:
222  LOG_RESULT << "log_" << LOG_PROB_WEIGHT_BASE << "( Prob( " << get_mgram_prob_str(query.m_gram) << " ) ) = " << SSTR(query.result.prob) << END_LOG;
223  LOG_INFO << "Prob( " << get_mgram_prob_str(query.m_gram) << " ) = " << SSTR(pow(LOG_PROB_WEIGHT_BASE, query.result.prob)) << END_LOG;
224  }
225  }
226 
227  //Stop the timer
228  endTime = StatisticsMonitor::getCPUTime();
229 
230  LOG_USAGE << "Total query execution time is " << (endTime - startTime) << " CPU seconds." << END_LOG;
231  }

243 
244  //Log the usage information
245  trie.log_trie_type_usage_info();
246 
247  LOG_USAGE << "Start creating and loading the Trie ..." << END_LOG;

249  StatisticsMonitor::getMemoryStatistics(memStatStart);

251  startTime = StatisticsMonitor::getCPUTime();
252  fill_in_trie(modelFile, trie);
253  LOG_DEBUG << "Getting the time statistics after creating the Trie ..." << END_LOG;

261  LOG_DEBUG << "Getting the memory statistics before closing the Model file ..." << END_LOG;
262  StatisticsMonitor::getMemoryStatistics(memStatStart);

265  LOG_DEBUG << "Getting the memory statistics after closing the Model file ..." << END_LOG;
266  StatisticsMonitor::getMemoryStatistics(memStatEnd);
267  //LOG_DEBUG << "Reporting on the memory consumption" << END_LOG;

269 
270  LOG_USAGE << "Start reading and executing the test queries ..." << END_LOG;
271  read_and_execute_queries(trie, testFile);

274  //Deallocate the trie

278  template<typename WordIndexType>
279  static void choose_trie_type_and_execute(const __Executor::TExecutionParams& params,
280  AFileReader &modelFile, AFileReader &testFile) {
281  switch (params.m_trie_type) {
282  case TrieTypesEnum::C2DH_TRIE:
283  execute < TrieDriver<LayeredTrieDriver<C2DHybridTrie<M_GRAM_LEVEL_MAX, WordIndexType>>> >(params, modelFile, testFile);
284  break;
285  case TrieTypesEnum::C2DM_TRIE:
286  execute < TrieDriver<LayeredTrieDriver<C2DMapTrie<M_GRAM_LEVEL_MAX, WordIndexType>>> >(params, modelFile, testFile);
287  break;
288  case TrieTypesEnum::C2WA_TRIE:
289  execute < TrieDriver<LayeredTrieDriver<C2WArrayTrie<M_GRAM_LEVEL_MAX, WordIndexType>>> >(params, modelFile, testFile);
290  break;
291  case TrieTypesEnum::G2DM_TRIE:
292  execute < TrieDriver<G2DMapTrie<M_GRAM_LEVEL_MAX, WordIndexType>> >(params, modelFile, testFile);
293  break;
294  case TrieTypesEnum::W2CA_TRIE:
295  execute < TrieDriver<LayeredTrieDriver<W2CArrayTrie<M_GRAM_LEVEL_MAX, WordIndexType>>> >(params, modelFile, testFile);
296  break;
297  case TrieTypesEnum::W2CH_TRIE:
298  execute < TrieDriver < LayeredTrieDriver<typename TW2CHybridTrie<M_GRAM_LEVEL_MAX, WordIndexType>::type>> > (params, modelFile, testFile);
299  break;
300  default:
301  stringstream msg;
302  msg << "Unrecognized trie type: " << params.m_trie_type;
303  throw Exception(msg.str());
304  }
305  }
