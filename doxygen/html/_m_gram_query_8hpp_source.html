111  void prepare_query() {

153  inline void store_unk_word_flags() {
154  const TModelLevel max_supp_level = (sizeof (PROB_UNK_MASKS) - 1);
155 
156  if (DO_SANITY_CHECKS && (N > max_supp_level)) {
157  stringstream msg;
158  msg << "store_unk_word_flags: Unsupported m-gram level: "
159  << SSTR(N) << ", must be <= " << SSTR(max_supp_level)
160  << "], insufficient m_unk_word_flags capacity!";
161  throw Exception(msg.str());
162  }
163 
164  //Re-initialize the flags with zero
165  m_unk_word_flags = 0;
166  //Fill in the values from the currently considered M-gram word ids
167  for (size_t idx = (N - m_gram.level); idx < N; ++idx) {
168  if (m_query_word_ids[idx] == AWordIndex::UNKNOWN_WORD_ID) {
169  m_unk_word_flags |= (1u << ((N - 1) - idx));
170  }
171  }
172 
173  LOG_DEBUG << "The query unknown word flags are: "
174  << bitset<NUM_BITS_IN_UINT_8>(m_unk_word_flags) << END_LOG;
175  }
