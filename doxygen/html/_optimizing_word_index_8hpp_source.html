337  inline void build_word_hash_buckets() {
338  uint32_t curr_idx = 0;
339  uint32_t next_idx = 0;
340 
341  //Go through the buckets array and initialize the begin indexes
342  for (size_t idx = 0; idx < m_num_bucket_maps; ++idx) {
343  if (DO_SANITY_CHECKS && (m_word_hash_buckets[idx] > 2)) {
344  LOG_WARNING << "A bucket with " << m_word_hash_buckets[idx] << " words is detected!" << END_LOG;
345  }
346 
347  next_idx = curr_idx + m_word_hash_buckets[idx];
348  m_word_hash_buckets[idx] = curr_idx;
