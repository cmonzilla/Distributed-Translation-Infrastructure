96  namespace __BitmapHashCache {
97  //The default number of buckets allocated for hash is equal to the
98  //number of M-grams in the Trie level. This is absolutely not enough
99  //As then the cache will be fully used and there will be only performance
100  //losses. To make the cache work we need much more buckets so that the
101  //queried M-grams have a very low chance to fall into the buckets with
102  //the Trie grams. So the bigger this number the better, yet the
103  //memory constraints. ALthough they are not crucial as we use bitmaps.
104  //NOTE: The experiments with C2WA showed a 5% performance improvement
105  //In the range of values 15-50. Yet, looking at memory consumption the
106  //optimum value was chosen to be 20 as with 15 it starts deteriorating
107  static const float BUCKET_MULTIPLIER_FACTOR = 20;
108  }
