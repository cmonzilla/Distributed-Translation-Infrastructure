176  inline TProbBackOffEntryPair& createMGramEntry(const TModelLevel level,
177  const TWordHashSize wordHash, const TReferenceHashSize contextHash) {
178  if (level == BGRAM_LEVEL_VALUE) {
179  //Store the Bi grams in a separate memory optimal place
180  TBGramEntryMap& bgamEntry = bGrams[wordHash];
181  //Get/Create the new Prob. and Back-Off entry pair in the map, for the context
182  return bgamEntry[static_cast<TWordHashSize> (contextHash)];
183  } else {
184  //Store the N-tires from length 3 on and indexing starts
185  //with 0, therefore "level-3". Get/Create the mapping for this
186  //word in the Trie level of the N-gram
187  TMGramEntryMap& ngamEntry = mGrams[level - MGRAM_IDX_OFFSET][wordHash];
188  //Get/Create the new Prob. and Back-Off entry pair in the map, for the context
189  return ngamEntry[contextHash];
190  }
191  }
