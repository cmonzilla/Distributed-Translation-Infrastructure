189  inline uint_fast32_t computePaulHsiehHash(const char * data, uint32_t len) {
190  uint_fast32_t hash = len, tmp;
191  int rem;
192 
193  if (len <= 0 || data == NULL) return 0;
194 
195  rem = len & 3;
196  len >>= 2;
197 
198  /* Main loop */
199  for (; len != 0; len--) {
200  hash += get16bits(data);
201  tmp = (get16bits(data + 2) << 11) ^ hash;
202  hash = (hash << 16) ^ tmp;
203  data += 2 * sizeof (uint16_t);
204  hash += hash >> 11;
205  }
206 
207  /* Handle end cases */
208  switch (rem) {
209  case 3: hash += get16bits(data);
210  hash ^= hash << 16;
211  hash ^= ((signed char) data[sizeof (uint16_t)]) << 18;
212  hash += hash >> 11;
213  break;
214  case 2: hash += get16bits(data);
215  hash ^= hash << 11;
216  hash += hash >> 17;
217  break;
218  case 1: hash += (signed char) *data;
219  hash ^= hash << 10;
220  hash += hash >> 1;
221  }
222 
223  /* Force "avalanching" of final 127 bits */
224  hash ^= hash << 3;
225  hash += hash >> 5;
226  hash ^= hash << 4;
227  hash += hash >> 17;
228  hash ^= hash << 25;
229  hash += hash >> 6;
230 
231  return hash;
232  }
