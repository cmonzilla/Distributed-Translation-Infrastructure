216  void query(const T_M_Gram & m_gram, TQueryResult & result) {
217  const TModelLevel level = m_gram.level;
218 
219  //Check the number of elements in the N-Gram
220  if (DO_SANITY_CHECKS && ((level < M_GRAM_LEVEL_1) || (level > N))) {
221  stringstream msg;
222  msg << "An improper N-Gram size, got " << level << ", must be between [1, " << N << "]!";
223  throw Exception(msg.str());
224  } else {
225  //Store the pointer to the M-gram
226  m_query_ptr = &m_gram;
227 
228  //Transform the given M-gram into word hashes.
229  store_m_gram_word_ids(m_gram);
230 
231  //Store unknown word flags
232  store_unk_word_flags();
233 
234  //Compute the probability in the loop fashion, should be faster that recursion.
235  //NOTE: Splitting this into two loops where one computes probability and the
236  //other one computes back-off weights if the probability is larger than log-zero
237  //probability did not show any significant performance improvements
238  TModelLevel curr_level = level;
239  result.prob = ZERO_PROB_WEIGHT;
240  while (!DO_SANITY_CHECKS || (curr_level != 0)) {
241  //Try to compute the next probability with decreased level
242  if (cache_check_add_prob_weight(curr_level, result.prob)) {
243  //If the probability has been finally computed stop
244  break;
245  } else {
246  //Decrease the level
247  curr_level--;
248  //Get the back_off
249  cache_check_add_back_off_weight(curr_level, result.prob);
250  }
251  }
252 
253  LOG_DEBUG << "The computed log_" << LOG_PROB_WEIGHT_BASE << " probability is: " << result.prob << END_LOG;
254  }
255  }

361  bool cache_check_add_prob_weight(const TModelLevel level, TLogProbBackOff & prob) {
362  LOG_DEBUG << "cache_check_add_prob_value(" << level << ") = " << prob << END_LOG;
363 
364  //Try getting the probability value.
365  //1. If the level is one go on: we can get smth
366  //even if the 1-Gram consists of just an unknown word.
367  //2. If the context length is more then one and there is
368  //an unknown word in the gram then it makes no sense to do
369  //searching as there are no M-grams with <unk> in them
370  if ((level == M_GRAM_LEVEL_1) ||
371  ((level > M_GRAM_LEVEL_1) && has_no_unk_words<false>(level)
372  && is_bitmap_hash_cache<false>(level))) {
373  //Let's look further, may be we will find something!
374  LOG_DEBUG1 << "All pre-checks are passed, calling add_prob_value(level, prob)!" << END_LOG;
375  return add_prob_weight(level, prob);
376  } else {
377  LOG_DEBUG << "Could try to get probs but it will not be "
378  << "successful due to the present unk words! "
379  << "Thus backing off right away!" << END_LOG;
380  //We already know we need to back-off
381  return false;
382  }
383  }

392  void cache_check_add_back_off_weight(const TModelLevel level, TLogProbBackOff & prob) {
393  LOG_DEBUG << "cache_check_add_back_off_weight(" << level << ") = " << prob << END_LOG;

395  //Try getting the back-off weight.
396  //1. If the context length is one go on: we can get smth
397  //even if the 1-Gram consists of just an unknown word.

458  inline void store_unk_word_flags() {
459  const TModelLevel max_supp_level = (sizeof (PROB_UNK_MASKS) - 1);
460 
461  if (DO_SANITY_CHECKS && (N > max_supp_level)) {
462  stringstream msg;
463  msg << "store_unk_word_flags: Unsupported m-gram level: "
464  << SSTR(N) << ", must be <= " << SSTR(max_supp_level)
465  << "], insufficient m_unk_word_flags capacity!";
466  throw Exception(msg.str());
467  }
468 
469  //Re-initialize the flags with zero
470  m_unk_word_flags = 0;
471  //Fill in the values from the currently considered M-gram word ids
472  for (size_t idx = (N - m_query_ptr->level); idx < N; ++idx) {
473  if (m_tmp_word_ids[idx] == AWordIndex::UNKNOWN_WORD_ID) {
474  m_unk_word_flags |= (1u << ((N - 1) - idx));
475  }
476  }
477 
478  LOG_DEBUG << "The query unknown word flags are: "
479  << bitset<NUM_BITS_IN_UINT_8>(m_unk_word_flags) << END_LOG;
