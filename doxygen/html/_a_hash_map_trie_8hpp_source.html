227  inline TReferenceHashSize computeHashContext(const TModelLevel contextLength, bool isBackOff) {
228  const TModelLevel mGramEndIdx = (isBackOff ? (N - 2) : (N - 1));
229  const TModelLevel eIdx = mGramEndIdx;
230  const TModelLevel bIdx = mGramEndIdx - contextLength;
231  TModelLevel idx = bIdx;
232 
233  LOG_DEBUG3 << "Computing context hash for context length " << SSTR(contextLength)
234  << " for a " << (isBackOff ? "back-off" : "probability")
235  << " computation" << END_LOG;
236 
237  //Compute the first words' hash
238  TReferenceHashSize contextHash = mGramWordHashes[idx];
239  LOG_DEBUG3 << "Word: " << SSTR(idx) << " hash == initial context hash: " << SSTR(contextHash) << END_LOG;
240  idx++;
241 
242  //Compute the subsequent hashes
243  for (; idx < eIdx;) {
244  contextHash = createContext(mGramWordHashes[idx], contextHash);
245  LOG_DEBUG3 << "Idx: " << SSTR(idx) << ", createContext(" << SSTR(mGramWordHashes[idx]) << ", prevContextHash) = " << SSTR(contextHash) << END_LOG;
246  idx++;
247  }
248 
249  LOG_DEBUG3 << "Resulting context hash for context length " << SSTR(contextLength)
250  << " of a " << (isBackOff ? "back-off" : "probability")
251  << " computation is: " << SSTR(contextHash) << END_LOG;
252 
253  return contextHash;
254  }

262  inline TReferenceHashSize computeHashContext(const vector<string> & tokens) {

362  void recordAndCheck(const TWordHashSize wordHash,
363  const TReferenceHashSize contextHash, const SBackOffNGram &gram) const {
364  //First try to get the entries for the given word
365  try {
366  unordered_map<TReferenceHashSize, vector < string>> &entries = ngRecorder.at(wordHash);
367  //Second try to get the entries for the given context
368  try {
369  vector<string> entry = entries.at(contextHash);
370 
371  //If we could get the values, then it is important to check
372  //that the length is the same. If it is then we have context
373  //hash collisions for the same length N-grams and this must
374  //not be happening! Then we will print a lot of debug info!
375  const TModelLevel size = entry.size();
376  if (size == gram.tokens.size()) {
377  LOG_WARNING << "N-gram collision/duplicates: '" << tokensToString(entry) << "' with '"
378  << tokensToString(gram.tokens) << "'! wordHash= " << SSTR(wordHash)
379  << ", contextHash= " << SSTR(contextHash) << END_LOG;
380 
381  TWordHashSize old[N], fresh[N];
382  AHashMapTrie<N>::tokensToHashes(entry, old);
383  AHashMapTrie<N>::tokensToHashes(gram.tokens, fresh);
384  for (int i = 0; i < size; i++) {
385  LOG_INFO << SSTR(i) << ") wordHash('" << SSTR(entry[i]) << "') = " << SSTR(old[N - size + i]) << END_LOG;
386  LOG_INFO << SSTR(i) << ") wordHash('" << SSTR(gram.tokens[i]) << "') = " << SSTR(fresh[N - size + i]) << END_LOG;
387  }
388  LOG_INFO << "-- First context computation: " << END_LOG;
389  AHashMapTrie<N>::template computeHashContext<Logger::INFO>(entry);
390  LOG_INFO << "-- Second context computation: " << END_LOG;
391  AHashMapTrie<N>::template computeHashContext<Logger::INFO>(gram.tokens);
392  }
393  } catch (out_of_range e) {
394  //If no entries for the context, add a new one
395  entries[contextHash] = gram.tokens;
396  }
397  } catch (out_of_range e) {
398  //If no entries for the given word and thus context add a new one
399  ngRecorder[wordHash][contextHash] = gram.tokens;
400  }
401  }
