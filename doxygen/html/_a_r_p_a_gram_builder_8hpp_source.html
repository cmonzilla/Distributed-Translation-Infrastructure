84  static inline bool gram_line_to_tokens(TextPieceReader &text, TextPieceReader * tokens, const TModelLevel level) {
85  TextPieceReader storage;
86  //First read text until the first tab, it should be present if it is a M-gram line
87  if (text.getTab(storage)) {
88  //There should be some text left it it is an M-gram
89  if (text.hasMore()) {
90  //Read until the next tab or end of line into the storage
91  if (text.getTab(storage)) {
92  //Note storage should contain all the space separated M-gram tokens, read them
93  TModelLevel idx = 0;
94  while (storage.getSpace(tokens[idx])) {
95  idx++;
96  }
97  //We should read exactly as many tokens as expected
98  if (idx == level) {
99  //We read as many tokens as there should be
100  return true;
101  } else {
102  //We read fewer tokens! This is not an M-gram we expected
103  LOG_WARNING << "Read only " << SSTR(idx) << " words from a "
104  << SSTR(level) << "-gram: [" << text.str()
105  << "], expected: " << SSTR(level) << END_LOG;
106  return false;
107  }
108  } else {
109  //Unexpected end of text
110  LOG_WARNING << "An unexpected end of line '" << text.str()
111  << "' when reading the " << level << "-gram!" << END_LOG;
112  return false;
113  }
114  } else {
115  LOG_DEBUG1 << " Encountered [" << SSTR(text.str())
116  << "] while trying to parse an " << SSTR(level)
117  << "-gram." << END_LOG;
118  //There is nothing left, this is definitely not an M-gram!
119  return false;
120  }
121  } else {
122  //Unexpected end of text
123  LOG_WARNING << "An unexpected end of line '" << text.str()
124  << "' when reading the " << level << "-gram!" << END_LOG;
125  return false;
126  }
127  }
