64 void Logger::updateProgressBar(){
65  if( currLEvel <= INFO ) {
66  const double currProgressUpdate = StatisticsMonitor::getCPUTime();
67  if( (currProgressUpdate - lastProgressUpdate ) > PROGRESS_UPDATE_PERIOD ) {
68  currProgCharIdx = (currProgCharIdx + 1) % numProgChars;
69  cout << progressChars[progressChars.size()-1] << progressChars[currProgCharIdx];
70  cout.flush();
71  lastProgressUpdate = currProgressUpdate;
72  }
73  }
74 }
