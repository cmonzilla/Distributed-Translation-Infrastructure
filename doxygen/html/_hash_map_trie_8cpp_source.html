53  void HashMapTrie<N, doCache>::addWords(vector<string> &tokens) {
54  //Add the words to the trie and update frequencies;
55  for (vector<string>::iterator it = tokens.begin(); it != tokens.end(); ++it) {
56  //Insert a new or get an existing entry
57  const string & token = *it;
58  TWordHashSize hash = computeHash(token);
59  TWordEntryPair & entry = words[hash];
60  if (entry.second == 0) {
61  entry.first = token;
62  } else {
63  if (entry.first.compare(token)) {
64  LOG_ERROR << "Hash collision: '" << token << "' and '" << entry.first << "' both have hash " << hash << END_LOG;
65  }
66  }
67  //Update/increase the frequency
68  entry.second++;
69  LOG_DEBUG << "freq( " << hash << " ) = " << entry.second << END_LOG;
70  }

107  template<TTrieSize N, bool doCache>
108  void HashMapTrie<N, doCache>::queryWordFreqs(TWordHashSize hash, SFrequencyResult<N> & wrap) {
109  TFrequencySize wordFreq = 0;
110  //First check if the given word is present at all, i.e. consider the 1-grams
111  try {
112  //Set the word's frequency, 0-gram
113  TWordEntryPair & entry = words.at(hash);
114  wrap.result[0] = entry.second;
115 
116  //Set the N-gram frequencies for N > 0, once an exception occurs this
117  //means the the next level's N-gram is not present, so we can stop
118  //searching already as there are definitely no occurrences of this word
119  //as the last one in higher level N-grams.
120  for (int idx = 1; idx < N; idx++) {
121  //Now go through all of the N-grams ending with
122  //the given word and sum-up their frequencies
123  TNTrieEntryPairsMap & entry = data[idx - 1].at(hash);
124  for (auto it = entry.cbegin(); it != entry.cend(); ++it) {
125  wrap.result[idx] += it->second;
126  }
127  }
128  } catch (out_of_range e) {
129  //DO NOTHING! This just means that the NGram is not found so we can simply stop searching
130  }
131  }

147  if (HashMapTrie<N, doCache>::doesQueryCache()) {
148  //Convert the word into it's cache
149  TWordHashSize hash = computeHash(word);

178  //Put it inside the hash

180 
181  LOG_DEBUG << "-- Computed the level's " << L << " next " << N - L << "-gram level word hash " << startWordHash << END_LOG;
182 
183  //Compute the context of the startWord on this level for this level's N-gram (L-gram))
184  TReferenceHashSize context = createContext(hashes, L);
185 
186  LOG_DEBUG << "-- The level " << L << " context is " << context << END_LOG;
187 
188  //BasicLogger::printDebug("Getting data[%u][%u][%u] = %u", L-1, endWordHash, context, data[L-MINIMUM_CONTEXT_LEVEL][endWordHash][context]);
189 
190  //Get the L-gram's frequency
191  freqs.result[N - L] = entry.at(context);
192 
193  LOG_DEBUG << "-- The level " << context << " frequency " << freqs.result[N - L] << " is found and stored at index " << N - L << END_LOG;
194 
195  //In case the maximum level N is not reached, do recursion
196  if (L < N) {
197  queryNGramFreqs(endWordHash, L + 1, ngram, hashes, freqs);

203  void HashMapTrie<N, doCache>::queryNGramFreqs(const vector<string> & ngram, SFrequencyResult<N> & freqs) {
204  //First just clean the array
205  fill(freqs.result, freqs.result + N, 0);
206  //This vector will store the N-gram's word hashes
207  //for being re-used in the recursive calls
208  vector<TWordHashSize> hashes;
209 
210  try {
211  //Compute the hash of the last word in the Ngram
212  TWordHashSize endWordHash = computeHash(ngram[N - 1]);
213 
214  //Get the last 1-gram's word frequency
215  freqs.result[N - 1] = words.at(endWordHash).second;
216 
217  //Now perform a recursive procedure for finding
218  //frequencies of all longer N-grams with N >= 2
219  queryNGramFreqs(endWordHash, MINIMUM_CONTEXT_LEVEL, ngram, hashes, freqs);
220  } catch (out_of_range e) {
