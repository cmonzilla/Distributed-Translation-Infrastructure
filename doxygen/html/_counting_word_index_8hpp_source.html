127  virtual void do_post_word_count() {
128  //All the words have been filled in, it is time to give them ids.
129  LOG_DEBUG1 << "Starting the post word counting actions!" << END_LOG;
130 
131  //00. Remove the <unk> word from the Map as it must get fixed index
132  LOG_DEBUG2 << "Remove the <unk> word from the Map as it must get fixed index" << END_LOG;
133  BasicWordIndex::_pWordIndexMap->erase(AWordIndex::UNKNOWN_WORD_STR);
134 
135  //01. Create an array of words info objects from BasicWordIndex::_pWordIndexMap
136  LOG_DEBUG2 << "Create an array of words info objects from BasicWordIndex::_pWordIndexMap" << END_LOG;
137  const size_t num_words = BasicWordIndex::_pWordIndexMap->size();
138  __CountingWordIndex::TWordInfo * word_infos = new __CountingWordIndex::TWordInfo[num_words];
139 
140  //02. Copy the word information from the map into that array.
141  LOG_DEBUG2 << "Copy the word information from the map into that array." << END_LOG;
142  BasicWordIndex::TWordIndexMap::const_iterator iter = BasicWordIndex::_pWordIndexMap->begin();
143  for (size_t idx = 0; iter != BasicWordIndex::_pWordIndexMap->end(); ++iter, ++idx) {
144  word_infos[idx].word = iter->first;
145  word_infos[idx].count = iter->second;
146  }
147 
148  //03. Sort the array of word info object in order to get
149  // the most used words in the beginning of the array
150  LOG_DEBUG2 << "Sort the array of word info object in order to get "
151  << "the most used words in the beginning of the array" << END_LOG;
152  my_sort<__CountingWordIndex::TWordInfo>(word_infos, num_words);
153 
154  //04. Iterate through the array and assign the new word ids
155  // into the _pWordIndexMap using the BasicWordIndex::_nextNewWordId
156  LOG_DEBUG2 << "Iterate through the array and assign the new word ids "
157  << "into the _pWordIndexMap using the BasicWordIndex::_nextNewWordId" << END_LOG;
158  for (size_t idx = 0; idx < num_words; ++idx) {
159  //Get the next word
160  string & word = word_infos[idx].word;
161  //Give it the next index
162  BasicWordIndex::_pWordIndexMap->operator[](word) = BasicWordIndex::_nextNewWordId++;
163  LOG_DEBUG4 << "Word [" << word << "], count: " << SSTR(word_infos[idx].count)
164  << " gets id: " << SSTR(BasicWordIndex::_nextNewWordId - 1) << END_LOG;
165  }
166 
167  //05. Delete the temporary sorted array
168  LOG_DEBUG2 << "Delete the temporary sorted array" << END_LOG;
169  delete[] word_infos;
170 
171  //06. Put back the <unk> word with its fixed index into the map
172  LOG_DEBUG2 << "Put back the <unk> word with its fixed index into the map" << END_LOG;
173  BasicWordIndex::_pWordIndexMap->operator[](UNKNOWN_WORD_STR) = UNKNOWN_WORD_ID;
174 
175  LOG_DEBUG1 << "Finished the post word counting actions!" << END_LOG;
176  };
