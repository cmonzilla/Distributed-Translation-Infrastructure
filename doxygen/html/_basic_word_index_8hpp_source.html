116  TShortId register_word(const TextPieceReader & token) {
117  //First get/create an existing/new word entry from from/in the word index
118  TShortId& hash = _pWordIndexMap->operator[](token.str());
119 
120  if (hash == UNDEFINED_WORD_ID) {
121  //If the word hash is not defined yet, then issue it a new hash id

124  }
125 
126  //Use the Prime numbers hashing algorithm as it outperforms djb2
127  return hash;
