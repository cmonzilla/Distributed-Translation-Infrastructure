194  inline bool getNext(TextPieceReader& out, const char delim) {
195  //The next line begins where we stopped
196  const char * out_m_beginPtr = m_cursorPtr;
197 
198  LOG_DEBUG3 << SSTR(static_cast<const void *> (out_m_beginPtr)) << END_LOG;
199 
200  //The next line length is first zero
201  size_t out_m_len = 0;
202 
203  //Search for the next new line symbol in the remainder of the file
204  const char * charPtr = static_cast<const char *> (memchr(m_cursorPtr, delim, m_restLen));
205 
206  LOG_DEBUG4 << "Searching for the character got: " << SSTR(static_cast<const void *> (charPtr)) << END_LOG;
207 
208  //Check if we found a pointer to the new line
209  if (charPtr != NULL) {
210  //Compute the line length
211  size_t lineLen = charPtr - m_cursorPtr;
212 
213  LOG_DEBUG4 << "The substring length is " << SSTR(lineLen) << END_LOG;
214 
215  //Store the pointer to the remaining of the file
216  m_cursorPtr = charPtr + 1;
217  //Store the remaining length of the file
218  m_restLen -= (lineLen + 1);
219 
220  LOG_DEBUG4 << "Resetting m_cursorPtr = "
221  << SSTR(static_cast<const void *> (m_cursorPtr))
222  << ", m_restLen = " << m_restLen << END_LOG;
223 
224  //Set the resulting length of the line
225  out_m_len = lineLen;
226 
227  //For Windows-format text files remove the '\r' as well
228  //in case we were looking for the end of file
229  if (lineLen && delim == '\n' && '\r' == out_m_beginPtr[lineLen - 1]) {
230  out_m_len--;
231  LOG_DEBUG4 << "A \\\\r detected, resetting m_restLen = "
232  << m_restLen << END_LOG;
233  }
234  } else {
235  //If the pointer is not found then the length of the
236  //last file line is the length of the file remains
237  out_m_len = m_restLen;
238 
239  //If the remaining file length is zero then return file
240  if (!m_restLen) {
241  return false;
242  } else {
243  //If there was something left then now we read it all.
244  //Set the remaining length to zero, as there is nothing left.
245  m_restLen = 0;
246  }
247  }
248 
249  //Set the return data
250  out.set(out_m_beginPtr, out_m_len);
251 
252  //Return true as we successfully read a new line
253  return true;
254  }
