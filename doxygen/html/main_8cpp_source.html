223 static double readAndExecuteQueries( ATrie<N,doCache> & trie, ifstream &testFile) {
224  //Declare time variables for CPU times in seconds
225  double totalTime, startTime, endTime;
226  //Will store the read line (word1 word2 word3 word4 word5)
227  string line;
228  //Will store the N-gram [word1 word2 word3 word4 word5] corresponding to the line
229  vector<string> ngram;
230  //Will store the N-gram frequencies for N-gram:
231  //freqs[0] = frequency( [word1 word2 word3 word4 word5] )
232  //freqs[1] = frequency( [word2 word3 word4 word5] )
233  //freqs[2] = frequency( [word3 word4 word5] )
234  //freqs[3] = frequency( [word4 word5] )
235  //freqs[4] = frequency( [word5] )
236  SFrequencyResult<N> freqs;
237 
238  //Read the test file line by line
239  while( getline(testFile, line) )
240  {
241  //First get the complete N-gram
242  ngrams::NGramBuilder<N,doCache>::buildNGram(line, N, TOKEN_DELIMITER_CHAR, ngram);
243 
244  LOGGER(Logger::DEBUG) << line << ":" << endl;
245 
246  //Second qury the Trie for the results
247  startTime = StatisticsMonitor::getCPUTime();
248  trie.queryNGramFreqs( ngram, freqs );
249  endTime = StatisticsMonitor::getCPUTime();
250 
251  //Print the results:
252  ;
253  unsigned idx = -1;
254  for(int i=0;i<N;i++){
255  LOGGER(Logger::RESULT) << "frequency( " << line << " ) = " << freqs.result[i] << endl;
256 
257  idx = line.find_first_of(TOKEN_DELIMITER_CHAR);
258  line = line.substr(idx+1);
259  }
260  LOGGER(Logger::RESULT) << "CPU Time needed: " << (endTime - startTime) << " sec." << endl;
261 
262  //update total time
263  totalTime += (endTime - startTime);
264  }
265 
266  return totalTime;
267 }
