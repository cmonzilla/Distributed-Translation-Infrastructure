116  MemoryMappedFileReader(const char * fileName) : AFileReader(), m_fileDesc(0) {
117  m_fileDesc = open(fileName, O_RDONLY);
118  LOG_DEBUG << "Opened the file '" << fileName << "' descriptor: " << SSTR(m_fileDesc) << END_LOG;
119 
120  if (m_fileDesc != UNDEFINED_FILE_DESCRIPTOR) {
121  // set the errno to default value
122  errno = 0;
123  //The statistics structure for the mapped file
124  struct stat fileStat;
125  if (fstat(m_fileDesc, &fileStat) < 0) {
126  LOG_ERROR << "Could not get the file '" << fileName << "' statistics after loading! ERROR: " << strerror(errno) << END_LOG;
127  //close the file
128  close();
129  } else {
130  //Get the file length
131  const size_t len = fileStat.st_size;
132  LOG_INFO << "Opened the file '" << fileName << "' size: " << SSTR(len) << " bytes." << END_LOG;
133 
134  //Map the file into memory
135  //NOTE: We populate the map as it is then immediately moved to
136  //the RAM and reading the file is then faster and it does not
137  //give a false impression of that we use too much memory in Trie.
138  // MAP_POPULATE Populate (prefault) page tables for a mapping.
139  //For a file mapping, this causes read-ahead on the file. Later
140  //accesses to the mapping will not be blocked by page faults.
141 #if __APPLE__
142  void * beginPtr = mmap(NULL, len, PROT_READ, MAP_PRIVATE, m_fileDesc, 0);
143 #else
144  void * beginPtr = mmap(NULL, len, PROT_READ, MAP_PRIVATE | MAP_POPULATE, m_fileDesc, 0);
145 #endif
146  LOG_DEBUG << "Memory mapping the file '" << fileName << "' gave: " << SSTR(beginPtr) << " pointer." << END_LOG;
147 
148  //Set the data to the base class
149  TextPieceReader::set(beginPtr, len);
150  }
151  }
152  }

180  virtual void close() {
181  // Release the memory (unnecessary because the program exits).
182  const void * filePtr = TextPieceReader::getBeginPtr();
183  const size_t len = TextPieceReader::getLen();
184  if (filePtr != NULL) {
185  LOG_DEBUG << "Releasing the Memory Mapped File memory: ptr = " <<
186  SSTR(filePtr) << ", len = " << len << END_LOG;
187  //Release the memory
188  int err = munmap(const_cast<void*>(filePtr), len);
189  LOG_DEBUG << "Result of munmap is: " << err << END_LOG;
190  //Re-set the internals!
191  set(NULL, 0);
192  }
193  // Close the file descriptor
194  if (m_fileDesc != UNDEFINED_FILE_DESCRIPTOR) {
195  LOG_DEBUG << "Closing the Memory Mapped File file!" << END_LOG;
196  ::close(m_fileDesc);
197  m_fileDesc = UNDEFINED_FILE_DESCRIPTOR;
198  }
199  };
