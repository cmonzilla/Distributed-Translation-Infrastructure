120  inline const size_t computeNewCapacity(const size_t capacity) const {
121  //Get the float capacity value, make it minimum of one element to avoid problems
122  const float fcap = (capacity > 0) ? (float) capacity : 1.0;
123  const size_t cap_inc = (size_t) (m_mem_inc_factor * m_get_capacity_inc_func(fcap));
124  return capacity + max(cap_inc, m_min_mem_inc);

135  inline MemIncreaseStrategy get_mem_incr_strat(const MemIncTypesEnum stype,
136  const size_t min_mem_inc, const float mem_inc_factor) {
137  TCapacityIncFunct inc_func;
138 
139  //ToDo: optimize this switch, it is pretty ugly, use a map or something
140  switch (stype) {
141  case MemIncTypesEnum::CONSTANT:
142  inc_func = [] (const float fcap) -> float {
143  //Return zero as then the minimum constant increase will be used!
144  return 0;
145  };
146  break;
147  case MemIncTypesEnum::LINEAR:
148  inc_func = [] (const float fcap) -> float {
149  return fcap;
150  };
151  break;
152  case MemIncTypesEnum::LOG_2:
153  inc_func = [] (const float fcap) -> float {
154  return fcap / log(fcap);
155  };
156  break;
157  case MemIncTypesEnum::LOG_10:
158  inc_func = [] (const float fcap) -> float {
159  //Get the float capacity value, make it minimum of one element to avoid problems
160  return fcap / log10(fcap);
161  };
162  break;
163  default:
164  stringstream msg;
165  msg << "Unrecognized memory allocation strategy: " << stype;
166  throw Exception(msg.str());
167  }
168 
169  //return the result object
170  return MemIncreaseStrategy(stype, inc_func, min_mem_inc, mem_inc_factor);
171  };

431  void reallocate(ELEMENT_TYPE_PTR & m_ptr, IDX_DATA_TYPE & m_capacity, IDX_DATA_TYPE new_capacity) {
432  LOG_DEBUG2 << "The new capacity is " << SSTR(new_capacity)

437  //Reallocate memory, potentially we get a new pointer!

441  if (new_ptr != m_ptr) {

446  LOG_DEBUG2 << "Memory is reallocated ptr: " << SSTR(m_ptr) << END_LOG;

482  template<bool IS_INC = true >
483  void reallocate(ELEMENT_TYPE_PTR & m_ptr, IDX_DATA_TYPE & m_capacity, IDX_DATA_TYPE & m_size) {
484  IDX_DATA_TYPE new_capacity;
485 
486  LOG_DEBUG2 << "Memory reallocation request: "
487  << ((IS_INC) ? "increase" : "decrease") << END_LOG;
488 
489  //Compute the new number of elements
490  if (IS_INC) {
491  //Compute the new capacity
492  size_t advised_capacity = ELEMENT_TYPE::m_mem_strat.computeNewCapacity(m_capacity);
493  //Check if the given capacity passes within the maximum allowed values for capacity etc
494  if (advised_capacity <= MAX_SIZE_TYPE_VALUE) {
495  //If it passes then set it in
496  new_capacity = (IDX_DATA_TYPE) advised_capacity;
497  } else {
498  //If it does not pass then try to see if we can still allocate something
499  if (m_capacity < MAX_SIZE_TYPE_VALUE) {
500  //If we can still allocate more, then go to the max!
501  new_capacity = MAX_SIZE_TYPE_VALUE;
502  } else {
503  //We are already at full capacity! Panic!!!!
504  stringstream msg;
505  msg << "Unable to increase the capacity, reached the maximum of "
506  << SSTR(MAX_SIZE_TYPE_VALUE) << " allowed by the data type!";
507  throw Exception(msg.str());
508  }
509  }
510  LOG_DEBUG2 << "Computed new capacity is: " << new_capacity << END_LOG;
511  } else {
512  //Decrease the capacity to the current size, remove the unneeded
513  new_capacity = m_size;
514  }
515 
516  //Reallocate to the computed new capacity
517  reallocate(m_ptr, m_capacity, new_capacity);
518  }
519  };
