155  inline void addGram(const TWordHashSize wordHash,
156  const TReferenceHashSize contextHash, const SBackOffNGram &gram) {
157  const TModelLevel levelIdx = gram.tokens.size()-1;
158  //Obtain the entry for the given N-gram
159  TProbBackOffEntryPair & pbData = ngRecorder[levelIdx][wordHash][contextHash];
160 
161  //Record the data for collision detection
162  AHashMapTrie<N>::recordAndCheck(wordHash, contextHash, gram);
163 
164  if (pbData.first != ZERO_LOG_PROB_WEIGHT) {
165  //If they are not the same then we have a collision!
166  REPORT_COLLISION_WARNING(gram.tokens, wordHash, contextHash,
167  pbData.first, pbData.second,
168  gram.prob, gram.back_off);
169  }
170 
171  //Store the probability and back-off values
172  pbData.first = gram.prob;
173  pbData.second = gram.back_off;

187  const size_t level = gram.tokens.size();
188  LOG_DEBUG << "Adding a " << level << "-Gram '" << tokensToString(gram.tokens) << "' to the Trie" << END_LOG;
189 
190  // 1. Compute the context hash defined by w1 w2 w3
191  TReferenceHashSize contextHash = AHashMapTrie<N>::template computeHashContext<Logger::DEBUG2>(gram.tokens);
192 
193  // 2. Compute the hash of w4
194  const string & endWord = *(--gram.tokens.end());
195  TWordHashSize wordHash = ( isCreate ? AHashMapTrie<N>::createUniqueIdHash(endWord ) : AHashMapTrie<N>::getUniqueIdHash(endWord));
196  LOG_DEBUG2 << "wordHash = computeHash('" << endWord << "') = " << wordHash << END_LOG;
197 
198  // 3. Insert the probability data into the trie
199  addGram(wordHash, contextHash, gram);
200  }
