192  virtual bool get_M_GramDataRef(const TModelLevel level, const TShortId wordId,
193  TLongId ctxId, const TProbBackOffEntry **ppData) {
194  //Get the next context id
195  if (getContextId(wordId, ctxId)) {
196  //Search for the map for that context id
197  const TModelLevel idx = (level - ALayeredTrie<N>::MGRAM_IDX_OFFSET);
198  TMGramsMap::const_iterator result = pMGramMap[idx]->find(ctxId);
199  if (result == pMGramMap[idx]->end()) {
200  //There is no data found under this context
201  return false;
202  } else {
203  //There is data found under this context
204  *ppData = &result->second;
205  return true;
206  }
207  } else {
208  //The context id could not be found
209  return false;
