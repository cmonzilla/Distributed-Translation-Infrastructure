246  template<typename WORD_ENTRY_TYPE>
247  void post_M_N_Grams(WORD_ENTRY_TYPE * wordsArray) {
248  //Define and initialize the current context index offset.
249  //The initial value is 1, although in this Trie it should
250  //not matter much, but it is better to reserve 0 for
251  //an undefined context value
252  TShortId cio = BASE::FIRST_VALID_CTX_ID;
253 
254  //Iterate through all the wordId sub-array mappings in the level and sort sub arrays
255  for (TShortId wordId = AWordIndex::UNDEFINED_WORD_ID; wordId < m_num_word_ids; wordId++) {
256  //First get the sub-array reference.
257  WORD_ENTRY_TYPE & ref = wordsArray[wordId];
258 
259  //Assign the context index offset
260  ref.cio = cio;
261  //Compute the next context index offset, for the next word
262  cio += ref.size();
263 
264  //Reduce capacity if there is unused memory
265  ref.shrink();
266 
267  //Order the N-gram array as it is unordered and we will binary search it later!
268  ref.sort();
269  }
270  }

415  template<typename WORD_ENTRY_TYPE>
416  void preAllocateWordsData(WORD_ENTRY_TYPE* & wordsArray, const TShortId numMGrams, const TShortId numWords) {
417  //Allocate dynamic array entries and initialize it with the memory increase strategy
418  wordsArray = new WORD_ENTRY_TYPE[m_num_word_ids];
419 
420  //Compute the initial capacity
421  size_t capacity = 0;
422  if (__W2CArrayTrie::PRE_ALLOCATE_MEMORY) {
423  //Compute the average number of data elements per word on the given M-gram level
424  const float avg_num_elems = ((float) numMGrams) / ((float) numWords);
425  //Compute the corrected number of elements to preallocate, minimum __W2COrderedArrayTrie::MIN_MEM_INC_NUM.
426  capacity = max(static_cast<size_t> (avg_num_elems * __W2CArrayTrie::INIT_MEM_ALLOC_PRCT),
427  __W2CArrayTrie::MIN_MEM_INC_NUM);
428 
429  //Pre-allocate capacity
430  for (TShortId wordId = AWordIndex::MIN_KNOWN_WORD_ID; wordId < m_num_word_ids; wordId++) {
431  wordsArray[wordId].pre_allocate(capacity);
432  }
433  }
434  }
435  };
