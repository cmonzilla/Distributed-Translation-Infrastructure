283  template<typename WORD_ENTRY_TYPE>
284  void post_M_N_Grams(WORD_ENTRY_TYPE * wordsArray) {
285  //Define and initialize the current context index offset.
286  //The initial value is 1, although in this Trie it should
287  //not matter much, but it is better to reserve 0 for
288  //an undefined context value
289  TShortId cio = ALayeredTrie<N>::FIRST_VALID_CTX_ID;
290 
291  //Iterate through all the wordId sub-array mappings in the level and sort sub arrays
292  for (TShortId wordId = AWordIndex::UNDEFINED_WORD_ID; wordId < m_num_word_ids; wordId++) {
293  //First get the sub-array reference.
294  WORD_ENTRY_TYPE & ref = wordsArray[wordId];
295 
296  //Assign the context index offset
297  ref.cio = cio;
298  //Compute the next context index offset, for the next word
299  cio += ref.size();
300 
301  //Reduce capacity if there is unused memory
302  ref.shrink();
303 
304  //Order the N-gram array as it is unordered and we will binary search it later!
305  ref.sort();
306  }
307  }

445  inline bool getContextId(const TShortId wordId, TLongId & ctxId, const TModelLevel level) {
446  //Compute the m-gram index
447  const TModelLevel mgram_idx = level - ALayeredTrie<N>::MGRAM_IDX_OFFSET;
448 
449  if (DO_SANITY_CHECKS && ((level == N) || (mgram_idx < 0))) {
450  stringstream msg;
451  msg << "Unsupported level id: " << level;
452  throw Exception(msg.str());
453  }
454 
455  LOG_DEBUG2 << "Searching next ctxId for " << SSTR(level)
456  << "-gram with wordId: " << SSTR(wordId) << ", ctxId: "
457  << SSTR(ctxId) << END_LOG;
458 
459  //First get the sub-array reference.
460  const T_M_GramWordEntry & ref = m_M_gram_word_2_data[mgram_idx][wordId];
461 
462  if (DO_SANITY_CHECKS && ref.has_data()) {
463  LOG_DEBUG3 << "ref.size: " << SSTR(ref.size()) << ", ref.cio: "
464  << SSTR(ref.cio) << ", ctxId range: [" << SSTR(ref[0].id) << ", "
465  << SSTR(ref[ref.size() - 1].id) << "]" << END_LOG;
466  }
467 
468  //Check that if this is the 2-Gram case and the previous context
469  //id is 0 then it is the unknown word id, at least this is how it
470  //is now in ATrie implementation, so we need to do a warning!
471  if (DO_SANITY_CHECKS && (level == M_GRAM_LEVEL_2) && (ctxId < AWordIndex::MIN_KNOWN_WORD_ID)) {
472  LOG_WARNING << "Perhaps we are being paranoid but there "
473  << "seems to be a problem! The " << SSTR(level) << "-gram ctxId: "
474  << SSTR(ctxId) << " is equal to an undefined(" << SSTR(AWordIndex::UNDEFINED_WORD_ID)
475  << ") or unknown(" << SSTR(AWordIndex::UNKNOWN_WORD_ID) << ") word ids!" << END_LOG;
476  }
477 
478  //Get the local entry index and then use it to compute the next context id
479  typename T_M_GramWordEntry::TIndexType localIdx;
480  //If the local entry index could be found then compute the next ctxId
481  if (get_M_N_GramLocalEntryIdx(ref, ctxId, localIdx)) {
482  LOG_DEBUG2 << "Got context mapping for ctxId: " << SSTR(ctxId)
483  << ", size = " << SSTR(ref.size()) << ", localIdx = "
484  << SSTR(localIdx) << ", resulting ctxId = "
485  << SSTR(ref.cio + localIdx) << END_LOG;
486 
487  //The next ctxId is the sum of the local index and the context index offset
488  ctxId = ref.cio + localIdx;
489  return true;
490  } else {
491  //The local index could not be found
492  return false;
493  }

510  void preAllocateWordsData(WORD_ENTRY_TYPE* & wordsArray, const TShortId numMGrams, const TShortId numWords) {

514  //Compute the initial capacity

519  //Compute the corrected number of elements to preallocate, minimum __W2COrderedArrayTrie::MIN_MEM_INC_NUM.

522
