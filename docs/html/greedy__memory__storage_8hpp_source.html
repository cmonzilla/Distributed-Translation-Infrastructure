123  void* allocate(size_type num) {
124  const size_type remains = getAvailableBytes();
125  void* ptr = NULL;
126 
127  //Check if there is more space needed!
128  if (remains > num) {
129  //Allocate the requested memory in the buffer
130  ptr = static_cast<void*> (static_cast<char*> (_pBuffer) + _allocBytes);
131  _allocBytes += num;
132  } else {
133  //WARNING: If we need more space we allocate it additionally in a not so efficient way.
134  LOG_DEBUG3 << "Allocating additional: " << num << " bytes!" << END_LOG;
135  ptr = static_cast<void*> (new TStorageData[num]);
136  _memoryBuffers.push_back(ptr);
137  }
138  return ptr;
139  }
