165 void select_reactor::run(bool block, op_queue<operation>& ops)
166 {
167  asio::detail::mutex::scoped_lock lock(mutex_);
168 
169 #if defined(ASIO_HAS_IOCP)
170  // Check if the thread is supposed to stop.
171  if (stop_thread_)
172  return;
173 #endif // defined(ASIO_HAS_IOCP)
174 
175  // Set up the descriptor sets.
176  for (int i = 0; i < max_select_ops; ++i)
177  fd_sets_[i].reset();
178  fd_sets_[read_op].set(interrupter_.read_descriptor());
179  socket_type max_fd = 0;
180  bool have_work_to_do = !timer_queues_.all_empty();
181  for (int i = 0; i < max_select_ops; ++i)
182  {
183  have_work_to_do = have_work_to_do || !op_queue_[i].empty();
184  fd_sets_[i].set(op_queue_[i], ops);
185  if (fd_sets_[i].max_descriptor() > max_fd)
186  max_fd = fd_sets_[i].max_descriptor();
187  }
188 
189 #if defined(ASIO_WINDOWS) || defined(__CYGWIN__)
190  // Connection operations on Windows use both except and write fd_sets.
191  have_work_to_do = have_work_to_do || !op_queue_[connect_op].empty();
192  fd_sets_[write_op].set(op_queue_[connect_op], ops);
193  if (fd_sets_[write_op].max_descriptor() > max_fd)
194  max_fd = fd_sets_[write_op].max_descriptor();
195  fd_sets_[except_op].set(op_queue_[connect_op], ops);
196  if (fd_sets_[except_op].max_descriptor() > max_fd)
197  max_fd = fd_sets_[except_op].max_descriptor();
198 #endif // defined(ASIO_WINDOWS) || defined(__CYGWIN__)
199 
200  // We can return immediately if there's no work to do and the reactor is
201  // not supposed to block.
202  if (!block && !have_work_to_do)
203  return;
204 
205  // Determine how long to block while waiting for events.
206  timeval tv_buf = { 0, 0 };
207  timeval* tv = block ? get_timeout(tv_buf) : &tv_buf;
208 
209  lock.unlock();
210 
211  // Block on the select call until descriptors become ready.
212  asio::error_code ec;
213  int retval = socket_ops::select(static_cast<int>(max_fd + 1),
214  fd_sets_[read_op], fd_sets_[write_op], fd_sets_[except_op], tv, ec);
215 
216  // Reset the interrupter.
217  if (retval > 0 && fd_sets_[read_op].is_set(interrupter_.read_descriptor()))
218  {
219  interrupter_.reset();
220  --retval;
221  }
222 
223  lock.lock();
224 
225  // Dispatch all ready operations.
226  if (retval > 0)
227  {
228 #if defined(ASIO_WINDOWS) || defined(__CYGWIN__)
229  // Connection operations on Windows use both except and write fd_sets.
230  fd_sets_[except_op].perform(op_queue_[connect_op], ops);
231  fd_sets_[write_op].perform(op_queue_[connect_op], ops);
232 #endif // defined(ASIO_WINDOWS) || defined(__CYGWIN__)
233 
234  // Exception operations must be processed first to ensure that any
235  // out-of-band data is read before normal data.
236  for (int i = max_select_ops - 1; i >= 0; --i)
237  fd_sets_[i].perform(op_queue_[i], ops);
238  }
239  timer_queues_.get_ready_timers(ops);
240 }

279 timeval* select_reactor::get_timeout(timeval& tv)
280 {
281  // By default we will wait no longer than 5 minutes. This will ensure that
282  // any changes to the system clock are detected after no longer than this.
283  long usec = timer_queues_.wait_duration_usec(5 * 60 * 1000 * 1000);
284  tv.tv_sec = usec / 1000000;
285  tv.tv_usec = usec % 1000000;
286  return &tv;
287 }

289 void select_reactor::cancel_ops_unlocked(socket_type descriptor,
290  const asio::error_code& ec)
291 {
292  bool need_interrupt = false;
293  op_queue<operation> ops;

295  need_interrupt = op_queue_[i].cancel_operations(
