46 void eventfd_select_interrupter::open_descriptors()
47 {
48 #if __GLIBC__ == 2 && __GLIBC_MINOR__ < 8
49  write_descriptor_ = read_descriptor_ = syscall(__NR_eventfd, 0);
50  if (read_descriptor_ != -1)
51  {
52  ::fcntl(read_descriptor_, F_SETFL, O_NONBLOCK);
53  ::fcntl(read_descriptor_, F_SETFD, FD_CLOEXEC);
54  }
55 #else // __GLIBC__ == 2 && __GLIBC_MINOR__ < 8
56 # if defined(EFD_CLOEXEC) && defined(EFD_NONBLOCK)
57  write_descriptor_ = read_descriptor_ =
58  ::eventfd(0, EFD_CLOEXEC | EFD_NONBLOCK);
59 # else // defined(EFD_CLOEXEC) && defined(EFD_NONBLOCK)
60  errno = EINVAL;
61  write_descriptor_ = read_descriptor_ = -1;
62 # endif // defined(EFD_CLOEXEC) && defined(EFD_NONBLOCK)
63  if (read_descriptor_ == -1 && errno == EINVAL)
64  {
65  write_descriptor_ = read_descriptor_ = ::eventfd(0, 0);
66  if (read_descriptor_ != -1)
67  {
68  ::fcntl(read_descriptor_, F_SETFL, O_NONBLOCK);
69  ::fcntl(read_descriptor_, F_SETFD, FD_CLOEXEC);
70  }
71  }
72 #endif // __GLIBC__ == 2 && __GLIBC_MINOR__ < 8
73 
74  if (read_descriptor_ == -1)
75  {
76  int pipe_fds[2];
77  if (pipe(pipe_fds) == 0)
78  {
79  read_descriptor_ = pipe_fds[0];
80  ::fcntl(read_descriptor_, F_SETFL, O_NONBLOCK);
81  ::fcntl(read_descriptor_, F_SETFD, FD_CLOEXEC);
82  write_descriptor_ = pipe_fds[1];
83  ::fcntl(write_descriptor_, F_SETFL, O_NONBLOCK);
84  ::fcntl(write_descriptor_, F_SETFD, FD_CLOEXEC);
85  }
86  else
87  {
88  asio::error_code ec(errno,
89  asio::error::get_system_category());
90  asio::detail::throw_error(ec, "eventfd_select_interrupter");
91  }
92  }
93 }

125 bool eventfd_select_interrupter::reset()
126 {
127  if (write_descriptor_ == read_descriptor_)
128  {
129  for (;;)
130  {
131  // Only perform one read. The kernel maintains an atomic counter.
132  uint64_t counter(0);
133  errno = 0;
134  int bytes_read = ::read(read_descriptor_, &counter, sizeof(uint64_t));
135  if (bytes_read < 0 && errno == EINTR)
136  continue;
137  bool was_interrupted = (bytes_read > 0);
138  return was_interrupted;
139  }
140  }
141  else
142  {
143  for (;;)
144  {
145  // Clear all data from the pipe.
146  char data[1024];
147  int bytes_read = ::read(read_descriptor_, data, sizeof(data));
148  if (bytes_read < 0 && errno == EINTR)
149  continue;
150  bool was_interrupted = (bytes_read > 0);
151  while (bytes_read == sizeof(data))
152  bytes_read = ::read(read_descriptor_, data, sizeof(data));
153  return was_interrupted;
154  }
155  }
156 }
