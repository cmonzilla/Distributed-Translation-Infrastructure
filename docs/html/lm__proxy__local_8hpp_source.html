178  const string & model_file_name = params.m_conn_string;
179 
180  //Declare time variables for CPU times in seconds

185  LOG_USAGE << "--------------------------------------------------------" << END_LOG;

193  //Attempt to open the model file
194  file_reader_type model_file(model_file_name.c_str());
195  model_file.log_reader_type_info();

197  stat_monitore::get_mem_stat(mem_stat_end);

202  LOG_DEBUG << "Getting the memory statistics before loading the " << model_name << " ..." << END_LOG;
203  stat_monitore::get_mem_stat(mem_stat_start);
