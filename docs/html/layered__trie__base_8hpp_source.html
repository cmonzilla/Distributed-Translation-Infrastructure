71  inline phrase_length search_m_gram_ctx_id(const TrieType & trie, const word_uid * const word_ids, TLongId & prev_ctx_id, TLongId & ctx_id) {
72  //Assert that this method is called for proper m-gram levels
73  ASSERT_SANITY_THROW(((CURR_LEVEL < M_GRAM_LEVEL_2) || (CURR_LEVEL > M_GRAM_LEVEL_5)), string("The level: ") + std::to_string(CURR_LEVEL) + string(" is not supported yet!"));
74 
75  //The initial context for anything larger than a unigram is the first word id
76  ctx_id = word_ids[0];
77 
78  //If we are at least at a level 3 m-gram
79  if (CURR_LEVEL >= M_GRAM_LEVEL_3) {
80  if (GET_BACK_OFF_CTX_ID && (CURR_LEVEL == M_GRAM_LEVEL_3)) prev_ctx_id = ctx_id;
81  if (trie.get_ctx_id(M_GRAM_LEVEL_2 - M_GRAM_LEVEL_2, word_ids[1], ctx_id)) {
82  //If we are at least at a level 4 m-gram
83  if (CURR_LEVEL >= M_GRAM_LEVEL_4) {
84  if (GET_BACK_OFF_CTX_ID && (CURR_LEVEL == M_GRAM_LEVEL_4)) prev_ctx_id = ctx_id;
85  if (trie.get_ctx_id(M_GRAM_LEVEL_3 - M_GRAM_LEVEL_2, word_ids[2], ctx_id)) {
86  //If we are at least at a level 5 m-gram
87  if (CURR_LEVEL >= M_GRAM_LEVEL_5) {
88  if (GET_BACK_OFF_CTX_ID && (CURR_LEVEL == M_GRAM_LEVEL_5)) prev_ctx_id = ctx_id;
89  if (trie.get_ctx_id(M_GRAM_LEVEL_4 - M_GRAM_LEVEL_2, word_ids[3], ctx_id)) {
90  return M_GRAM_LEVEL_5;
91  }
92  }
93  return M_GRAM_LEVEL_4;
94  }
95  }
96  return M_GRAM_LEVEL_3;
97  }
98  }
99  return M_GRAM_LEVEL_2;

112  template<typename TrieType, phrase_length CURR_LEVEL, debug_levels_enum LOG_LEVEL>
113  inline void get_context_id(TrieType & trie, const model_m_gram &gram, TLongId & ctx_id) {
114  //Perform sanity check for the level values they should be the same!
115  ASSERT_SANITY_THROW(CURR_LEVEL != gram.get_num_words(),
116  string("The improper level values! Template level parameter = ") + std::to_string(CURR_LEVEL) +
117  string(" but the m-gram level value is: ") + std::to_string(gram.get_num_words()));
118 
119  //Try to retrieve the context from the cache, if not present then compute it
120  if (trie.template get_cached_context_id<CURR_LEVEL>(gram, ctx_id)) {
121  //Compute the context id, check on the level
122  const phrase_length ctx_level = search_m_gram_ctx_id<TrieType, CURR_LEVEL, false, LOG_LEVEL>(trie, gram.word_ids(), ctx_id, ctx_id);
123 
124  LOG_DEBUG4 << "The context level is: " << SSTR(ctx_level) << ", the current level is " << SSTR(CURR_LEVEL) << END_LOG;
125 
126  //Do sanity check if needed
127  ASSERT_SANITY_THROW(CURR_LEVEL != ctx_level,
128  string("The m-gram context could not be computed!"));
129 
130  //Cache the newly computed context id for the given n-gram context
131  trie.template set_cache_context_id<CURR_LEVEL>(gram, ctx_id);
132 
133  //The context Id was found in the Trie
134  LOGGER(LOG_LEVEL) << "The ctx_id could be computed, " << "it's value is: " << SSTR(ctx_id) << END_LOG;
135  } else {
136  //The context Id was found in the cache
137  LOGGER(LOG_LEVEL) << "The ctx_id was found in cache, " << "it's value is: " << SSTR(ctx_id) << END_LOG;
138  }

201  template<phrase_length CURR_LEVEL>
202  inline bool get_cached_context_id(const model_m_gram &gram, TLongId & result) const {
203  //Compute the context level
204  constexpr phrase_length CONTEXT_LEVEL = CURR_LEVEL - 1;
205  //Check if this is the same m-gram
206  if (memcmp(m_cached_ctx[CONTEXT_LEVEL].m_word_ids, gram.word_ids(), CONTEXT_LEVEL * sizeof (TShortId)) == 0) {
207  result = m_cached_ctx[CONTEXT_LEVEL].m_ctx_id;
208  //There was something cached, so no need to search further!
209  return false;
210  }
211  //There was nothing cached, so return keep searching!
212  return true;

241  inline void ensure_context(m_gram_query & query, MGramStatusEnum & status) const {
242  //Get the context id reference for convenience
243  TLongId & ctx_id = query.get_curr_ctx_ref();
244 
245  LOG_DEBUG << "Ensuring context for sub-m-gram : " << query
246  << ", the last computed context value is: " << ctx_id << END_LOG;
247 
248  //The first context is the first word id
249  ctx_id = query.get_curr_begin_word_id();
250 
251  LOG_DEBUG << "Setting the first context value to the first word id: " << ctx_id << END_LOG;
252 
253  //Decrement the end word index to get down to the back-off level
254  query.m_curr_end_word_idx--;
255 
256  //Set the result status to true
257  status = MGramStatusEnum::GOOD_PRESENT_MGS;
258 
259  //Check if the back-off m-gram is a unigram or not
260  if (query.m_curr_begin_word_idx == query.m_curr_end_word_idx) {
261  //The the back-off sub-m-gram is a uni-gram obtain its payload
262  static_cast<const TrieType*> (this)->get_unigram_payload(query);
263  } else {
264  //If the back-off sub-m-gram is not a uni-gram then do the context
265  for (phrase_length word_idx = query.m_curr_begin_word_idx + 1; word_idx < query.m_curr_end_word_idx; ++word_idx) {
266  LOG_DEBUG2 << "Getting the context id for sub-m-gram: [" << SSTR(query.m_curr_begin_word_idx) << ", " << SSTR(word_idx) << "]" << END_LOG;
267  const phrase_length & level_idx = CURR_LEVEL_MIN_2_MAP[query.m_curr_begin_word_idx][word_idx];
268  if (!static_cast<const TrieType*> (this)->get_ctx_id(level_idx, query[word_idx], ctx_id)) {
269  //If the next context could not be computed, we stop with a bad status
270  status = MGramStatusEnum::BAD_NO_PAYLOAD_MGS;
271  break;
272  }
273  }
274 
275  //If the context of the back-off sub-m-gram could be computed then retrieve its payload
276  if (status == MGramStatusEnum::GOOD_PRESENT_MGS) {
277  //The back-off sub-m-gram is at least a bi-gram and never an n-gram
278  static_cast<const TrieType*> (this)->get_m_gram_payload(query, status);
279  }
280 
281  //If the back-off payload could not be computed, set the zero payload
282  if (status != MGramStatusEnum::GOOD_PRESENT_MGS) {
283  //Set the back-off payload to zero payload!
284 
285  query.set_curr_payload(&m_nothing_payload);
286  }
287  }
288  //Increment the end word index to get back to the original sub-m-gram
289  query.m_curr_end_word_idx++;
290 
291  LOG_DEBUG << "Ensuring context or sub-m-gram: " << query << " status is: "
292  << status_to_string(status) << END_LOG;
293  }
