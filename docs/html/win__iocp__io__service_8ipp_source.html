90 void win_iocp_io_service::shutdown_service()
91 {
92  ::InterlockedExchange(&shutdown_, 1);
93 
94  if (timer_thread_.get())
95  {
96  LARGE_INTEGER timeout;
97  timeout.QuadPart = 1;
98  ::SetWaitableTimer(waitable_timer_.handle, &timeout, 1, 0, 0, FALSE);
99  }
100 
101  while (::InterlockedExchangeAdd(&outstanding_work_, 0) > 0)
102  {
103  op_queue<win_iocp_operation> ops;
104  timer_queues_.get_all_timers(ops);
105  ops.push(completed_ops_);
106  if (!ops.empty())
107  {
108  while (win_iocp_operation* op = ops.front())
109  {
110  ops.pop();
111  ::InterlockedDecrement(&outstanding_work_);
112  op->destroy();
113  }
114  }
115  else
116  {
117  DWORD bytes_transferred = 0;
118  dword_ptr_t completion_key = 0;
119  LPOVERLAPPED overlapped = 0;
120  ::GetQueuedCompletionStatus(iocp_.handle, &bytes_transferred,
121  &completion_key, &overlapped, gqcs_timeout_);
122  if (overlapped)
123  {
124  ::InterlockedDecrement(&outstanding_work_);
125  static_cast<win_iocp_operation*>(overlapped)->destroy();
126  }
127  }
128  }
129

131  timer_thread_->join();

344 size_t win_iocp_io_service::do_one(bool block, asio::error_code& ec)
345 {
346  for (;;)
347  {
348  // Try to acquire responsibility for dispatching timers and completed ops.
349  if (::InterlockedCompareExchange(&dispatch_required_, 0, 1) == 1)
350  {
351  mutex::scoped_lock lock(dispatch_mutex_);
352 
353  // Dispatch pending timers and operations.
354  op_queue<win_iocp_operation> ops;
355  ops.push(completed_ops_);
356  timer_queues_.get_ready_timers(ops);
357  post_deferred_completions(ops);
358  update_timeout();
359  }
360 
361  // Get the next operation from the queue.
362  DWORD bytes_transferred = 0;
363  dword_ptr_t completion_key = 0;
364  LPOVERLAPPED overlapped = 0;
365  ::SetLastError(0);
366  BOOL ok = ::GetQueuedCompletionStatus(iocp_.handle, &bytes_transferred,
367  &completion_key, &overlapped, block ? gqcs_timeout_ : 0);
368  DWORD last_error = ::GetLastError();
369 
370  if (overlapped)
371  {
372  win_iocp_operation* op = static_cast<win_iocp_operation*>(overlapped);
373  asio::error_code result_ec(last_error,
374  asio::error::get_system_category());
375 
376  // We may have been passed the last_error and bytes_transferred in the
377  // OVERLAPPED structure itself.
378  if (completion_key == overlapped_contains_result)
379  {
380  result_ec = asio::error_code(static_cast<int>(op->Offset),
381  *reinterpret_cast<asio::error_category*>(op->Internal));
382  bytes_transferred = op->OffsetHigh;
383  }
384 
385  // Otherwise ensure any result has been saved into the OVERLAPPED
386  // structure.
387  else
388  {
389  op->Internal = reinterpret_cast<ulong_ptr_t>(&result_ec.category());
390  op->Offset = result_ec.value();
391  op->OffsetHigh = bytes_transferred;
392  }
393 
394  // Dispatch the operation only if ready. The operation may not be ready
395  // if the initiating function (e.g. a call to WSARecv) has not yet
396  // returned. This is because the initiating function still wants access
397  // to the operation's OVERLAPPED structure.
398  if (::InterlockedCompareExchange(&op->ready_, 1, 0) == 1)
399  {
400  // Ensure the count of outstanding work is decremented on block exit.
401  work_finished_on_block_exit on_exit = { this };
402  (void)on_exit;
403 
404  op->complete(*this, result_ec, bytes_transferred);
405  ec = asio::error_code();
406  return 1;
407  }
408  }
409  else if (!ok)
410  {
411  if (last_error != WAIT_TIMEOUT)
412  {
413  ec = asio::error_code(last_error,
414  asio::error::get_system_category());
415  return 0;
416  }
417 
418  // If we're not polling we need to keep going until we get a real handler.
419  if (block)
420  continue;
421 
422  ec = asio::error_code();
423  return 0;
424  }
425  else if (completion_key == wake_for_dispatch)
426  {
427  // We have been woken up to try to acquire responsibility for dispatching
428  // timers and completed operations.
429  }
430  else
431  {
432  // Indicate that there is no longer an in-flight stop event.
433  ::InterlockedExchange(&stop_event_posted_, 0);
434 
435  // The stopped_ flag is always checked to ensure that any leftover
436  // stop events from a previous run invocation are ignored.
437  if (::InterlockedExchangeAdd(&stopped_, 0) != 0)
438  {
439  // Wake up next thread that is blocked on GetQueuedCompletionStatus.
440  if (::InterlockedExchange(&stop_event_posted_, 1) == 0)
441  {
442  if (!::PostQueuedCompletionStatus(iocp_.handle, 0, 0, 0))
443  {
444  last_error = ::GetLastError();
445  ec = asio::error_code(last_error,
446  asio::error::get_system_category());
447  return 0;
448  }
449  }
450 
451  ec = asio::error_code();
452  return 0;
453  }
454  }
455  }
