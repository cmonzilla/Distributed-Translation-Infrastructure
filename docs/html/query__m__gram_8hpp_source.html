78  inline uint64_t get_hash(phrase_length begin_word_idx, const phrase_length end_word_idx) const {
79  //Define the reference to the previous level
80  phrase_length & prev_level_ref = const_cast<phrase_length &> (m_hash_level_row[begin_word_idx]);
81 
82  LOG_DEBUG1 << "Getting hash values for begin/end index: " << SSTR(begin_word_idx)
83  << "/" << SSTR(end_word_idx) << ", the previous computed begin level "
84  << "is: " << SSTR(prev_level_ref) << END_LOG;
85 
86  //Define the reference to the hash row
87  uint64_t(& hash_row_ref)[QUERY_M_GRAM_MAX_LEN] = const_cast<uint64_t(&)[QUERY_M_GRAM_MAX_LEN]> (m_hash_matrix[begin_word_idx]);
88 
89  //Compute the current level
90  const phrase_length curr_level = CURR_LEVEL_MAP[begin_word_idx][end_word_idx];
91  //Check if the given hash is already available.
92  if (curr_level > prev_level_ref) {
93  //Check if there has its been computed before for this row
94  if (prev_level_ref == M_GRAM_LEVEL_UNDEF) {
95  //If there has not been anything computed yet,
96  //then first initialize the starting word
97  hash_row_ref[begin_word_idx] = BASE::operator [](begin_word_idx);
98 
99  LOG_DEBUG1 << "word[" << SSTR(begin_word_idx) << "] = "
100  << BASE::operator [](begin_word_idx)
101  << ", hash[" << SSTR(begin_word_idx) << "] = "
102  << hash_row_ref[begin_word_idx] << END_LOG;
103 
104  ++begin_word_idx;
105  } else {
106  //This is the case of at least a bi-gram, but the actual
107  //begin word index is the one stored from before
108  begin_word_idx += prev_level_ref;
109  }
110 
111  //Iterate on and compute the subsequent hashes, if any
112  for (; begin_word_idx <= end_word_idx; ++begin_word_idx) {
113  //Incrementally build up hash, using the previous hash value and the next word id
114  hash_row_ref[begin_word_idx] = combine_phrase_uids(hash_row_ref[begin_word_idx - 1], BASE::operator [](begin_word_idx));
115 
116  LOG_DEBUG1 << "hash[" << SSTR(begin_word_idx) << "] = combine( word["
117  << SSTR(begin_word_idx) << "] = " << BASE::operator [](begin_word_idx)
118  << ", hash[" << SSTR(begin_word_idx - 1) << "] = "
119  << hash_row_ref[begin_word_idx - 1] << " ) = "
120  << hash_row_ref[begin_word_idx] << END_LOG;
121  }
122  //Set the processed level
123  prev_level_ref = curr_level;
124  }
125 
126  LOG_DEBUG1 << "Resulting hash value: " << hash_row_ref[end_word_idx] << END_LOG;
127 
128  //Return the hash value that must have been pre-computed
129  return hash_row_ref[end_word_idx];
130  }
