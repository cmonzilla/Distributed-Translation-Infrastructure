216  void on_message(websocketpp::connection_hdl hdl, server::message_ptr msg) {
217  LOG_DEBUG << "Received a message!" << END_LOG;
218 
219  //Declare the translation job request pointer
220  trans_job_request_ptr request_ptr = NULL;
221  try {
222  //Extract the translation job request
223  request_ptr = new trans_job_request(msg->get_payload());
224 
225  //Schedule a translation job
226  m_manager.translate(hdl, request_ptr);
227  } catch (Exception & ex) {
228  //Locally report error
229  LOG_ERROR << ex.get_message() << END_LOG;
230 
231  //Create the reply message, with or without job id
232  const job_id_type job_id = (request_ptr == NULL) ? job_id::UNDEFINED_JOB_ID : request_ptr->get_job_id();
233  trans_job_response response(job_id, trans_job_code::RESULT_ERROR, ex.get_message());
234 
235  //Send the response
236  send_response(hdl, response);
237  }
238  }
