195 const asio::error_code& engine::map_error_code(
196  asio::error_code& ec) const
197 {
198  // We only want to map the error::eof code.
199  if (ec != asio::error::eof)
200  return ec;
201 
202  // If there's data yet to be read, it's an error.
203  if (BIO_wpending(ext_bio_))
204  {
205  ec = asio::error_code(
206  ERR_PACK(ERR_LIB_SSL, 0, SSL_R_SHORT_READ),
207  asio::error::get_ssl_category());
208  return ec;
209  }
210 
211  // SSL v2 doesn't provide a protocol-level shutdown, so an eof on the
212  // underlying transport is passed through.
213  if (ssl_->version == SSL2_VERSION)
214  return ec;
215 
216  // Otherwise, the peer should have negotiated a proper shutdown.
217  if ((::SSL_get_shutdown(ssl_) & SSL_RECEIVED_SHUTDOWN) == 0)
218  {
219  ec = asio::error_code(
220  ERR_PACK(ERR_LIB_SSL, 0, SSL_R_SHORT_READ),
221  asio::error::get_ssl_category());
222  }
223 
224  return ec;
225 }

233 engine::want engine::perform(int (engine::* op)(void*, std::size_t),
234  void* data, std::size_t length, asio::error_code& ec,
235  std::size_t* bytes_transferred)
236 {
237  std::size_t pending_output_before = ::BIO_ctrl_pending(ext_bio_);
238  ::ERR_clear_error();
239  int result = (this->*op)(data, length);
240  int ssl_error = ::SSL_get_error(ssl_, result);
241  int sys_error = static_cast<int>(::ERR_get_error());
242  std::size_t pending_output_after = ::BIO_ctrl_pending(ext_bio_);
243 
244  if (ssl_error == SSL_ERROR_SSL)
245  {
246  ec = asio::error_code(sys_error,
247  asio::error::get_ssl_category());
248  return want_nothing;
249  }
250 
251  if (ssl_error == SSL_ERROR_SYSCALL)
252  {
253  ec = asio::error_code(sys_error,
254  asio::error::get_system_category());
255  return want_nothing;
256  }
257 
258  if (result > 0 && bytes_transferred)
259  *bytes_transferred = static_cast<std::size_t>(result);
260 
261  if (ssl_error == SSL_ERROR_WANT_WRITE)
262  {
263  ec = asio::error_code();
264  return want_output_and_retry;
265  }
266  else if (pending_output_after > pending_output_before)
267  {
268  ec = asio::error_code();
269  return result > 0 ? want_output : want_output_and_retry;
270  }
271  else if (ssl_error == SSL_ERROR_WANT_READ)
272  {
273  ec = asio::error_code();
274  return want_input_and_retry;
275  }
276  else if (::SSL_get_shutdown(ssl_) & SSL_RECEIVED_SHUTDOWN)
277  {
278  ec = asio::error::eof;
279  return want_nothing;
280  }
281  else
282  {
283  ec = asio::error_code();
284  return want_nothing;
285  }
286 }
