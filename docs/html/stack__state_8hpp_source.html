182  inline void get_translation(string & target_sent) const {
183  //Go through parents until the root state and append translations when going back!
184  if (m_parent != NULL) {
185  //Go recursively to the parent
186  m_parent->get_translation(target_sent);
187  }
188 
189  //Check that the target is not NULL if it is then
190  //it is either the begin <s> or end </s> state
191  if (m_state_data.m_target != NULL) {
192  //Append the space plus the current state translation
193  if (m_state_data.m_target->is_unk_trans()) {
194  //If this is an unknown translation then just copy the original source text
195 
196  //Get the begin and end source phrase word indexes
197  const phrase_length begin_word_idx = m_state_data.m_s_begin_word_idx;
198  const phrase_length end_word_idx = m_state_data.m_s_end_word_idx;
199 
200  //Get the begin and end source phrase character indexes
201  const uint32_t begin_ch_idx = m_state_data.m_stack_data.m_sent_data[begin_word_idx][begin_word_idx].m_begin_ch_idx;
202  const uint32_t end_ch_idx = m_state_data.m_stack_data.m_sent_data[end_word_idx][end_word_idx].m_end_ch_idx;

208  target_sent += m_state_data.m_target->get_target_phrase();

318  recomb_from->m_prev = NULL;
319  recomb_from->m_next = other_state->m_recomb_from;
320  //If there was something in the other's state recombine from then link the list
321  if (other_state->m_recomb_from != NULL) {
322  LOG_DEBUG << "State " << other_state << " recombined from "
323  << "ptr = " << other_state->m_recomb_from << END_LOG;
324  other_state->m_recomb_from->m_prev = recomb_from;
325  //Clean the recombined from data
326  other_state->m_recomb_from = NULL;
327  other_state->m_recomb_from_count = 0;
328  }
329 
330  //Attempt to insert the other new state into the recombined from list.
331  if (m_recomb_from == NULL) {
332  LOG_DEBUG << "State " << other_state << " is the first"
333  << " state recombined into " << this << END_LOG;
334 
335  //If this state has no recombined-from states
336  //then just set the other list into this state
337  m_recomb_from = recomb_from;
338  m_recomb_from_count = recomb_from_count;
339 
340  //Do not prune the extra recombined from states, as
341  //there an be just one extra. We can postpone this
342  //until more states are recombined into this one.
343  } else {
344  LOG_DEBUG << "State " << this << " has more recombined-from states, merging "
345  << " state " << other_state << " into the list" << END_LOG;
346 
347  //If this state has recombined-from states
348  //then merge this and the other list.
349  merge_recomb_from(recomb_from, recomb_from_count);
350  }
351  LOG_DEBUG1 << "Recombining " << other_state << " into " << this << " is done!" << END_LOG;
352  LOG_DEBUG2 << "====================================================================" << END_LOG;
353  }

415  inline void merge_recomb_from(const stack_state_ptr recomb_from, const size_t recomb_from_count) {
416  //Store the counter for the number of states in the resulting list
417  size_t state_count = 0;
418 
419  //The current state from the target list
420  stack_state_ptr target_state = m_recomb_from;
421  //The current state from the source list
422  stack_state_ptr source_state = recomb_from;
423  //Will store the pointer to the last target state
424  stack_state_ptr last_target_state = NULL;
425 
426  //Iterate until we have enough states in the resulting
427  //list or we reach the end of one of the lists
428  bool is_go_on = true;
429  while (is_go_on) {
430  if (source_state != NULL) {
431  if (target_state != NULL) {
432  //Case: (source_state != NULL) && (target_state != NULL)
433 
434  //Both source and target are not null, so we can compare them
435  if (*source_state < *target_state) {
436  //If this is the last target state, remember it.
437  if (target_state->m_next == NULL) {
438  last_target_state = target_state;
439  }
440  //Move to the next element within the target list
441  target_state = target_state->m_next;
442  //We move to the next state in the target
443  ++state_count;
444  } else {
445  //Remember the next source state
446  stack_state_ptr next_source_state = source_state->m_next;
447  //Insert the current source state before the current target state.
448  if (target_state->m_prev == NULL) {
449  //There is no previous, set this as the first one
450  m_recomb_from = source_state;
451  source_state->m_next = target_state;
452  target_state->m_prev = source_state;
453  } else {
454  //Set the new one in between the two existing states
455  target_state->m_prev->m_next = source_state;
456  source_state->m_prev = target_state->m_prev;
457  source_state->m_next = target_state;
458  target_state->m_prev = source_state;
459  }
460  //We added a new state to the target
461  ++state_count;
462  //Move on to the next source state
463  source_state = next_source_state;
464  }
465 
466  //Check if we need to stop due to reaching the capacity
467  if (state_count == m_state_data.m_stack_data.m_params.m_num_best_trans) {
468  //We've reached the capacity, need to stop now
469  is_go_on = false;
470 
471  //If we have enough states delete both tails
472  cut_the_tail(target_state);
473  cut_the_tail(source_state);
474  }
475  } else {
476  //Case: (source_state != NULL) && (target_state == NULL)
477 
478  //The target state is NULL, we moved past the last element
479  //of the target list, all the subsequent source list states
480  //are to be added to the end of the target list.
481  last_target_state->m_next = source_state;
482  source_state->m_prev = last_target_state;
483 
484  //Count the remaining source list and delete the extra ones.
485  count_and_prune(state_count, source_state);
486 
487  //No need to proceed, we can stop
488  is_go_on = false;
489  }
490  } else {
491  //The source state is NULL, there is no more elements to merge into the target list
492  if (target_state != NULL) {
493  //Case: (source_state == NULL) && (target_state != NULL)
494 
495  //Count the remaining target list and delete the extra ones.
496  count_and_prune(state_count, target_state);
497  } else {
498  //Case: (source_state == NULL) && (target_state == NULL)
499 
500  //There is nothing to be done here, both lists are exhausted
501  }
502 
503  //No need to proceed, we can stop
504  is_go_on = false;
505  }
506  }

537  inline void expand_left() {
538  LOG_DEBUG1 << ">>>>> expand left from [" << m_state_data.m_s_begin_word_idx
539  << "," << m_state_data.m_s_end_word_idx << "]" << END_LOG;
540 
541  //Store the shorthand to the minimum possible word index
542  const int32_t & MIN_WORD_IDX = m_state_data.m_stack_data.m_sent_data.m_min_idx;
543 
544  //Compute the minimum position to consider, based on distortion
545  int32_t min_pos;
546  if (m_state_data.m_stack_data.m_params.m_is_dist) {
547  //Compute the normal minimum position for distortion
548  min_pos = (m_state_data.m_s_end_word_idx - m_state_data.m_stack_data.m_params.m_distortion);
549  //Bound the position by the minimum word index
550  min_pos = max(min_pos, MIN_WORD_IDX);
551  } else {
552  //There is no distortion limit
553  min_pos = MIN_WORD_IDX;
554  }
555 
556  //We shall start expanding from the first word before
557  //the beginning of the last translated phrase
558  int32_t curr_pos = (m_state_data.m_s_begin_word_idx - 1);
559 
560  LOG_DEBUG << "start pos = " << curr_pos << ", min pos = " << min_pos << END_LOG;
561 
562  //Store the number of left expansions made
563  size_t num_exp = 0;
564 
565  //Iterate to the left of the last begin positions until the
566  //position is valid and the distortion is within the limits
567  while (curr_pos >= min_pos) {
568  //Expand the state to the words of the last phrase if not covered
569  expand_length_if_not_covered(curr_pos, num_exp);
570  }
571 
572  LOG_DEBUG << "Number of regular left expansion position: " << num_exp << END_LOG;
573 
574  //If we could not expand to the left at all then use the
575  //extra distortion limit. At this point there is no need
576  //to check for whether the distortion is on, if it is not
577  //then we have already gone past the MIN_WORD_IDX.
578  if (num_exp == 0) {
579  while ((curr_pos >= MIN_WORD_IDX) &&
580  (num_exp < m_state_data.m_stack_data.m_params.m_ext_dist_left)) {
581  //Expand the state to the words of the last phrase if not covered
582  expand_length_if_not_covered(curr_pos, num_exp);
583  }
584  LOG_DEBUG << "Number of extra left expansion position: " << num_exp << END_LOG;
585  }
586 
587  LOG_DEBUG1 << "<<<<< expand left" << END_LOG;
588  }

636  inline void expand_length(const size_t start_pos) {
637  LOG_DEBUG1 << ">>>>> [start_pos] = [" << start_pos << "]" << END_LOG;
638 
639  //Always take the one word translation even if
640  //It is an unknown entry.
641  size_t end_pos = start_pos;
642  //Just expand the single word, this is always needed and possible
643  expand_trans<true>(start_pos, end_pos);
644 
645  //Iterate through lengths > 1 until the maximum possible source phrase length is
646  //reached or we hit the end of the sentence or the end of the uncovered region
647  for (size_t len = 2; len <= m_state_data.m_stack_data.m_params.m_max_s_phrase_len; ++len) {
648  //Move to the next end position
649  ++end_pos;
650  //Check if we are in a good state
651  if ((end_pos < m_state_data.m_stack_data.m_sent_data.get_dim()) && (!m_state_data.m_covered[end_pos])) {
652  //Expand the source phrase which is not a single word
653  expand_trans<false>(start_pos, end_pos);
654  } else {
655  //We've hit the end possible length here
656 
657  break;
658  }
659  }
660 
661  LOG_DEBUG1 << "<<<<< [start_pos] = [" << start_pos << "]" << END_LOG;
662  }

667  template<bool single_word>
668  inline void expand_trans(const size_t start_pos, const size_t end_pos) {
669  LOG_DEBUG1 << ">>>>> [start_pos, end_pos] = [" << start_pos << ", " << end_pos << "]" << END_LOG;
670 
671  //Obtain the source entry for the currently considered source phrase
672  tm_const_source_entry_ptr entry = m_state_data.m_stack_data.m_sent_data[start_pos][end_pos].m_source_entry;
673 
674  ASSERT_SANITY_THROW((entry == NULL),
675  string("The source entry [") + to_string(start_pos) +
676  string(", ") + to_string(end_pos) + string("] is NULL!"));
677 
678  //Check if we are in the situation of a single word
679  if (single_word || entry->has_translations()) {
680  //Get the targets
681  tm_const_target_entry* targets = entry->get_targets();
682 
683  //Initialize the new covered vector, take the old one plus enable the new states
684  typename state_data::covered_info covered(m_state_data.m_covered);
685  for (phrase_length idx = start_pos; idx <= end_pos; ++idx) {
686  covered.set(idx);
687  }
688 
689  //Iterate through all the available target translations
