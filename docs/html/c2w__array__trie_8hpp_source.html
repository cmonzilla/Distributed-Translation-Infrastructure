95  inline int8_t compare(const TCtxIdProbData & one, const TCtxIdProbData & two) {
96  if (one.word_id < two.word_id) {
97  return -1;
98  } else {
99  if (one.word_id == two.word_id) {
100  if (one.ctx_id < two.ctx_id) {
101  return -1;
102  } else {
103  if (one.ctx_id == two.ctx_id) {
104  return 0;
105  } else {
106  return +1;
107  }
108  }
109  } else {
110  return +1;
111  }
112  }
113  };

166  inline bool get_ctx_id(const phrase_length level_idx, const TShortId word_id, TLongId & ctx_id) const {
167  //Compute back the current level pure for debug purposes.
168  const phrase_length curr_level = level_idx + BASE::MGRAM_IDX_OFFSET;
169 
170  //Perform sanity checks if needed
171  ASSERT_SANITY_THROW(((curr_level == LM_M_GRAM_LEVEL_MAX) || (curr_level < M_GRAM_LEVEL_2)), string("Unsupported level id: ") + std::to_string(curr_level));
172 
173  LOG_DEBUG2 << "Searching for the next ctx_id of " << SSTR(curr_level)
174  << "-gram with word_id: " << SSTR(word_id) << ", ctx_id: "
175  << SSTR(ctx_id) << END_LOG;
176 
177  //First get the sub-array reference. Note that, even if it is the 2-Gram
178  //case and the previous word is unknown (ctx_id == 0) we still can use
179  //the ctx_id to get the data entry. The reason is that we allocated memory
180  //for it but being for an unknown word context it should have no data!
181  TSubArrReference & ref = m_m_gram_ctx_2_data[level_idx][ctx_id];
182 
183  LOG_DEBUG2 << "Got context mapping for ctx_id: " << SSTR(ctx_id)
184  << ", with beginIdx: " << SSTR(ref.begin_idx) << ", endIdx: "
185  << SSTR(ref.end_idx) << END_LOG;
186 
187  //Check that there is data for the given context available
188  if (ref.begin_idx != BASE::UNDEFINED_ARR_IDX) {
189  //The data is available search for the word index in the array
190  //WARNING: The linear search here is much slower!!!
191  if (my_bsearch_id<TWordIdPBEntry>(m_m_gram_data[level_idx], ref.begin_idx, ref.end_idx, word_id, ctx_id)) {
192  LOG_DEBUG2 << "The next ctx_id for word_id: " << SSTR(word_id) << ", is: " << SSTR(ctx_id) << END_LOG;
193  return true;
194  }
195  }
196  return false;
197  }

258  template<phrase_length CURR_LEVEL>
259  inline void add_m_gram(const model_m_gram & gram) {
260  const TShortId word_id = gram.get_last_word_id();
261  if (CURR_LEVEL == M_GRAM_LEVEL_1) {
262  //Store the payload
263  m_1_gram_data[word_id] = gram.m_payload;
264  } else {
265  //Register the m-gram in the hash cache
266  this->register_m_gram_cache(gram);
267 
268  //Define the context id variable
269  TLongId ctx_id = UNKNOWN_WORD_ID;
270  //Obtain the m-gram context id
271  __LayeredTrieBase::get_context_id<C2WArrayTrie<WordIndexType>, CURR_LEVEL, DebugLevelsEnum::DEBUG2>(*this, gram, ctx_id);
272 
273  if (CURR_LEVEL == LM_M_GRAM_LEVEL_MAX) {
274  //Get the new n-gram index
275  const TShortId n_gram_idx = m_m_n_gram_next_ctx_id[BASE::N_GRAM_IDX_IN_M_N_ARR]++;
276 
277  //Store the context and word ids
278  m_n_gram_data[n_gram_idx].ctx_id = ctx_id;
279  m_n_gram_data[n_gram_idx].word_id = word_id;
280 
281  //Store the payload
282  m_n_gram_data[n_gram_idx].prob = gram.m_payload.m_prob;
283  } else {
284  //Compute the m-gram index
285  const phrase_length m_gram_idx = CURR_LEVEL - BASE::MGRAM_IDX_OFFSET;
286 
287  //First get the sub-array reference.
288  TSubArrReference & ref = m_m_gram_ctx_2_data[m_gram_idx][ctx_id];
289 
290  //Get the new index and increment - this will be the new end index
291  ref.end_idx = m_m_n_gram_next_ctx_id[m_gram_idx]++;
292 
293  //Check if there are yet no elements for this context
294  if (ref.begin_idx == BASE::UNDEFINED_ARR_IDX) {
295  //There was no elements put into this context, the begin index is then equal to the end index
296  ref.begin_idx = ref.end_idx;
297  }
298 
299  //Store the word id
300  m_m_gram_data[m_gram_idx][ref.end_idx].id = word_id;
301 
302  //Store the payload
303  m_m_gram_data[m_gram_idx][ref.end_idx].payload = gram.m_payload;
304  }
305  }

328  inline void get_m_gram_payload(m_gram_query & query, MGramStatusEnum & status) const {
329  LOG_DEBUG << "Getting the payload for sub-m-gram : " << query << END_LOG;
330 
331  //First ensure the context of the given sub-m-gram
332  LAYERED_BASE_ENSURE_CONTEXT(query, status);
333 
334  //If the context is successfully ensured, then move on to the m-gram and try to obtain its payload
335  if (status == MGramStatusEnum::GOOD_PRESENT_MGS) {
336  //Store the shorthand for the context and end word id
337  TLongId & ctx_id = query.get_curr_ctx_ref();
338  const TShortId & word_id = query.get_curr_end_word_id();
339 
340  //Compute the distance between words
341  const phrase_length & curr_level = query.get_curr_level();
342  LOG_DEBUG << "curr_level: " << SSTR(curr_level) << ", ctx_id: " << ctx_id << ", m_end_word_idx: "
343  << SSTR(query.m_curr_end_word_idx) << ", end word id: " << word_id << END_LOG;
344 
345  //Get the next context id
346  const phrase_length & level_idx = query.get_curr_level_m2();
347  if (get_ctx_id(level_idx, word_id, ctx_id)) {
348  LOG_DEBUG << "level_idx: " << SSTR(level_idx) << ", ctx_id: " << ctx_id << END_LOG;
349  //There is data found under this context
350  query.set_curr_payload(&m_m_gram_data[level_idx][ctx_id].payload);
351  LOG_DEBUG << "The payload is retrieved: " << m_m_gram_data[level_idx][ctx_id].payload << END_LOG;
352  } else {
353  //The payload could not be found
354  LOG_DEBUG1 << "Unable to find m-gram data for ctx_id: " << SSTR(ctx_id)
355  << ", word_id: " << SSTR(word_id) << END_LOG;
356  status = MGramStatusEnum::BAD_NO_PAYLOAD_MGS;
357  }
358  LOG_DEBUG << "Context ensure status is: " << status_to_string(status) << END_LOG;
359  }

366  inline void get_n_gram_payload(m_gram_query & query, MGramStatusEnum & status) const {
367  //First ensure the context of the given sub-m-gram
368  LAYERED_BASE_ENSURE_CONTEXT(query, status);
369 
370  //If the context is successfully ensured, then move on to the m-gram and try to obtain its payload
371  if (status == MGramStatusEnum::GOOD_PRESENT_MGS) {
372  //Store the shorthand for the context and end word id
373  TLongId & ctx_id = query.get_curr_ctx_ref();
374  const TShortId & word_id = query.get_curr_end_word_id();
375 
376  LOG_DEBUG2 << "Getting " << SSTR(LM_M_GRAM_LEVEL_MAX) << "-gram with word_id: "
377  << SSTR(word_id) << ", ctx_id: " << SSTR(ctx_id) << END_LOG;
378 
379  //Create the search key by combining ctx and word ids, see TCtxIdProbEntryPair
380  const TLongId key = put_32_32_in_64(word_id, ctx_id);
381  LOG_DEBUG4 << "Searching N-Gram: TShortId_TShortId_2_TLongId(word_id = " << SSTR(word_id)
382  << ", ctx_id = " << SSTR(ctx_id) << ") = " << SSTR(key) << END_LOG;
383 
384  //Search for the index using binary search
385  TShortId idx = BASE::UNDEFINED_ARR_IDX;
386  if (my_bsearch_wordId_ctxId<TCtxIdProbEntry>(m_n_gram_data, BASE::FIRST_VALID_CTX_ID,
387  m_m_n_gram_num_ctx_ids[BASE::N_GRAM_IDX_IN_M_N_ARR] - 1, word_id, ctx_id, idx)) {
388  //Return the data
389  query.set_curr_payload(&m_n_gram_data[idx].prob);
390  LOG_DEBUG << "The payload is retrieved: " << m_n_gram_data[idx].prob << END_LOG;
391  } else {
392  //The payload could not be found
393  LOG_DEBUG1 << "Unable to find " << SSTR(LM_M_GRAM_LEVEL_MAX) << "-gram data for ctx_id: " << SSTR(ctx_id)
394  << ", word_id: " << SSTR(word_id) << ", key " << SSTR(key) << END_LOG;
395  status = MGramStatusEnum::BAD_NO_PAYLOAD_MGS;
396  }

426  template<phrase_length CURR_LEVEL>
427  inline void post_m_grams() {
428  //Compute the m-gram index
429  constexpr phrase_length mgram_idx = (CURR_LEVEL - BASE::MGRAM_IDX_OFFSET);
430 
431  LOG_DEBUG2 << "Running post actions on " << CURR_LEVEL << "-grams, m-gram array index: " << mgram_idx << END_LOG;
432 
433  //Sort the entries per context with respect to the word index
434  //the order could be arbitrary if a non-basic word index is used!
435  const size_t num_prev_ctx = (CURR_LEVEL == M_GRAM_LEVEL_2) ? m_one_gram_arr_size : m_m_n_gram_num_ctx_ids[mgram_idx - 1];
436  LOG_DEBUG2 << "Number of previous contexts: " << num_prev_ctx << END_LOG;
437  for (size_t ctx_id = 0; ctx_id < num_prev_ctx; ++ctx_id) {
438  const TSubArrReference & info = m_m_gram_ctx_2_data[mgram_idx][ctx_id];
439  if (info.begin_idx != BASE::UNDEFINED_ARR_IDX) {
440  LOG_DEBUG3 << "Sorting for context id: " << ctx_id << ", info.beginIdx: "
441  << info.begin_idx << ", info.endIdx: " << info.end_idx << END_LOG;
442  my_sort<TWordIdPBEntry>(&m_m_gram_data[mgram_idx][info.begin_idx], (info.end_idx - info.begin_idx) + 1);
443  }
444  }
445  }

447  inline void post_n_grams() {
448  LOG_DEBUG2 << "Sorting the N-gram's data: ptr: " << m_n_gram_data
449  << ", size: " << m_m_n_gram_num_ctx_ids[BASE::N_GRAM_IDX_IN_M_N_ARR] << END_LOG;
450 
451  //Order the N-gram array as it is unordered and we will binary search it later!
452  //Note: We dot not use Q-sort as it needs quite a lot of extra memory!
453  //Also, I did not yet see any performance advantages compared to sort!
454  //Actually the qsort provided here was 50% slower on a 20 Gb language
455  //model when compared to the str::sort!
456  my_sort<TCtxIdProbEntry>(m_n_gram_data, m_m_n_gram_num_ctx_ids[BASE::N_GRAM_IDX_IN_M_N_ARR]);
457  };
