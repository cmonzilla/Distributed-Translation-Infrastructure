70  static inline void * memrchr(const void * s, const char c, size_t n) {
71  if (n != 0) {
72  const unsigned char * cp = (unsigned char *) s + n;
73  do {
74  if (*(--cp) == (unsigned char) c)
75  return ((void *) cp);
76  } while (--n != 0);
77  }
78  return ((void *) 0);
79  }

224  static inline std::string& punctuate(std::string& str,
225  const std::string& fill = UTF8_SPACE_STRING,
226  const std::string& puncts = UTF8_ASCII_PUNCTUATIONS) {
227  LOG_DEBUG4 << "Tokenizing the string '" << str << "'" << END_LOG;
228  //Find the delimiters and surround them with fill symbols
229  size_t delim_pos = str.find_first_of(puncts);
230  while (delim_pos != std::string::npos) {
231  //Surround the delimiter with two fill elements
232  str.replace(delim_pos, 1, fill + str[delim_pos] + fill);
233  //Search for the next delimiter, skip the positions we just put our new values into
234  delim_pos = str.find_first_of(puncts, delim_pos + 2 * fill.length() + 1);
235  }
236 
237  LOG_DEBUG4 << "The string after tokenizing is '" << str << "'" << END_LOG;
238  return str;

345  static inline bool fast_s_to_f(float & res, const char *p) {
346  //Store the original ptr
347  const char * original = p;
348 
349  uint_fast32_t int_part = 0.0;
350  uint_fast32_t c = 0; // counter to check how many numbers we got!
351 
352  // Get the sign!
353  bool neg = false;
354  if (*p == '-') {
355  neg = true;
356  ++p;
357  } else if (*p == '+') {
358  neg = false;
359  ++p;
360  }
361 
362  // Get the digits before decimal point
363  while (valid_digit(*p)) {
364  int_part = (int_part * 10) + (*p - '0');
365  ++p;
366  ++c;
367  }
368  res = int_part;
369 
370  // Get the digits after decimal point
371  if (*p == '.') {
372  uint_fast32_t dec_part = 0;
373  uint_fast32_t scale = 1;
374  ++p;
375  while (valid_digit(*p)) {
376  dec_part = (dec_part * 10) + (*p - '0');
377  ++p;
378  scale *= 10;
379  ++c;
380  }
381  res += dec_part * (1.0f / scale);
382  }
383 
384  // FIRST CHECK:
385  if (c == 0) {
386  return false;
387  } // we got no dezimal places! this cannot be any number!
388 
389  // Get the digits after the "e"/"E" (exponenet)
390  if (*p == 'e' || *p == 'E') {
391  uint_fast64_t e = 0;
392 
393  LOG_DEBUG4 << "Got exponent in: " << original << END_LOG;
394 
395  bool negE = false;
396  ++p;
397  if (*p == '-') {
398  negE = true;
399  ++p;
400  LOG_DEBUG4 << "Exponent is Negative" << END_LOG;
401  } else if (*p == '+') {
402  negE = false;
403  ++p;
404  LOG_DEBUG4 << "Exponent is Positive" << END_LOG;
405  }
406 
407  // Get exponent
408  c = 0;
409  while (valid_digit(*p)) {
410  e = (e * 10) + (*p - '0');
411  ++p;
412  ++c;
413  }
414  LOG_DEBUG4 << "Exponent value is: " << e << END_LOG;
415 
416  //Get the min and max exponent values, both should be taken as positive values
417  static constexpr uint max_exponent10 = std::numeric_limits<float>::min_exponent10;
418  static constexpr uint min_exponent10 = -std::numeric_limits<float>::min_exponent10;
419 
420  if (!neg && e > max_exponent10) {
421  e = max_exponent10;
422  } else if (e > min_exponent10) {
423  e = min_exponent10;
424  }
425 
426  LOG_DEBUG4 << "Adjusted exponent value is: " << e << END_LOG;
427 
428  // SECOND CHECK:
429  if (c == 0) {
430  return false;
431  } // we got no exponent! this was not intended!!
432 
433  double scale_exp = 1.0;
434  // Calculate scaling factor.
435  while (e > 0) {
436  scale_exp *= 10;
437  e -= 1;
438  }
439 
440  LOG_DEBUG4 << "Exponent multiplier is: " << scale_exp << END_LOG;
441 
442  if (negE) {
443  res *= 1.0 / scale_exp;
444  } else {
445  res *= scale_exp;
446  }
447 
448  LOG_DEBUG4 << "The result is: " << ((neg) ? -res : res) << END_LOG;
449  }
450 
451  // Apply sign to number
452  if (neg) {
453  res = -res;
454  }
455 
456  return true;
457  }
458  }
