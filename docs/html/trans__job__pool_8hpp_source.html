101  void stop() {
102  LOG_DEBUG << "Request stopping the job pool!" << END_LOG;
103 
104  //Make sure this does not interfere with any adding new job activity
105  {
106  scoped_guard guard_stopping(m_stopping_lock);
107 
108  //If we are not stopping yet, do a stop, else return
109  if (!m_is_stopping) {
110  //Set the stopping flag
111  m_is_stopping = true;
112  } else {
113  return;
114  };
115  }
116 
117  LOG_DEBUG << "The stopping flag is set!" << END_LOG;
118 
119  //Cancel all the remaining jobs, do that without the stopping lock synchronization
120  //If put inside the above synchronization block will most likely cause a deadlock(?).
121  cancel_all_jobs();
122 
123  LOG_DEBUG << "All the existing jobs are canceled!" << END_LOG;
124 
125  //Wake up the jobs thread for the case there is no jobs being processed
126  wake_up_jobs_thread();
127 
128  //Wait until the job processing thread finishes
129  m_jobs_thread.join();
130 
131  LOG_DEBUG << "The result processing thread is finished!" << END_LOG;
132 
133  //In case that we have stopped with running jobs - report an error
134  if (m_job_count != 0) {
135  LOG_ERROR << "The jobs pool has stopped but there are still "
136  << m_job_count << " running jobs left!" << END_LOG;

201  void cancel_jobs(const session_id_type session_id) {
202  recursive_guard guard_all_jobs(m_all_jobs_lock);
203 
204  LOG_DEBUG << "Canceling the jobs of session: " << session_id << END_LOG;
205 
206  //Get the jobs jobs for the given session
207  if (m_sessions_map.find(session_id) != m_sessions_map.end()) {
208  //Get the known jobs map for the given session
209  jobs_map_type & m_jobs_map = m_sessions_map[session_id];
210 
211  //Iterate through the available jobs and cancel them
212  for (jobs_map_iter_type iter = m_jobs_map.begin(); iter != m_jobs_map.end(); ++iter) {
213  //Get the translation job for future use
214  trans_job_ptr job = iter->second;
215  LOG_DEBUG << "Canceling the translation job " << this << " with id " << job->get_job_id() << END_LOG;
216  //Call the canceling method on the job
217  job->cancel();
218  }
219  LOG_DEBUG << "Finished canceling the jobs of session: " << session_id << END_LOG;
220  } else {
221  LOG_DEBUG << "The session with id: " << session_id << " has no jobs!" << END_LOG;
222  }

253  LOG_DEBUG << "Adding the job with ptr: " << trans_job << " to the job pool" << END_LOG;

259 
260  //Check that the job with the same id does not exist
261  ASSERT_CONDITION_THROW((m_sessions_map[session_id][job_id] != NULL),

264 
265  //The session is present, so we need to add it into the pool
266  m_sessions_map[session_id][job_id] = trans_job;
267

269  m_job_count++;
270

274  m_tasks_pool.plan_new_task(*it);
275  }
276

384  void process_finished_jobs() {
385  unique_guard guard_finished_jobs(m_finished_jobs_lock);
386 
387  //Stop iteration only when we are stopping and there are no jobs left
388  while (!is_stop_running()) {
389  //Wait the thread to be notified
390  m_is_job_done.wait(guard_finished_jobs);
391 
392  LOG_DEBUG << "Processing finished jobs!" << END_LOG;
393 
394  //The thread is notified, process the finished jobs
395  for (jobs_list_iter_type iter = m_done_jobs_list.begin(); iter != m_done_jobs_list.end();
396  /*The shift is done by itself when erasing the element!*/) {
397 
398  //Get the translation job pointer
399  trans_job_ptr trans_job = *iter;
400 
401  LOG_DEBUG << "Got the finished job ptr: " << trans_job << " to process." << END_LOG;
402 
403  LOG_DEBUG << "The job " << trans_job << " id is " << trans_job->get_job_id() <<
404  " session id is " << trans_job->get_session_id() << END_LOG;
405 
406  //Do the sanity check assert
407  ASSERT_SANITY_THROW(!m_notify_job_finished_func,
408  "The job pool's result setting function is not set!");
409 
410  //Send the job response
411  m_notify_job_finished_func(trans_job);
412 
413  LOG_DEBUG << "Erasing the job " << trans_job << " from the done jobs list" << END_LOG;
414 
415  //Erase the processed job from the list of finished jobs
416  m_done_jobs_list.erase(iter);
417 
418  LOG_DEBUG << "Deleting the job " << trans_job << " instance" << END_LOG;
419 
420  //Remove the job from the pool's administration and destroy
421  delete_job(trans_job);
422 
423  LOG_DEBUG << "The job " << trans_job << " is processed and deleted!" << END_LOG;
424  }
425 
426  LOG_DEBUG << "Finished processing finished jobs, back to sleep or stop!" << END_LOG;
427  }
