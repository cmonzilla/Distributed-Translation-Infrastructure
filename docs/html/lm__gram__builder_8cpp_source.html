75  template<typename WordIndexType, phrase_length CURR_LEVEL, bool is_mult_weight>
76  bool lm_gram_builder<WordIndexType, CURR_LEVEL, is_mult_weight>::parse_to_gram(text_piece_reader &line) {
77  //Read the first element until the tab, we read until the tab because it should be the probability
78  if (line.get_first_tab(m_token)) {
79  //Try to parse the probability to float
80  if (fast_s_to_f(m_m_gram.m_payload.m_prob, m_token.get_rest_c_str())) {
81  LOG_DEBUG2 << "Parsed the N-gram probability: " << m_m_gram.m_payload.m_prob << END_LOG;
82 
83  //In case we need to multiply with the lambda weight do it now;
84  if (is_mult_weight) {
85  m_m_gram.m_payload.m_prob *= m_params.get_lm_weight();
86  LOG_DEBUG2 << "Weighted N-gram probability: " << m_m_gram.m_payload.m_back << END_LOG;
87  }
88 
89  //Start the new m-gram
90  m_m_gram.start_new_m_gram();
91 
92  //Read the first N-1 tokens of the N-gram - space separated
93  for (int i = 0; i < (CURR_LEVEL - 1); i++) {
94  if (!line.get_first_space(m_m_gram.get_next_new_token())) {
95  LOG_WARNING << "An unexpected end of line '" << line.str()
96  << "' when reading the " << (i + 1)
97  << "'th " << CURR_LEVEL << "-gram token!" << END_LOG;
98  //The unexpected end of file, broken file format (?)
99  return false;
100  }
101  }
102 
103  //Read the last token of the N-gram, which is followed by the new line or a tab
104  if (!line.get_first_tab(m_m_gram.get_next_new_token())) {
105  LOG_WARNING << "An unexpected end of line '" << line.str()
106  << "' when reading the " << CURR_LEVEL << "'th "
107  << CURR_LEVEL << "-gram token!" << END_LOG;
108  //The unexpected end of file, broken file format (?)
109  return false;
110  }
111 
112  //Now if there is something left it should be the back-off weight, otherwise we are done
113  if (line.has_more()) {
114  //Take the remainder of the line and try to parse it!
115  if (!fast_s_to_f(m_m_gram.m_payload.m_back, line.get_rest_c_str())) {
116  LOG_WARNING << "Could not parse the remainder of the line '" << line.str()
117  << "' as a back-off weight!" << END_LOG;
118  //The first token was not a float, need to skip to another N-Gram section(?)
119  return false;
120  }
121  LOG_DEBUG2 << "Parsed the N-gram back-off weight: " << m_m_gram.m_payload.m_back << END_LOG;
122 
123  //In case we need to multiply with the lambda weight do it now;
124  if (is_mult_weight) {
125  m_m_gram.m_payload.m_back *= m_params.get_lm_weight();
126  LOG_DEBUG2 << "Weighted N-gram back-off weight: " << m_m_gram.m_payload.m_back << END_LOG;
127  }
128  } else {
129  //There is no back-off so set it to zero
130  m_m_gram.m_payload.m_back = 0.0;
131  LOG_DEBUG2 << "The parsed N-gram '" << line.str()
132  << "' does not have back-off using: " << m_m_gram.m_payload.m_back << END_LOG;
133  }
134  return true;
135  } else {
136  //NOTE: Do it as a debug3 level and not a warning because
137  //this will happen each time we need to move on to a new section!
138  LOG_DEBUG3 << "Could not parse the the string '" << m_token.str()
139  << "' as a probability!" << END_LOG;
140  //The first token was not a float, need to skip to another N-Gram section(?)
141 
142  //Take the line and convert it into a string, then trim it.
143  string line = m_token.str();
144  trim(line);
145  //Skip to the next section only if we are dealing with a non-empty line!
146  return (line == "");
147  }
148  } else {
149  LOG_DEBUG3 << "An unexpected end of line '" << line.str()
150  << "', an empty line detected!" << END_LOG;
151  //The unexpected end of line, it is an empty line so we skip it but keep reading this N-gram section
152  return true;
153  }
154  }

156  template<typename WordIndexType, phrase_length CURR_LEVEL, bool is_mult_weight>
157  bool lm_gram_builder<WordIndexType, CURR_LEVEL, is_mult_weight>::parse_line(text_piece_reader & line) {
158  LOG_DEBUG << "Processing the " << CURR_LEVEL << "-Gram (?) line: '" << line << "'" << END_LOG;
159  //We expect a good input, so the result is set to false by default.
160  bool result = false;
161 
162  //First tokenize as a pattern "prob \t gram \t back-off"
163  if (parse_to_gram(line)) {
164  //Prepare the N-gram and for being added to the trie
165  m_m_gram.template prepare_for_adding<WordIndexType>(m_word_idx);
166 
167  LOG_DEBUG << "Adding a " << SSTR(CURR_LEVEL) << "-Gram "
168  << m_m_gram << " to the Trie" << END_LOG;
169 
170  //Add the obtained N-gram data to the Trie
171  m_add_garm_func(m_m_gram);
172  } else {
173  //If we could not parse the line to gram then it should
174  //be the beginning of the next m-gram section
175  result = true;
176  }
177 
178  LOG_DEBUG << "Finished processing the " << CURR_LEVEL << "-Gram (?) line: '"
179  << line << "', it is " << (result ? "NOT " : "") << "accepted" << END_LOG;
180 
181  return result;
182  }
