65  template<typename WordIndexType>
66  void G2DMapTrie<WordIndexType>::pre_allocate(const size_t counts[LM_M_GRAM_LEVEL_MAX]) {
67  //Call the base-class
68  BASE::pre_allocate(counts);
69 
70  //Pre-allocate the 1-Gram data
71  const size_t num_words = BASE::get_word_index().get_number_of_words(counts[0]);
72  m_1_gram_data = new m_gram_payload[num_words];
73  memset(m_1_gram_data, 0, num_words * sizeof (m_gram_payload));
74 
75  //Initialize the m-gram maps
76  for (phrase_length idx = 1; idx <= BASE::NUM_M_GRAM_LEVELS; ++idx) {
77  m_m_gram_data[idx - 1] = new TProbBackMap(__G2DMapTrie::BUCKETS_FACTOR, counts[idx]);
78  }
79 
80  //Initialize the n-gram's map
81  m_n_gram_data = new TProbMap(__G2DMapTrie::BUCKETS_FACTOR, counts[LM_M_GRAM_LEVEL_MAX - 1]);
82  };
