146  inline prob_weight & initialize_future_costs(const size_t & start_idx, const size_t & end_idx) {
147  //Get the phrase data entry
148  phrase_data_entry & phrase_data = m_sent_data[start_idx][end_idx];
149 
150  //Get the entry pointer
151  const tm_source_entry * source_entry = phrase_data.m_source_entry;
152 
153  //Get the reference to the future cost
154  prob_weight & cost = phrase_data.future_cost;
155 
156  LOG_DEBUG1 << "Initializing future cost for [" << start_idx << ", " << end_idx << "]" << END_LOG;
157 
158  //Check if the source entry is present, the entry should be there!
159  if (source_entry != NULL) {
160  LOG_DEBUG1 << "The source entry of phrase [" << start_idx << ", " << end_idx << "] is present." << END_LOG;
161  //Check if this is a phrase with translation
162  if (source_entry->has_translations()) {
163  LOG_DEBUG1 << "The source entry [" << start_idx << ", " << end_idx << "] has translations." << END_LOG;
164  //Set the value with the pre-computed minimum cost
165  cost = source_entry->get_min_cost();
166  LOG_DEBUG1 << "Initialize phrase cost [" << start_idx << ", " << end_idx << "] = " << cost << END_LOG;
167  } else {
168  LOG_DEBUG1 << "The source entry of phrase [" << start_idx << ", " << end_idx << "] is UNK translation." << END_LOG;
169  //Check if this just a single unknown word
170  if (start_idx == end_idx) {
171  //For a single UNK word we take its minimum cost which will be actually
172  //just the UNK translation cost plus the UNK language model probability
173  cost = source_entry->get_min_cost();
174  LOG_DEBUG1 << "Initialize UNK word cost [" << start_idx << ", " << end_idx << "] = " << cost << END_LOG;
175  } else {
176  //The undefined log probability value "-1000" is set in the phrase data entry in its constructor!
177  LOG_DEBUG1 << "Initialize UNK phrase cost [" << start_idx << ", " << end_idx << "] = " << cost << END_LOG;
178  }
179  }
180  } else {
181  //The longer phrases do not have translations, this is normal!
182  LOG_DEBUG1 << "Initialize TOO-LONG phrase cost [" << start_idx << ", " << end_idx << "] = " << cost << END_LOG;
183  }
184 
185  //Return the reference to the future cost, it will be needed in the caller
186  return cost;
187  }

192  inline void compute_futue_costs() {
193  //Obtain the number of words/tokens in the phrase it is equal to the number of dimensions
194  const phrase_length num_words = m_sent_data.get_dim();
195 
196  //Iterate through all the lengths, the minimum length is one word
197  for (phrase_length length = 1; (length <= num_words); ++length) {
198  LOG_DEBUG1 << "Phrase length: " << length << END_LOG;
199 
200  //Iterate from the first until the last possible index for the given length
201  for (phrase_length start_idx = 0; (start_idx <= (num_words - length)); ++start_idx) {
202  //Compute the end index
203  const phrase_length end_idx = start_idx + length - 1;
204  LOG_DEBUG1 << "CFC start/end word idx: " << start_idx << "/" << end_idx << END_LOG;
205 
206  //Initialize the interval with the TM/LM based value first
207  //and get the reference to the complete cost then
208  prob_weight & phrase_cost = initialize_future_costs(start_idx, end_idx);
209 
210  //Iterate the middle point between start and end indexes
211  for (phrase_length mid_idx = start_idx; (mid_idx < end_idx); ++mid_idx) {
212  LOG_DEBUG1 << "CFC middle word idx: " << mid_idx << END_LOG;
213 
214  //Get the costs of the phrase one and two
215  const prob_weight ph1_cost = m_sent_data[start_idx][mid_idx].future_cost;
216  const prob_weight ph2_cost = m_sent_data[mid_idx + 1][end_idx].future_cost;
217  //Compute the cost of two sub-phrases
218  const prob_weight sub_cost = ph1_cost + ph2_cost;
219 
220  LOG_DEBUG1 << "\t cost [" << start_idx << ", " << mid_idx << "] = " << ph1_cost
221  << " + cost [" << (mid_idx + 1) << ", " << end_idx << "] = " << ph2_cost
222  << " <?> cost [" << start_idx << ", " << end_idx << "] = " << phrase_cost << END_LOG;
223 
224  //If the sub cost that is a logarithmic value of probability (a negative value) is
225  //larger than than of the future cost for the entire phrase then use the sub cost.
226  if (sub_cost > phrase_cost) {
227  //Set the sum ofthe sub-costs as a new cost
228  phrase_cost = sub_cost;
229 
230  LOG_DEBUG << "Changing the cost [" << start_idx << ", " << end_idx
231  << "] = [" << start_idx << ", " << mid_idx << "] + ["
232  << (mid_idx + 1) << ", " << end_idx << "] = " << phrase_cost << END_LOG;
233  } else {
234  LOG_DEBUG << "Keeping the cost [" << start_idx << ", " << end_idx
235  << "] = " << phrase_cost << END_LOG;
236  }
237 
238  //Check if we need to stop, if yes, then return
239  if (m_is_stop) return;
240  }
241  }
242  }

252 
253  //Compute the number of tokens in the sentence

259  string(") exceeds the maximum allowed number of words per sentence (") +
260  to_string(MAX_WORDS_PER_SENTENCE));
261 
262  LOG_DEBUG1 << "The sentence: __" << sentence << "___ has " << num_tokens << " tokens" << END_LOG;
263

270  inline void query_translation_model() {
271  //Fill in the matrix with the phrases and their uids
272  int32_t end_wd_idx = MIN_SENT_WORD_INDEX;
273  //Declare the begin and end character index variables
274  size_t ch_b_idx = 0, ch_e_idx = m_source_sent.find_first_of(UTF8_SPACE_STRING);
275 
276  while (ch_e_idx <= std::string::npos) {
277  //Get the appropriate map entry reference
278  phrase_data_entry & end_word_data = m_sent_data[end_wd_idx][end_wd_idx];
279 
280  //Get the token from the source phrase
281  const string token = m_source_sent.substr(ch_b_idx, ch_e_idx - ch_b_idx);
282 
283  LOG_DEBUG1 << "Found the new token @ [" << ch_b_idx << ","
284  << ch_e_idx << "): ___" << token << "___" << END_LOG;
285 
286  //Store the phrase begin and end character indexes
287  end_word_data.m_begin_ch_idx = ch_b_idx;
288  end_word_data.m_end_ch_idx = ch_e_idx;
289 
290  //Compute the phrase id and store it as well
291  end_word_data.m_phrase_uid = get_phrase_uid<true>(token);
292 
293  LOG_DEBUG1 << "The token ___" << token << "___ @ [" << ch_b_idx << ","
294  << ch_e_idx << ") uid is: " << end_word_data.m_phrase_uid << END_LOG;
295 
296  LOG_DEBUG1 << "Considering the token [" << end_wd_idx << ", " << end_wd_idx << "] translation." << END_LOG;
297 
298  //Get the uni-gram phrase (word) translations
299  m_tm_query.execute(end_word_data.m_phrase_uid, end_word_data.m_source_entry);
300 
301  LOG_DEBUG1 << "The token ___" << token << "___ @ [" << ch_b_idx << ","
302  << ch_e_idx << ") HAS" << (end_word_data.m_source_entry->has_translations() ? "" : " NO")
303  << " translation(s), num entries: " << end_word_data.m_source_entry->num_targets() << END_LOG;
304 
305  //Compute the new phrases and phrase ids for the new column elements,
306  //Note that, the longest phrase length to consider is defined by the
307  //decoding parameters. It is the end word plus several previous.
308  int32_t begin_wd_idx = max(MIN_SENT_WORD_INDEX, end_wd_idx - m_params.m_max_s_phrase_len + 1);
309  for (; (begin_wd_idx < end_wd_idx); ++begin_wd_idx) {
310  LOG_DEBUG1 << "Considering the phrase [" << begin_wd_idx << ", " << end_wd_idx << "] translation." << END_LOG;
311 
312  //Get the previous column entry
313  phrase_data_entry & prev_entry = m_sent_data[begin_wd_idx][end_wd_idx - 1];
314  //Get the new column entry
315  phrase_data_entry & new_entry = m_sent_data[begin_wd_idx][end_wd_idx];
316 
317  //Store the phrase begin and end character indexes
318  new_entry.m_begin_ch_idx = prev_entry.m_begin_ch_idx; // All the phrases in the row begin at the same place
319  new_entry.m_end_ch_idx = end_word_data.m_end_ch_idx; //All the phrases in the column end at the same place
320 
321  //Get the phrase for logging
322  const string phrase = m_source_sent.substr(new_entry.m_begin_ch_idx, new_entry.m_end_ch_idx - new_entry.m_begin_ch_idx);
323 
324  LOG_DEBUG1 << "The phrase [" << begin_wd_idx << ", " << end_wd_idx << "] is ___" << phrase << "___" << END_LOG;
325 
326  //Compute the phrase uid
327  new_entry.m_phrase_uid = combine_phrase_uids(prev_entry.m_phrase_uid, end_word_data.m_phrase_uid);
328 
329  LOG_DEBUG1 << "The phrase ___" << phrase << "___ uid = combine(" << prev_entry.m_phrase_uid
330  << "," << end_word_data.m_phrase_uid << ") = " << new_entry.m_phrase_uid << END_LOG;
331 
332  //Add the m-gram phrase to the query
333  m_tm_query.execute(new_entry.m_phrase_uid, new_entry.m_source_entry);
334 
335  LOG_DEBUG1 << "Phrase: ___" << phrase << "___ uid: " << new_entry.m_phrase_uid << " HAS"
336  << (new_entry.m_source_entry->has_translations() ? "" : " NO") << " translation(s),"
337  << " num entries: " << new_entry.m_source_entry->num_targets() << END_LOG;
338 
339  //Check if we need to stop, if yes, then return
340  if (m_is_stop) return;
341  }
342 
343  //Check on the stop condition
344  if (ch_e_idx == std::string::npos) {
345  break;
346  }
347 
348  //Increment the end word index
349  ++end_wd_idx;
350 
351  //Search for the next delimiter
352  ch_b_idx = ch_e_idx + 1;
353  ch_e_idx = m_source_sent.find_first_of(UTF8_SPACE_STRING, ch_b_idx);
354  }
355  }
