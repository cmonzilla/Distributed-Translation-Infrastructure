157  template<phrase_length CURR_LEVEL>
158  inline void add_m_gram(const model_m_gram & gram) {
159  const TShortId word_id = gram.get_last_word_id();
160  if (CURR_LEVEL == M_GRAM_LEVEL_1) {
161  //Store the payload
162  m_1_gram_data[word_id] = gram.m_payload;
163  } else {
164  //Register the m-gram in the hash cache
165  this->register_m_gram_cache(gram);
166 
167  //Define the context id variable
168  TLongId ctx_id = UNKNOWN_WORD_ID;
169  //Obtain the m-gram context id
170  __LayeredTrieBase::get_context_id<C2DMapTrie<WordIndexType>, CURR_LEVEL, DebugLevelsEnum::DEBUG2>(*this, gram, ctx_id);
171 
172  //Obtain this m-gram id
173  (void) get_ctx_id(CURR_LEVEL - BASE::MGRAM_IDX_OFFSET, word_id, ctx_id);
174 
175  //Store the payload
176  if (CURR_LEVEL == LM_M_GRAM_LEVEL_MAX) {
177  m_n_gram_map_ptr->operator[](ctx_id) = gram.m_payload.m_prob;
178  } else {
179  m_m_gram_map_ptrs[CURR_LEVEL - BASE::MGRAM_IDX_OFFSET]->operator[](ctx_id) = gram.m_payload;
180  }
181  }

204  inline void get_m_gram_payload(m_gram_query & query, MGramStatusEnum & status) const {
205  LOG_DEBUG << "Getting the payload for sub-m-gram : "<< query << END_LOG;
206 
207  //First ensure the context of the given sub-m-gram
208  LAYERED_BASE_ENSURE_CONTEXT(query, status);
209 
210  //If the context is successfully ensured, then move on to the m-gram and try to obtain its payload
211  if (status == MGramStatusEnum::GOOD_PRESENT_MGS) {
212  //Store the shorthand for the context and end word id
213  TLongId & ctx_id = query.get_curr_ctx_ref();
214  const TShortId & word_id = query.get_curr_end_word_id();
215 
216  //Compute the distance between words
217  const phrase_length & curr_level =query.get_curr_level();
218  LOG_DEBUG << "curr_level: " << SSTR(curr_level) << ", ctx_id: " << ctx_id << ", m_end_word_idx: "
219  << SSTR(query.m_curr_end_word_idx) << ", end word id: " << word_id << END_LOG;
220 
221  //Get the next context id
222  const phrase_length & level_idx = query.get_curr_level_m2();
223  if (get_ctx_id(level_idx, word_id, ctx_id)) {
224  LOG_DEBUG << "level_idx: " << SSTR(level_idx) << ", ctx_id: " << ctx_id << END_LOG;
225  TMGramsMap::const_iterator result = m_m_gram_map_ptrs[level_idx]->find(ctx_id);
226  if (result == m_m_gram_map_ptrs[level_idx]->end()) {
227  //The payload could not be found
228  LOG_DEBUG1 << "Unable to find m-gram data for ctx_id: " << SSTR(ctx_id)
229  << ", word_id: " << SSTR(word_id) << END_LOG;
230  status = MGramStatusEnum::BAD_NO_PAYLOAD_MGS;
231  } else {
232  //There is data found under this context
233  query.set_curr_payload(&result->second);
234  LOG_DEBUG << "The payload is retrieved: " << result->second << END_LOG;
235  }
236  } else {
237  //The payload could not be found
238  LOG_DEBUG1 << "Unable to find m-gram data for ctx_id: " << SSTR(ctx_id)
239  << ", word_id: " << SSTR(word_id) << END_LOG;
240  status = MGramStatusEnum::BAD_NO_PAYLOAD_MGS;
241  }
242  LOG_DEBUG << "Context ensure status is: " << status_to_string(status) << END_LOG;
243  }
244  }
