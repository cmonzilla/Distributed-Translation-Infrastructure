102  virtual void get_word_ids(text_piece_reader phrase, phrase_length & num_words,
103  word_uid word_ids[tm::TM_MAX_TARGET_PHRASE_LEN]) const {
104  //Initialize with zero words
105  num_words = 0;
106 
107  //Declare the text piece reader for storing words
108  text_piece_reader word;
109 
110  LOG_DEBUG1 << "Getting word uids for phrase: ___" << phrase << "___" << END_LOG;
111 
112  //Read the tokens one by one backwards and decrement the index
113  while (phrase.get_first_space(word)) {
114  //Check that we do not get too many words!
115  ASSERT_SANITY_THROW((num_words >= tm::TM_MAX_TARGET_PHRASE_LEN),
116  string("The target phrase: ___") + phrase.str() +
117  string("___ has too many words, the allowed maximum is: ") +
118  to_string(tm::TM_MAX_TARGET_PHRASE_LEN));
119 
120  //Obtain the word id from the word index
121  word_ids[num_words] = m_word_idx.get_word_id(word);
122 
123  LOG_DEBUG1 << "Word: ___" << word << "___ has uid: " << word_ids[num_words] << END_LOG;
124 
125  //Increase the number of read words
126  num_words++;
127  }

149  virtual prob_weight execute(const phrase_length num_words,
150  const word_uid * word_ids, phrase_length & min_level) {
151  //Re-initialize the joint prob reault with zero
152  m_joint_prob = 0.0;
153 
154  //Set the words data into the query object
155  m_query.set_data<m_is_ctx>(num_words, word_ids);
156 
157  //Compute the maximum to consider m-gram level
158  const phrase_length max_m_gram_level = std::min<phrase_length>(num_words, LM_M_GRAM_LEVEL_MAX);
159 
160  //Initialize the begin and end word indexes
161  phrase_length begin_word_idx = 0;
162  phrase_length sub_end_word_idx = min_level - 1;
163  phrase_length end_word_idx = max_m_gram_level - 1;
164 
165  //Check that the minimum level is actually possible!
166  ASSERT_SANITY_THROW((sub_end_word_idx > end_word_idx),
167  string("Impossible min_level: ") + to_string(min_level) +
168  string(" the maximum possible level is: ") + to_string(max_m_gram_level));
169 
170  //Set the m-gram values for the first query execution
171  m_query.set_word_indxes(begin_word_idx, sub_end_word_idx, end_word_idx);
172 
173  //Execute the first part of the query
174  m_trie.execute(m_query);
175 
176  //Report the partial results, and update the total
177  get_report_interm_results(begin_word_idx, sub_end_word_idx, end_word_idx);
178 
179  //Now do the sliding window and compute more probabilities,
180  //Note that if the end_word_idx is smaller than the query
181  //last word idx then it means that:
182  // (max_m_gram_level == LM_M_GRAM_LEVEL_MAX)
183  //and there is still m-grams to compute
184  while (end_word_idx < m_query.get_query_end_word_idx()) {
185  //Slide the window one step forward
186  begin_word_idx++;
187  end_word_idx++;
188 
189  //Set the window value inside, this time we need a single probability and not the joint
190  m_query.set_word_indxes(begin_word_idx, end_word_idx);
191 
192  //Execute the query
193  m_trie.execute(m_query);
194 
195  //Report the partial result, and update the total
196  get_report_interm_results(begin_word_idx, end_word_idx, end_word_idx);
197  }
198 
199  //Report the total result
200  report_final_result();
201 
202  //Compute the next minimum level to consider, it is either one level higher or we are at the maximum
203  min_level = std::min<phrase_length>(max_m_gram_level + 1, LM_M_GRAM_LEVEL_MAX);
204 
205  LOG_DEBUG << "Computed log10(Prob(" << m_query << ")) = " << m_joint_prob << ", next min_level: " << min_level << END_LOG;
206 
207  //Return the final result;
208  return m_joint_prob;
209  }

224  inline string get_m_gram_str(const phrase_length begin_word_idx, const phrase_length end_word_idx) const {
225  if (begin_word_idx > end_word_idx) {
226  return "<none>";
227  } else {
228  if (begin_word_idx == end_word_idx) {
229  return to_string(m_query[begin_word_idx]);
230  } else {
231  string result = to_string(m_query[end_word_idx]) + " |";
232  for (phrase_length idx = begin_word_idx; idx != end_word_idx; ++idx) {
233  result += string(" ") + to_string(m_query[idx]);
234  }
235  return result;
236  }
237  }
238  }

247  inline string get_query_str() const {
248  const phrase_length begin_idx = m_query.get_query_begin_word_idx();
249  const phrase_length end_idx = m_query.get_query_end_word_idx();
250  if (begin_idx == end_idx) {
251  return to_string(m_query[begin_idx]);
252  } else {
253  string result;
254  for (phrase_length idx = begin_idx; idx <= end_idx; ++idx) {
255  result += to_string(m_query[idx]) + string(" ");
256  }
257  return result.substr(0, result.length() - 1);
258  }

267  void get_report_interm_results(
268  const phrase_length begin_word_idx,
269  const phrase_length first_end_word_idx,
270  const phrase_length last_end_word_idx) {
271  //Print the intermediate results
272  for (phrase_length end_word_idx = first_end_word_idx; end_word_idx <= last_end_word_idx; ++end_word_idx) {
273  const string gram_str = get_m_gram_str(begin_word_idx, end_word_idx);
274 
275  LOG_DEBUG << " log_" << LOG_PROB_WEIGHT_BASE << "( Prob( " << gram_str
276  << " ) ) = " << SSTR(m_query.m_probs[end_word_idx]) << END_LOG;
277  LOG_DEBUG1 << " Prob( " << gram_str << " ) = "
278  << SSTR(pow(LOG_PROB_WEIGHT_BASE, m_query.m_probs[end_word_idx])) << END_LOG;
279 
280  if (m_query.m_probs[end_word_idx] > ZERO_LOG_PROB_WEIGHT) {
281  m_joint_prob += m_query.m_probs[end_word_idx];
282  }
283  }
