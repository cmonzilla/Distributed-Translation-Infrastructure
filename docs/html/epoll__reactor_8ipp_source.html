92 {
93  if (fork_ev == asio::io_service::fork_child)
94  {
95  if (epoll_fd_ != -1)
96  ::close(epoll_fd_);
97  epoll_fd_ = -1;

99 
100  if (timer_fd_ != -1)
101  ::close(timer_fd_);
102  timer_fd_ = -1;
103  timer_fd_ = do_timerfd_create();
104 
105  interrupter_.recreate();
106 
107  // Add the interrupter's descriptor to epoll.
108  epoll_event ev = { 0, { 0 } };
109  ev.events = EPOLLIN | EPOLLERR | EPOLLET;
110  ev.data.ptr = &interrupter_;
111  epoll_ctl(epoll_fd_, EPOLL_CTL_ADD, interrupter_.read_descriptor(), &ev);
112  interrupter_.interrupt();
113 
114  // Add the timer descriptor to epoll.
115  if (timer_fd_ != -1)
116  {
117  ev.events = EPOLLIN | EPOLLERR;
118  ev.data.ptr = &timer_fd_;
119  epoll_ctl(epoll_fd_, EPOLL_CTL_ADD, timer_fd_, &ev);
120  }
121 
122  update_timeout();
123 
124  // Re-register all descriptors with epoll.
125  mutex::scoped_lock descriptors_lock(registered_descriptors_mutex_);
126  for (descriptor_state* state = registered_descriptors_.first();
127  state != 0; state = state->next_)
128  {
129  ev.events = state->registered_events_;
130  ev.data.ptr = state;
131  int result = epoll_ctl(epoll_fd_, EPOLL_CTL_ADD, state->descriptor_, &ev);
132  if (result != 0)
133  {
134  asio::error_code ec(errno,
135  asio::error::get_system_category());
136  asio::detail::throw_error(ec, "epoll re-registration");
137  }
138  }
139  }

205 void epoll_reactor::start_op(int op_type, socket_type descriptor,
206  epoll_reactor::per_descriptor_data& descriptor_data, reactor_op* op,
207  bool is_continuation, bool allow_speculative)
208 {
209  if (!descriptor_data)
210  {
211  op->ec_ = asio::error::bad_descriptor;
212  post_immediate_completion(op, is_continuation);
213  return;
214  }
215 
216  mutex::scoped_lock descriptor_lock(descriptor_data->mutex_);
217 
218  if (descriptor_data->shutdown_)
219  {
220  post_immediate_completion(op, is_continuation);
221  return;
222  }
223 
224  if (descriptor_data->op_queue_[op_type].empty())
225  {
226  if (allow_speculative
227  && (op_type != read_op
228  || descriptor_data->op_queue_[except_op].empty()))
229  {
230  if (op->perform())
231  {
232  descriptor_lock.unlock();
233  io_service_.post_immediate_completion(op, is_continuation);
234  return;
235  }
236 
237  if (op_type == write_op)
238  {
239  if ((descriptor_data->registered_events_ & EPOLLOUT) == 0)
240  {
241  epoll_event ev = { 0, { 0 } };
242  ev.events = descriptor_data->registered_events_ | EPOLLOUT;
243  ev.data.ptr = descriptor_data;
244  if (epoll_ctl(epoll_fd_, EPOLL_CTL_MOD, descriptor, &ev) == 0)
245  {
246  descriptor_data->registered_events_ |= ev.events;
247  }
248  else
249  {
250  op->ec_ = asio::error_code(errno,
251  asio::error::get_system_category());
252  io_service_.post_immediate_completion(op, is_continuation);
253  return;
254  }
255  }
256  }
257  }
258  else
259  {
260  if (op_type == write_op)
261  {
262  descriptor_data->registered_events_ |= EPOLLOUT;
263  }
264 
265  epoll_event ev = { 0, { 0 } };
266  ev.events = descriptor_data->registered_events_;
267  ev.data.ptr = descriptor_data;
268  epoll_ctl(epoll_fd_, EPOLL_CTL_MOD, descriptor, &ev);
269  }
270  }
271 
272  descriptor_data->op_queue_[op_type].push(op);
273  io_service_.work_started();
274 }

371 void epoll_reactor::run(bool block, op_queue<operation>& ops)
372 {
373  // This code relies on the fact that the task_io_service queues the reactor
374  // task behind all descriptor operations generated by this function. This
375  // means, that by the time we reach this point, any previously returned
376  // descriptor operations have already been dequeued. Therefore it is now safe
377  // for us to reuse and return them for the task_io_service to queue again.
378 
379  // Calculate a timeout only if timerfd is not used.
380  int timeout;
381  if (timer_fd_ != -1)
382  timeout = block ? -1 : 0;
383  else
384  {
385  mutex::scoped_lock lock(mutex_);
386  timeout = block ? get_timeout() : 0;
387  }
388 
389  // Block on the epoll descriptor.
390  epoll_event events[128];
391  int num_events = epoll_wait(epoll_fd_, events, 128, timeout);
392 
393 #if defined(ASIO_HAS_TIMERFD)
394  bool check_timers = (timer_fd_ == -1);
395 #else // defined(ASIO_HAS_TIMERFD)
396  bool check_timers = true;
397 #endif // defined(ASIO_HAS_TIMERFD)
398 
399  // Dispatch the waiting events.
400  for (int i = 0; i < num_events; ++i)
401  {
402  void* ptr = events[i].data.ptr;
403  if (ptr == &interrupter_)
404  {
405  // No need to reset the interrupter since we're leaving the descriptor
406  // in a ready-to-read state and relying on edge-triggered notifications
407  // to make it so that we only get woken up when the descriptor's epoll
408  // registration is updated.
409 
410 #if defined(ASIO_HAS_TIMERFD)
411  if (timer_fd_ == -1)
412  check_timers = true;
413 #else // defined(ASIO_HAS_TIMERFD)
414  check_timers = true;
415 #endif // defined(ASIO_HAS_TIMERFD)
416  }
417 #if defined(ASIO_HAS_TIMERFD)
418  else if (ptr == &timer_fd_)
419  {
420  check_timers = true;
421  }
422 #endif // defined(ASIO_HAS_TIMERFD)
423  else
424  {
425  // The descriptor operation doesn't count as work in and of itself, so we
426  // don't call work_started() here. This still allows the io_service to
427  // stop if the only remaining operations are descriptor operations.
428  descriptor_state* descriptor_data = static_cast<descriptor_state*>(ptr);
429  descriptor_data->set_ready_events(events[i].events);
430  ops.push(descriptor_data);
431  }
432  }
433 
434  if (check_timers)
435  {
436  mutex::scoped_lock common_lock(mutex_);
437  timer_queues_.get_ready_timers(ops);
438 
439 #if defined(ASIO_HAS_TIMERFD)
440  if (timer_fd_ != -1)
441  {
442  itimerspec new_timeout;
443  itimerspec old_timeout;
444  int flags = get_timeout(new_timeout);
445  timerfd_settime(timer_fd_, flags, &new_timeout, &old_timeout);
446  }
447 #endif // defined(ASIO_HAS_TIMERFD)
448  }
449 }

568 struct epoll_reactor::perform_io_cleanup_on_block_exit
569 {
570  explicit perform_io_cleanup_on_block_exit(epoll_reactor* r)
571  : reactor_(r), first_op_(0)
572  {
573  }
574 
575  ~perform_io_cleanup_on_block_exit()
576  {
577  if (first_op_)
578  {
579  // Post the remaining completed operations for invocation.
580  if (!ops_.empty())
581  reactor_->io_service_.post_deferred_completions(ops_);
582 
583  // A user-initiated operation has completed, but there's no need to
584  // explicitly call work_finished() here. Instead, we'll take advantage of
585  // the fact that the task_io_service will call work_finished() once we
586  // return.
587  }
588  else
589  {
590  // No user-initiated operations have completed, so we need to compensate
591  // for the work_finished() call that the task_io_service will make once
592  // this operation returns.
593  reactor_->io_service_.work_started();
594  }
595  }

598  op_queue<operation> ops_;
599  operation* first_op_;

607 operation* epoll_reactor::descriptor_state::perform_io(uint32_t events)
608 {
609  mutex_.lock();
610  perform_io_cleanup_on_block_exit io_cleanup(reactor_);
611  mutex::scoped_lock descriptor_lock(mutex_, mutex::scoped_lock::adopt_lock);
612 
613  // Exception operations must be processed first to ensure that any
614  // out-of-band data is read before normal data.
615  static const int flag[max_ops] = { EPOLLIN, EPOLLOUT, EPOLLPRI };
616  for (int j = max_ops - 1; j >= 0; --j)
617  {
618  if (events & (flag[j] | EPOLLERR | EPOLLHUP))
619  {
620  while (reactor_op* op = op_queue_[j].front())
621  {
622  if (op->perform())
623  {
624  op_queue_[j].pop();
625  io_cleanup.ops_.push(op);
626  }
627  else
628  break;
629  }
630  }
631  }
632 
633  // The first operation will be returned for completion now. The others will
634  // be posted for later by the io_cleanup object's destructor.
635  io_cleanup.first_op_ = io_cleanup.ops_.front();
636  io_cleanup.ops_.pop();
637  return io_cleanup.first_op_;
638 }
