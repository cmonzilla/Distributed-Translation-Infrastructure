85  static void execute_queries(TFileReaderQuery &test_file) {
86  //Declare time variables for CPU times in seconds
87  double start_time = 0.0, end_time = 0.0;
88 
89  //Will store the read line (word1 word2 word3 word4 word5)
90  text_piece_reader line;
91 
92  //Get the query executor proxy object
93  lm_slow_query_proxy & query = lm_configurator::allocate_slow_query_proxy();
94 
95  LOG_USAGE << "Start reading and executing the test queries ..." << END_LOG;
96 
97  //Start the timer
98  start_time = stat_monitore::get_cpu_time();
99 
100  //Read the test file line by line
101  while (test_file.get_first_line(line)) {
102  LOG_DEBUG << "Got query line [ " << line.str() << " ]" << END_LOG;
103 
104  try {
105  //Query the Trie for the results and log them
106  query.execute(line);
107  } catch (uva_exception & ex) {
108  //The query has failed! print an exception and proceed!
109  LOG_ERROR << ex.get_message() << END_LOG;
110  }
111  }
112 
113  //Stop the timer
114  end_time = stat_monitore::get_cpu_time();
115 
116  //Dispose the query
117  lm_configurator::dispose_slow_query_proxy(query);
118 
119  LOG_USAGE << "Total query execution time is " << (end_time - start_time) << " CPU seconds." << END_LOG;
120  }
