89  inline word_uid get_word_id(const text_piece_reader & token) const {
90  //Return the word index making sure that it is at least
91  //equal to two. So that the undefined and unknown word
92  //indexes are not used and no overflow or other checks.
93  const uint64_t hash_value = compute_hash(token.get_begin_c_str(), token.length());
94  const word_uid word_id = hash_value | (1 << 1);
95  LOG_DEBUG2 << "Hashing '" << token << "' into: " << hash_value << ", resulting id is: " << word_id << END_LOG;
96  return word_id;
97  };
