84  inline void pre_allocate(const size_t num_elems, const uint8_t buckets_factor) {
85  if (DO_SANITY_CHECKS && (m_data_ptr != NULL)) {
86  throw Exception("The bitset is already pre-allocated!");
87  }
88 
89  if (num_elems != 0) {
90  //Compute the number of buckets as a power of two, so that we do not need to use %
91  m_num_buckets = const_expr::power(2, const_expr::ceil(const_expr::log2(buckets_factor * num_elems)));
92  m_buckets_capacity = m_num_buckets - 1;
93  size_t num_bytes = NUM_BYTES_4_BITS(m_num_buckets);
94 
95  LOG_DEBUG << "num_elems: " << num_elems
96  << " m_num_buckets: " << m_num_buckets
97  << " bytes: " << num_bytes << END_LOG;
98 
99  m_data_ptr = new uint8_t[num_bytes];
100  fill_n(m_data_ptr, num_bytes, 0u);
101  } else {
102  throw Exception("Trying to pre-allocate 0 elements for a BitmaphashCache!");
103  }
104  }
