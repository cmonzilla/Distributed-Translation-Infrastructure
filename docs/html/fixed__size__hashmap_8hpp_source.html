77  explicit fixed_size_hashmap(const double buckets_factor, const IDX_TYPE num_elems) : MAX_ELEMENT_INDEX(num_elems) {
78  //Compute and set the number of buckets and the buckets divider
79  set_number_of_elements(buckets_factor, num_elems);

144  ELEMENT_TYPE * get_element(const uint_fast64_t key_uid, const KEY_TYPE & key) const {
145  //Get the bucket index from the hash
146  uint_fast64_t bucket_idx = get_bucket_idx(key_uid);
147 
148  LOG_DEBUG3 << "Got bucket_idx: " << bucket_idx << " for uid value: " << key_uid << END_LOG;
149 
150  //Search for the first empty bucket
151  while (m_buckets[bucket_idx] != NO_ELEMENT_INDEX) {
152  //Check if the element is equal to the key
153  if (m_elems[m_buckets[bucket_idx]] == key) {
154  //The element is found, return
155  LOG_DEBUG3 << "Found the element index: " << m_buckets[bucket_idx]
156  << " in bucket: " << bucket_idx << END_LOG;
157  return &m_elems[m_buckets[bucket_idx]];
158  }
159  //The element is not found, move to the next one
160  get_next_bucket_idx(bucket_idx);
161  }
162 
163  LOG_DEBUG3 << "Could not find an element for key uid: " << key_uid
164  << ", last bucket index: " << bucket_idx << END_LOG;
165 
166  return NULL;
167  }

202  inline void set_number_of_elements(const double buckets_factor, const IDX_TYPE num_elems) {
203  //Do a compulsory assert on the buckets factor
204 
205  ASSERT_CONDITION_THROW((buckets_factor < 1.0), string("buckets_factor: ") +
206  std::to_string(buckets_factor) + string(", must be >= 1.0"));
207 
208  //Compute the number of buckets
209  m_num_buckets = const_expr::power(2, const_expr::ceil(const_expr::log2(buckets_factor * (num_elems + 1))));
210  //Compute the buckets divider
211  m_buckets_capacity = m_num_buckets - 1;
212 
213  ASSERT_CONDITION_THROW((num_elems > m_buckets_capacity), string("Insufficient buckets capacity: ") +
214  std::to_string(m_buckets_capacity) + string(" need at least ") + std::to_string(num_elems));
215 
216  LOG_DEBUG << "FSHM: num_elems: " << num_elems << ", m_num_buckets: " << m_num_buckets
217  << ", m_buckets_capacity: " << m_buckets_capacity << END_LOG;
218  }

225  inline uint_fast64_t get_bucket_idx(uint_fast64_t key_value) const {
226  //Compute the bucket index, note that since m_capacity is the power of two,
227  //we can compute ( hash_value % m_num_buckets ) as ( hash_value & m_capacity )
228  //where m_capacity = ( m_num_buckets - 1);
229  const uint_fast64_t bucket_idx = mix_fasthash(key_value) & m_buckets_capacity;
230 
231  LOG_DEBUG3 << "The mixed key value is: " << key_value
232  << ", bucket_idx: " << SSTR(bucket_idx) << END_LOG;
233 
234  //If the sanity check is on then check on that the id is within the range
235  ASSERT_SANITY_THROW((bucket_idx > m_buckets_capacity),
236  string("The hash value got a bad bucket index: ") + std::to_string(bucket_idx) +
237  string(", must be within [0, ") + std::to_string(m_buckets_capacity) + "]");
238 
239  return bucket_idx;
240  }

246  inline void get_next_bucket_idx(uint_fast64_t & bucket_idx) const {
