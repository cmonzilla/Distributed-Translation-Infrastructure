116  void process_entry_weights(text_piece_reader & rest, rm_entry & entry) {
117  //Declare the token to store weights
118  text_piece_reader token;
119 
120  //Skip the first space
121  ASSERT_CONDITION_THROW(!rest.get_first_space(token), "Could not skip the first space!");
122 
123  //Read the subsequent weights, check that the number of weights is as expected
124  size_t idx = 0;
125  while (rest.get_first_space(token) && (idx < rm_entry::NUM_FEATURES)) {
126  //Parse the token into the entry weight
127  ASSERT_CONDITION_THROW(!fast_s_to_f(entry[idx], token.str().c_str()),
128  string("Could not parse the token: ") + token.str());
129  //Now convert to the log probability and multiply with the appropriate weight
130  entry[idx] = log10(entry[idx]) * m_params.m_lambdas[idx];
131  //Increment the index
132  ++idx;
133  }
134  }

145  template<bool count_or_build>
146  inline void parse_rm_file(tm_query_proxy & query) {
147  //Declare the text piece reader for storing the read line and source phrase
148  text_piece_reader line, source, target;
149 
150  //Store the cached source string and its uid values
151  string source_str = "";
152  phrase_uid source_uid = UNDEFINED_PHRASE_ID;
153  phrase_uid target_uid = UNDEFINED_PHRASE_ID;
154  tm_const_source_entry * source_entry = NULL;
155  bool is_good_source = false;
156  bool is_good_target = false;
157 
158  //Start reading the translation model file line by line
159  while (m_reader.get_first_line(line)) {
160  //Read the source phrase
161  line.get_first<RM_DELIMITER, RM_DELIMITER_CDTY>(source);
162 
163  //Get the current source phrase
164  string next_source_str = source.str();
165  trim(next_source_str);
166 
167  //Check if this is a new source phrase
168  if (source_str != next_source_str) {
169  //Store the new source string
170  source_str = next_source_str;
171  //Compute the new source string uid
172  source_uid = get_phrase_uid(source_str);
173 
174  //Obtain the source entry
175  source_entry = query.get_source_entry(source_uid);
176 
177  //Check if the a know the source entry
178  is_good_source = ((source_entry != NULL) ||
179  (source_uid == m_model.SOURCE_UNK_UID) ||
180  (source_uid == m_model.BEGIN_SENT_TAG_UID) ||
181  (source_uid == m_model.END_SENT_TAG_UID));
182 
183  LOG_DEBUG1 << "The RM source entry '" << source_str << "' with id " << source_uid
184  << " is " << (is_good_source ? "" : "not") << " good!" << END_LOG;
185  }
186 
187  //If the source is present then count the entry
188  if (is_good_source) {
189  //Read the target phrase
190  line.get_first<RM_DELIMITER, RM_DELIMITER_CDTY>(target);
191  string target_str = target.str();
192  trim(target_str);
193 
194  LOG_DEBUG1 << "Got the RM source/target entry: " << source_str << " / " << target_str << END_LOG;
195 
196  //Parse the rest of the target entry
197  target_uid = get_phrase_uid<true>(target_str);
198 
199  //Check if the given translation phrase is known
200  is_good_target = (((source_entry != NULL) && source_entry->has_target(target_uid)) ||
201  (target_uid == m_model.TARGET_UNK_UID) ||
202  (target_uid == m_model.BEGIN_SENT_TAG_UID) ||
203  (target_uid == m_model.END_SENT_TAG_UID));
204 
205  //If the target is known then do the thing based on what we need
206  if (is_good_target) {
207  if (count_or_build) {
208  //Increment the counter
209  ++m_num_entries;
210  } else {
211  //If the target translation is present then process the features
212  process_entry_weights(line, m_model.add_entry(source_uid, target_uid));
213  }
214  }
215 
216  LOG_DEBUG1 << "The TM target entry '" << target_str << "' with id " << target_uid
217  << " is " << (is_good_target ? "" : "not") << " good!" << END_LOG;
218  }
219 
220  //Update the progress bar status
221  logger::update_progress_bar();
222  }
223  }
