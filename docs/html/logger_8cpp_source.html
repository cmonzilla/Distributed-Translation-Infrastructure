163  string logger::compute_time_string(const clock_t elapsedClockTime, size_t & timeStrLen) {
164  const float timeSec = (((float) elapsedClockTime) / CLOCKS_PER_SEC);
165  const uint minute = (((uint) timeSec) % 3600) / 60;
