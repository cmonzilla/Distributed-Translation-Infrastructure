80  reactive_socket_service_base::base_implementation_type& impl)
81 {
82  if (impl.socket_ != invalid_socket)

84  ASIO_HANDLER_OPERATION(("socket", &impl, "close"));

94 asio::error_code reactive_socket_service_base::close(
95  reactive_socket_service_base::base_implementation_type& impl,
96  asio::error_code& ec)
97 {
98  if (is_open(impl))
99  {
100  ASIO_HANDLER_OPERATION(("socket", &impl, "close"));
101 
102  reactor_.deregister_descriptor(impl.socket_, impl.reactor_data_,
103  (impl.state_ & socket_ops::possible_dup) == 0);
104  }
105 
106  socket_ops::close(impl.socket_, impl.state_, false, ec);
107 
108  // The descriptor is closed by the OS even if close() returns an error.
109  //
110  // (Actually, POSIX says the state of the descriptor is unspecified. On
111  // Linux the descriptor is apparently closed anyway; e.g. see
112  // http://lkml.org/lkml/2005/9/10/129
113  // We'll just have to assume that other OSes follow the same behaviour. The
114  // known exception is when Windows's closesocket() function fails with
115  // WSAEWOULDBLOCK, but this case is handled inside socket_ops::close().
116  construct(impl);
117 
118  return ec;
119 }

138 asio::error_code reactive_socket_service_base::do_open(
139  reactive_socket_service_base::base_implementation_type& impl,
140  int af, int type, int protocol, asio::error_code& ec)
141 {
142  if (is_open(impl))
143  {
144  ec = asio::error::already_open;
145  return ec;
146  }
147 
148  socket_holder sock(socket_ops::socket(af, type, protocol, ec));
149  if (sock.get() == invalid_socket)
150  return ec;
151 
152  if (int err = reactor_.register_descriptor(sock.get(), impl.reactor_data_))
153  {
154  ec = asio::error_code(err,
155  asio::error::get_system_category());
156  return ec;
157  }
158 
159  impl.socket_ = sock.release();
160  switch (type)
161  {
162  case SOCK_STREAM: impl.state_ = socket_ops::stream_oriented; break;
163  case SOCK_DGRAM: impl.state_ = socket_ops::datagram_oriented; break;
164  default: impl.state_ = 0; break;
165  }
166  ec = asio::error_code();
167  return ec;
168 }

186  return ec;

188 
189  impl.socket_ = native_socket;
190  switch (type)
191  {
192  case SOCK_STREAM: impl.state_ = socket_ops::stream_oriented; break;
193  case SOCK_DGRAM: impl.state_ = socket_ops::datagram_oriented; break;
194  default: impl.state_ = 0; break;
195  }
196  impl.state_ |= socket_ops::possible_dup;
197  ec = asio::error_code();
198  return ec;

234 void reactive_socket_service_base::start_connect_op(
235  reactive_socket_service_base::base_implementation_type& impl,
236  reactor_op* op, bool is_continuation,
237  const socket_addr_type* addr, size_t addrlen)
238 {
239  if ((impl.state_ & socket_ops::non_blocking)
240  || socket_ops::set_internal_non_blocking(
241  impl.socket_, impl.state_, true, op->ec_))
242  {
243  if (socket_ops::connect(impl.socket_, addr, addrlen, op->ec_) != 0)
244  {
245  if (op->ec_ == asio::error::in_progress

247  {
248  op->ec_ = asio::error_code();
249  reactor_.start_op(reactor::connect_op, impl.socket_,
250  impl.reactor_data_, op, is_continuation, false);
251  return;
252  }
253  }
254  }
255 
256  reactor_.post_immediate_completion(op, is_continuation);
257 }
