205  m_handlers.erase(session_id);
206  }
207  }
208  LOG_DEBUG << "Session : " << session_id << " internal mappings "
209  << "are cleaned, canceling the job! " << END_LOG;
210 
211  //Request cancellation of all the translation jobs associated with this connection.
212  if (session_id != session_id::UNDEFINED_SESSION_ID) {
213  //NOTE: This can be done outside the synchronization block
214  m_job_pool.cancel_jobs(session_id);
215  }
216 
217  LOG_DEBUG << "The session with id: " << session_id << " is closed!" << END_LOG;

235  void notify_job_finished(trans_job_ptr trans_job) {
236  //Declare and initialize session and job id for future use
237  const job_id_type job_id = trans_job->get_job_id();
238  const job_id_type session_id = trans_job->get_session_id();
239 
240  //Create the translation job response
241  trans_job_response response(job_id, trans_job->get_code(), trans_job->get_text());
242 
243  LOG_DEBUG << "Created the job response: " << &response << " for job "
244  << trans_job->get_job_id() << " from session "
245  << trans_job->get_session_id() << END_LOG;
246 
247  //Do the sanity check assert
248  ASSERT_SANITY_THROW(!m_sender_func,
249  "The sender function of the translation manager is not set!");
250 
251  //Declare the connection handler
252  websocketpp::connection_hdl hdl;
253 
254  //Retrieve the connection handler based on the session id
255  {
256  //Use the scoped mutex lock to avoid race conditions
257  scoped_guard guard(m_lock);
258 
259  //Get the connection handler for the session
260  hdl = m_handlers[session_id];
261  }
262 
263  //If the sender function is present, and the handler is not expired
264  if (!hdl.expired()) {
265  m_sender_func(hdl, response);
266  } else {
267  LOG_ERROR << "Could not send the translation response for " << session_id
268  << "/" << job_id << " as the connection handler has expired!" << END_LOG;
269  }
270  }
