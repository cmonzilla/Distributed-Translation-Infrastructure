67  void stat_monitore::get_mem_stat(TMemotyUsage & memStat) {
68  char *line;
69  char *vmsize = NULL;
70  char *vmpeak = NULL;
71  char *vmrss = NULL;
72  char *vmhwm = NULL;
73  size_t len;
74  FILE *f;
75 
76  line = (char*) malloc(128);
77  len = 128;
78 
79  f = fopen("/proc/self/status", "r");
80  if (!f) throw uva_exception("Unable to open /proc/self/status for reading!");
81 
82  /* Read memory size data from /proc/self/status */
83  while (!vmsize || !vmpeak || !vmrss || !vmhwm) {
84  if (getline(&line, &len, f) == -1) {
85  /* Some of the information isn't there, die */
86  throw uva_exception("Unable to read memory statistics data from /proc/self/status");
87  }
88 
89  /* Find VmPeak */
90  if (!strncmp(line, "VmPeak:", 7)) {
91  vmpeak = strdup(&line[7]);
92  }/* Find VmSize */
93  else if (!strncmp(line, "VmSize:", 7)) {
94  vmsize = strdup(&line[7]);
95  }/* Find VmRSS */
96  else if (!strncmp(line, "VmRSS:", 6)) {
97  vmrss = strdup(&line[7]);
98  }/* Find VmHWM */
99  else if (!strncmp(line, "VmHWM:", 6)) {
100  vmhwm = strdup(&line[7]);
101  }
102  }
103  free(line);
104 
105  fclose(f);
106 
107  /* Get rid of " kB\n"*/
108  len = strlen(vmsize);
109  vmsize[len - 4] = 0;
110  len = strlen(vmpeak);
111  vmpeak[len - 4] = 0;
112  len = strlen(vmrss);
113  vmrss[len - 4] = 0;
114  len = strlen(vmhwm);
115  vmhwm[len - 4] = 0;
116 
117  /* Parse the results */
118  memStat.vmpeak = atoi(vmpeak);
119  memStat.vmsize = atoi(vmsize);
120  memStat.vmrss = atoi(vmrss);
121  memStat.vmhwm = atoi(vmhwm);

123  /* Print some info and debug information */

144  double stat_monitore::get_cpu_time() {
145 #if defined(_WIN32)
146  /* Windows -------------------------------------------------- */
147  FILETIME createTime;
148  FILETIME exitTime;
149  FILETIME kernelTime;
150  FILETIME userTime;
151  if (GetProcessTimes(GetCurrentProcess(),
152  &createTime, &exitTime, &kernelTime, &userTime) != -1) {
153  SYSTEMTIME userSystemTime;
154  if (FileTimeToSystemTime(&userTime, &userSystemTime) != -1)
155  return (double) userSystemTime.wHour * 3600.0 +
156  (double) userSystemTime.wMinute * 60.0 +
157  (double) userSystemTime.wSecond +
158  (double) userSystemTime.wMilliseconds / 1000.0;
159  }
160 
161 #elif defined(__unix__) || defined(__unix) || defined(unix) || (defined(__APPLE__) && defined(__MACH__))
162  /* AIX, BSD, Cygwin, HP-UX, Linux, OSX, and Solaris --------- */
163 
164 #if defined(_POSIX_TIMERS) && (_POSIX_TIMERS > 0)
165  /* Prefer high-res POSIX timers, when available. */
166  {
167  clockid_t id;
168  struct timespec ts;
169 #if _POSIX_CPUTIME > 0
170  /* Clock ids vary by OS. Query the id, if possible. */
171  if (clock_getcpuclockid(0, &id) == -1)
172 #endif
173 #if defined(CLOCK_PROCESS_CPUTIME_ID)
174  /* Use known clock id for AIX, Linux, or Solaris. */
175  id = CLOCK_PROCESS_CPUTIME_ID;
176 #elif defined(CLOCK_VIRTUAL)
177  /* Use known clock id for BSD or HP-UX. */
178  id = CLOCK_VIRTUAL;
179 #else
180  id = (clockid_t) - 1;
181 #endif
182  if (id != (clockid_t) - 1 && clock_gettime(id, &ts) != -1)
183  return (double) ts.tv_sec +
184  (double) ts.tv_nsec / 1000000000.0;
185  }
186 #endif
187 
188 #if defined(RUSAGE_SELF)
189  {
190  struct rusage rusage;
191  if (getrusage(RUSAGE_SELF, &rusage) != -1)
192  return (double) rusage.ru_utime.tv_sec +
193  (double) rusage.ru_utime.tv_usec / 1000000.0;
194  }
195 #endif
196 
197 #if defined(_SC_CLK_TCK)
198  {
199  const double ticks = (double) sysconf(_SC_CLK_TCK);
200  struct tms tms;
201  if (times(&tms) != (clock_t) - 1)
202  return (double) tms.tms_utime / ticks;
203  }
204 #endif
205
