141  void expand() {
142  //Define the max stack level constant
143  const size_t MAX_STACK_LEVEL = (m_num_levels - 1);
144  //Stores the current stack level index
145  uint32_t curr_level = MIN_STACK_LEVEL;
146 
147  //Iterate the stack levels and expand them one by one
148  //until the last one or until we are requested to stop
149  //Note: the last stack level is for the end state </s>
150  //it should not be expanded!
151  while (!m_data.m_is_stop && (curr_level < MAX_STACK_LEVEL)) {
152  LOG_DEBUG << ">>>>> Start LEVEL (" << curr_level << "/ " << MAX_STACK_LEVEL
153  << ") expansion, #states=" << m_levels[curr_level]->get_size() << END_LOG;
154 
155  //Here we expand the stack level and then
156  //increment the current level index variable
157  m_levels[curr_level]->expand();
158 
159  LOG_DEBUG << "<<<<< End LEVEL (" << curr_level << "/ " << MAX_STACK_LEVEL
160  << ") expansion, #states=" << m_levels[curr_level]->get_size() << END_LOG;
161 
162  //Move to the next level
163  ++curr_level;
164  }
