193  inline void is_m_gram_potentially_present(m_gram_query& query,
194  MGramStatusEnum &status) const {
195  //Do sanity check if needed
196  ASSERT_SANITY_THROW(query.is_curr_uni_gram(),
197  "Trying to check the bitmap hash cache for a uni-gram!");
198 
199  //Check if the caching is enabled and needed
200  if (NEEDS_BITMAP_HASH_CACHE) {
201  //Check if the end word is unknown, if not proceed to the cache check
202  if (query.get_curr_end_word_id() != UNKNOWN_WORD_ID) {
203  //Check if the begin word is unknown, if not proceed to the cache check
204  if (query.get_curr_begin_word_id() != UNKNOWN_WORD_ID) {
205  //Compute the level array index
206  const phrase_length level_idx = query.get_curr_level_m2();
207 
208  //If the caching is enabled, the higher sub-m-gram levels always require checking
209  const BitmapHashCache & ref = m_bitmap_hash_cach[level_idx];
210 
211  //Get the m-gram's hash
212  const uint64_t hash = query.get_curr_m_gram_hash();
213 
214  if (ref.is_hash_cached(hash)) {
215  //The m-gram hash is cached, so potentially a payload data
216  status = MGramStatusEnum::GOOD_PRESENT_MGS;
217  } else {
218  //The m-gram hash is not in cache, so definitely no data
219  status = MGramStatusEnum::BAD_NO_PAYLOAD_MGS;
220  }
221  } else {
222  //The m-gram hash is not in cache, so definitely no data
223  status = MGramStatusEnum::BAD_NO_PAYLOAD_MGS;
224  }
225  } else {
226  //The end word is unknown, so definitely no m-gram data
227  status = MGramStatusEnum::BAD_END_WORD_UNKNOWN_MGS;
228  }
229  } else {
230  //If caching is not enabled then we always check the trie
231  status = MGramStatusEnum::GOOD_PRESENT_MGS;
232  }
233  }

240  inline void execute(m_gram_query & query) const {
241  //Declare the stream-compute result status variable
242  MGramStatusEnum status = MGramStatusEnum::GOOD_PRESENT_MGS;
243 
244  //Iterate until the query is fully executed
245  while (query.is_not_finished()) {
246  LOG_DEBUG << "-----> Streaming cumulative sub-m-gram: " << query << END_LOG;
247  //Check whether this is a uni-gram case
248  if (!query.is_curr_uni_gram()) {
249  //If this is not a uni-gram case then
250  switch (status) {
251  case MGramStatusEnum::BAD_NO_PAYLOAD_MGS: //The previous status was: unknown payload
252  {
253  //The m-gram payload was not found, so this is not a uni-gram. The method will
254  //increment the begin index to move to the next row, the column remains.
255  process_unknown(query);
256 
257  //The status is always good as back-off can not fail, even if the back-off payload
258  //is not found, we then just count as if the back-off weight is zero and move on to
259  //computing the next row probability.
260  status = MGramStatusEnum::GOOD_PRESENT_MGS;
261  break;
262  }
263  case MGramStatusEnum::BAD_END_WORD_UNKNOWN_MGS: //The previous status was: unknown end word
264  {
265  //The m-gram end word was not found, so this is not a uni-gram The method will
266  //increment the begin index to move to the next row, while the column remains.
267  process_unknown(query);
268 
269  //The status is always an unknown end word, so that we keep iterating down the column
270  status = MGramStatusEnum::BAD_END_WORD_UNKNOWN_MGS;
271  break;
272  }
273  case MGramStatusEnum::GOOD_PRESENT_MGS: //The previous status was good: process the next sub-m-gram
274  {
275  //Retrieve the next sub-m-gram payload
276  process_m_n_gram(query, status);
277  //If the sub-m-gram was found then just move on
278  if (status == MGramStatusEnum::GOOD_PRESENT_MGS) {
279  //Move on to the longer sub-m-gram
280  ++query.m_curr_end_word_idx;
281  }
282  break;
283  }
284  default:
285  {
286  THROW_EXCEPTION(string("Unsupported status value: ") + status_to_string(status));
287  }
288  }
289  } else {
290  //Retrieve the payload
291  process_unigram(query, status);
292  //Move on to the longer sub-m-gram
293  ++query.m_curr_end_word_idx;
294  }
295  }

360  inline void process_unigram(m_gram_query & query, MGramStatusEnum & status) const {
361  //Get the uni-gram word index
362  const phrase_length & word_idx = query.m_curr_end_word_idx;
363  //Get the REFERENCE to the payload for convenience
364  const void * & payload_ref = query.get_curr_payload_ref();
365 
366  //Retrieve the payload
367  static_cast<const TrieType*> (this)->get_unigram_payload(query);
368  LOG_DEBUG << "The 1-gram is found, payload: "
369  << *((const m_gram_payload *) payload_ref) << END_LOG;
370 
371  //No need to check on the status, it is always good for the uni-gram
372  query.m_probs[word_idx] += ((const m_gram_payload *) payload_ref)->m_prob;
373  LOG_DEBUG << "probs[" << SSTR(word_idx) << "] += "
374  << ((const m_gram_payload *) payload_ref)->m_prob << END_LOG;
375 
376  //The payload of a uni-gram is always present, even if
377  //it is an unknown word, the data is still available.
378  status = MGramStatusEnum::GOOD_PRESENT_MGS;

387  inline void process_m_n_gram(m_gram_query & query, MGramStatusEnum & status) const {
388  //If this is at least a bi-gram, continue iterations, otherwise we are done!
389  LOG_DEBUG << "Considering the sub-m-gram: " << query << END_LOG;
390 
391  //First check if it makes sense to look into the trie
392  is_m_gram_potentially_present(query, status);
393  LOG_DEBUG << "The payload availability status for sub-m-gram : "
394  << query << " is: " << status_to_string(status) << END_LOG;
395 
396  //If the status says that the m-gram is potentially present then we try to retrieve it from the trie
397  if (status == MGramStatusEnum::GOOD_PRESENT_MGS) {
398  //Compute the current sub-m-gram level
399  const phrase_length curr_level = query.get_curr_level();
400  LOG_DEBUG << "The current sub-m-gram level is: " << SSTR(curr_level) << END_LOG;
401 
402  //Just for convenience get the REFERENCE to the payload element
403  const void * & payload_ref = query.get_curr_payload_ref();
404 
405  //Obtain the payload, depending on the sub-m-gram level
406  if (curr_level == LM_M_GRAM_LEVEL_MAX) {
407  LOG_DEBUG << "Calling the get_" << SSTR(curr_level) << "_gram_payload function." << END_LOG;
408  //We are at the last trie level, retrieve the payload
409  static_cast<const TrieType*> (this)->get_n_gram_payload(query, status);
410 
411  //Append the probability if the retrieval was successful
412  if (status == MGramStatusEnum::GOOD_PRESENT_MGS) {
413  LOG_DEBUG << "The n-gram is found, probability: "
414  << *((const prob_weight *) payload_ref) << END_LOG;
415  query.m_probs[query.m_curr_end_word_idx] += *((const prob_weight *) payload_ref);
416  LOG_DEBUG << "probs[" << SSTR(query.m_curr_begin_word_idx) << "] += "
417  << *((const prob_weight *) payload_ref) << END_LOG;
418  }
419  } else {
420  LOG_DEBUG << "Calling the get_" << SSTR(curr_level) << "_gram_payload function." << END_LOG;
421  //We are at one of the intermediate trie level, retrieve the payload
422  static_cast<const TrieType*> (this)->get_m_gram_payload(query, status);
423 
424  //Append the probability if the retrieval was successful
425  if (status == MGramStatusEnum::GOOD_PRESENT_MGS) {
426  LOG_DEBUG << "The m-gram is found, payload: "
427  << *((const m_gram_payload *) payload_ref) << END_LOG;
428  query.m_probs[query.m_curr_end_word_idx] += ((const m_gram_payload *) payload_ref)->m_prob;
429  LOG_DEBUG << "probs[" << SSTR(query.m_curr_begin_word_idx) << "] += "
430  << ((const m_gram_payload *) payload_ref)->m_prob << END_LOG;
431  }
432  }
433  }
434 
435  LOG_DEBUG << "The result for the sub-m-gram: " << query << " is : " << status_to_string(status) << END_LOG;
436  }

443  inline MGramStatusEnum get_uni_m_gram_payload(m_gram_query & query) const {
444  LOG_DEBUG << "The payload for sub-m-gram: " << query << " needs to be retrieved!" << END_LOG;
445  //Try to retrieve the back-off sub-m-gram
446  if (query.m_curr_begin_word_idx == query.m_curr_end_word_idx) {
447  //If the back-off sub-m-gram is a uni-gram then
448  static_cast<const TrieType*> (this)->get_unigram_payload(query);
449 
450  //Set the result status to good
451  return MGramStatusEnum::GOOD_PRESENT_MGS;
452  } else {
453  //Declare the status variable
454  MGramStatusEnum status;
455 
456  //Check if the back-off sub-m-gram is potentially available
457  is_m_gram_potentially_present(query, status);
458  LOG_DEBUG << "The payload availability status for sub-m-gram: "
459  << query << " is: " << status_to_string(status) << END_LOG;
460 
461  if (status == MGramStatusEnum::GOOD_PRESENT_MGS) {
462  //The back-off sub-m-gram has a level M: 1 < M < N
463  static_cast<const TrieType*> (this)->get_m_gram_payload(query, status);
464  }
465 
466  return status;
467  }

474  inline void process_unknown(m_gram_query & query) const {
475  LOG_DEBUG << " Current query - in: " << query << END_LOG;
476 
477  //Decrease the end word index, as we need the back-off data
478  query.m_curr_end_word_idx--;
479 
480  //Define the REFERENCE to the payload pointer, just for convenience
481  const void * & bo_payload_ref = query.get_curr_payload_ref();
482 
483  //If the back-off payload is present, then take it into account, else try to retrieve it and take into account
484  if ((bo_payload_ref != NULL) || (get_uni_m_gram_payload(query) == MGramStatusEnum::GOOD_PRESENT_MGS)) {
485  LOG_DEBUG << "The m-gram is found, payload: "
486  << *((const m_gram_payload *) bo_payload_ref) << END_LOG;
487  query.m_probs[query.m_curr_end_word_idx + 1] += ((const m_gram_payload *) bo_payload_ref)->m_back;
488  LOG_DEBUG << "probs[" << SSTR(query.m_curr_end_word_idx + 1) << "] += "
489  << ((const m_gram_payload *) bo_payload_ref)->m_back << END_LOG;
490  }
491 
492  //Increase the end word index as we are going back to normal
493  query.m_curr_end_word_idx++;
494 
495  //Next we shift to the next row, it is possible as it is not a uni-gram case, the latter
496  //always get MGramStatusEnum::GOOD_PRESENT_MGS result when their payload is retrieved
497  query.m_curr_begin_word_idx++;
498 
499  LOG_DEBUG << " Current query - out: " << query << END_LOG;
500  }
