64  template<typename WordIndexType>
65  void c2w_array_trie<WordIndexType>::pre_allocate(const size_t counts[LM_M_GRAM_LEVEL_MAX]) {
66  //01) Pre-allocate the word index super class call
67  BASE::pre_allocate(counts);
68 
69  //02) Compute and store the M-gram level sizes in terms of the number of M-grams per level
70  //Also initialize the M-gram index counters, for issuing context indexes
71  for (phrase_length i = 0; i < BASE::NUM_M_N_GRAM_LEVELS; i++) {
72  //The index counts must start with one as zero is reserved for the UNDEFINED_ARR_IDX
73  m_m_n_gram_next_ctx_id[i] = BASE::FIRST_VALID_CTX_ID;
74  //Due to the reserved first index, make the array sizes one element larger, to avoid extra computations
75  m_m_n_gram_num_ctx_ids[i] = counts[i + 1] + BASE::FIRST_VALID_CTX_ID;
76  }
77 
78  //03) Pre-allocate the 1-Gram data
79  //The size of this array is made two elements larger than the number
80  //of 1-Grams is since we want to account for the word indexes that start
81  //from 2, as 0 is given to UNDEFINED and 1 to UNKNOWN (<unk>)
82  m_one_gram_arr_size = BASE::get_word_index().get_number_of_words(counts[0]);
83  m_1_gram_data = new m_gram_payload[m_one_gram_arr_size];
84  memset(m_1_gram_data, 0, m_one_gram_arr_size * sizeof (m_gram_payload));
85 
86  //04) Allocate data for the M-grams
87 
88  //First allocate the contexts to data mappings for the 2-grams (stored under index 0)
89  //The number of contexts is the number of words in previous level 1 i.e. counts[0]
90  //Yet we know that the word index begins with 2, due to UNDEFINED and UNKNOWN word ids
91  //Therefore for the 2-gram level contexts array we add two more elements, just to simplify computations
92  m_m_gram_ctx_2_data[0] = new TSubArrReference[m_one_gram_arr_size];
93  memset(m_m_gram_ctx_2_data[0], 0, m_one_gram_arr_size * sizeof (TSubArrReference));
94 
95  //Now also allocate the data for the 2-Grams, the number of 2-grams is m_MN_gram_size[0]
96  m_m_gram_data[0] = new TWordIdPBEntry[m_m_n_gram_num_ctx_ids[0]];
97  memset(m_m_gram_data[0], 0, m_m_n_gram_num_ctx_ids[0] * sizeof (TWordIdPBEntry));
98 
99  //Now the remaining elements can be added in a loop
100  for (phrase_length i = 1; i < BASE::NUM_M_GRAM_LEVELS; i++) {
101  //Here i is the index of the array, the corresponding M-gram
102  //level M = i + 2. The m_MN_gram_size[i-1] stores the number of elements
103  //on the previous level - the maximum number of possible contexts.
104  m_m_gram_ctx_2_data[i] = new TSubArrReference[m_m_n_gram_num_ctx_ids[i - 1]];
105  memset(m_m_gram_ctx_2_data[i], 0, m_m_n_gram_num_ctx_ids[i - 1] * sizeof (TSubArrReference));
106  //The m_MN_gram_size[i] stores the number of elements
107  //on the current level - the number of M-Grams.
108  m_m_gram_data[i] = new TWordIdPBEntry[m_m_n_gram_num_ctx_ids[i]];
109  memset(m_m_gram_data[i], 0, m_m_n_gram_num_ctx_ids[i] * sizeof (TWordIdPBEntry));
110  }
111 
112  //05) Allocate the data for the M-Grams.
113  m_n_gram_data = new TCtxIdProbEntry[m_m_n_gram_num_ctx_ids[BASE::N_GRAM_IDX_IN_M_N_ARR]];
114  memset(m_n_gram_data, 0, m_m_n_gram_num_ctx_ids[BASE::N_GRAM_IDX_IN_M_N_ARR] * sizeof (TCtxIdProbEntry));
115  }
