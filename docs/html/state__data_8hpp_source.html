231  prob_weight get_lm_probability() {
232  //The number of new words that came into translation is either the
233  //number of words in the target or one, for the <s> or </s> tags
234  const size_t num_new_words = ((m_target != NULL) ? m_target->get_num_words() : 1);
235 
236  //Do the sanity check
237  ASSERT_SANITY_THROW(((MAX_HISTORY_LENGTH + num_new_words) > MAX_M_GRAM_QUERY_LENGTH ),
238  string("MAX_HISTORY_LENGTH (") + to_string(MAX_HISTORY_LENGTH) +
239  string(") + num_new_words (") + to_string(num_new_words) +
240  string(") > MAX_M_GRAM_QUERY_LENGTH (") + to_string(MAX_M_GRAM_QUERY_LENGTH) + string(")"));
241 
242  //It is only for these new words that we need to compute the lm probabilities
243  //for. So compute the current number of elements in the words' history.
244  const size_t all_hist_words = m_trans_frame.get_size() - num_new_words;
245  //The number of interesting words from the history is bounded by MAX_HISTORY_LENGTH
246  const size_t act_hist_words = min(all_hist_words, MAX_HISTORY_LENGTH);
247 
248  //Compute the query length to consider
249  const size_t num_query_words = act_hist_words + num_new_words;
250 
251  //Compute the number of words we need to skip in the query from the translation frame
252  const size_t num_words_to_skip = all_hist_words - act_hist_words;
253  //Compute the pointer to the beginning of the query words array
254  const word_uid * query_word_ids = m_trans_frame.get_elems() + num_words_to_skip;
255 
256  LOG_DEBUG1 << "Begin lm_level: " << m_begin_lm_level << ", hist_words: "
257  << act_hist_words << ", new_words: " << num_new_words << ", query words: "
258  << array_to_string<word_uid>(num_query_words, query_word_ids) << END_LOG;
259 
260  //Execute the query and return the value
261  //Logger::get_reporting_level() = DebugLevelsEnum::INFO2;
262  const prob_weight prob = m_stack_data.m_lm_query.execute(num_query_words, query_word_ids, m_begin_lm_level);
263  //Logger::get_reporting_level() = DebugLevelsEnum::DEBUG2;
264 
265  return prob;
266  }

273  inline reordering_orientation get_reordering_orientation(const state_data_templ & prev_state_data) {
274  LOG_DEBUG2 << "Previous state end_word_idx: " << prev_state_data.m_s_end_word_idx
275  << ", new state begin_word_idx: " << m_s_begin_word_idx << END_LOG;
276 
277  if (m_s_begin_word_idx < prev_state_data.m_s_end_word_idx) {
278  //We went to the left from the previous translation
279  if ((prev_state_data.m_s_begin_word_idx - m_s_end_word_idx) == 1) {
280  LOG_DEBUG2 << "SWAP_ORIENT" << END_LOG;
281  //The current phrase is right next to the previous
282  return reordering_orientation::SWAP_ORIENT;
283  } else {
284  LOG_DEBUG2 << "DISCONT_LEFT_ORIENT" << END_LOG;
285  //We have a discontinuous jump from the previous
286  return reordering_orientation::DISCONT_LEFT_ORIENT;
287  }
288  } else {
289  //We went to the right from the previous translation
290  if ((m_s_begin_word_idx - prev_state_data.m_s_end_word_idx) == 1) {
291  LOG_DEBUG2 << "MONOTONE_ORIENT" << END_LOG;
292  //The current phrase is right next to the previous
293  return reordering_orientation::MONOTONE_ORIENT;
294  } else {
295  LOG_DEBUG2 << "DISCONT_RIGHT_ORIENT" << END_LOG;
296  //We have a discontinuous jump from the previous
297  return reordering_orientation::DISCONT_RIGHT_ORIENT;
298  }
299  }

306  inline void compute_final_score(const state_data_templ & prev_state_data) {
307  //After the construction the partial score is to stay fixed,
308  //thus it is declared as constant and here we do a const_cast
309  prob_weight & partial_score = const_cast<prob_weight &> (m_partial_score);
310 
311  LOG_DEBUG1 << "Initial partial score is: " << partial_score << END_LOG;
312 
313  //Add the language model probability
314  partial_score += get_lm_probability();
315 
316  LOG_DEBUG1 << "partial score + LM is: " << partial_score << END_LOG;
317 
318  //Add the distance based reordering penalty
319  partial_score += -abs(m_s_begin_word_idx - prev_state_data.m_s_end_word_idx - 1);
320 
321  LOG_DEBUG1 << "partial score + RM discrete is: " << partial_score << END_LOG;
322 
323  //Add the lexicolized reordering costs
324  const reordering_orientation orient = get_reordering_orientation(prev_state_data);
325  partial_score += prev_state_data.rm_entry_data.template get_weight<true>(orient);
326  partial_score += rm_entry_data.template get_weight<false>(orient);
327 
328  LOG_DEBUG1 << "partial score + RM lexicolized is: " << partial_score << END_LOG;
329 
330  //After the construction the total score is to stay fixed,
331  //thus it is declared as constant and here we do a const_cast
332  prob_weight & total_score = const_cast<prob_weight &> (m_total_score);
333 
334  //Set the total score to be equal to the partial score as the translation is finished
335  total_score = partial_score;
336 
337  LOG_DEBUG1 << "Total score: " << total_score << END_LOG;
338  }

345  inline void update_partial_score(const state_data_templ & prev_state_data) {
346  //After the construction the partial score is to stay fixed,
347  //thus it is declared as constant and here we do a const_cast
348  prob_weight & partial_score = const_cast<prob_weight &> (m_partial_score);
349 
350  LOG_DEBUG1 << "Initial partial score is: " << partial_score << END_LOG;
351 
352  //Add the phrase translation probability
353  partial_score += m_target->get_total_weight();
354 
355  LOG_DEBUG1 << "partial score + TM is: " << partial_score << END_LOG;
356 
357  //Add the language model probability
358  partial_score += get_lm_probability();
359 
360  LOG_DEBUG1 << "partial score + LM is: " << partial_score << END_LOG;
361 
362  //Add the phrase penalty
363  partial_score += m_stack_data.m_params.m_phrase_penalty;
364 
365  LOG_DEBUG1 << "partial score + phrase penalty is: " << partial_score << END_LOG;
366 
367  //Add the word penalty
368  partial_score += m_stack_data.m_params.m_word_penalty * m_target->get_num_words();
369 
370  LOG_DEBUG1 << "partial score + word penalty is: " << partial_score << END_LOG;
371 
372  //Add the distance based reordering penalty
373  partial_score += -abs(m_s_begin_word_idx - prev_state_data.m_s_end_word_idx - 1);
374 
375  LOG_DEBUG1 << "partial score + RM discrete is: " << partial_score << END_LOG;
376 
377  //Add the lexicolized reordering costs
378  const reordering_orientation orient = get_reordering_orientation(prev_state_data);
379  partial_score += prev_state_data.rm_entry_data.template get_weight<true>(orient);
380  partial_score += rm_entry_data.template get_weight<false>(orient);
381 
382  LOG_DEBUG1 << "partial score + RM lexicolized is: " << partial_score << END_LOG;
383  }

389  inline void compute_total_score() {
390  //After the construction the total score is to stay fixed,
391  //thus it is declared as constant and here we do a const_cast
392  prob_weight & total_score = const_cast<prob_weight &> (m_total_score);
393 
394  //Set the total score to the current partial score and then add the future costs
395  total_score = m_partial_score;
396 
397  LOG_DEBUG1 << "Initial total score: " << total_score << END_LOG;
398 
399  //Iterate through all the non-translated phrase spans and add the future costs thereof
400  phrase_length begin_idx = m_stack_data.m_sent_data.m_min_idx;
401  while (begin_idx <= m_stack_data.m_sent_data.m_max_idx) {
402  LOG_DEBUG3 << "m_covered[" << begin_idx << "] = "
403  << (m_covered[begin_idx] ? "true" : "false") << END_LOG;
404 
405  if (!m_covered[begin_idx]) {
406  //Start from the begin word, as may be this is the only uncovered word left.
407  phrase_length end_idx = begin_idx;
408 
409  LOG_DEBUG3 << "first end_idx: " << end_idx << END_LOG;
410 
411  //Iterate until we are beyond the last word or the word is covered
412  while ((end_idx <= m_stack_data.m_sent_data.m_max_idx) && !m_covered[end_idx]) {
413  ++end_idx;
414  LOG_DEBUG3 << "Considering end_idx: " << end_idx << END_LOG;
415  }
416 
417  LOG_DEBUG3 << "First bad end_idx: " << end_idx << " the previous was good!" << END_LOG;
418 
419  //The previous end word was the last good one, add the span's costs
420  total_score += m_stack_data.m_sent_data[begin_idx][end_idx - 1].future_cost;
421 
422  LOG_DEBUG3 << "total score + future_cost[" << begin_idx << ", "
423  << (end_idx - 1) << "]: " << total_score << END_LOG;
424 
425  //Start searching further from the first word after the bad one we just found
426  begin_idx = end_idx + 1;
427  } else {
428  //Move on to the next begin index
429  ++begin_idx;
430  }
431 
432  LOG_DEBUG3 << "next begin_idx: " << begin_idx << END_LOG;
433  }
434 
435  LOG_DEBUG1 << "Total score: " << total_score << END_LOG;
436  }
