125  inline word_uid register_word(const text_piece_reader & token) {
126  //First get/create an existing/new word entry from from/in the word index
127  word_uid& hash = m_word_index_map_ptr->operator[](token.str());
128 
129  if (hash == UNDEFINED_WORD_ID) {
130  //If the word hash is not defined yet, then issue it a new hash id
131  hash = m_next_new_word_id++;
132  LOG_DEBUG2 << "Word: '" << token.str() << "' is not known yet, issuing it a new id: " << SSTR(hash) << END_LOG;
133  }
134 
135  //Use the Prime numbers hashing algorithm as it outperforms djb2
136  return hash;
137  }
