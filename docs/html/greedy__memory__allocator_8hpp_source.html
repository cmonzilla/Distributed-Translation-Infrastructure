64  template<typename TContaner, typename TAllocator>
65  void allocate_container(TContaner ** ppContainer, TAllocator ** ppAllocator,
66  const size_t numEntries, const string ctName,
67  const float factor = UNORDERED_MAP_MEMORY_FACTOR) {
68  //Compute the number of bytes needed to store these words
69  const size_t factoredNumElems = numEntries * factor;
70 
71  LOG_DEBUG4 << "Computing the required storage for " << ctName
72  << ", numEntries=" << numEntries << ", factor=" << factor
73  << ", result=" << factoredNumElems << END_LOG;
74 
75  //Allocate the allocator
76  LOG_DEBUG4 << "Allocating the " << ctName << " allocator for " << factoredNumElems << " elements!" << END_LOG;
77  *ppAllocator = new TAllocator(factoredNumElems);
78 
79  //Allocate the map with the given allocator
80  LOG_DEBUG4 << "Allocating the " << ctName << " container with the created allocator!" << END_LOG;
81  *ppContainer = new TContaner(**ppAllocator);
82  }

94  template<typename TContaner, typename TAllocator>
95  void reserve_mem_unordered_map(TContaner ** ppContainer, TAllocator ** ppAllocator,
96  const size_t numEntries, const string ctName,
97  const float factor = UNORDERED_MAP_MEMORY_FACTOR) {
98  //Call the generic allocation function with the appropriate factor
99  allocate_container<TContaner, TAllocator>(ppContainer, ppAllocator, numEntries, ctName, factor);
100 
101  //Reserve some memory for the buckets!
