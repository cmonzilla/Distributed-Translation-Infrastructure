136 static void extract_arguments(const uint argc, char const * const * const argv, client_config & params) {
137  //Parse the arguments
138  try {
139  p_cmd_args->parse(argc, argv);
140  } catch (ArgException &e) {
141  THROW_EXCEPTION(string("Error: ") + e.error() + string(", for argument: ") + e.argId());
142  }
143 
144  //Set the logging level right away
145  Logger::set_reporting_level(p_debug_level_arg->getValue());
146 
147  //Store the parsed parameter values
148  params.m_source_file = p_source_file_arg->getValue();
149  params.m_source_lang = p_source_lang_arg->getValue();
150  LOG_USAGE << "Given input file: '" << params.m_source_file << "', language: '" << params.m_source_lang << "'" << END_LOG;
151 
152  params.m_target_file = p_target_file_arg->getValue();
153  params.m_target_lang = p_target_lang_arg->getValue();
154  LOG_USAGE << "Given output file: '" << params.m_target_file << "', language: '" << params.m_target_lang << "'" << END_LOG;
155 
156  params.m_server = p_server_arg->getValue();
157  params.m_port = p_port_arg->getValue();
158  LOG_USAGE << "Using server address: '" << params.m_server << "', port: '" << params.m_port << "'" << END_LOG;
159 
160  params.m_min_sent = p_min_sent->getValue();
161  params.m_max_sent = p_max_sent->getValue();
162  LOG_USAGE << "The min/max number of sentences per request: '" << params.m_min_sent << "/" << params.m_max_sent << "'" << END_LOG;
163 
164  params.is_pre_process = p_pre_process_arg->getValue();
165  LOG_USAGE << "The source sentence pre-processing is " << ( params.is_pre_process ? "ON" : "OFF") << END_LOG;
166 }

171 int main(int argc, char** argv) {
172  //Declare the return code
173  int return_code = 0;
174 
175  //Set the uncaught exception handler
176  std::set_terminate(handler);
177 
178  //First print the program info
179  print_info();
180 
181  //Set up possible program arguments
182  create_arguments_parser();
183 
184  try {
185  //Define en empty parameters structure
186  client_config params = {};
187 
188  //Attempt to extract the program arguments
189  extract_arguments(argc, argv, params);
190 
191  //Create the translation manager
192  trans_manager manager(params);
193 
194  //Start the translation manager
195  LOG_USAGE << "Starting the translation process ..." << END_LOG;
196  manager.start();
197 
198  //Wait until the translations are done
199  LOG_USAGE << "Waiting for the translation process to finish ..." << END_LOG;
200  manager.wait();
201 
202  //Stop the translation manager
203  LOG_USAGE << "Finishing the translation process ..." << END_LOG;
204  manager.stop();

207  } catch (Exception & ex) {

209  LOG_ERROR << ex.get_message() << END_LOG;
210  return_code = 1;
211  }
212 
213  //Destroy the command line parameters parser
214  destroy_arguments_parser();
215
