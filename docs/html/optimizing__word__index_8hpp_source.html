146  inline word_uid get_word_id(const text_piece_reader & token) const {
147  //Compute the bucket id
148  uint_fast64_t bucket_idx = get_bucket_idx(token);
149 
150  LOG_DEBUG3 << "Searching for: " << token.str() << ", the " <<
151  "initial bucket index is: " << bucket_idx << END_LOG;
152 
153  //Search until an empty bucket or the word is found
154  while (m_word_buckets[bucket_idx].m_word != NULL) {
155  //Search within the bucket
156  if (IS_EQUAL(token, m_word_buckets[bucket_idx])) {
157  return m_word_buckets[bucket_idx].m_word_id;
158  }
159  get_next_bucket_idx(bucket_idx);
160  }
161  LOG_DEBUG3 << "Encountered an empty bucket, the word is unknown!" << END_LOG;
162 
163  return UNKNOWN_WORD_ID;
164  };

297  inline void set_number_of_elements(const double buckets_factor, const size_t num_elems) {
298  //Do a compulsory assert on the buckets factor
299  ASSERT_CONDITION_THROW((buckets_factor < 1.0), string("buckets_factor: ") +
300  std::to_string(buckets_factor) + string(", must be >= 1.0"));
301 
302  //Compute the number of buckets
303  m_num_buckets = const_expr::power(2, const_expr::ceil(const_expr::log2(buckets_factor * (num_elems + 1))));
304  //Compute the buckets divider
305  m_capacity = m_num_buckets - 1;
306 
307  ASSERT_CONDITION_THROW((num_elems > m_capacity), string("Insufficient buckets capacity: ") +
308  std::to_string(m_capacity) + string(" need at least ") + std::to_string(num_elems));
309 
310  LOG_DEBUG << "OWI: num_elems: " << num_elems << ", m_num_buckets: " << m_num_buckets
311  << ", m_capacity: " << m_capacity << END_LOG;
312  }

360  inline void fill_buckets_with_data() {
361  LOG_DEBUG2 << "Start filling the buckets with data" << END_LOG;
362 
363  basic_word_index::TWordIndexMapConstIter curr = m_disp_word_index_ptr->begin();
364  const basic_word_index::TWordIndexMapConstIter end = m_disp_word_index_ptr->end();
365  uint_fast64_t bucket_idx = 0;
366 
367  //Go through all the words and fill in the buckets
368  while (curr != end) {
369  const basic_word_index::TWordIndexEntry & word_data = *curr;
370 
371  LOG_DEBUG2 << "Converting word: '" << word_data.first
372  << "' with word id: " << word_data.second << END_LOG;
373 
374  bucket_idx = get_bucket_idx(word_data.first);
375 
376  LOG_DEBUG2 << "Got bucket_idx: " << bucket_idx << END_LOG;
377 
378  //Search for the first empty bucket
379  while (m_word_buckets[bucket_idx].m_word != NULL) {
380  LOG_DEBUG2 << "The bucket: " << bucket_idx <<
381  " is full, skipping to the next." << END_LOG;
382  get_next_bucket_idx(bucket_idx);
383  }
384 
385  LOG_DEBUG2 << "Adding word: '" << word_data.first
386  << "' to bucket: " << bucket_idx << END_LOG;
387 
388  //Copy the data into the bucket
389  TBucketEntry & entry = m_word_buckets[bucket_idx];
390 
391  //Copy the string but without the terminating null character
392  entry.m_len = word_data.first.length();
393  entry.m_word = new char[entry.m_len];
394  strncpy(entry.m_word, word_data.first.c_str(), entry.m_len);
395 
396  //Copy the word id
397  entry.m_word_id = word_data.second;
398 
399  ++curr;
400  }
