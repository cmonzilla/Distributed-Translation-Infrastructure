252  void combine_job_result() {
253  LOG_DEBUG << "Combining the job " << this << " result!" << END_LOG;
254 
255  //Declare the variable for counting the number of CANCELED tasks
256  uint32_t num_canceled = 0;
257 
258  //Iterate through the translation tasks and combine the results
259  for (tasks_iter_type it = m_tasks.begin(); it != m_tasks.end(); ++it) {
260  //Get the task pointer for future use
261  trans_task_ptr task = *it;
262  //Count the number of canceled tasks and leave text as an empty line then
263  if (task->get_code() == trans_job_code::RESULT_CANCELED) {
264  num_canceled++;
265  }
266  //Append the next task result
267  m_target_text += task->get_target_text() + "\n";
268  LOG_DEBUG1 << "The target text of task: " << task->get_task_id() << " has been retrieved!" << END_LOG;
269  }
270 
271  LOG_DEBUG2 << "The translation job " << this << " result is:\n" << m_target_text << END_LOG;
272 
273  //Remove the last sentence new line symbol
274  m_target_text.substr(0, m_target_text.size() - 1);
275 
276  //Decide on the result code
277  if (num_canceled == 0) {
278  //If there is no canceled jobs then the result is good
279  m_code = trans_job_code::RESULT_OK;
280  } else {
281  if (num_canceled == m_done_tasks_count) {
282  //All of the tasks have been canceled
283  m_code = trans_job_code::RESULT_CANCELED;
284  } else {
285  //Some of the sentences were translated but not all of them
286  m_code = trans_job_code::RESULT_PARTIAL;
287  }
288  }
289 
290  LOG_DEBUG << "The translation job " << this << " result is ready!" << END_LOG;
291  }
