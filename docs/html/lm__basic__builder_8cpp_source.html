72  void lm_basic_builder<TrieType, TFileReaderModel>::read_headers() {
73  LOG_DEBUG << "Start reading ARPA headers." << END_LOG;
74 
75  while (true) {
76  LOG_DEBUG1 << "Read header (?) line: '" << m_line.str() << "'" << END_LOG;
77 
78  //If the line is empty then we keep reading
79  if (m_line.has_more()) {
80  //If the line begins with "<" then it could be a header
81  //or something else that we do not need to read or
82  //interpret! Therefore we skip it and read on, otherwise
83  //it is potentially a meaningful data so we should stop
84  //and go on to the next section, namely" data
85  if (m_line[0] != '<') {
86  LOG_DEBUG1 << "Is something meaningful, moving to data section!" << END_LOG;
87  break;
88  } else {
89  LOG_DEBUG1 << "Is something meaningless, starting with <, skipping forward" << END_LOG;
90  }
91  } else {
92  LOG_DEBUG1 << "Is an empty line, skipping forward" << END_LOG;
93  }
94 
95  //Update the progress bar status
96  Logger::update_progress_bar();
97 
98  //Read the next line from the file if it is there
99  if (!m_file.get_first_line(m_line)) {
100  throw Exception("Incorrect ARPA format: An unexpected end of file while reading the ARPA headers!");
101  }
102  }
103 
104  LOG_DEBUG << "Finished reading ARPA headers." << END_LOG;
105  }

125  template<typename TrieType, typename TFileReaderModel>
126  void lm_basic_builder<TrieType, TFileReaderModel>::read_data(size_t counts[LM_M_GRAM_LEVEL_MAX]) {
127  LOG_DEBUG << "Start reading ARPA data." << END_LOG;
128 
129  //If we are here then it means we just finished reading the
130  //ARPA headers and we stumbled upon something meaningful,
131  //that actually must be the begin of the data section
132  if (m_line != END_OF_ARPA_FILE) {
133  phrase_length level = M_GRAM_LEVEL_1;
134  while (true) {
135  if (m_file.get_first_line(m_line)) {
136  LOG_DEBUG1 << "Read data (?) line: '" << m_line.str() << "'" << END_LOG;
137 
138  //Update the progress bar status
139  Logger::update_progress_bar();
140 
141  //If the line is empty then we keep reading
142  if (m_line.has_more()) {
143  //Check that the next line contains the meaningful N-gram amount information!
144  if (regex_match(m_line.str(), m_ng_amount_reg_exp)) {
145  //This is a valid data section entry, there is no need to do anything with it.
146  //Later we might want to read the numbers and then check them against the
147  //actual number of provided n-grams but for now it is not needed.
148  LOG_DEBUG1 << "Reading " << level << "-gram amount, got: '" << m_line << "'!" << END_LOG;
149 
150  //Read the number of N-grams in order to have enough data
151  //for the pre-allocation of the memory in the trie! For
152  //large language models that should both improve the memory
153  //usage and the performance of adding new N-Grams!
154 
155  //First tokenize the string
156  vector<string> elems;
157  tokenize(m_line.str(), elems, NGRAM_COUNTS_DELIM);
158 
159  //Parse the second (last) value and store it as the amount
160  string & amount = *(--elems.end());
161  try {
162  counts[level - 1] = stoull(amount);
163  LOG_DEBUG << "Expected number of " << level << "-grams is: " << counts[level - 1] << END_LOG;
164  } catch (invalid_argument) {
165  stringstream msg;
166  msg << "Incorrect ARPA format: Can not parse the "
167  << level << "-gram amount: '" << amount
168  << "' from string: '" << m_line << "'";
169  throw Exception(msg.str());
170  }
171  } else {
172  LOG_DEBUG1 << "Is something other than n-gram amount, moving to n-gram sections!" << END_LOG;
173  break;
174  }
175  } else {
176  LOG_DEBUG1 << "Is an empty line, skipping forward" << END_LOG;
177  }
178  } else {
179  throw Exception("Incorrect ARPA format: An unexpected end of file while reading the ARPA data section!");
180  }
181  level++;
182  }
183  } else {
184  stringstream msg;
185  msg << "Incorrect ARPA format: Got '" << m_line << "' instead of '"
186  << END_OF_ARPA_FILE << "' when starting on the data section!";
187  throw Exception(msg.str());
188  }
189 
190  LOG_DEBUG << "Finished reading ARPA data." << END_LOG;
191  }

193  template<typename TrieType, typename TFileReaderModel>
194  template<phrase_length CURR_LEVEL, bool is_mult_weight>
195  void lm_basic_builder<TrieType, TFileReaderModel>::read_m_gram_level() {
196  //Declare the pointer to the N-Grma builder
197  lm_gram_builder<WordIndexType, CURR_LEVEL, is_mult_weight> *gram_builder_ptr = NULL;
198  lm_gram_builder_factory<TrieType>::template get_builder<CURR_LEVEL, is_mult_weight>(m_params, m_trie, &gram_builder_ptr);
199 
200  try {
201  //The counter of the N-grams
202  uint numNgrams = 0;
203  //Read the current level N-grams and add them to the trie
204  while (true) {
205  //Try to read the next line
206  if (m_file.get_first_line(m_line)) {
207  LOG_DEBUG1 << "Read " << CURR_LEVEL << "-Gram (?) line: '" << m_line.str() << "'" << END_LOG;
208 
209  //Empty lines will just be skipped
210  if (m_line.has_more()) {
211  //Pass the given N-gram string to the N-Gram Builder. If the
212  //N-gram is not matched then stop the loop and move on
213  if (gram_builder_ptr->parse_line(m_line)) {
214  //If there was no match then it is something else
215  //than the given level N-gram so we move on
216  LOG_DEBUG << "Actual number of " << CURR_LEVEL << "-grams is: " << numNgrams << END_LOG;
217 
218  //Now stop reading this level N-grams and move on
219  break;
220  }
221  numNgrams++;
222  }
223 
224  //Update the progress bar status
225  Logger::update_progress_bar();
226  } else {
227  //If the next line does not exist then it an error as we expect the end of data section any way
228  stringstream msg;
229  msg << "Incorrect ARPA format: Unexpected end of file, missing the '" << END_OF_ARPA_FILE << "' tag!";
230  throw Exception(msg.str());
231  }
232  }
233  } catch (...) {
234  //Free the allocated N-gram builder in case of an exception
235  delete gram_builder_ptr;
236  //Re-throw the exception
237  throw;
238  }
239  //Free the allocated N-gram builder in case of no exception
240  delete gram_builder_ptr;
241  gram_builder_ptr = NULL;
242 
243  LOG_DEBUG << "Finished reading ARPA " << CURR_LEVEL << "-Grams." << END_LOG;
244  //Stop the progress bar in case of no exception
245  Logger::stop_progress_bar();
246  }

311  template<typename TrieType, typename TFileReaderModel>
312  template<phrase_length CURR_LEVEL>
313  void lm_basic_builder<TrieType, TFileReaderModel>::read_grams() {
314  stringstream msg;
315  //Do the progress bard indicator
316  msg << "Reading ARPA " << CURR_LEVEL << "-Grams";
317  Logger::start_progress_bar(msg.str());
318 
319  //The regular expression for matching the n-grams section
320  stringstream regexpStr;
321  regexpStr << "\\\\" << CURR_LEVEL << "\\-grams\\:";
322  LOG_DEBUG1 << "The N-gram section reg-exp: '" << regexpStr.str() << "'" << END_LOG;
323  const regex n_gram_sect_reg_exp(regexpStr.str());
324 
325  //Check if the line that was input is the header of the N-grams section for N=level
326  if (regex_match(m_line.str(), n_gram_sect_reg_exp)) {
327  //Check if we need to multiply with the m-gram weight
328  if (m_params.is_lm_weight()) {
329  //Read the M-grams of the given level
330  read_m_gram_level<CURR_LEVEL, true>();
331  } else {
332  //Read the M-grams of the given level
333  read_m_gram_level<CURR_LEVEL, false>();
334  }
335 
336  //If the first M-gram level has been read then do
337  //the word index post-actions if needed.
338  if (CURR_LEVEL == M_GRAM_LEVEL_1) {
339  do_word_index_post_1_gram_actions();
340  }
341 
342  //Perform the post-M-gram actions if needed
343  do_post_m_gram_actions<CURR_LEVEL>();
344 
345  //Check if we need to keep reading and recurse or we are done
346  check_and_go_m_grams<CURR_LEVEL>();
347  } else {
348  //The obtained string is something else than the next n-grams section header
349  //So the only thing it is allowed to be is the end of file, let's check on
350  //it and otherwise report an error
351 
352  ASSERT_CONDITION_THROW((m_line != END_OF_ARPA_FILE),
353  string("Incorrect ARPA format: Got '") + m_line.str() +
354  string("' when trying to read the ") + to_string(CURR_LEVEL) +
355  string("-grams section!"));
356  }
357  }

365  template<typename TrieType, typename TFileReaderModel>
366  void lm_basic_builder<TrieType, TFileReaderModel>::get_word_counts_from_unigrams() {
367  typename TrieType::WordIndexType & word_index = m_trie.get_word_index();
368 
369  //The regular expression for matching the n-grams section
370  stringstream regexpStr;
371  regexpStr << "\\\\" << M_GRAM_LEVEL_1 << "\\-grams\\:";
372  LOG_DEBUG1 << "The N-gram section reg-exp: '" << regexpStr.str() << "'" << END_LOG;
373  const regex n_gram_sect_reg_exp(regexpStr.str());
374 
375  //Check if the line that was input is the header of the N-grams section for N=level
376  if (regex_match(m_line.str(), n_gram_sect_reg_exp)) {
377  //Declare the variables needed to get the word counts
378  TextPieceReader word;
379  prob_weight prob;
380 
381  //Read the current level N-grams and add them to the trie
382  while (m_file.get_first_line(m_line)) {
383  LOG_DEBUG1 << "Reading " << SSTR(M_GRAM_LEVEL_1) << "-gram, got: [" << m_line.str() << "]" << END_LOG;
384  //If this is not an empty line
385  if (m_line.has_more()) {
386  //Parse line to words without probabilities and back-offs
387  //If it is not the M-gram line then we stop break
388  if (lm_gram_builder<WordIndexType, M_GRAM_LEVEL_1, false>::unigram_to_prob(m_line, word, prob)) {
389  //Set the word with its probability into the word index
390  word_index.count_word(word, prob);
391  //Update the progress bar status
392  Logger::update_progress_bar();
393  } else {
394  //This is not an expected M-gram
395  LOG_DEBUG1 << "Stopping words counting in M-gram level: " << SSTR(M_GRAM_LEVEL_1) << END_LOG;
396  break;
397  }
398  }
399  }
400  LOG_DEBUG3 << "Line : " << m_line.str() << END_LOG;
401  LOG_DEBUG1 << "Finished counting words in " << SSTR(M_GRAM_LEVEL_1) << "-grams" << END_LOG;
402  } else {
403 
404  THROW_EXCEPTION("Could not count words, did not get a match with for the beginning of the 1-gram section!");
405  }
406 
407  //Update the progress bar status
408  Logger::update_progress_bar();
409  }

456  LOG_DEBUG << "Starting to read the file and build the trie ..." << END_LOG;
457 
458  //Declare an array of N-Gram counts, that is to be filled from the
459  //headers. This data will be used to pre-allocate memory for the Trie

461  memset(counts, 0, LM_M_GRAM_LEVEL_MAX * sizeof (size_t));
462

475

481

486  Logger::stop_progress_bar();

488  }
