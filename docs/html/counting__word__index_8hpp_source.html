151  void do_post_word_count() {
152  //All the words have been filled in, it is time to give them ids.
153  LOG_DEBUG1 << "Starting the post word counting actions!" << END_LOG;
154 
155  //00. Remove the <unk> word from the Map as it must get fixed index
156  LOG_DEBUG2 << "Remove the <unk> word from the Map as it must get fixed index" << END_LOG;
157  basic_word_index::m_word_index_map_ptr->erase(UNKNOWN_WORD_STR);
158 
159  //01. Create an array of words info objects from BasicWordIndex::_pWordIndexMap
160  LOG_DEBUG2 << "Create an array of words info objects from BasicWordIndex::_pWordIndexMap" << END_LOG;
161  const size_t num_words = basic_word_index::m_word_index_map_ptr->size();
162  __counting_word_index::TWordInfo * word_infos = new __counting_word_index::TWordInfo[num_words];
163 
164  //02. Copy the word information from the map into that array.
165  LOG_DEBUG2 << "Copy the word information from the map into that array." << END_LOG;
166  basic_word_index::TWordIndexMap::const_iterator iter = basic_word_index::m_word_index_map_ptr->begin();
167  for (size_t idx = 0; iter != basic_word_index::m_word_index_map_ptr->end(); ++iter, ++idx) {
168  word_infos[idx].word = iter->first;
169  memcpy(&word_infos[idx].prob, &iter->second, sizeof (prob_weight));
170  }
171 
172  //03. Sort the array of word info object in order to get
173  // the most used words in the beginning of the array
174  LOG_DEBUG2 << "Sort the array of word info object in order to get "
175  << "the most used words in the beginning of the array" << END_LOG;
176  my_sort<__counting_word_index::TWordInfo>(word_infos, num_words);
177 
178  //04. Iterate through the array and assign the new word ids
179  // into the _pWordIndexMap using the BasicWordIndex::_nextNewWordId
180  LOG_DEBUG2 << "Iterate through the array and assign the new word ids "
181  << "into the _pWordIndexMap using the BasicWordIndex::_nextNewWordId" << END_LOG;
182  for (size_t idx = 0; idx < num_words; ++idx) {
183  //Get the next word
184 
185  string & word = word_infos[idx].word;
186  //Give it the next index
187  basic_word_index::m_word_index_map_ptr->operator[](word) = basic_word_index::m_next_new_word_id++;
188  LOG_DEBUG4 << "Word [" << word << "], count: " << word_infos[idx].prob << " gets id: "
189  << SSTR(basic_word_index::m_next_new_word_id - 1) << END_LOG;
190  }
191 
192  //05. Delete the temporary sorted array
193  LOG_DEBUG2 << "Delete the temporary sorted array" << END_LOG;
194  delete[] word_infos;
195 
196  //06. Put back the <unk> word with its fixed index into the map
197  LOG_DEBUG2 << "Put back the <unk> word with its fixed index into the map" << END_LOG;
198  basic_word_index::m_word_index_map_ptr->operator[](UNKNOWN_WORD_STR) = UNKNOWN_WORD_ID;
199 
200  LOG_DEBUG1 << "Finished the post word counting actions!" << END_LOG;
201  };
