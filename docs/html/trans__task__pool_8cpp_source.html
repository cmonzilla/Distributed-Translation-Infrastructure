52  void trans_task_pool::set_num_threads(const size_t new_num_threads) {
53  ASSERT_CONDITION_THROW((new_num_threads == 0),
54  string("The new number of threads is 0, must be > 0!"));
55 
56  const size_t curr_num_threads = m_threads.size();
57  if (new_num_threads > curr_num_threads) {
58  //We are to add more worker threads
59  for (size_t count = curr_num_threads; count < new_num_threads; ++count) {
60  //Add the new worker
61  m_workers.emplace_back(new trans_task_pool_worker(*this));
62  //Add the worker thread
63  m_threads.emplace_back(thread(&trans_task_pool_worker::operator(), m_workers.back()));
64  }
65  } else {
66  if (new_num_threads < curr_num_threads) {
67  //The number of workers to delete
68  const size_t num_to_delete = (curr_num_threads - new_num_threads);
69 
70  LOG_WARNING << num_to_delete << " worker threads are about to be deleted, please wait!" << END_LOG;
71 
72  //Delete the requested number of workers
73  size_t count = 0;
74  int32_t idx = 0;
75  while (count < num_to_delete) {
76  LOG_DEBUG1 << "Checking worker " << idx << "/" << m_workers.size() << END_LOG;
77 
78  //If the worker is not busy, try to delete it
79  if (!m_workers[idx]->is_busy()) {
80  LOG_DEBUG1 << "The worker " << idx << " is free!" << END_LOG;
81 
82  //Ask the worker to stop
83  m_workers[idx]->stop();
84  //Wake up all sleeping threads as the notify one might
85  //wake up some other thread than the stopped one
86  m_condition.notify_all();
87  //Wait until the thread is finished
88  m_threads[idx].join();
89 
90  //Delete the worker
91  delete m_workers[idx];
92 
93  //Erase the worker and its thread
94  m_workers.erase(m_workers.begin() + idx);
95  m_threads.erase(m_threads.begin() + idx);
96 
97  //Increment the deleted workers count
98  ++count;
99  }
100 
101  //Move on through the workers
102  idx = (idx + 1) % m_workers.size();
103  }
104  }
105  }
106  }
