387 {
388  std::size_t total_buffer_size = 0;
389 
390  typename BufferSequence::const_iterator iter = b.begin();
391  typename BufferSequence::const_iterator end = b.end();
392  for (; iter != end; ++iter)

395  return total_buffer_size;
