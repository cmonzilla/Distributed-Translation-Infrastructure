68 void strand_service::construct(strand_service::implementation_type& impl)
69 {
70  asio::detail::mutex::scoped_lock lock(mutex_);
71 
72  std::size_t salt = salt_++;
73 #if defined(ASIO_ENABLE_SEQUENTIAL_STRAND_ALLOCATION)
74  std::size_t index = salt;
75 #else // defined(ASIO_ENABLE_SEQUENTIAL_STRAND_ALLOCATION)

77  index += (reinterpret_cast<std::size_t>(&impl) >> 3);

93 bool strand_service::do_dispatch(implementation_type& impl, operation* op)
94 {
95  // If we are running inside the io_service, and no other handler already
96  // holds the strand lock, then the handler can run immediately.
97  bool can_dispatch = io_service_.can_dispatch();
98  impl->mutex_.lock();
99  if (can_dispatch && !impl->locked_)
100  {
101  // Immediate invocation is allowed.
102  impl->locked_ = true;
103  impl->mutex_.unlock();
104  return true;
105  }
106 
107  if (impl->locked_)
108  {
109  // Some other handler already holds the strand lock. Enqueue for later.
110  impl->waiting_queue_.push(op);
111  impl->mutex_.unlock();
112  }
113  else
114  {
115  // The handler is acquiring the strand lock and so is responsible for
116  // scheduling the strand.
117  impl->locked_ = true;
118  impl->mutex_.unlock();
119  impl->ready_queue_.push(op);
120  io_service_.post_immediate_completion(impl, false);
121  }
122 
123  return false;
124 }

147 void strand_service::do_complete(io_service_impl* owner, operation* base,
148  const asio::error_code& ec, std::size_t /*bytes_transferred*/)
149 {
150  if (owner)
151  {
152  strand_impl* impl = static_cast<strand_impl*>(base);
153 
154  // Indicate that this strand is executing on the current thread.
155  call_stack<strand_impl>::context ctx(impl);
156 
157  // Ensure the next handler, if any, is scheduled on block exit.
158  on_do_complete_exit on_exit = { owner, impl };
159  (void)on_exit;
160 
161  // Run all ready handlers. No lock is required since the ready queue is
162  // accessed only within the strand.
163  while (operation* o = impl->ready_queue_.front())
164  {
165  impl->ready_queue_.pop();
166  o->complete(*owner, ec, 0);
167  }
168  }
169 }
