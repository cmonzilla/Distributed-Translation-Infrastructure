71  template<typename WordIndexType, template<phrase_length > class StorageFactory, class StorageContainer>
72  void W2CHybridTrie<WordIndexType, StorageFactory, StorageContainer>::pre_allocate(const size_t counts[LM_M_GRAM_LEVEL_MAX]) {
73  //01) Pre-allocate the word index super class call
74  BASE::pre_allocate(counts);
75 
76  //Compute the number of words to be stored
77  m_word_arr_size = BASE::get_word_index().get_number_of_words(counts[0]);
78 
79  //02) Allocate the factory
80  m_storage_factory = new StorageFactory<LM_M_GRAM_LEVEL_MAX>(counts);
81 
82  //03) Allocate the main arrays of pointers where probs/back-offs will be stored
83 
84  //First allocate the memory for the One-grams, add an extra
85  //element for the unknown word and initialize it!
86  m_mgram_data[0] = new m_gram_payload[m_word_arr_size];
87  memset(m_mgram_data[0], 0, m_word_arr_size * sizeof (m_gram_payload));
88 
89  //Allocate more memory for probabilities and back off weight for
90  //the remaining M-gram levels until M < N. For M==N there is no
91  //back-off weights and thus we will store the probabilities just
92  //Inside the C container class values.
93  for (int idx = 1; idx < (LM_M_GRAM_LEVEL_MAX - 1); idx++) {
94  m_mgram_data[idx] = new m_gram_payload[counts[idx]];
95  memset(m_mgram_data[idx], 0, counts[idx] * sizeof (m_gram_payload));
96  }
97 
98  //04) Allocate the word map arrays per level There is N-1 levels to have
99  //as the for M == 0 - the One Grams, we do not need this mappings
100  for (int idx = 0; idx < (LM_M_GRAM_LEVEL_MAX - 1); idx++) {
101  m_mgram_mapping[idx] = new StorageContainer*[m_word_arr_size];
102  memset(m_mgram_mapping[idx], 0, m_word_arr_size * sizeof (StorageContainer*));
103  }
104  }

116  W2CHybridTrie<WordIndexType, StorageFactory, StorageContainer>::~W2CHybridTrie() {
117  //Delete the probability and back-off data
118  for (phrase_length idx = 0; idx < (LM_M_GRAM_LEVEL_MAX - 1); idx++) {
119  //Delete the prob/back-off arrays per level
120  if (m_mgram_data[idx] != NULL) {
121  delete[] m_mgram_data[idx];
122  }
123  }
124  //Delete the mapping data
125  for (phrase_length idx = 0; idx < (LM_M_GRAM_LEVEL_MAX - 1); idx++) {
126  //Delete the word arrays per level
127  if (m_mgram_mapping[idx] != NULL) {
128  for (TShortId widx = 0; widx < m_word_arr_size; widx++) {
129  //Delete the C containers per word index
130  if (m_mgram_mapping[idx][widx] != NULL) {
131  delete m_mgram_mapping[idx][widx];
132  }
133  }
134  delete[] m_mgram_mapping[idx];
135  }
136  }
137  if (m_storage_factory != NULL) {
138  delete m_storage_factory;
139  }
140  }
