107  void de_serialize(const string & message) {
108  //Initialize the reader
109  TextPieceReader reader(message.c_str(), message.length());
110  LOG_DEBUG3 << "De-serializing reply message (" << &message << "): '" << reader.str() << "'" << END_LOG;
111 
112  //The text will contain the read text from the reader
113  TextPieceReader text;
114 
115  //First get the job id
116  if (reader.get_first<HEADER_DELIMITER>(text)) {
117  LOG_DEBUG1 << "Message " << &message << ", read the job id: " << text.str() << END_LOG;
118  m_job_id = stoi(text.str());
119  //Second get the result code
120  if (reader.get_first<NEW_LINE_HEADER_ENDING>(text)) {
121  LOG_DEBUG1 << "Message " << &message << ", read the job code: " << text.str() << END_LOG;
122  m_code = trans_job_code(stoi(text.str()));
123 
124  //Now the rest is the translated text or the error message
125  m_text = reader.get_rest_str();
126 
127  LOG_DEBUG << "Received message " << &message << ", \nm_job_id = " << m_job_id
128  << ", m_code = " << m_code << ", m_text = \n" << m_text << END_LOG;
129  } else {
130  THROW_EXCEPTION(string("Could not find result code in the job reply header!"));
131  }
132  } else {
133  THROW_EXCEPTION(string("Could not find job_id in the job reply header!"));
134  }
135  }
