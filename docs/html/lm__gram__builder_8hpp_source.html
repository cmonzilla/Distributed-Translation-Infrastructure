89  static inline bool unigram_to_prob(text_piece_reader &text, text_piece_reader & word, prob_weight & prob) {
90  //There should be some text left it it is an M-gram
91  if (text.has_more()) {
92  //First read text until the first tab, it should be present if it is a uni-gram line
93  if (text.get_first_tab(word)) {
94  if (fast_s_to_f(prob, word.get_rest_c_str())) {
95  //If the uni-gram probability has been parsed, then read the next token which should be the word itself
96  if (text.get_first_tab(word)) {
97  //We read the word and so we are done reading this uni-gram the rest is irrelevant
98  return true;
99  } else {
100  //It looks like we are done with reading uni-grams
101  LOG_DEBUG1 << "Could not read the uni-gram word, looks like we are done!" << END_LOG;
102  return false;
103  }
104  } else {
105  //It looks like we are done with reading uni-grams
106  LOG_DEBUG1 << "Could not parse the uni-gram probability (" << word.str() << "), looks like we are done!" << END_LOG;
107  return false;
108  }
109  } else {
110  //It looks like we are done with reading uni-grams
111  LOG_DEBUG1 << "Could read the uni-gram probability as string, looks like we are done!" << END_LOG;
112  return false;
113  }
114  } else {
115  //Unexpected end of text
116  LOG_WARNING << "An unexpected end of line '" << text.str() << "' when reading the uni-gram!" << END_LOG;
117  return false;
118  }
119  }
