115  inline bool get_ctx_id(const phrase_length level_idx, const TShortId word_id, TLongId & ctx_id) const {
116  const TLongId key = put_32_32_in_64(ctx_id, word_id);
117 
118  //Search for the map for that context id
119  TMGramsMap::const_iterator result = m_m_gram_map_ptrs[level_idx]->find(key);
120  if (result == m_m_gram_map_ptrs[level_idx]->end()) {
121  //There is no data found under this context
122  return false;
123  } else {
124  //Update the context with the found value uf the next context
125  ctx_id = result->second;
126  //The context can always be computed
127  return true;
128  }

151  template<phrase_length CURR_LEVEL>
152  inline void add_m_gram(const model_m_gram & gram) {
153  const TShortId word_id = gram.get_last_word_id();
154  if (CURR_LEVEL == M_GRAM_LEVEL_1) {
155  //Store the payload
156  m_1_gram_data[word_id] = gram.m_payload;
157  } else {
158  //Register the m-gram in the hash cache
159  this->register_m_gram_cache(gram);
160 
161  //Define the context id variable
162  TLongId ctx_id = UNKNOWN_WORD_ID;
163  //Obtain the m-gram context id
164  __LayeredTrieBase::get_context_id<C2DHybridTrie<WordIndexType>, CURR_LEVEL, DebugLevelsEnum::DEBUG2>(*this, gram, ctx_id);
165 
166  //Obtain the context key and then create a new mapping
167  const TLongId key = put_32_32_in_64(ctx_id, word_id);
168 
169  //Store the payload
170  if (CURR_LEVEL == LM_M_GRAM_LEVEL_MAX) {
171  m_n_gram_map_ptr->operator[](key) = gram.m_payload.m_prob;
172  } else {
173  //Get the next context id
174  const phrase_length level_idx = (CURR_LEVEL - BASE::MGRAM_IDX_OFFSET);
175  TShortId next_ctx_id = m_M_gram_next_ctx_id[level_idx]++;
176 
177  //Store the context mapping inside the map
178  m_m_gram_map_ptrs[level_idx]->operator[](key) = next_ctx_id;
179 
180  //Return the reference to the piece of memory
181  m_m_gram_data[level_idx][next_ctx_id] = gram.m_payload;
182  }
183  }

206  inline void get_m_gram_payload(m_gram_query & query, MGramStatusEnum & status) const {
207  LOG_DEBUG << "Getting the payload for sub-m-gram : "<< query << END_LOG;
208 
209  //First ensure the context of the given sub-m-gram
210  LAYERED_BASE_ENSURE_CONTEXT(query, status);
211 
212  //If the context is successfully ensured, then move on to the m-gram and try to obtain its payload
213  if (status == MGramStatusEnum::GOOD_PRESENT_MGS) {
214  //Store the shorthand for the context and end word id
215  TLongId & ctx_id = query.get_curr_ctx_ref();
216  const TShortId & word_id = query.get_curr_end_word_id();
217 
218  //Compute the distance between words
219  const phrase_length & curr_level =query.get_curr_level();
220  LOG_DEBUG << "curr_level: " << SSTR(curr_level) << ", ctx_id: " << ctx_id << ", m_end_word_idx: "
221  << SSTR(query.m_curr_end_word_idx) << ", end word id: " << word_id << END_LOG;
222 
223  //Get the next context id
224  const phrase_length & level_idx = query.get_curr_level_m2();
225  if (get_ctx_id(level_idx, word_id, ctx_id)) {
226  LOG_DEBUG << "level_idx: " << SSTR(level_idx) << ", ctx_id: " << ctx_id << END_LOG;
227  //There is data found under this context
228  query.set_curr_payload(&m_m_gram_data[level_idx][ctx_id]);
229  LOG_DEBUG << "The payload is retrieved: " << m_m_gram_data[level_idx][ctx_id] << END_LOG;
230  } else {
231  //The payload could not be found
232  LOG_DEBUG1 << "Unable to find m-gram data for ctx_id: " << SSTR(ctx_id)
233  << ", word_id: " << SSTR(word_id) << END_LOG;
234  status = MGramStatusEnum::BAD_NO_PAYLOAD_MGS;
235  }
236  LOG_DEBUG << "Context ensure status is: " << status_to_string(status) << END_LOG;
237  }
