129  inline bool get_ctx_id(const phrase_length level_idx, const TShortId word_id, TLongId & ctx_id) const {
130  //Compute back the current level pure for debug purposes.
131  const phrase_length curr_level = level_idx + BASE::MGRAM_IDX_OFFSET;
132 
133  ASSERT_SANITY_THROW((curr_level == LM_M_GRAM_LEVEL_MAX) || (level_idx < 0),
134  string("Unsupported level id: ") + std::to_string(curr_level));
135 
136  LOG_DEBUG2 << "Searching next ctx_id for " << SSTR(curr_level)
137  << "-gram with word_id: " << SSTR(word_id) << ", ctx_id: "
138  << SSTR(ctx_id) << END_LOG;
139 
140  //First get the sub-array reference.
141  const T_M_GramWordEntry & ref = m_m_gram_word_2_data[level_idx][word_id];
142 
143  if (DO_SANITY_CHECKS && ref.has_data()) {
144  LOG_DEBUG3 << "ref.size: " << SSTR(ref.size()) << ", ref.cio: "
145  << SSTR(ref.cio) << ", ctx_id range: [" << SSTR(ref[0].id) << ", "
146  << SSTR(ref[ref.size() - 1].id) << "]" << END_LOG;
147  }
148 
149  //Check that if this is the 2-Gram case and the previous context
150  //id is 0 then it is the unknown word id, at least this is how it
151  //is now in ATrie implementation, so we need to do a warning!
152  if (DO_SANITY_CHECKS && (curr_level == M_GRAM_LEVEL_2) && (ctx_id < MIN_KNOWN_WORD_ID)) {
153  LOG_WARNING << "Perhaps we are being paranoid but there "
154  << "seems to be a problem! The " << SSTR(curr_level) << "-gram ctx_id: "
155  << SSTR(ctx_id) << " is equal to an undefined(" << SSTR(UNDEFINED_WORD_ID)
156  << ") or unknown(" << SSTR(UNKNOWN_WORD_ID) << ") word ids!" << END_LOG;
157  }
158 
159  //Get the local entry index and then use it to compute the next context id
160  typename T_M_GramWordEntry::TIndexType local_idx;
161  //If the local entry index could be found then compute the next ctx_id
162  if (get_m_n_gram_local_entry_idx(ref, ctx_id, local_idx)) {
163  LOG_DEBUG2 << "Got context mapping for ctx_id: " << SSTR(ctx_id)
164  << ", size = " << SSTR(ref.size()) << ", localIdx = "
165  << SSTR(local_idx) << ", resulting ctx_id = "
166  << SSTR(ref.cio + local_idx) << END_LOG;
167 
168  //The next ctx_id is the sum of the local index and the context index offset
169  ctx_id = ref.cio + local_idx;
170  return true;
171  } else {
172  //The local index could not be found
173  return false;
174  }
175  }

240  template<phrase_length CURR_LEVEL>
241  inline void add_m_gram(const model_m_gram & gram) {
242  const TShortId word_id = gram.get_last_word_id();
243  if (CURR_LEVEL == M_GRAM_LEVEL_1) {
244  //Store the payload
245  m_1_gram_data[word_id] = gram.m_payload;
246  } else {
247  //Register the m-gram in the hash cache
248  this->register_m_gram_cache(gram);
249 
250  //Define the context id variable
251  TLongId ctx_id = UNKNOWN_WORD_ID;
252  //Obtain the m-gram context id
253  __LayeredTrieBase::get_context_id<W2CArrayTrie<WordIndexType>, CURR_LEVEL, debug_levels_enum::DEBUG2>(*this, gram, ctx_id);
254 
255  //Store the payload
256  if (CURR_LEVEL == LM_M_GRAM_LEVEL_MAX) {
257  //Get the sub-array reference.
258  typename T_N_GramWordEntry::TElemType & ref = make_m_n_gram_entry<T_N_GramWordEntry>(m_n_gram_word_2_data, word_id);
259  //Store the context and word ids
260  ref.id = ctx_id;
261  //Return the reference to the probability
262  ref.payload = gram.m_payload.m_prob;
263  } else {
264  //Get the sub-array reference.
265  typename T_M_GramWordEntry::TElemType & ref = make_m_n_gram_entry<T_M_GramWordEntry>(m_m_gram_word_2_data[CURR_LEVEL - BASE::MGRAM_IDX_OFFSET], word_id);
266  //Store the context and word ids
267  ref.id = ctx_id;
268  //Return the reference to the newly allocated element
269  ref.payload = gram.m_payload;
270  }

295  LOG_DEBUG << "Getting the payload for sub-m-gram: " << query << END_LOG;
296 
297  //First ensure the context of the given sub-m-gram
298  LAYERED_BASE_ENSURE_CONTEXT(query, status);
299 
300  //If the context is successfully ensured, then move on to the m-gram and try to obtain its payload
301  if (status == MGramStatusEnum::GOOD_PRESENT_MGS) {
302  //Store the shorthand for the context and end word id
303  TLongId & ctx_id = query.get_curr_ctx_ref();
304  const TShortId & word_id = query.get_curr_end_word_id();
305 
306  LOG_DEBUG2 << "Getting m-gram with word_id: " << SSTR(word_id)
307  << ", ctx_id: " << SSTR(ctx_id) << END_LOG;
308 
309  //Get the entry
310  const typename T_M_GramWordEntry::TElemType * entry_ptr;
311  const phrase_length level_idx = (query.m_curr_end_word_idx - query.m_curr_begin_word_idx) + 1 - BASE::MGRAM_IDX_OFFSET;
312  const T_M_GramWordEntry * ptr = m_m_gram_word_2_data[level_idx];
313  if (get_m_n_gram_entry<T_M_GramWordEntry>(ptr, word_id, ctx_id, &entry_ptr)) {
314  //Return the data
315  query.set_curr_payload(&entry_ptr->payload);
316  LOG_DEBUG << "The payload is retrieved: " << entry_ptr->payload << END_LOG;

318  //The payload could not be found
319  LOG_DEBUG1 << "Unable to find m-gram data for ctx_id: " << SSTR(ctx_id)

394  template<typename WORD_ENTRY_TYPE>
395  void post_M_N_Grams(WORD_ENTRY_TYPE * wordsArray) {
396  //Define and initialize the current context index offset.
397  //The initial value is 1, although in this Trie it should
398  //not matter much, but it is better to reserve 0 for
399  //an undefined context value
400  TShortId cio = BASE::FIRST_VALID_CTX_ID;
401 
402  //Iterate through all the word_id sub-array mappings in the level and sort sub arrays
403  for (TShortId word_id = UNDEFINED_WORD_ID; word_id < m_num_word_ids; word_id++) {
404  //First get the sub-array reference.
405  WORD_ENTRY_TYPE & ref = wordsArray[word_id];
406 
407  //Assign the context index offset
408  ref.cio = cio;
409  //Compute the next context index offset, for the next word
410  cio += ref.size();
411 
412  //Reduce capacity if there is unused memory
413  ref.shrink();
414 
415  //Order the N-gram array as it is unordered and we will binary search it later!
416  ref.sort();
417  }
418  }

548  template<typename WORD_ENTRY_TYPE>
549  void preAllocateWordsData(WORD_ENTRY_TYPE* & wordsArray, const TShortId numMGrams, const TShortId numWords) {
550  //Allocate dynamic array entries and initialize it with the memory increase strategy
551  wordsArray = new WORD_ENTRY_TYPE[m_num_word_ids];
552 
553  //Compute the initial capacity
554  size_t capacity = 0;
555  if (__W2CArrayTrie::PRE_ALLOCATE_MEMORY) {
556  //Compute the average number of data elements per word on the given M-gram level
557  const float avg_num_elems = ((float) numMGrams) / ((float) numWords);
558  //Compute the corrected number of elements to preallocate, minimum __W2COrderedArrayTrie::MIN_MEM_INC_NUM.
559  capacity = max(static_cast<size_t> (avg_num_elems * __W2CArrayTrie::INIT_MEM_ALLOC_PRCT),
560  __W2CArrayTrie::MIN_MEM_INC_NUM);
561 
562  //Pre-allocate capacity
563  for (TShortId word_id = MIN_KNOWN_WORD_ID; word_id < m_num_word_ids; word_id++) {
564  wordsArray[word_id].pre_allocate(capacity);
565  }
566  }
567  }
