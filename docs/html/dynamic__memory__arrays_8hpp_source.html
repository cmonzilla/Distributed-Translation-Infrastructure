146  inline MemIncreaseStrategy get_mem_incr_strat(const MemIncTypesEnum stype,
147  const size_t min_mem_inc, const size_t mem_inc_factor) {
148  TCapacityIncFunct inc_func;
149 
150  //ToDo: optimize this switch, it is pretty ugly, use a map or something
151  switch (stype) {
152  case MemIncTypesEnum::CONSTANT:
153  inc_func = [] (const size_t fcap) -> size_t {
154  //Return zero as then the minimum constant increase will be used!
155  return 0;
156  };
157  break;
158  case MemIncTypesEnum::LINEAR:
159  inc_func = [] (const size_t fcap) -> size_t {
160  return fcap;
161  };
162  break;
163  case MemIncTypesEnum::LOG_2:
164  inc_func = [] (const size_t fcap) -> size_t {
165  return fcap * (1 / log(fcap));
166  };
167  break;
168  case MemIncTypesEnum::LOG_10:
169  inc_func = [] (const size_t fcap) -> size_t {
170  return fcap * (1 / log10(fcap));
171  };
172  break;
173  default:
174  stringstream msg;
175  msg << "Unrecognized memory allocation strategy: " << stype;
176  throw Exception(msg.str());
177  }
178 
179  //return the result object
180  return MemIncreaseStrategy(stype, inc_func, min_mem_inc, mem_inc_factor);
181  };

410  template<bool IS_INC = true >
411  static inline void reallocate(ELEMENT_TYPE_PTR & m_ptr, IDX_DATA_TYPE & m_capacity, size_t new_capacity) {
412  LOG_DEBUG2 << "The new capacity is " << SSTR(new_capacity)
413  << ", the old capacity was " << SSTR(m_capacity)
414  << ", ptr: " << SSTR((void*) m_ptr) << ", elem size: "
415  << SSTR(sizeof (ELEMENT_TYPE)) << END_LOG;
416 
417  //Check that there is still space to be allocated
418  ASSERT_CONDITION_THROW(IS_INC && (m_capacity == MAX_SIZE_TYPE_VALUE),
419  string("Unable to increase the capacity, reached the maximum of ") +
420  std::to_string(MAX_SIZE_TYPE_VALUE) + " allowed by the data type!");
421 
422  //Reallocate memory, potentially we get a new pointer!
423  new_capacity = min(new_capacity, MAX_SIZE_TYPE_VALUE);
424  ASSERT_SANITY_THROW((m_capacity == new_capacity), "The new capacity is equal to the old one!");
425  m_ptr = (ELEMENT_TYPE_PTR) realloc(m_ptr, new_capacity * sizeof (ELEMENT_TYPE));
426 
427  LOG_DEBUG2 << "Memory is reallocated ptr: " << SSTR(m_ptr) << END_LOG;
428 
429  //Initialize the newly allocated memory
430  if (IS_INC) {
431  for (IDX_DATA_TYPE idx = m_capacity; idx < new_capacity; ++idx) {
432  new(static_cast<void *> (&m_ptr[idx])) ELEMENT_TYPE();
433  LOG_DEBUG3 << "Creating a new element [" << SSTR(idx)
434  << "]: " << SSTR((void *) &m_ptr[idx]) << END_LOG;
435  }
436  }
437 
438  //Set the new capacity in
439  m_capacity = new_capacity;
440 
441  LOG_DEBUG2 << "The end capacity is " << SSTR(m_capacity)
442  << ", ptr: " << SSTR(m_ptr) << END_LOG;
443 
444  //Do the null pointer check if sanity
445  ASSERT_SANITY_THROW((m_ptr == NULL),
446  string("Ran out of memory when trying to allocate ") +
447  std::to_string(new_capacity) + " data elements for a word_id");
448  }
449  };
