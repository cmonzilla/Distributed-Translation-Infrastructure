531 asio::error_code context::use_certificate_chain(
532  const const_buffer& chain, asio::error_code& ec)
533 {
534  ::ERR_clear_error();
535 
536  bio_cleanup bio = { make_buffer_bio(chain) };
537  if (bio.p)
538  {
539  x509_cleanup cert = {
540  ::PEM_read_bio_X509_AUX(bio.p, 0,
541  handle_->default_passwd_callback,
542  handle_->default_passwd_callback_userdata) };
543  if (!cert.p)
544  {
545  ec = asio::error_code(ERR_R_PEM_LIB,
546  asio::error::get_ssl_category());
547  return ec;
548  }
549 
550  int result = ::SSL_CTX_use_certificate(handle_, cert.p);
551  if (result == 0 || ::ERR_peek_error() != 0)
552  {
553  ec = asio::error_code(
554  static_cast<int>(::ERR_get_error()),
555  asio::error::get_ssl_category());
556  return ec;
557  }
558 
559  if (handle_->extra_certs)
560  {
561  ::sk_X509_pop_free(handle_->extra_certs, X509_free);
562  handle_->extra_certs = 0;
563  }
564 
565  while (X509* cacert = ::PEM_read_bio_X509(bio.p, 0,
566  handle_->default_passwd_callback,
567  handle_->default_passwd_callback_userdata))
568  {
569  if (!::SSL_CTX_add_extra_chain_cert(handle_, cacert))
570  {
571  ec = asio::error_code(
572  static_cast<int>(::ERR_get_error()),
573  asio::error::get_ssl_category());
574  return ec;
575  }
576  }
577 
578  result = ::ERR_peek_last_error();
579  if ((ERR_GET_LIB(result) == ERR_LIB_PEM)
580  && (ERR_GET_REASON(result) == PEM_R_NO_START_LINE))
581  {
582  ::ERR_clear_error();
583  ec = asio::error_code();
584  return ec;
585  }
586  }
587 
588  ec = asio::error_code(
589  static_cast<int>(::ERR_get_error()),
590  asio::error::get_ssl_category());
591  return ec;
592 }
