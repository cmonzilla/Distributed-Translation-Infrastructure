95  circular_queue(const circular_queue & other, const size_t num_elems, const elem_type * elems) : m_size(0) {
96  LOG_DEBUG3 << "Creating with queue: " << other << ", and array: "
97  << array_to_string<elem_type>(num_elems, elems)
98  << ", the queue capacity: " << capacity << END_LOG;
99 
100  if ((num_elems >= capacity) || (other.m_size == 0)) {
101  LOG_DEBUG3 << "The number of array elements: " << num_elems
102  << ", is larget or equal to the queue capacity: " << capacity
103  << ", or the other queue size (" << other.m_size << ") is zero" << END_LOG;
104 
105  //The number of elements in the array is larger or equal
106  //to the queue capacity or the other queue is empty, then
107  //we can just ignore the other queue.
108  push_back(num_elems, elems);
109 
110  LOG_DEBUG3 << "The queue state: " << *this << END_LOG;
111  } else {
112  //The number of elements in the array is smaller than the
113  //capacity so we will fit both elements from the queue and
114  //from the array.
115 
116  //Compute the maximum number of elements from the queue that we can keep
117  const size_t max_num_from_queue = (capacity - num_elems);
118  //Compute the number of elements from the queue to keep
119  const size_t num_from_queue = min(other.m_size, max_num_from_queue);
120 
121  LOG_DEBUG3 << "The max num of elem(s) to copy from the other queue is: " << max_num_from_queue
122  << ", the number of elem(s) that will be copied is: " << num_from_queue << END_LOG;
123 
124  //Add the other queue elements, only the ones that we need to take over
125  push_back(num_from_queue, other.m_elems + (other.m_size - num_from_queue));
126 
127  LOG_DEBUG3 << "The queue state: " << *this << END_LOG;
128 
129  //Add the array elements
130  push_back(num_elems, elems);

168  if ((m_size >= num_elems) && (other.m_size >= num_elems)) {

230  void push_back(const size_t num_elems, const elem_type * elems) {
231  LOG_DEBUG3 << "Pushing array: " << array_to_string<elem_type>(num_elems, elems)
232  << " to the queue: " << *this << ", capacity: " << capacity << END_LOG;
233 
234  //Compute the new size right away
235  const size_t new_size = min((m_size + num_elems), capacity);
236 
237  LOG_DEBUG3 << "The new queue size will be " << new_size << END_LOG;
238 
239  //Check if we can fit the data without shifting or not
240  if ((m_size + num_elems) <= capacity) {
241  //There is space to accommodate everything without shifting
242 
243  LOG_DEBUG3 << "Copying " << num_elems << " elem(s) from "
244  << "position 0 to position: " << m_size << END_LOG;
245 
246  //Copy the new data to the remainder of the queue
247  memcpy(m_elems + m_size, elems, num_elems * sizeof (elem_type));
248  } else {
249  //The number of elements will become larger than the queue capacity
250 
251  //Check if the number of elements we have to add larger than the capacity
252  if (num_elems >= capacity) {
253  //The number of elements we are to add is larger than
254  //the queue capacity so we need to push all of the old
255  //elements out and may be some of the new elements.
256 
257  LOG_DEBUG3 << "The added number of elements " << num_elems
258  << " is larger the capacity: " << capacity << END_LOG;
259 
260  //Compute the sub-array element index to copy from
261  const size_t begin_idx = (num_elems - capacity);
262 
263  LOG_DEBUG3 << "Copying " << capacity << " elem(s) from position: "
264  << begin_idx << " to position 0 of the target " << END_LOG;
265 
266  //Compute the position to start copying from, we only need last "capacity" elements
267  const elem_type * sub_arr = elems + begin_idx;
268 
269  //Copy the last part of the data, overwriting everything there was
270  memcpy(m_elems, sub_arr, capacity * sizeof (elem_type));
271  } else {
272  //The number of elements we are to add is smaller than
273  //the queue capacity so they will all fit, but we need
274  //to push some of the old elements out
275 
276  //Compute the number of free slots
277 
278  const size_t num_empty = capacity - m_size;
279  //Compute the number of slots we need to free
280  const size_t num_push_out = num_elems - num_empty;
281  //Compute the number of elements to keep
282  const size_t num_keep = m_size - num_push_out;
283 
284  LOG_DEBUG3 << "Number of free slots is: " << num_empty << ", yet to free "
285  << num_push_out << " slots, the number of elements to keep is: "
286  << num_keep << END_LOG;
287 
288  LOG_DEBUG << "Moving " << num_keep << " elements from position: "
289  << num_push_out << " to position 0" << END_LOG;
290 
291  //Move the elements we have to keep to the beginning of the queue
292  memmove(m_elems, m_elems + num_push_out, num_keep * sizeof (elem_type));
293 
294  LOG_DEBUG << "Copying " << num_elems << " elem(s) from "
295  << "position 0 to position: " << num_keep << END_LOG;
296 
297  //Copy the elements into the freed remainder of the array
298  memcpy(m_elems + num_keep, elems, num_elems * sizeof (elem_type));
299  }
300  }
301 
302  //Store the new number of elements
303  m_size = new_size;
304  }
