94  template<bool is_need_ctx_ids>
95  inline void set_data(const phrase_length num_words, const word_uid * word_ids) {
96  //Check that the number of words has a valid value
97  ASSERT_SANITY_THROW((num_words > QUERY_M_GRAM_MAX_LEN),
98  string("The number of words in the query: ") + to_string(num_words) +
99  string(" exceeds the allowed maximum: ") + to_string(QUERY_M_GRAM_MAX_LEN));
100 
101  //Clean the probability and payload data
102  memset(m_probs, 0, sizeof (prob_weight) * QUERY_M_GRAM_MAX_LEN);
103  memset(m_payloads, 0, sizeof (void*) * QUERY_M_GRAM_MAX_LEN * QUERY_M_GRAM_MAX_LEN);
104 
105  //Clean the contexts if needed
106  if (is_need_ctx_ids) {
107  memset(m_last_ctx_ids, UNDEFINED_WORD_ID, sizeof (TLongId) * QUERY_M_GRAM_MAX_LEN);
108  }
109 
110  //Store the values
111  m_gram.set_m_gram(num_words, word_ids);
112  }
