108  bool connect() {
109  // Create a new connection to the given URI
110  websocketpp::lib::error_code ec;
111  client::connection_ptr con = m_client.get_connection(m_uri, ec);
112  ASSERT_CONDITION_THROW(ec, string("Get Connection (") + m_uri + string(") Error: ") + ec.message());
113 
114  // Grab a handle for this connection so we can talk to it in a thread
115  // safe manor after the event loop starts.
116  m_hdl = con->get_handle();
117 
118  // Queue the connection. No DNS queries or network connections will be
119  // made until the io_service event loop is run.
120  m_client.connect(con);
121 
122  // Create a thread to run the ASIO io_service event loop
123  m_asio_thread = thread(&client::run, &m_client);
124 
125  //Set the client as started
126  m_started = true;
127 
128  return wait_connect();
129  }

134  void disconnect() {
135  LOG_DEBUG << "Stopping the client: m_open = " << to_string(m_opened) << ", m_done = " << to_string(m_closed) << END_LOG;
136 
137  if (m_opened && !m_closed) {
138  //Invalidate the connection close notifier
139  m_notify_conn_close = NULL;
140 
141  LOG_INFO << "Closing the server connection..." << END_LOG;
142 
143  //Close the connection to the server
144  m_client.close(m_hdl, websocketpp::close::status::normal, "The needed translations are finished.");
145 
146  //Set the done flag to true as we are now done
147  m_closed = true;
148  }
149

151  if (m_started && !m_stopped) {

263  //Declare the variable to store the local connection status
264  bool is_connecting = false;
265 
266  //Wait until the connection is established
267  while (1) {
268  //Check the connection status

274  //If we we are still connecting then sleep, otherwise move on
275  if (is_connecting) {
276  LOG_DEBUG << "Going to sleep, m_open = " << to_string(m_opened)

278  sleep(1);

280  } else {
