74  static inline uint8_t get_number_of_bytes(const uint64_t word_id) {

76  LOG_DEBUG4 << "# bytes to store " << word_id << " is: 1, as it is <= " << MAX_VALUE_UINT8_T << END_LOG;

172  template<typename TWordIdType>
173  uint8_t Byte_M_Gram_Id<TWordIdType>::create_m_gram_id(const TWordIdType * word_ids,
174  const uint8_t num_word_ids, TM_Gram_Id_Value_Ptr & m_p_gram_id) {
175  //Do the sanity check if needed
176  ASSERT_SANITY_THROW((num_word_ids < M_GRAM_LEVEL_2) || (num_word_ids > M_GRAM_LEVEL_6),
177  string("create_m_gram_id: Unsupported m-gram level: ") + std::to_string(num_word_ids) +
178  string(", must be within [") + std::to_string(M_GRAM_LEVEL_2) + string(", ") +
179  std::to_string(M_GRAM_LEVEL_6) + string("]"));
180 
181  //Get the id len in bytes
182  const uint8_t TYPE_LEN_BYTES = ID_TYPE_LEN_BYTES[num_word_ids];
183 
184  uint8_t id_len_bytes = TYPE_LEN_BYTES;
185 
186  LOG_DEBUG3 << "Creating the " << SSTR(num_word_ids) << "-gram id with "
187  << "id type length: " << SSTR((uint32_t) TYPE_LEN_BYTES) << END_LOG;
188 
189  //Obtain the word ids and their lengths in bytes and
190  //the total length in bytes needed to store the key
191  uint8_t len_bytes[num_word_ids];
192  id_len_bytes += compute_byte_lengths(word_ids, num_word_ids, len_bytes);
193 
194  LOG_DEBUG3 << "Total len. in bytes: " << SSTR((uint32_t) id_len_bytes) << END_LOG;
195 
196  //Allocate the id memory
197  m_p_gram_id = new uint8_t[id_len_bytes];
198  LOG_DEBUG3 << "Created a Byte_M_Gram_Id: " << SSTR((void *) m_p_gram_id) << END_LOG;
199 
200  //NOTE: There is no need to clean the memory!
201  //We do pure assignments of all the bytes
202  //memset(m_p_gram_id, 0, id_len_bytes);
203 
204  //Determine the type id value from the bit lengths of the words
205  const uint32_t & id_type_value = gram_id_byte_len_2_type(num_word_ids, len_bytes);
206  LOG_DEBUG3 << "ID_TYPE_LEN_BYTES: " << (uint32_t) TYPE_LEN_BYTES
207  << ", id_type_value: " << id_type_value << END_LOG;
208 
209  //Set the id type to the very beginning of the M-gram id
210  copy_end_bytes_to_pos(id_type_value, TYPE_LEN_BYTES, m_p_gram_id, 0);
211 
212  //Append the word id meaningful bits to the id in reverse order
213  uint8_t to_byte_pos = TYPE_LEN_BYTES;
214  for (size_t idx = 0; idx < num_word_ids; ++idx) {
215  copy_end_bytes_to_pos(word_ids[idx], len_bytes[idx], m_p_gram_id, to_byte_pos);
216  to_byte_pos += len_bytes[idx];
217  }
218 
219  LOG_DEBUG3 << "Finished making the " << SSTR(num_word_ids) << "-gram id with "
220  << "id type length: " << SSTR((uint32_t) TYPE_LEN_BYTES) << END_LOG;
221  LOG_DEBUG3 << "Id bits: " << bytes_to_bit_string(m_p_gram_id, id_len_bytes) << END_LOG;
222 
223  return id_len_bytes;
