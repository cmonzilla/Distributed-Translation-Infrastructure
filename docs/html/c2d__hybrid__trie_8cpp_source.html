121  template<typename WordIndexType>
122  void C2DHybridTrie<WordIndexType>::pre_allocate(const size_t counts[LM_M_GRAM_LEVEL_MAX]) {
123  //Call the super class pre-allocator!
124  BASE::pre_allocate(counts);
125 
126  //Compute and store the M-gram level sizes in terms of the number of M-gram indexes per level
127  //Also initialize the M-gram index counters, for issuing context indexes
128  for (phrase_length i = 0; i < BASE::NUM_M_GRAM_LEVELS; i++) {
129  //The index counts must start with one as zero is reserved for the UNDEFINED_ARR_IDX
130  m_M_gram_next_ctx_id[i] = BASE::FIRST_VALID_CTX_ID;
131  //Due to the reserved first index, make the array sizes one element larger, to avoid extra computations
132  m_M_gram_num_ctx_ids[i] = counts[i + 1] + BASE::FIRST_VALID_CTX_ID;
133  }
134 
135  //Pre-allocate 0-Grams
136  preAllocateOGrams(counts);
137 
138  //Pre-allocate M-Grams
139  preAllocateMGrams(counts);
140 
141  //Pre-allocate N-Grams
142  preAllocateNGrams(counts);
