166  template<phrase_length CURR_LEVEL>
167  inline void add_m_gram(const model_m_gram & gram) {
168  if (CURR_LEVEL == M_GRAM_LEVEL_1) {
169  //Get the word id of this unigram, so there is just one word in it and its the end one
170  const TShortId word_id = gram.get_last_word_id();
171  //Store the probability data in the one gram data storage, under its id
172  m_1_gram_data[word_id] = gram.m_payload;
173  } else {
174  //Register the m-gram in the hash cache
175  this->register_m_gram_cache(gram);
176 
177  if (CURR_LEVEL == LM_M_GRAM_LEVEL_MAX) {
178  //Create a new M-Gram data entry
179  T_M_Gram_Prob_Entry & data = m_n_gram_data->add_new_element(gram.get_hash());
180  //Create the N-gram id from the word ids
181  gram.create_phrase_id(gram.get_first_word_idx(), CURR_LEVEL, data.m_id);
182  //Set the probability data
183  data.m_payload = gram.m_payload.m_prob;
184  } else {
185  //Compute the M-gram level index
186  constexpr phrase_length LEVEL_IDX = (CURR_LEVEL - BASE::MGRAM_IDX_OFFSET);
187  //Create a new M-Gram data entry
188  T_M_Gram_PB_Entry & data = m_m_gram_data[LEVEL_IDX]->add_new_element(gram.get_hash());
189  //Create the M-gram id from the word ids.
190  gram.create_phrase_id(gram.get_first_word_idx(), CURR_LEVEL, data.m_id);
191  //Set the probability and back-off data
192  data.m_payload = gram.m_payload;
193  }
194  }
195  }

271  template<typename STORAGE_MAP>
272  static inline MGramStatusEnum get_payload(const STORAGE_MAP * map,
273  m_gram_query & query) {
274 
275  LOG_DEBUG << "Getting the bucket id for: " << query << END_LOG;
276 
277  //Obtain the m-gram key
278  T_Gram_Id_Key key;
279  key.m_id = query.get_curr_m_gram_id(key.m_len_bytes);
280 
281  //Get the hash value
282  const uint64_t hash_value = query.get_curr_m_gram_hash();
283 
284  LOG_DEBUG << "Retrieving payload for " << query << END_LOG;
285 
286  //Get the element from the map
287  const typename STORAGE_MAP::TElemType * elem = map->get_element(hash_value, key);
288  if (elem != NULL) {
289  //We are now done, the payload is found, can return!
290  query.set_curr_payload(&elem->m_payload);
291  return MGramStatusEnum::GOOD_PRESENT_MGS;
292  } else {
293  //Could not retrieve the payload for the given sub-m-gram
294  LOG_DEBUG << "Unable to find the payload for sub-m-gram : " << query << END_LOG;
295  return MGramStatusEnum::BAD_NO_PAYLOAD_MGS;
296  }
297  }
