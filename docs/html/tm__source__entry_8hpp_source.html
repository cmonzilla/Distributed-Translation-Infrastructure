147  inline void add_target(const string & target, const phrase_uid target_uid,
148  const size_t num_features, const prob_weight * features,
149  const phrase_length num_words, const word_uid * word_ids,
150  const prob_weight lm_weight) {
151  //Perform a sanity check
152  ASSERT_SANITY_THROW((m_next_idx >= m_capacity),
153  string("Exceeding the source entry capacity: ") + to_string(m_capacity));
154 
155  //Get the next free entry for the target phrase
156  tm_target_entry & entry = m_targets[m_next_idx++];
157 
158  //Set the entry's target phrase and its id
159  entry.set_data(m_source_uid, target, target_uid,
160  num_features, features, num_words, word_ids);
161 
162  //Compute the minimum cost which in log space is a maximum value
163  m_min_cost = max(m_min_cost, (entry.get_t_c_s() + lm_weight));
164  }
