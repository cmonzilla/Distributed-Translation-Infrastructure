108  static inline void report_memory_usage(const char* action, TMemotyUsage ms_start, TMemotyUsage ms_end, const bool is_do_info) {
109  LOG_USAGE << "Action: \'" << action << "\' memory change:" << END_LOG;
110  LOG_DEBUG << "\tmemory before: vmsize=" << SSTR(ms_start.vmsize) << " Kb, vmpeak="
111  << SSTR(ms_start.vmpeak) << " Kb, vmrss=" << SSTR(ms_start.vmrss)
112  << " Kb, vmhwm=" << SSTR(ms_start.vmhwm) << " Kb" << END_LOG;
113  LOG_DEBUG << "memory after: vmsize=" << SSTR(ms_end.vmsize) << " Kb, vmpeak="
114  << SSTR(ms_end.vmpeak) << " Kb, vmrss=" << SSTR(ms_end.vmrss)
115  << " Kb, vmhwm=" << SSTR(ms_end.vmhwm) << " Kb" << END_LOG;
116 
117  int vmsize = ((ms_end.vmsize < ms_start.vmsize) ? 0 : ms_end.vmsize - ms_start.vmsize) / BYTES_ONE_MB;
118  int vmpeak = ((ms_end.vmpeak < ms_start.vmpeak) ? 0 : ms_end.vmpeak - ms_start.vmpeak) / BYTES_ONE_MB;
119  int vmrss = ((ms_end.vmrss < ms_start.vmrss) ? 0 : ms_end.vmrss - ms_start.vmrss) / BYTES_ONE_MB;
120  int vmhwm = ((ms_end.vmhwm < ms_start.vmhwm) ? 0 : ms_end.vmhwm - ms_start.vmhwm) / BYTES_ONE_MB;
121  LOG_USAGE << showpos << "vmsize=" << vmsize << " Mb, vmpeak=" << vmpeak
122  << " Mb, vmrss=" << vmrss << " Mb, vmhwm=" << vmhwm
123  << " Mb" << noshowpos << END_LOG;
124  if (is_do_info) {
125  LOG_INFO3 << " vmsize - Virtual memory size; vmpeak - Peak virtual memory size" << END_LOG;
126  LOG_INFO3 << " Virtual memory size is how much virtual memory the process has in total (RAM+SWAP)" << END_LOG;
127  LOG_INFO3 << " vmrss - Resident set size; vmhwm - Peak resident set size" << END_LOG;
128  LOG_INFO3 << " Resident set size is how much memory this process currently has in main memory (RAM)" << END_LOG;
129  }
