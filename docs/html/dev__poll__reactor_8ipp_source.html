70 void dev_poll_reactor::fork_service(asio::io_service::fork_event fork_ev)
71 {
72  if (fork_ev == asio::io_service::fork_child)
73  {
74  detail::mutex::scoped_lock lock(mutex_);
75 
76  if (dev_poll_fd_ != -1)
77  ::close(dev_poll_fd_);
78  dev_poll_fd_ = -1;
79  dev_poll_fd_ = do_dev_poll_create();
80 
81  interrupter_.recreate();
82 
83  // Add the interrupter's descriptor to /dev/poll.
84  ::pollfd ev = { 0, 0, 0 };
85  ev.fd = interrupter_.read_descriptor();
86  ev.events = POLLIN | POLLERR;
87  ev.revents = 0;
88  ::write(dev_poll_fd_, &ev, sizeof(ev));
89 
90  // Re-register all descriptors with /dev/poll. The changes will be written
91  // to the /dev/poll descriptor the next time the reactor is run.
92  for (int i = 0; i < max_ops; ++i)
93  {
94  reactor_op_queue<socket_type>::iterator iter = op_queue_[i].begin();
95  reactor_op_queue<socket_type>::iterator end = op_queue_[i].end();
96  for (; iter != end; ++iter)
97  {
98  ::pollfd& pending_ev = add_pending_event_change(iter->first);
99  pending_ev.events |= POLLERR | POLLHUP;
100  switch (i)
101  {
102  case read_op: pending_ev.events |= POLLIN; break;
103  case write_op: pending_ev.events |= POLLOUT; break;
104  case except_op: pending_ev.events |= POLLPRI; break;
105  default: break;
106  }
107  }
108  }
109  interrupter_.interrupt();
110  }
111 }

149 void dev_poll_reactor::start_op(int op_type, socket_type descriptor,
150  dev_poll_reactor::per_descriptor_data&, reactor_op* op,
151  bool is_continuation, bool allow_speculative)
152 {
153  asio::detail::mutex::scoped_lock lock(mutex_);
154 
155  if (shutdown_)
156  {
157  post_immediate_completion(op, is_continuation);
158  return;
159  }
160 
161  if (allow_speculative)
162  {
163  if (op_type != read_op || !op_queue_[except_op].has_operation(descriptor))
164  {
165  if (!op_queue_[op_type].has_operation(descriptor))
166  {
167  if (op->perform())
168  {
169  lock.unlock();
170  io_service_.post_immediate_completion(op, is_continuation);
171  return;
172  }
173  }
174  }
175  }
176 
177  bool first = op_queue_[op_type].enqueue_operation(descriptor, op);
178  io_service_.work_started();
179  if (first)
180  {
181  ::pollfd& ev = add_pending_event_change(descriptor);
182  ev.events = POLLERR | POLLHUP;
183  if (op_type == read_op
184  || op_queue_[read_op].has_operation(descriptor))
185  ev.events |= POLLIN;
186  if (op_type == write_op
187  || op_queue_[write_op].has_operation(descriptor))
188  ev.events |= POLLOUT;
189  if (op_type == except_op
190  || op_queue_[except_op].has_operation(descriptor))
191  ev.events |= POLLPRI;
192  interrupter_.interrupt();
193  }
194 }

237 void dev_poll_reactor::run(bool block, op_queue<operation>& ops)
238 {
239  asio::detail::mutex::scoped_lock lock(mutex_);
240 
241  // We can return immediately if there's no work to do and the reactor is
242  // not supposed to block.
243  if (!block && op_queue_[read_op].empty() && op_queue_[write_op].empty()
244  && op_queue_[except_op].empty() && timer_queues_.all_empty())
245  return;
246 
247  // Write the pending event registration changes to the /dev/poll descriptor.
248  std::size_t events_size = sizeof(::pollfd) * pending_event_changes_.size();
249  if (events_size > 0)
250  {
251  errno = 0;
252  int result = ::write(dev_poll_fd_,
253  &pending_event_changes_[0], events_size);
254  if (result != static_cast<int>(events_size))
255  {
256  asio::error_code ec = asio::error_code(
257  errno, asio::error::get_system_category());
258  for (std::size_t i = 0; i < pending_event_changes_.size(); ++i)
259  {
260  int descriptor = pending_event_changes_[i].fd;
261  for (int j = 0; j < max_ops; ++j)
262  op_queue_[j].cancel_operations(descriptor, ops, ec);
263  }
264  }
265  pending_event_changes_.clear();
266  pending_event_change_index_.clear();
267  }
268 
269  int timeout = block ? get_timeout() : 0;
270  lock.unlock();
271 
272  // Block on the /dev/poll descriptor.
273  ::pollfd events[128] = { { 0, 0, 0 } };
274  ::dvpoll dp = { 0, 0, 0 };
275  dp.dp_fds = events;
276  dp.dp_nfds = 128;
277  dp.dp_timeout = timeout;
278  int num_events = ::ioctl(dev_poll_fd_, DP_POLL, &dp);
279 
280  lock.lock();
281 
282  // Dispatch the waiting events.
283  for (int i = 0; i < num_events; ++i)
284  {
285  int descriptor = events[i].fd;
286  if (descriptor == interrupter_.read_descriptor())
287  {
288  interrupter_.reset();
289  }
290  else
291  {
292  bool more_reads = false;
293  bool more_writes = false;
294  bool more_except = false;
295 
296  // Exception operations must be processed first to ensure that any
297  // out-of-band data is read before normal data.
298  if (events[i].events & (POLLPRI | POLLERR | POLLHUP))
299  more_except =
300  op_queue_[except_op].perform_operations(descriptor, ops);
301  else
302  more_except = op_queue_[except_op].has_operation(descriptor);
303 
304  if (events[i].events & (POLLIN | POLLERR | POLLHUP))
305  more_reads = op_queue_[read_op].perform_operations(descriptor, ops);
306  else
307  more_reads = op_queue_[read_op].has_operation(descriptor);
308 
309  if (events[i].events & (POLLOUT | POLLERR | POLLHUP))
310  more_writes = op_queue_[write_op].perform_operations(descriptor, ops);
311  else
312  more_writes = op_queue_[write_op].has_operation(descriptor);
313 
314  if ((events[i].events & (POLLERR | POLLHUP)) != 0
315  && !more_except && !more_reads && !more_writes)
316  {
317  // If we have an event and no operations associated with the
318  // descriptor then we need to delete the descriptor from /dev/poll.
319  // The poll operation can produce POLLHUP or POLLERR events when there
320  // is no operation pending, so if we do not remove the descriptor we
321  // can end up in a tight polling loop.
322  ::pollfd ev = { 0, 0, 0 };
323  ev.fd = descriptor;
324  ev.events = POLLREMOVE;
325  ev.revents = 0;
326  ::write(dev_poll_fd_, &ev, sizeof(ev));
327  }
328  else
329  {
330  ::pollfd ev = { 0, 0, 0 };
331  ev.fd = descriptor;
332  ev.events = POLLERR | POLLHUP;
333  if (more_reads)
334  ev.events |= POLLIN;
335  if (more_writes)
336  ev.events |= POLLOUT;
337  if (more_except)
338  ev.events |= POLLPRI;
339  ev.revents = 0;
340  int result = ::write(dev_poll_fd_, &ev, sizeof(ev));
341  if (result != sizeof(ev))
342  {
343  asio::error_code ec(errno,
344  asio::error::get_system_category());
345  for (int j = 0; j < max_ops; ++j)
346  op_queue_[j].cancel_operations(descriptor, ops, ec);
347  }
348  }
349  }
350  }
351  timer_queues_.get_ready_timers(ops);
352 }

403 ::pollfd& dev_poll_reactor::add_pending_event_change(int descriptor)
404 {
405  hash_map<int, std::size_t>::iterator iter
406  = pending_event_change_index_.find(descriptor);
407  if (iter == pending_event_change_index_.end())
408  {
409  std::size_t index = pending_event_changes_.size();
410  pending_event_changes_.reserve(pending_event_changes_.size() + 1);
411  pending_event_change_index_.insert(std::make_pair(descriptor, index));
412  pending_event_changes_.push_back(::pollfd());
413  pending_event_changes_[index].fd = descriptor;
414  pending_event_changes_[index].revents = 0;
415  return pending_event_changes_[index];
416  }
417  else
418  {
419  return pending_event_changes_[iter->second];
420  }
421 }
