82  void add_state(stack_state_ptr new_state) {
83  LOG_DEBUG1 << "Adding a new state (" << new_state << ") to the "
84  << "level with " << m_size << " state(s)." << END_LOG;
85 
86  //If there is no states in the level yet, then set this one as the first
87  if (m_first_state == NULL) {
88  LOG_DEBUG1 << "Setting (" << new_state
89  << ") as the first/last in the level!" << END_LOG;
90 
91  //Insert this state as the first one
92  insert_as_first(new_state);
93  } else {
94  //Do not even consider the new state unless it is above the threshold
95  if (new_state->is_above_threshold(m_score_bound)) {
96  //Declare the state pointer that shall point to the
97  //position prior to which the new state is to be added.
98  stack_state_ptr curr_state = NULL;
99 
100  //Find the position the new state is to be inserted into
101  //or possibly recombine, if needed, with an existing state.
102  if (m_params.m_is_recombine && find_recombine(curr_state, *new_state)) {
103  //The new state was recombined into an existing one, no need to proceed.
104  return;
105  }
106 
107  LOG_DEBUG1 << "The last considered state is: " << curr_state << END_LOG;
108 
109  //Check if we found a state which is less probable than the new one
110  if (curr_state != NULL) {
111  //We need to add the new state before some existing state
112  add_before(curr_state, new_state);
113  } else {
114  //Add the new state as the last state inside the level
115  add_last(new_state);
116  }
117  } else {
118  //The new state is below the threshold, so delete it
119  LOG_DEBUG1 << "Deleting (threshold pruning) the ("

121  delete new_state;
122  }

186  inline bool find_recombine(stack_state_ptr & curr_state, stack_state & new_state) {
187  LOG_DEBUG1 << "Searching for the place for the " << &new_state
188  << " state within the level!" << END_LOG;
189 
190  //Initialize the reference to the pointer to the
191  //state place where we should put the new one
192  curr_state = m_first_state;
193 
194  //Search for the proper position of the state,
195  //the state we could recombine into or until
196  //we find an empty position, then stop.
197  while ((curr_state != NULL) && (new_state < *curr_state)) {
198  if (new_state == *curr_state) {
199  //The two states are equal meaning that they can be recombined.
200  //Since the new state is smaller than the current one it has
201  //smaller probability, therefore we need to recombine it into
202  //the current state and then we can just abort the process as
203  curr_state->recombine_from(&new_state);
204  //The number of elements in the level does not change, so return.
205  return true;
206  } else {
207  //Move further to the next state
208  curr_state = curr_state->m_next;
209  }
210  }
211  return false;
212  }

222  inline void add_last(stack_state_ptr new_state) {
223  //All other states are more probable and non is equivalent to the new one
224  LOG_DEBUG1 << "The new state is to be added to the end of the level" << END_LOG;
225 
226  //Check if there is free space left in the level
227  if (is_space_left()) {
228  //If there is free space left and the
229  //state satisfies the threshold pruning
230  //then we add it to the list of states
231  insert_as_last(new_state);
232  } else {
233  //There is no place in the existence for this poor fellow,
234  //destroy it. This is part of histogram pruning method.
235  LOG_DEBUG1 << "Deleting (histogram pruning) the ("
236  << new_state << ") state!" << END_LOG;
237  delete new_state;
238  }

252  inline void add_before(stack_state_ptr curr_state, stack_state_ptr new_state) {
253  //There is a less probable state
254  LOG_DEBUG1 << "The new state is to be added before: " << curr_state << END_LOG;
255 
256  //Insert this state before the
257  insert_before(curr_state, new_state);
258 
259  //There is a less probable state
260  LOG_DEBUG1 << "Checking for states equivalent to " << new_state
261  << " starting from " << curr_state << END_LOG;
262 
263  //If needed, search further from the curr_state, including
264  //curr_state, for a state that could be recombined into the
265  //newly added one. There can not be more than one of such
266  //states due to incremental nature of the stack building.
267  if (m_params.m_is_recombine) {
268 
269  //Search for the state to be recombined into this one
270  while ((curr_state != NULL) && (*new_state != *curr_state)) {
271  LOG_DEBUG << "Moving from " << curr_state << " to " << curr_state->m_next << END_LOG;
272  //Move further to the next state
273  curr_state = curr_state->m_next;
274  }
275 
276  //We found a state that is to be recombined into the new one.
277  if (curr_state != NULL) {
278  LOG_DEBUG << "Found an equivalent state " << curr_state
279  << " == " << new_state << " !" << END_LOG;
280 
281  //Remove the current state from the level
282  remove_from_level(curr_state);
283 
284  //Recombine the current state state into the new one
285  new_state->recombine_from(curr_state);
286  }
287  }
288 
289  //Perform pruning techniques, we might not have added a new
290  //state but we could have gotten the new best scoring state.
291  prune_states();
292  }

365  state->m_next = m_first_state;

374

480  inline void remove_from_level(stack_state_ptr state) {
481  ASSERT_SANITY_THROW((state == NULL),
482  "Bad pointer state == NULL");
483 
484  if (m_first_state == m_last_state) {
485  LOG_DEBUG1 << "Removing the only state " << state << " from the list " << END_LOG;
486  ASSERT_SANITY_THROW((state != m_first_state),
487  "The provided state is not in the list!");
488 
489  //This is the only element in the list
490  m_first_state = NULL;
491  m_last_state = NULL;
492  } else {
493  //There is more elements in the list
494  if (state == m_last_state) {
495  LOG_DEBUG1 << "Removing the last state " << state << "!" << END_LOG;
496 
497  //We are deleting the last element of the list, and
498  //there is more than one element in the list
499  ASSERT_SANITY_THROW((state->m_prev == NULL),
500  "The previous state can not be NULL!");
501 
502  //The new last element will be the previous of this one
503  m_last_state = state->m_prev;
504  //The new last element shall have no next element
505  m_last_state->m_next = NULL;
506  } else {
507  if (state == m_first_state) {
508  LOG_DEBUG1 << "Removing the first state " << state << "!" << END_LOG;
509 
510  //We are deleting the first element of the list, and
511  //there is more than one element in the list
512  ASSERT_SANITY_THROW((state->m_next == NULL),
513  "The next state can not be NULL!");
514 
515  //The new first element will be the next of this one
516  m_first_state = state->m_next;
517  //The new first element shall have no previous element
518  m_first_state->m_prev = NULL;
519  } else {
520  //We are deleting some intermediate element
521  LOG_DEBUG1 << "Removing an intermediate state " << state << "!" << END_LOG;
522 
523  //The previous of this shall now point to the next of this as next
524  state->m_prev->m_next = state->m_next;
525 
526  //The next of this shall now point to the previous of this as previous
527  state->m_next->m_prev = state->m_prev;
528  }
529  }
530  }
531 
532  //Now it is time to decrement the count
533  m_size--;
534  }
