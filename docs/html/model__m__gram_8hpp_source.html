115  template<typename WordIndexType>
116  inline void prepare_for_adding(WordIndexType & word_index) {
117  LOG_DEBUG1 << "Preparing the " << SSTR(BASE::get_num_words()) << "-gram for adding to the trie." << END_LOG;
118 
119  //If we have a unigram then add it to the index otherwise get the word ids
120  if (BASE::get_num_words() == M_GRAM_LEVEL_1) {
121  const phrase_length begin_word_idx = BASE::get_first_word_idx();
122  if (word_index.is_word_registering_needed()) {
123  //Register the word if it is needed
124  m_word_ids[begin_word_idx] = word_index.register_word(m_tokens[begin_word_idx]);
125  } else {
126  //Otherwise jut get its id
127  m_word_ids[begin_word_idx] = word_index.get_word_id(m_tokens[begin_word_idx]);
128  }
129  //The Unigram's hash value is equal to the word id
130  m_hash_values[begin_word_idx] = m_word_ids[begin_word_idx];
131 
132  LOG_DEBUG1 << "word[" << SSTR(begin_word_idx) << "] = "
133  << m_word_ids[begin_word_idx]
134  << ", hash[" << SSTR(begin_word_idx) << "] = "
135  << m_hash_values[begin_word_idx] << END_LOG;
136  } else {
137  phrase_length curr_idx = BASE::get_first_word_idx();
138  //Start with the first word
139  m_word_ids[curr_idx] = word_index.get_word_id(m_tokens[curr_idx]);
140  //The Unigram's hash value is equal to the word id
141  m_hash_values[curr_idx] = m_word_ids[curr_idx];
142 
143  //Store the word ids without the unknown word flags and pre-compute the m-gram hash values
144  for (++curr_idx; curr_idx <= BASE::get_last_word_idx(); ++curr_idx) {
145  //Get the next word id
146  m_word_ids[curr_idx] = word_index.get_word_id(m_tokens[curr_idx]);
147  //Compute the next hash value
148  m_hash_values[curr_idx] = combine_phrase_uids(m_hash_values[curr_idx - 1], m_word_ids[curr_idx]);
149 
150  LOG_DEBUG1 << "hash[" << SSTR(curr_idx) << "] = combine( word["
151  << SSTR(curr_idx) << "] = " << m_word_ids[curr_idx]
152  << ", hash[" << SSTR(curr_idx - 1) << "] = "
153  << m_hash_values[curr_idx - 1] << " ) = "
154  << m_hash_values[curr_idx] << END_LOG;
155  }
156  }
157  }
