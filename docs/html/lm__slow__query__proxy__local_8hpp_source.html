83  virtual void execute(TextPieceReader & line) {
84  //Re-initialize the joint prob reault with zero
85  m_joint_prob = 0.0;
86 
87  //Parse the query line into tokens and get the word ids thereof
88  set_tokens_and_word_ids(line);
89 
90  //Check that there was something on the line!
91  if (m_num_words > 0) {
92  //Set the words data into the query object
93  m_query.set_data<m_is_ctx>(m_num_words, m_word_ids);
94 
95  //Compute the maximum to consider m-gram level
96  const phrase_length max_m_gram_level = std::min<phrase_length>(m_num_words, LM_M_GRAM_LEVEL_MAX);
97 
98  //Initialize the begin and end word indexes
99  phrase_length begin_word_idx = 0;
100  phrase_length end_word_idx = begin_word_idx + max_m_gram_level - 1;
101 
102  //Set the m-gram values for the first query execution
103  m_query.set_word_indxes(begin_word_idx, begin_word_idx, end_word_idx);
104 
105  //Execute the first part of the query
106  m_trie.execute(m_query);
107 
108  //Report the partial results, and update the total
109  get_report_interm_results(begin_word_idx, begin_word_idx, end_word_idx);
110 
111  //Now do the sliding window and compute more probabilities,
112  //Note that if the end_word_idx is smaller than the query
113  //last word idx then it means that:
114  // (max_m_gram_level == LM_M_GRAM_LEVEL_MAX)
115  //and there is still m-grams to compute
116  while (end_word_idx < m_query.get_query_end_word_idx()) {
117  //Slide the window one step forward
118  begin_word_idx++;
119  end_word_idx++;
120 
121  //Set the window value inside, this time we need a single probability and not the joint
122  m_query.set_word_indxes(begin_word_idx, end_word_idx);
123 
124  //Execute the query
125  m_trie.execute(m_query);
126 
127  //Report the partial result, and update the total
128  get_report_interm_results(begin_word_idx, end_word_idx, end_word_idx);
129  }
130 
131  //Report the total result
132  report_final_result();
133  } else {
134  LOG_WARNING << "Got an empty line query request: Ignoring!" << END_LOG;
135  }
136 
137  LOG_RESULT << "-------------------------------------------" << END_LOG;
138  }

148  inline void get_report_interm_results(
149  const phrase_length begin_word_idx,
150  const phrase_length first_end_word_idx,
151  const phrase_length last_end_word_idx) {
152  //Print the intermediate results
153  for (phrase_length end_word_idx = first_end_word_idx; end_word_idx <= last_end_word_idx; ++end_word_idx) {
154  const string gram_str = get_m_gram_str(begin_word_idx, end_word_idx);
155 
156  LOG_RESULT << " log_" << LOG_PROB_WEIGHT_BASE << "( Prob( " << gram_str
157  << " ) ) = " << SSTR(m_query.m_probs[end_word_idx]) << END_LOG;
158  LOG_INFO << " Prob( " << gram_str << " ) = "
159  << SSTR(pow(LOG_PROB_WEIGHT_BASE, m_query.m_probs[end_word_idx])) << END_LOG;
160 
161  if (m_query.m_probs[end_word_idx] > ZERO_LOG_PROB_WEIGHT) {
162  m_joint_prob += m_query.m_probs[end_word_idx];
163  }
164  }

191  inline string get_m_gram_str(const phrase_length begin_word_idx, const phrase_length end_word_idx) const {
192  if (begin_word_idx > end_word_idx) {
193  return "<none>";
194  } else {
195  if (begin_word_idx == end_word_idx) {
196  const TextPieceReader & token = m_tokens[begin_word_idx];
197  return token.str().empty() ? "<empty>" : token.str();
198  } else {
199  string result = m_tokens[end_word_idx].str() + " |";
200  for (phrase_length idx = begin_word_idx; idx != end_word_idx; ++idx) {
201  result += string(" ") + m_tokens[idx].str();
202  }
203  return result;
204  }
205  }
206  }

215  inline string get_query_str() const {
216  if (m_num_words == 0) {
217  return "<none>";
218  } else {
219  if (m_num_words == 1) {
220  const TextPieceReader & token = m_tokens[0];
221  return token.str().empty() ? "<empty>" : token.str();
222  } else {
223  string result;
224  for (phrase_length idx = 0; idx < m_num_words; ++idx) {
225  result += m_tokens[idx].str() + string(" ");
226  }
227  return result.substr(0, result.length() - 1);
228  }
229  }

235  virtual void set_tokens_and_word_ids(TextPieceReader phrase) {
236  //Initialize with zero words
237  m_num_words = 0;
238 
239  LOG_DEBUG1 << "Getting word uids for phrase: ___" << phrase << "___" << END_LOG;
240 
241  //Read the tokens one by one backwards and decrement the index
242  while ((m_num_words < LM_MAX_QUERY_LEN) && phrase.get_first_space(m_tokens[m_num_words])) {
243  //Obtain the word id from the word index
244  m_word_ids[m_num_words] = m_word_idx.get_word_id(m_tokens[m_num_words]);
245 
246  LOG_DEBUG1 << "Word: ___" << m_tokens[m_num_words]
247  << "___ uid: " << m_word_ids[m_num_words] << END_LOG;
248 
249  //Increase the number of read words
250  m_num_words++;
251  }
252 
253  //Check that we do not get too many words!
254  ASSERT_CONDITION_THROW(phrase.has_more(),
255  string("The m-gram query: ___") + phrase.str() +
256  string("___ is too long, the maximum length is: ") +
257  to_string(LM_MAX_QUERY_LEN));
258  };
