51 void endpoint::resize(std::size_t new_size)
52 {
53  if (new_size > sizeof(asio::detail::sockaddr_un_type))
54  {
55  asio::error_code ec(asio::error::invalid_argument);
56  asio::detail::throw_error(ec);
57  }
58  else if (new_size == 0)
59  {
60  path_length_ = 0;

69  --path_length_;

99 void endpoint::init(const char* path_name, std::size_t path_length)
100 {
101  if (path_length > sizeof(data_.local.sun_path) - 1)
102  {
103  // The buffer is not large enough to store this address.
104  asio::error_code ec(asio::error::name_too_long);
105  asio::detail::throw_error(ec);
106  }
107 
108  using namespace std; // For memcpy.
109  data_.local = asio::detail::sockaddr_un_type();
110  data_.local.sun_family = AF_UNIX;
111  memcpy(data_.local.sun_path, path_name, path_length);
112  path_length_ = path_length;
113 
114  // NUL-terminate normal path names. Names that start with a NUL are in the
115  // UNIX domain protocol's "abstract namespace" and are not NUL-terminated.
116  if (path_length > 0 && data_.local.sun_path[0] == 0)
117  data_.local.sun_path[path_length] = 0;
118 }
