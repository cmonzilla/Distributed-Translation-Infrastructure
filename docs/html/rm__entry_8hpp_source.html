102  template<bool is_from>
103  const prob_weight get_weight(const reordering_orientation orient) const {
104  //Compute the static position correction for the from/to cases
105  static constexpr uint32_t pos_corr = (is_from ? 0 : num_features / 2);
106  static constexpr uint32_t mon_pos = pos_corr;
107  static constexpr uint32_t swap_pos = ((num_features <= TWO_RM_FEATURES) ? mon_pos : mon_pos + 1);
108  static constexpr uint32_t disc_left_pos = ((num_features <= FOUR_RM_FEATURES) ? swap_pos : swap_pos + 1);
109  static constexpr uint32_t disc_right_pos = ((num_features <= SIX_RM_FEATURES) ? disc_left_pos : disc_left_pos + 1);
110 
111  //Return the proper weight based on the orientation
112  switch (orient) {
113  case reordering_orientation::MONOTONE_ORIENT:
114  LOG_DEBUG1 << "MONOTONE_ORIENT " << (is_from ? "'from'" : "'to'")
115  << " position: " << mon_pos << ", value: " << m_weights[mon_pos] << END_LOG;
116  return m_weights[mon_pos];
117  case reordering_orientation::SWAP_ORIENT:
118  LOG_DEBUG1 << "SWAP_ORIENT " << (is_from ? "'from'" : "'to'")
119  << " position: " << swap_pos << ", value: " << m_weights[swap_pos] << END_LOG;
120  return m_weights[swap_pos];
121  case reordering_orientation::DISCONT_RIGHT_ORIENT:
122  LOG_DEBUG1 << "DISCONT_RIGHT_ORIENT " << (is_from ? "'from'" : "'to'")
123  << " position: " << disc_left_pos << ", value: " << m_weights[disc_left_pos] << END_LOG;
124  return m_weights[disc_left_pos];
125  case reordering_orientation::DISCONT_LEFT_ORIENT:
126  LOG_DEBUG1 << "DISCONT_LEFT_ORIENT " << (is_from ? "'from'" : "'to'")
127  << " position: " << disc_right_pos << ", value: " << m_weights[disc_right_pos] << END_LOG;
128  return m_weights[disc_right_pos];
129  default:
130  THROW_EXCEPTION(string("Unsupported orientation value: ") + to_string(orient));
131  }
132  }
