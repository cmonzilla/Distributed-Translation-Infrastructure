110 static void extract_arguments(const uint argc, char const * const * const argv, server_parameters & params) {
111  //Parse the arguments
112  try {
113  p_cmd_args->parse(argc, argv);
114  } catch (ArgException &e) {
115  THROW_EXCEPTION(string("Error: ") + e.error() + string(", for argument: ") + e.argId());
116  }
117 
118  //Set the logging level right away
119  logger::set_reporting_level(p_debug_level_arg->getValue());
120 
121  //Get the configuration file name and read the config values from the file
122  const string config_file_name = p_config_file_arg->getValue();
123  LOG_USAGE << "Loading the server configuration option from: " << config_file_name << END_LOG;
124  INI<> ini(config_file_name, false);
125 
126  //Parse the configuration file
127  if (ini.parse()) {
128  //Get the configuration options from the file
129  string section = "Server Options";
130  params.m_server_port = get_integer<uint16_t>(ini, section, "server_port");
131  params.m_num_threads = get_integer<uint16_t>(ini, section, "num_threads");
132  params.m_source_lang = get_string(ini, section, "source_lang");
133  params.m_target_lang = get_string(ini, section, "target_lang");
134  params.verify();
135 
136  LOG_USAGE << "Translation server from '" << params.m_source_lang << "' into '"
137  << params.m_target_lang << "' on port: '" << params.m_server_port
138  << "' translation threads: '" << params.m_num_threads << "'" << END_LOG;
139 
140  section = "Language Models";
141  params.m_lm_params.m_conn_string = get_string(ini, section, "connection_string");
142  tokenize_s_t_f<NUM_LM_FEATURES>("lm_feature_weights",
143  get_string(ini, section, "lm_feature_weights"),
144  params.m_lm_params.m_lambdas,
145  params.m_lm_params.m_num_lambdas,
146  LM_FEATURE_WEIGHTS_DELIMITER_STR);
147  params.m_lm_params.finalize();
148  LOG_INFO << params.m_lm_params << END_LOG;
149 
150  section = "Translation Models";
151  params.m_tm_params.m_conn_string = get_string(ini, section, "connection_string");
152  tokenize_s_t_f<NUM_TM_FEATURES>("tm_feature_weights",
153  get_string(ini, section, "tm_feature_weights"),
154  params.m_tm_params.m_lambdas,
155  params.m_tm_params.m_num_lambdas,
156  TM_FEATURE_WEIGHTS_DELIMITER_STR);
157  tokenize_s_t_f<NUM_TM_FEATURES>("tm_unk_features",
158  get_string(ini, section, "tm_unk_features"),
159  params.m_tm_params.m_unk_features,
160  params.m_tm_params.m_num_unk_features,
161  TM_FEATURE_WEIGHTS_DELIMITER_STR);
162  params.m_tm_params.m_trans_limit = get_integer<size_t>(ini, section, "translation_limit");
163  params.m_tm_params.m_min_tran_prob = get_float(ini, section, "min_trans_prob");
164  params.m_tm_params.finalize();
165  LOG_INFO << params.m_tm_params << END_LOG;
166 
167  section = "Reordering Models";
168  params.m_rm_params.m_conn_string = get_string(ini, section, "connection_string");
169  tokenize_s_t_f<NUM_RM_FEATURES>("rm_feature_weights",
170  get_string(ini, section, "rm_feature_weights"),
171  params.m_rm_params.m_lambdas,
172  params.m_rm_params.m_num_lambdas,
173  RM_FEATURE_WEIGHTS_DELIMITER_STR);
174  params.m_rm_params.finalize();
175  LOG_INFO << params.m_rm_params << END_LOG;
176 
177  section = "Decoding Options";
178  params.m_de_params.m_num_best_trans = get_integer<uint32_t>(ini, section, "num_best_trans");
179  params.m_de_params.m_distortion = get_integer<int32_t>(ini, section, "distortion");
180  params.m_de_params.m_ext_dist_left = get_integer<uint32_t>(ini, section, "ext_dist_left");
181  params.m_de_params.m_pruning_threshold = get_float(ini, section, "pruning_threshold");
182  params.m_de_params.m_stack_capacity = get_integer<uint32_t>(ini, section, "stack_capacity");
183  params.m_de_params.m_max_s_phrase_len = get_integer<phrase_length>(ini, section, "max_source_phrase_length");
184  params.m_de_params.m_max_t_phrase_len = get_integer<phrase_length>(ini, section, "max_target_phrase_length");
185  params.m_de_params.m_word_penalty = get_float(ini, section, "word_penalty");
186  params.m_de_params.m_phrase_penalty = get_float(ini, section, "phrase_penalty");
187  params.m_de_params.finalize();
188  LOG_INFO << params.m_de_params << END_LOG;
189 
190  LOG_INFO3 << "Sanity checks are: " << (DO_SANITY_CHECKS ? "ON" : "OFF") << " !" << END_LOG;
191  } else {
192  //We could not parse the configuration file, report an error
193  THROW_EXCEPTION(string("Could not find or parse the configuration file: ") + config_file_name);
194  }
195 }

235 int main(int argc, char** argv) {
236  //Declare the return code
237  int return_code = 0;
238 
239  //Set the uncaught exception handler
240  std::set_terminate(handler);
241 
242  //First print the program info
243  print_info();
244 
245  //Set up possible program arguments
246  create_arguments_parser();
247 
248  try {
249  //Define en empty parameters structure
250  server_parameters params;
251 
252  //Attempt to extract the program arguments
253  extract_arguments(argc, argv, params);
254 
255  //Initialize connections to the used models
256  connect_to_models(params);
257 
258  //Instantiate the translation server
259  translation_server server(params.m_server_port, params.m_num_threads);
260 
261  //Run the translation server in a separate thread
262  thread server_thread(bind(&translation_server::run, &server));
263 
264  LOG_USAGE << "The server is started!" << END_LOG;
265 
266  //Override the reporting level for testing purposes
267  //Logger::get_reporting_level() = DebugLevelsEnum::DEBUG2;
268 
269  //Wait until the server is stopped by pressing and exit button
270  perform_command_loop(params, server, server_thread);
271 
272  } catch (uva_exception & ex) {
273  //The argument's extraction has failed, print the error message and quit
274  LOG_ERROR << ex.get_message() << END_LOG;
275  return_code = 1;
276  }
277 
278  //Disconnect from the used models
279  disconnect_from_models();
280 
281  //Destroy the command line parameters parser
282  destroy_arguments_parser();
283 
284  return return_code;
285 }
