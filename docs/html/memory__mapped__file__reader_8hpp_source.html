115  MemoryMappedFileReader(const char * fileName) : AFileReader(), m_fileDesc(0) {
116  m_fileDesc = open(fileName, O_RDONLY);
117  LOG_DEBUG << "Opened the file '" << fileName << "' descriptor: " << SSTR(m_fileDesc) << END_LOG;
118 
119  if (m_fileDesc != UNDEFINED_FILE_DESCRIPTOR) {
120  // set the errno to default value
121  errno = 0;
122  //The statistics structure for the mapped file
123  struct stat fileStat;
124  if (fstat(m_fileDesc, &fileStat) < 0) {
125  LOG_ERROR << "Could not get the file '" << fileName << "' statistics after loading! ERROR: " << strerror(errno) << END_LOG;
126  //close the file
127  close();
128  } else {
129  //Get the file length
130  const size_t len = fileStat.st_size;
131  LOG_INFO << "Opened the file '" << fileName << "' size: " << SSTR(len) << " bytes." << END_LOG;
132 
133  //Map the file into memory
134  //NOTE: We populate the map as it is then immediately moved to
135  //the RAM and reading the file is then faster and it does not
136  //give a false impression of that we use too much memory in Trie.
137  // MAP_POPULATE Populate (prefault) page tables for a mapping.
138  //For a file mapping, this causes read-ahead on the file. Later
139  //accesses to the mapping will not be blocked by page faults.
140 #if __APPLE__
141  void * beginPtr = mmap(NULL, len, PROT_READ, MAP_PRIVATE, m_fileDesc, 0);
142 #else
143  void * beginPtr = mmap(NULL, len, PROT_READ, MAP_PRIVATE | MAP_POPULATE, m_fileDesc, 0);
144 #endif
145  LOG_DEBUG << "Memory mapping the file '" << fileName << "' gave: " << SSTR(beginPtr) << " pointer." << END_LOG;
146 
147  //Set the data to the base class
148  TextPieceReader::set(beginPtr, len);
149  }
150  }
151  }

190  //Release the memory

193  //Re-set the internals!
194  set(NULL, 0);

197  if (m_fileDesc != UNDEFINED_FILE_DESCRIPTOR) {
198  LOG_DEBUG << "Closing the Memory Mapped File file!" << END_LOG;
