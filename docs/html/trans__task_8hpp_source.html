208  void process_task_result() {
209  //Set the task is not canceled then set the result, otherwise set the canceled code.
210  if (m_code == trans_job_code::RESULT_ERROR) {
211  //If there was an error during translation send back the error message stored in the target texts
212  m_target_text = string("<error>: ") + m_target_text;
213  } else {
214  //Unless it was an error while translating there should be an undefined status
215  ASSERT_SANITY_THROW((m_code != trans_job_code::RESULT_UNDEFINED),
216  string("Unexpected translation code: ") + to_string(m_code) +
217  string(" must be UNDEFINED!"));
218 
219  //Check if we were interrupted or not
220  if (m_is_interrupted) {
221  //If the translation has been canceled just send back the source
222  m_code = trans_job_code::RESULT_CANCELED;
223  m_target_text = string("<canceled>: ") + m_source_text;
224  } else {
225  //If the translation has been finished send back the target
226  m_code = trans_job_code::RESULT_OK;
227  m_target_text = string("<finished>: ") + m_target_text;
228  }
229  }
230 
231  LOG_DEBUG1 << "The task " << m_task_id << " translation is done, notifying!" << END_LOG;
232 
233  //Call the task-done notification function to report that we are finished!s
234  m_notify_task_done_func(this);
235  }
