158  template<phrase_length CURR_LEVEL>
159  inline void add_m_gram(const model_m_gram & gram) {
160  //If not a uni-gram then register in the cache
161  if (CURR_LEVEL != M_GRAM_LEVEL_1) {
162  //Register the m-gram in the hash cache
163  this->register_m_gram_cache(gram);
164  }
165 
166  //Compute the M-gram level index, here we store m-grams for 1 <=m < n in one structure
167  constexpr phrase_length LEVEL_IDX = (CURR_LEVEL - LEVEL_IDX_OFFSET);
168 
169  //Get the bucket index
170  const uint64_t hash_value = gram.get_hash();
171  LOG_DEBUG << "Getting the bucket id for the m-gram: " << gram << " hash value: " << hash_value << END_LOG;
172 
173  if (CURR_LEVEL == LM_M_GRAM_LEVEL_MAX) {
174  //Create a new M-Gram data entry
175  T_M_Gram_Prob_Entry & data = m_n_gram_data->add_new_element(hash_value);
176  //The n-gram id is equal to its hash value
177  data.m_id = hash_value;
178  //Set the probability data
179  data.m_payload = gram.m_payload.m_prob;
180  } else {
181  //Check if this is an <unk> unigram, in this case we store the payload elsewhere
182  if ((CURR_LEVEL == M_GRAM_LEVEL_1) && gram.is_unk_unigram()) {
183  //Store the uni-gram payload - overwrite the default values.
184  m_unk_data = gram.m_payload;
185  } else {
186  //Create a new M-Gram data entry
187  T_M_Gram_PB_Entry & data = m_m_gram_data[LEVEL_IDX]->add_new_element(hash_value);
188  //The m-gram id is equal to its hash value
189  data.m_id = hash_value;
190  //Set the probability and back-off data
191  data.m_payload = gram.m_payload;
192  }
193  }
194  }

277  template<typename STORAGE_MAP>
278  static inline MGramStatusEnum get_payload(const STORAGE_MAP * map,
279  m_gram_query & query) {
280  LOG_DEBUG << "Getting the bucket id for the sub-m-gram " << query << END_LOG;
281 
282  const uint64_t hash_value = query.get_curr_m_gram_hash();
283 
284  LOG_DEBUG << "Retrieving payload for a sub-m-gram " << query << END_LOG;
285 
286  //Get the element from the map, note that the key is the hash value
287  const typename STORAGE_MAP::TElemType * elem = map->get_element(hash_value, hash_value);
288  if (elem != NULL) {
289  //We are now done, the payload is found, can return!
290  query.set_curr_payload(&elem->m_payload);
291  return MGramStatusEnum::GOOD_PRESENT_MGS;
292  } else {
293  //Could not retrieve the payload for the given sub-m-gram
294  LOG_DEBUG << "Unable to find the payload for the sub-m-gram " << query << END_LOG;
295  return MGramStatusEnum::BAD_NO_PAYLOAD_MGS;
296  }
297  }
