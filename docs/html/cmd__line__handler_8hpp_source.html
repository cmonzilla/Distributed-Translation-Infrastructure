251  inline bool process_input_cmd(server_parameters & params, translation_server &server,
252  thread &server_thread, char command[CMD_BUFF_SIZE]) {
253  //Convert the buffer into string
254  string cmd(command);
255 
256  //We are to print the command prompt
257  if (cmd == PROGRAM_NOTHING_CMD) {
258  return false;
259  }
260 
261  //Stop the server
262  if (cmd == PROGRAM_EXIT_CMD) {
263  stop(server, server_thread);
264  return true;
265  }
266 
267  //Print the server commands menu
268  if (cmd == PROGRAM_INFO_CMD) {
269  print_server_commands();
270  return false;
271  }
272 
273  //Report the runtime information to the console
274  if (cmd == PROGRAM_RUNTIME_CMD) {
275  server.report_run_time_info();
276  return false;
277  }
278 
279  //Lor parameters
280  if (cmd == PROGRAM_PARAMS_CMD) {
281  LOG_USAGE << "Log level: " << logger::get_curr_level_str()
282  << ", " << params << END_LOG;
283  return false;
284  }
285 
286  //Set the debug level
287  if (begins_with(cmd, PROGRAM_SET_LL_CMD)) {
288  set_log_level(cmd, PROGRAM_SET_LL_CMD);
289  return false;
290  }
291 
292  //Set the number of threads
293  if (begins_with(cmd, PROGRAM_SET_NT_CMD)) {
294  set_num_threads(params, server, cmd, PROGRAM_SET_LL_CMD);
295  return false;
296  }
297 
298  //Set other decoder parameters
299  set_decoder_params(cmd, params.m_de_params);
300 
301  //Continue to the next command.
302  return false;
303  }
