164  win_iocp_socket_service_base::base_implementation_type& impl,
165  asio::error_code& ec)
166 {
167  if (is_open(impl))
168  {
169  ASIO_HANDLER_OPERATION(("socket", &impl, "close"));
170 
171  // Check if the reactor was created, in which case we need to close the
172  // socket on the reactor as well to cancel any operations that might be
173  // running there.

175  interlocked_compare_exchange_pointer(

179  }
180

182 
183  impl.socket_ = invalid_socket;
184  impl.state_ = 0;
185  impl.cancel_token_.reset();
186 #if defined(ASIO_ENABLE_CANCELIO)
187  impl.safe_cancellation_thread_id_ = 0;
188 #endif // defined(ASIO_ENABLE_CANCELIO)

193 asio::error_code win_iocp_socket_service_base::cancel(
194  win_iocp_socket_service_base::base_implementation_type& impl,
195  asio::error_code& ec)
196 {
197  if (!is_open(impl))
198  {
199  ec = asio::error::bad_descriptor;
200  return ec;
201  }
202 
203  ASIO_HANDLER_OPERATION(("socket", &impl, "cancel"));
204 
205  if (FARPROC cancel_io_ex_ptr = ::GetProcAddress(
206  ::GetModuleHandleA("KERNEL32"), "CancelIoEx"))
207  {
208  // The version of Windows supports cancellation from any thread.
209  typedef BOOL (WINAPI* cancel_io_ex_t)(HANDLE, LPOVERLAPPED);
210  cancel_io_ex_t cancel_io_ex = (cancel_io_ex_t)cancel_io_ex_ptr;
211  socket_type sock = impl.socket_;
212  HANDLE sock_as_handle = reinterpret_cast<HANDLE>(sock);
213  if (!cancel_io_ex(sock_as_handle, 0))
214  {
215  DWORD last_error = ::GetLastError();
216  if (last_error == ERROR_NOT_FOUND)
217  {
218  // ERROR_NOT_FOUND means that there were no operations to be
219  // cancelled. We swallow this error to match the behaviour on other
220  // platforms.
221  ec = asio::error_code();
222  }
223  else
224  {
225  ec = asio::error_code(last_error,
226  asio::error::get_system_category());
227  }
228  }
229  else
230  {
231  ec = asio::error_code();
232  }
233  }
234 #if defined(ASIO_ENABLE_CANCELIO)
235  else if (impl.safe_cancellation_thread_id_ == 0)
236  {
237  // No operations have been started, so there's nothing to cancel.
238  ec = asio::error_code();
239  }
240  else if (impl.safe_cancellation_thread_id_ == ::GetCurrentThreadId())
241  {
242  // Asynchronous operations have been started from the current thread only,
243  // so it is safe to try to cancel them using CancelIo.
244  socket_type sock = impl.socket_;
245  HANDLE sock_as_handle = reinterpret_cast<HANDLE>(sock);
246  if (!::CancelIo(sock_as_handle))
247  {
248  DWORD last_error = ::GetLastError();
249  ec = asio::error_code(last_error,
250  asio::error::get_system_category());
251  }
252  else
253  {
254  ec = asio::error_code();
255  }
256  }
257  else
258  {
259  // Asynchronous operations have been started from more than one thread,
260  // so cancellation is not safe.
261  ec = asio::error::operation_not_supported;
262  }
263 #else // defined(ASIO_ENABLE_CANCELIO)
264  else
265  {
266  // Cancellation is not supported as CancelIo may not be used.
267  ec = asio::error::operation_not_supported;
268  }
269 #endif // defined(ASIO_ENABLE_CANCELIO)
270 
271  // Cancel any operations started via the reactor.
272  if (!ec)
273  {
274  reactor* r = static_cast<reactor*>(
275  interlocked_compare_exchange_pointer(
276  reinterpret_cast<void**>(&reactor_), 0, 0));
277  if (r)
278  r->cancel_ops(impl.socket_, impl.reactor_data_);
279  }
280 
281  return ec;
282 }

284 asio::error_code win_iocp_socket_service_base::do_open(
285  win_iocp_socket_service_base::base_implementation_type& impl,
286  int family, int type, int protocol, asio::error_code& ec)
287 {
288  if (is_open(impl))
289  {
290  ec = asio::error::already_open;
291  return ec;
292  }
293 
294  socket_holder sock(socket_ops::socket(family, type, protocol, ec));
295  if (sock.get() == invalid_socket)
296  return ec;
297 
298  HANDLE sock_as_handle = reinterpret_cast<HANDLE>(sock.get());
299  if (iocp_service_.register_handle(sock_as_handle, ec))
300  return ec;
301 
302  impl.socket_ = sock.release();
303  switch (type)
304  {
305  case SOCK_STREAM: impl.state_ = socket_ops::stream_oriented; break;
306  case SOCK_DGRAM: impl.state_ = socket_ops::datagram_oriented; break;
307  default: impl.state_ = 0; break;
308  }
309  impl.cancel_token_.reset(static_cast<void*>(0), socket_ops::noop_deleter());
310  ec = asio::error_code();
311  return ec;
312 }

341  win_iocp_socket_service_base::base_implementation_type& impl,
342  WSABUF* buffers, std::size_t buffer_count,
343  socket_base::message_flags flags, bool noop, operation* op)
344 {
345  update_cancellation_thread_id(impl);
346  iocp_service_.work_started();
347

546 void win_iocp_socket_service_base::start_connect_op(
547  win_iocp_socket_service_base::base_implementation_type& impl,
548  int family, int type, const socket_addr_type* addr,
549  std::size_t addrlen, win_iocp_socket_connect_op_base* op)
550 {
551  // If ConnectEx is available, use that.
552  if (family == ASIO_OS_DEF(AF_INET)
553  || family == ASIO_OS_DEF(AF_INET6))
554  {
555  if (connect_ex_fn connect_ex = get_connect_ex(impl, type))
556  {
557  union address_union
558  {
559  socket_addr_type base;
560  sockaddr_in4_type v4;
561  sockaddr_in6_type v6;
562  } a;
563 
564  using namespace std; // For memset.
565  memset(&a, 0, sizeof(a));
566  a.base.sa_family = family;
567 
568  socket_ops::bind(impl.socket_, &a.base,
569  family == ASIO_OS_DEF(AF_INET)
570  ? sizeof(a.v4) : sizeof(a.v6), op->ec_);
571  if (op->ec_ && op->ec_ != asio::error::invalid_argument)
572  {
573  iocp_service_.post_immediate_completion(op, false);
574  return;
575  }
576 
577  op->connect_ex_ = true;
578  update_cancellation_thread_id(impl);
579  iocp_service_.work_started();
580 
581  BOOL result = connect_ex(impl.socket_,
582  addr, static_cast<int>(addrlen), 0, 0, 0, op);
583  DWORD last_error = ::WSAGetLastError();
584  if (!result && last_error != WSA_IO_PENDING)
585  iocp_service_.on_completion(op, last_error);
586  else
587  iocp_service_.on_pending(op);
588  return;
589  }
590  }
591 
592  // Otherwise, fall back to a reactor-based implementation.
593  reactor& r = get_reactor();
594  update_cancellation_thread_id(impl);
595 
596  if ((impl.state_ & socket_ops::non_blocking) != 0
597  || socket_ops::set_internal_non_blocking(
598  impl.socket_, impl.state_, true, op->ec_))
599  {
600  if (socket_ops::connect(impl.socket_, addr, addrlen, op->ec_) != 0)
601  {
602  if (op->ec_ == asio::error::in_progress
603  || op->ec_ == asio::error::would_block)
604  {
605  op->ec_ = asio::error_code();
606  r.start_op(reactor::connect_op, impl.socket_,
607  impl.reactor_data_, op, false, false);
608  return;
609  }
610  }
611  }
612 
613  r.post_immediate_completion(op, false);
614 }
