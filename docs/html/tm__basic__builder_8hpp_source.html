135  inline float post_process_feature(const float feature, const float lambda) {
136  //Now convert to the log probability and multiply with the appropriate weight
137  float result = log10(feature) * lambda;
138  LOG_DEBUG << "log10(" << feature << ") * " << lambda << " = " << result << END_LOG;
139  return result;

153  template<bool is_get_weights>
154  inline bool process_features(TextPieceReader weights, size_t & num_features, prob_weight * storage) {
155  //Declare the token
156  TextPieceReader token;
157  //Store the read probability weight
158  size_t idx = 0;
159  //Store the read weight value
160  float feature;
161 
162  LOG_DEBUG << "Reading the features from: " << weights << END_LOG;
163 
164  //Skip the first space
165  weights.get_first_space(token);
166 
167  LOG_DEBUG3 << "TM features to parse: " << weights << END_LOG;
168 
169  //Read the subsequent weights, check that the number of weights is as expected
170  while (weights.get_first_space(token) && (idx < tm_target_entry::NUM_FEATURES)) {
171  //Parse the token into the entry weight
172  ASSERT_CONDITION_THROW(!fast_s_to_f(feature, token.str().c_str()),
173  string("Could not parse the token: ") + token.str());
174 
175  LOG_DEBUG3 << "parsed: " << token << " -> " << feature << END_LOG;
176 
177  //Check the probabilities at the indexes for the bound
178  if (((idx == 0) || (idx == 2)) && (feature < m_params.m_min_tran_prob)) {
179  LOG_DEBUG1 << "The feature[" << idx << "] = " << feature
180  << " < " << m_params.m_min_tran_prob << END_LOG;
181  return false;
182  } else {
183  if (is_get_weights) {
184  //Now convert to the log probability and multiply with the appropriate weight
185  storage[idx] = post_process_feature(feature, m_params.m_lambdas[idx]);
186  }
187  }
188 
189  //Increment the index
190  ++idx;
191  }
192 
193  //Update the number of features
194  num_features = idx;
195 
196  LOG_DEBUG1 << "Got " << num_features << " good features!" << END_LOG;
197 
198  return true;
199  }

235  inline void process_target_entry(tm_source_entry * source_entry, TextPieceReader &rest,
236  size_t & count_ref, size_t & tmp_features_size, prob_weight * tmp_features) {
237  LOG_DEBUG2 << "Got translation line to parse: ___" << rest << "___" << END_LOG;
238 
239  //Declare the target entry storing reader
240  TextPieceReader target;
241 
242  //Skip the first space symbol that follows the delimiter with the source
243  rest.get_first_space(target);
244 
245  //Read the target phrase, it is surrounded by spaces
246  rest.get_first<TM_DELIMITER, TM_DELIMITER_CDTY>(target);
247 
248  //Check that the weights are good and retrieve them if they are
249  if (process_features<true>(rest, tmp_features_size, tmp_features)) {
250  LOG_DEBUG2 << "The target phrase is: " << target << END_LOG;
251  //Add the translation entry to the model
252  string target_str = target.str();
253  trim(target_str);
254  //Compute the target phrase and its uid
255  const phrase_uid target_uid = get_phrase_uid<true>(target_str);
256 
257  //Use the language model to get the target translation word ids
258  m_lm_query.get_word_ids(target, m_tmp_num_words, m_tmp_word_ids);
259 
260  LOG_DEBUG << "The phrase: ___" << target << "__ got "
261  << m_tmp_num_words << " word ids: " <<
262  array_to_string<word_uid>(m_tmp_num_words, m_tmp_word_ids) << END_LOG;
263 
264  //Get the language model weights for the target translation
265  const prob_weight lm_weight = m_lm_query.execute(m_tmp_num_words, m_tmp_word_ids);
266 
267  LOG_DEBUG << "The phrase: ___" << target << "__ lm-weight: " << lm_weight << END_LOG;
268 
269  //Initiate a new target entry
270  source_entry->add_target(target_str, target_uid,
271  tmp_features_size, tmp_features,
272  m_tmp_num_words, m_tmp_word_ids, lm_weight);
273 
274  //Reduce the counter
275  count_ref--;
276 
277  LOG_DEBUG1 << "The source/target (" << source_entry->get_source_uid()
278  << "/" << target_uid << ") counter is " << count_ref << END_LOG;
279  }

290  template<bool count_or_build>
291  inline void parse_tm_file() {
292 
293  //Declare the text piece reader for storing the read line and source phrase
294  TextPieceReader line, source;
295 
296  //Store the source entry
297  tm_source_entry * source_entry = NULL;
298 
299  //Store the cached source string and its uid values
300  string source_str = "";
301  phrase_uid source_uid = UNDEFINED_PHRASE_ID;
302  bool is_good_source = false;
303 
304  //Declare an array of weights for temporary use
305  feature_array tmp_features;
306  size_t tmp_features_size = 0;
307 
308  //Start reading the translation model file line by line
309  while (m_reader.get_first_line(line)) {
310  //Read the source phrase
311  line.get_first<TM_DELIMITER, TM_DELIMITER_CDTY>(source);
312 
313  //Get the current source phrase uid
314  string next_source_str = source.str();
315  trim(next_source_str);
316 
317  LOG_DEBUG << "Got the source phrase: ___" << next_source_str << "___" << END_LOG;
318 
319  if (source_str != next_source_str) {
320  size_t size_val = m_sizes[source_uid];
321  if (count_or_build) {
322  //Remove the previous entry if the count is zero
323  if (size_val == 0) {
324  m_sizes.erase(source_uid);
325  }
326  } else {
327  //Finalize the previous entry if there was one
328  if (is_good_source && (source_uid != UNDEFINED_PHRASE_ID)) {
329  LOG_DEBUG1 << "Finishing a source entry for: ___" << source_str
330  << "___ uid: " << source_uid << ", count: " << size_val << END_LOG;
331 
332  m_model.finalize_entry(source_uid);
333  }
334  }
335 
336  //Store the new source string
337  source_str = next_source_str;
338 
339  //Compute the new source string uid
340  source_uid = get_phrase_uid(source_str);
341 
342  //Get the current value for the new source id
343  size_val = m_sizes[source_uid];
344 
345  LOG_DEBUG1 << "The new source ___" << source_str << "___ id is: "
346  << source_uid << ", count is: " << size_val << END_LOG;
347 
348  if (count_or_build) {
349  //Nothing to be done here when counting entries
350  LOG_DEBUG1 << "Counting the valid translations for " << source_uid << END_LOG;
351  } else {
352  //Check if the new entry is to be entirely skipped
353  is_good_source = (size_val != 0);
354 
355  if (is_good_source) {
356  //Open the new source entry
357  source_entry = m_model.begin_entry(source_uid, size_val);
358 
359  LOG_DEBUG1 << "Starting a new source entry for: ___" << source_str
360  << "___ uid: " << source_uid << ", count: " << size_val << END_LOG;
361  } else {
362  //This source is skipped and if it is the
363  //last one then we do not need to finalize it
364  source_uid = UNDEFINED_PHRASE_ID;
365  }
366  }
367  }
368 
369  //Get the reference to the current size
370  size_t & size_ref = m_sizes[source_uid];
371  if (count_or_build) {
372  //Check that the filter conditions hold
373  if ((size_ref < m_params.m_trans_limit) &&
374  is_good_features(line, tmp_features_size, tmp_features)) {
375  //Increment the count for the given source uid
376  ++size_ref;
377 
378  LOG_DEBUG1 << "The new source " << source_uid << " count is " << size_ref << END_LOG;
379  }
380  } else {
381  //If the source entry is not to be skipped, parse
382  if (is_good_source && (size_ref > 0)) {
383  //Parse the rest of the target entry
384  process_target_entry(source_entry, line, size_ref, tmp_features_size, tmp_features);
385  } else {
386  LOG_DEBUG << "Skipping source-target entry: " << line << END_LOG;
387  }
388  }
389 
390  //Update the progress bar status
391  Logger::update_progress_bar();
392  }
393 
394  if (count_or_build) {
395  //Check if the last entry resulted in zero score, if yes then remove it
396  if (m_sizes[source_uid] == 0) {
397  m_sizes.erase(source_uid);
398  }
399  } else {
400  //Finalize the previous entry if there was one
401  if (source_uid != UNDEFINED_PHRASE_ID) {
402  m_model.finalize_entry(source_uid);
403  }
404 
405  //Finalize the model
406  m_model.finalize();
407  }
408  }

447  inline void add_unk_translation() {
448  //Declare an array of features, zero-valued
449  feature_array unk_features = {};
450

452  << array_to_string<prob_weight>(m_params.m_num_unk_features, m_params.m_unk_features) << END_LOG;
453  LOG_DEBUG << "The UNK lambdas: "
454  << array_to_string<prob_weight>(m_params.m_num_unk_features, m_params.m_lambdas) << END_LOG;
455 
456  //Copy the values of the unk features to the writable array
457  for (size_t idx = 0; idx < m_params.m_num_unk_features; ++idx) {
458  //Now convert to the log probability and multiply with the appropriate weight
459  unk_features[idx] = post_process_feature(m_params.m_unk_features[idx], m_params.m_lambdas[idx]);
460  }
461 
462  //Set the unk features to the model
463  m_model.set_unk_entry(UNKNOWN_WORD_ID, m_params.m_num_unk_features,
464  unk_features, m_lm_query.get_unk_word_prob());
