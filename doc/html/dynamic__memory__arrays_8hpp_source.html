147  inline mem_increase_strategy get_mem_incr_strat(const mem_inc_types_enum stype,
148  const size_t min_mem_inc, const size_t mem_inc_factor) {
149  TCapacityIncFunct inc_func;
150  switch (stype) {
151  case mem_inc_types_enum::CONSTANT:
152  inc_func = [] (const size_t fcap) -> size_t {
153  //Return zero as then the minimum constant increase will be used!
154  return 0;
155  };
156  break;
157  case mem_inc_types_enum::LINEAR:
158  inc_func = [] (const size_t fcap) -> size_t {
159  return fcap;
160  };
161  break;
162  case mem_inc_types_enum::LOG_2:
163  inc_func = [] (const size_t fcap) -> size_t {
164  return fcap * (1 / log(fcap));
165  };
166  break;
167  case mem_inc_types_enum::LOG_10:
168  inc_func = [] (const size_t fcap) -> size_t {
169  return fcap * (1 / log10(fcap));
170  };
171  break;
172  default:
173  stringstream msg;
174  msg << "Unrecognized memory allocation strategy: " << stype;
175  throw uva_exception(msg.str());
176  }
177 
178  //return the result object
179  return mem_increase_strategy(stype, inc_func, min_mem_inc, mem_inc_factor);
180  };

409  template<bool IS_INC = true >
410  static inline void reallocate(ELEMENT_TYPE_PTR & m_ptr, IDX_DATA_TYPE & m_capacity, size_t new_capacity) {
411  LOG_DEBUG2 << "The new capacity is " << SSTR(new_capacity)
412  << ", the old capacity was " << SSTR(m_capacity)
413  << ", ptr: " << SSTR((void*) m_ptr) << ", elem size: "
414  << SSTR(sizeof (ELEMENT_TYPE)) << END_LOG;
415 
416  //Check that there is still space to be allocated
417  ASSERT_CONDITION_THROW(IS_INC && (m_capacity == MAX_SIZE_TYPE_VALUE),
418  string("Unable to increase the capacity, reached the maximum of ") +
419  std::to_string(MAX_SIZE_TYPE_VALUE) + " allowed by the data type!");
420 
421  //Reallocate memory, potentially we get a new pointer!
422  new_capacity = min(new_capacity, MAX_SIZE_TYPE_VALUE);
423  ASSERT_SANITY_THROW((m_capacity == new_capacity), "The new capacity is equal to the old one!");
424  m_ptr = (ELEMENT_TYPE_PTR) realloc(m_ptr, new_capacity * sizeof (ELEMENT_TYPE));
425 
426  LOG_DEBUG2 << "Memory is reallocated ptr: " << SSTR(m_ptr) << END_LOG;
427 
428  //Initialize the newly allocated memory
429  if (IS_INC) {
430  for (IDX_DATA_TYPE idx = m_capacity; idx < new_capacity; ++idx) {
431  new(static_cast<void *> (&m_ptr[idx])) ELEMENT_TYPE();
432  LOG_DEBUG3 << "Creating a new element [" << SSTR(idx)
433  << "]: " << SSTR((void *) &m_ptr[idx]) << END_LOG;
434  }
435  }
436 
437  //Set the new capacity in
438  m_capacity = new_capacity;
439 
440  LOG_DEBUG2 << "The end capacity is " << SSTR(m_capacity)
441  << ", ptr: " << SSTR(m_ptr) << END_LOG;
442 
443  //Do the null pointer check if sanity
444  ASSERT_SANITY_THROW((m_ptr == NULL),
445  string("Ran out of memory when trying to allocate ") +
446  std::to_string(new_capacity) + " data elements for a word_id");
447  }
448  };
