35 bool rfc2818_verification::operator()(
36  bool preverified, verify_context& ctx) const
37 {
38  using namespace std; // For memcmp.
39 
40  // Don't bother looking at certificates that have failed pre-verification.
41  if (!preverified)
42  return false;
43 
44  // We're only interested in checking the certificate at the end of the chain.
45  int depth = X509_STORE_CTX_get_error_depth(ctx.native_handle());
46  if (depth > 0)
47  return true;
48 
49  // Try converting the host name to an address. If it is an address then we
50  // need to look for an IP address in the certificate rather than a host name.
51  asio::error_code ec;
52  ip::address address = ip::address::from_string(host_, ec);
53  bool is_address = !ec;
54 
55  X509* cert = X509_STORE_CTX_get_current_cert(ctx.native_handle());
56 
57  // Go through the alternate names in the certificate looking for matching DNS
58  // or IP address entries.
59  GENERAL_NAMES* gens = static_cast<GENERAL_NAMES*>(
60  X509_get_ext_d2i(cert, NID_subject_alt_name, 0, 0));
61  for (int i = 0; i < sk_GENERAL_NAME_num(gens); ++i)
62  {
63  GENERAL_NAME* gen = sk_GENERAL_NAME_value(gens, i);
64  if (gen->type == GEN_DNS && !is_address)
65  {
66  ASN1_IA5STRING* domain = gen->d.dNSName;
67  if (domain->type == V_ASN1_IA5STRING && domain->data && domain->length)
68  {
69  const char* pattern = reinterpret_cast<const char*>(domain->data);
70  std::size_t pattern_length = domain->length;
71  if (match_pattern(pattern, pattern_length, host_.c_str()))
72  {
73  GENERAL_NAMES_free(gens);
74  return true;
75  }
76  }
77  }
78  else if (gen->type == GEN_IPADD && is_address)
79  {
80  ASN1_OCTET_STRING* ip_address = gen->d.iPAddress;
81  if (ip_address->type == V_ASN1_OCTET_STRING && ip_address->data)
82  {
83  if (address.is_v4() && ip_address->length == 4)
84  {
85  ip::address_v4::bytes_type bytes = address.to_v4().to_bytes();
86  if (memcmp(bytes.data(), ip_address->data, 4) == 0)
87  {
88  GENERAL_NAMES_free(gens);
89  return true;
90  }
91  }
92  else if (address.is_v6() && ip_address->length == 16)
93  {
94  ip::address_v6::bytes_type bytes = address.to_v6().to_bytes();
95  if (memcmp(bytes.data(), ip_address->data, 16) == 0)
96  {
97  GENERAL_NAMES_free(gens);
98  return true;
99  }
100  }
101  }
102  }
103  }
104  GENERAL_NAMES_free(gens);
105 
106  // No match in the alternate names, so try the common names. We should only
107  // use the "most specific" common name, which is the last one in the list.
108  X509_NAME* name = X509_get_subject_name(cert);
109  int i = -1;
110  ASN1_STRING* common_name = 0;
111  while ((i = X509_NAME_get_index_by_NID(name, NID_commonName, i)) >= 0)
112  {
113  X509_NAME_ENTRY* name_entry = X509_NAME_get_entry(name, i);
114  common_name = X509_NAME_ENTRY_get_data(name_entry);
115  }
116  if (common_name && common_name->data && common_name->length)
117  {
118  const char* pattern = reinterpret_cast<const char*>(common_name->data);
119  std::size_t pattern_length = common_name->length;
120  if (match_pattern(pattern, pattern_length, host_.c_str()))
121  return true;
122  }
123 
124  return false;
125 }

127 bool rfc2818_verification::match_pattern(const char* pattern,
128  std::size_t pattern_length, const char* host)
129 {
130  using namespace std; // For tolower.
131 
132  const char* p = pattern;
133  const char* p_end = p + pattern_length;
134  const char* h = host;
135 
136  while (p != p_end && *h)
137  {
138  if (*p == '*')
139  {
140  ++p;
141  while (*h && *h != '.')
142  if (match_pattern(p, p_end - p, h++))
143  return true;
144  }
145  else if (tolower(*p) == tolower(*h))
146  {
147  ++p;
148  ++h;
149  }
150  else
151  {
152  return false;
153  }
154  }
155 
156  return p == p_end && !*h;
157 }
