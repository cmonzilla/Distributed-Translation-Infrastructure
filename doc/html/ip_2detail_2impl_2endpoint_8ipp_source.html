138 asio::ip::address endpoint::address() const
139 {
140  using namespace std; // For memcpy.
141  if (is_v4())
142  {
143  return asio::ip::address_v4(
144  asio::detail::socket_ops::network_to_host_long(
145  data_.v4.sin_addr.s_addr));
146  }
147  else
148  {
149  asio::ip::address_v6::bytes_type bytes;
150 #if defined(ASIO_HAS_STD_ARRAY)
151  memcpy(bytes.data(), data_.v6.sin6_addr.s6_addr, 16);
152 #else // defined(ASIO_HAS_STD_ARRAY)
153  memcpy(bytes.elems, data_.v6.sin6_addr.s6_addr, 16);
154 #endif // defined(ASIO_HAS_STD_ARRAY)
155  return asio::ip::address_v6(bytes, data_.v6.sin6_scope_id);
156  }
