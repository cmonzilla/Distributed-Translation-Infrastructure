42  hEvent = ::CreateEvent(0, TRUE, FALSE, 0);

46  // bit of this event prevents our synchronous writes from being treated

226 asio::error_code win_iocp_handle_service::cancel(
227  win_iocp_handle_service::implementation_type& impl,
228  asio::error_code& ec)
229 {
230  if (!is_open(impl))
231  {
232  ec = asio::error::bad_descriptor;
233  return ec;
234  }
235 
236  ASIO_HANDLER_OPERATION(("handle", &impl, "cancel"));
237 
238  if (FARPROC cancel_io_ex_ptr = ::GetProcAddress(
239  ::GetModuleHandleA("KERNEL32"), "CancelIoEx"))
240  {
241  // The version of Windows supports cancellation from any thread.
242  typedef BOOL (WINAPI* cancel_io_ex_t)(HANDLE, LPOVERLAPPED);
243  cancel_io_ex_t cancel_io_ex = (cancel_io_ex_t)cancel_io_ex_ptr;
244  if (!cancel_io_ex(impl.handle_, 0))
245  {
246  DWORD last_error = ::GetLastError();
247  if (last_error == ERROR_NOT_FOUND)
248  {
249  // ERROR_NOT_FOUND means that there were no operations to be
250  // cancelled. We swallow this error to match the behaviour on other
251  // platforms.
252  ec = asio::error_code();
253  }
254  else
255  {
256  ec = asio::error_code(last_error,
257  asio::error::get_system_category());
258  }
259  }
260  else
261  {
262  ec = asio::error_code();
263  }
264  }
265  else if (impl.safe_cancellation_thread_id_ == 0)
266  {
267  // No operations have been started, so there's nothing to cancel.
268  ec = asio::error_code();
269  }
270  else if (impl.safe_cancellation_thread_id_ == ::GetCurrentThreadId())
271  {
272  // Asynchronous operations have been started from the current thread only,
273  // so it is safe to try to cancel them using CancelIo.
274  if (!::CancelIo(impl.handle_))
275  {
276  DWORD last_error = ::GetLastError();
277  ec = asio::error_code(last_error,
278  asio::error::get_system_category());
279  }
280  else
281  {
282  ec = asio::error_code();
283  }
284  }
285  else
286  {
287  // Asynchronous operations have been started from more than one thread,
288  // so cancellation is not safe.
289  ec = asio::error::operation_not_supported;
290  }
291 
292  return ec;
293 }

295 size_t win_iocp_handle_service::do_write(
296  win_iocp_handle_service::implementation_type& impl, uint64_t offset,
297  const asio::const_buffer& buffer, asio::error_code& ec)
298 {
299  if (!is_open(impl))
300  {
301  ec = asio::error::bad_descriptor;
302  return 0;
303  }
304 
305  // A request to write 0 bytes on a handle is a no-op.
306  if (asio::buffer_size(buffer) == 0)
307  {
308  ec = asio::error_code();
309  return 0;
310  }
311 
312  overlapped_wrapper overlapped(ec);
313  if (ec)
314  {
315  return 0;
316  }
317 
318  // Write the data.
319  overlapped.Offset = offset & 0xFFFFFFFF;
320  overlapped.OffsetHigh = (offset >> 32) & 0xFFFFFFFF;
321  BOOL ok = ::WriteFile(impl.handle_,
322  asio::buffer_cast<LPCVOID>(buffer),
323  static_cast<DWORD>(asio::buffer_size(buffer)), 0, &overlapped);
324  if (!ok)
325  {
326  DWORD last_error = ::GetLastError();
327  if (last_error != ERROR_IO_PENDING)
328  {
329  ec = asio::error_code(last_error,
330  asio::error::get_system_category());
331  return 0;
332  }
333  }
334 
335  // Wait for the operation to complete.
336  DWORD bytes_transferred = 0;
337  ok = ::GetOverlappedResult(impl.handle_,
338  &overlapped, &bytes_transferred, TRUE);
339  if (!ok)
340  {
341  DWORD last_error = ::GetLastError();
342  ec = asio::error_code(last_error,
343  asio::error::get_system_category());
344  return 0;
345  }
346 
347  ec = asio::error_code();
348  return bytes_transferred;
349 }

377  if (!ok && last_error != ERROR_IO_PENDING

389 size_t win_iocp_handle_service::do_read(
390  win_iocp_handle_service::implementation_type& impl, uint64_t offset,
391  const asio::mutable_buffer& buffer, asio::error_code& ec)
392 {
393  if (!is_open(impl))
394  {
395  ec = asio::error::bad_descriptor;
396  return 0;
397  }
398 
399  // A request to read 0 bytes on a stream handle is a no-op.
400  if (asio::buffer_size(buffer) == 0)
401  {
402  ec = asio::error_code();
403  return 0;
404  }
405 
406  overlapped_wrapper overlapped(ec);
407  if (ec)
408  {
409  return 0;
410  }
411 
412  // Read some data.
413  overlapped.Offset = offset & 0xFFFFFFFF;
414  overlapped.OffsetHigh = (offset >> 32) & 0xFFFFFFFF;
415  BOOL ok = ::ReadFile(impl.handle_,
416  asio::buffer_cast<LPVOID>(buffer),
417  static_cast<DWORD>(asio::buffer_size(buffer)), 0, &overlapped);
418  if (!ok)
419  {
420  DWORD last_error = ::GetLastError();
421  if (last_error != ERROR_IO_PENDING && last_error != ERROR_MORE_DATA)
422  {
423  if (last_error == ERROR_HANDLE_EOF)
424  {
425  ec = asio::error::eof;
426  }
427  else
428  {
429  ec = asio::error_code(last_error,
430  asio::error::get_system_category());
431  }
432  return 0;
433  }
434  }
435 
436  // Wait for the operation to complete.
437  DWORD bytes_transferred = 0;
438  ok = ::GetOverlappedResult(impl.handle_,
439  &overlapped, &bytes_transferred, TRUE);
440  if (!ok)
441  {
442  DWORD last_error = ::GetLastError();
443  if (last_error == ERROR_HANDLE_EOF)
444  {
445  ec = asio::error::eof;
446  }
447  else
448  {
449  ec = asio::error_code(last_error,
450  asio::error::get_system_category());
451  }
452  return (last_error == ERROR_MORE_DATA) ? bytes_transferred : 0;
453  }
454 
455  ec = asio::error_code();
456  return bytes_transferred;
457 }
