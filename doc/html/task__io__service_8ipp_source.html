33 struct task_io_service::task_cleanup
34 {
35  ~task_cleanup()
36  {
37  if (this_thread_->private_outstanding_work > 0)
38  {
39  asio::detail::increment(
40  task_io_service_->outstanding_work_,

51  }

59 {
60  ~work_cleanup()

62  if (this_thread_->private_outstanding_work > 1)
63  {
64  asio::detail::increment(
65  task_io_service_->outstanding_work_,
66  this_thread_->private_outstanding_work - 1);

81  }

327 std::size_t task_io_service::do_run_one(mutex::scoped_lock& lock,
328  task_io_service::thread_info& this_thread,
329  const asio::error_code& ec)
330 {
331  while (!stopped_)
332  {
333  if (!op_queue_.empty())
334  {
335  // Prepare to execute first handler from queue.
336  operation* o = op_queue_.front();
337  op_queue_.pop();
338  bool more_handlers = (!op_queue_.empty());
339 
340  if (o == &task_operation_)
341  {
342  task_interrupted_ = more_handlers;
343 
344  if (more_handlers && !one_thread_)
345  wakeup_event_.unlock_and_signal_one(lock);
346  else
347  lock.unlock();
348 
349  task_cleanup on_exit = { this, &lock, &this_thread };
350  (void)on_exit;
351 
352  // Run the task. May throw an exception. Only block if the operation
353  // queue is empty and we're not polling, otherwise we want to return
354  // as soon as possible.
355  task_->run(!more_handlers, this_thread.private_op_queue);
356  }
357  else
358  {
359  std::size_t task_result = o->task_result_;
360 
361  if (more_handlers && !one_thread_)
362  wake_one_thread_and_unlock(lock);
363  else
364  lock.unlock();
365 
366  // Ensure the count of outstanding work is decremented on block exit.
367  work_cleanup on_exit = { this, &lock, &this_thread };
368  (void)on_exit;
369 
370  // Complete the operation. May throw an exception. Deletes the object.
371  o->complete(*this, ec, task_result);
372 
373  return 1;
374  }
375  }
376  else
377  {
378  wakeup_event_.clear(lock);
379  wakeup_event_.wait(lock);
380  }
381  }
382 
383  return 0;
384 }

386 std::size_t task_io_service::do_poll_one(mutex::scoped_lock& lock,
387  task_io_service::thread_info& this_thread,
388  const asio::error_code& ec)
389 {
390  if (stopped_)
391  return 0;
392 
393  operation* o = op_queue_.front();
394  if (o == &task_operation_)
395  {
396  op_queue_.pop();
397  lock.unlock();
398 
399  {
400  task_cleanup c = { this, &lock, &this_thread };
401  (void)c;
402 
403  // Run the task. May throw an exception. Only block if the operation
404  // queue is empty and we're not polling, otherwise we want to return
405  // as soon as possible.
406  task_->run(false, this_thread.private_op_queue);
407  }
408 
409  o = op_queue_.front();
410  if (o == &task_operation_)
411  {
412  wakeup_event_.maybe_unlock_and_signal_one(lock);
413  return 0;
414  }
415  }
416 
417  if (o == 0)
418  return 0;
419 
420  op_queue_.pop();
421  bool more_handlers = (!op_queue_.empty());
422 
423  std::size_t task_result = o->task_result_;
424 
425  if (more_handlers && !one_thread_)
426  wake_one_thread_and_unlock(lock);
427  else
428  lock.unlock();
429 
430  // Ensure the count of outstanding work is decremented on block exit.
431  work_cleanup on_exit = { this, &lock, &this_thread };
432  (void)on_exit;
433 
434  // Complete the operation. May throw an exception. Deletes the object.
435  o->complete(*this, ec, task_result);
436 
437  return 1;
438 }
