39 address_v6::address_v6(const address_v6::bytes_type& bytes,
40  unsigned long scope)
41  : scope_id_(scope)
42 {
43 #if UCHAR_MAX > 0xFF

48  std::out_of_range ex("address_v6 from bytes_type");

50  }
51  }
52 #endif // UCHAR_MAX > 0xFF
53 
54  using namespace std; // For memcpy.
55  memcpy(addr_.s6_addr, bytes.data(), 16);
56 }
