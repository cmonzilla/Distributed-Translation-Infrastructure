91  void de_serialize(const string & message) {
92  //Initialize the reader
93  text_piece_reader reader(message.c_str(), message.length());
94  LOG_DEBUG1 << "De-serializing request message: '" << reader.str() << "'" << END_LOG;
95 
96  //The text will contain the read text from the reader
97  text_piece_reader text;
98 
99  //First get the job id
100  if (reader.get_first<HEADER_DELIMITER>(text)) {
101  m_job_id = stoi(text.str());
102  //Second get the source language string
103  if (reader.get_first<HEADER_DELIMITER>(text)) {
104  m_source_lang = text.str();
105  //Third get the target language string
106  if (reader.get_first<NEW_LINE_HEADER_ENDING>(text)) {
107  m_target_lang = text.str();
108 
109  //Now the rest is the text to be translated.
110  m_text = reader.get_rest_str();
111 
112  LOG_DEBUG << "\nm_job_id = " << m_job_id << ", m_source_lang = "
113  << m_source_lang << ", m_target_lang = " << m_target_lang
114  << ", m_text = \n" << m_text << END_LOG;
115  } else {
116  THROW_EXCEPTION(string("Could not find result code in the job reply header!"));
117  }
118  } else {
119  THROW_EXCEPTION(string("Could not find result code in the job reply header!"));
120  }
121  } else {
122  THROW_EXCEPTION(string("Could not find job_id in the job reply header!"));
123  }
124  }
