194  static inline void copy_one_bit(const uint8_t source, const uint8_t sbit_idx, uint8_t & target, const uint8_t tbit_idx) {
195  LOG_DEBUG4 << "Source bits: " << bitset<NUM_BITS_IN_UINT_8>(source) << END_LOG;
196  LOG_DEBUG4 << "Target bits: " << bitset<NUM_BITS_IN_UINT_8>(target) << END_LOG;

199  LOG_DEBUG4 << "Copying bit " << SSTR((uint32_t) sbit_idx) << ", it is ON" << END_LOG;

204  }

266  static inline void copy_single_bits(const uint8_t * p_source, uint32_t from_pos_bit,
267  uint8_t * p_target, uint32_t to_pos_bit, const uint8_t num_bits) {
268  //Copy bits one by one
269  const uint8_t end_pos_bit = (from_pos_bit + num_bits);
270 
271  LOG_DEBUG4 << "from_pos_bit = " << (uint32_t) from_pos_bit
272  << ", end_pos_bit = " << (uint32_t) end_pos_bit << END_LOG;
273 
274  while (from_pos_bit < end_pos_bit) {
275  const uint8_t from_byte = BYTE_IDX(from_pos_bit);
276  const uint8_t from_bit = REMAINING_BIT_IDX(from_pos_bit);
277  const uint8_t to_byte = BYTE_IDX(to_pos_bit);
278  const uint8_t to_bit = REMAINING_BIT_IDX(to_pos_bit);
279 
280  //Track the number of copied bits
281  uint8_t num_copied_bits = 0;
282  if (from_bit > to_bit) {
283  //Need to shift to the left
284  num_copied_bits = (NUM_BITS_IN_UINT_8 - from_bit);
285  p_target[to_byte] |= ((p_source[from_byte] & clean_prefix_bits_array[from_bit]) << (from_bit - to_bit));
286  } else {
287  if (from_bit == to_bit) {
288  //No need to shift
289  num_copied_bits = (NUM_BITS_IN_UINT_8 - from_bit);
290  p_target[to_byte] |= (p_source[from_byte] & clean_prefix_bits_array[from_bit]);
291  } else {
292  //Need to shift to the right
293  num_copied_bits = (NUM_BITS_IN_UINT_8 - to_bit);
294  p_target[to_byte] |= ((p_source[from_byte] & clean_prefix_bits_array[from_bit]) >> (to_bit - from_bit));
295  }
296  }
297  //Increment the positions
298  from_pos_bit += num_copied_bits;
299  to_pos_bit += num_copied_bits;
300  }
301  };

311  static inline void copy_all_bits(const uint8_t * p_source, uint32_t from_pos_bit,
312  uint8_t * p_target, uint32_t to_pos_bit, const uint8_t num_bits) {
313 
314  //Depending on whether the bits are byte aligned, we have two copying strategies.
315  if ((REMAINING_BIT_IDX(from_pos_bit) > 0) || (REMAINING_BIT_IDX(to_pos_bit) > 0)) {
316  //If there is no byte alignment: Copy bits one by one
317  copy_single_bits(p_source, from_pos_bit, p_target, to_pos_bit, num_bits);
318  } else {
319  //If there is byte alignment then we can fast copy
320  //some whole bytes and then the remaining bits.
321 
322  //1. Copy as many whole bytes as possible
323  const uint8_t num_full_bytes = NUM_FULL_BYTES(num_bits);
324  if (num_full_bytes > 0) {
325  const uint8_t from_pos_byte = BYTE_IDX(from_pos_bit);
326  const uint8_t to_pos_byte = BYTE_IDX(to_pos_bit);
327  memcpy(p_target + to_pos_byte, p_source + from_pos_byte, num_full_bytes);
328  }
329 
330  //2. Copy the remaining bits one by one
331  const uint8_t num_rem_bits = NUM_BITS_REMAINDER(num_bits);
332  if (num_rem_bits > 0) {
333  //Compute the number of remaining bytes to copy
334  const uint8_t num_copied_bits = BYTES_TO_BITS(num_full_bytes);
335  //Copy the remaining bytes
336  copy_single_bits(p_source, from_pos_bit + num_copied_bits,
337  p_target, to_pos_bit + num_copied_bits,
338  num_rem_bits);
339  }
340  }
341  };
