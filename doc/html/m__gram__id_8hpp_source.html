213  static inline int compare(const uint8_t id_len_bytes,
214  const TM_Gram_Id_Value_Ptr & m_p_gram_id_one,
215  const TM_Gram_Id_Value_Ptr & m_p_gram_id_two) {
216  //Compare with the fast system function. Note that, the
217  //First element of an id is its type, and all of the type
218  //lengths for the m-gram of a given level is the same.
219  //The id type defines the lengths of the rest of the id.
220  //Thus, if the m-gram level is equal then we only get to
221  //compare the bytes after the id type if the id types are equal.
222  LOG_DEBUG2 << "Comparing: " << bytes_to_bit_string(m_p_gram_id_one, id_len_bytes) << END_LOG;

224  return memcmp(m_p_gram_id_one, m_p_gram_id_two, id_len_bytes);
225  }

258  template<phrase_length CURR_LEVEL>
259  static inline const uint8_t & gram_id_type_2_byte_len(uint32_t id_type) {
260  LOG_DEBUG3 << "Computing the " << SSTR(CURR_LEVEL) << "-gram id len in bytes" << END_LOG;
261 
262  //Depending on the level return the pre-computed value
263  switch (CURR_LEVEL) {
264  case M_GRAM_LEVEL_5:
265  return LEVEL_5_GRAM_TO_BYTE_LEN[id_type];
266  case M_GRAM_LEVEL_4:
267  return LEVEL_4_GRAM_TO_BYTE_LEN[id_type];
268  case M_GRAM_LEVEL_3:
269  return LEVEL_3_GRAM_TO_BYTE_LEN[id_type];
270  case M_GRAM_LEVEL_2:
271  return LEVEL_2_GRAM_TO_BYTE_LEN[id_type];
272  case M_GRAM_LEVEL_6:
273  return LEVEL_6_GRAM_TO_BYTE_LEN[id_type];
274  default:
275  THROW_EXCEPTION(string("Unsupported m-gram level: ") + std::to_string(CURR_LEVEL) +
276  string(", must be within [") + std::to_string(M_GRAM_LEVEL_2) + string(", ") +
277  std::to_string(M_GRAM_LEVEL_6) + string("], insufficient data!"));

288  static inline const uint32_t & gram_id_byte_len_2_type(const phrase_length gram_level, uint8_t * len_bytes) {
289  LOG_DEBUG3 << "Computing the " << SSTR(gram_level) << "-gram id type" << END_LOG;
290 
291  //Depending on the level return the pre-computed value
292  switch (gram_level) {

295  case M_GRAM_LEVEL_4:

297  case M_GRAM_LEVEL_3:
298  return LEVEL_3_GRAM_TO_TYPE_LEN[len_bytes[0]][len_bytes[1]][len_bytes[2]];
299  case M_GRAM_LEVEL_2:
300  return LEVEL_2_GRAM_TO_TYPE_LEN[len_bytes[0]][len_bytes[1]];
301  case M_GRAM_LEVEL_6:
302  return LEVEL_6_GRAM_TO_TYPE_LEN[len_bytes[0]][len_bytes[1]][len_bytes[2]][len_bytes[3]][len_bytes[4]][len_bytes[5]];
303  default:
304  THROW_EXCEPTION(string("Unsupported m-gram level: ") + std::to_string(gram_level) +

317  template<phrase_length CURR_LEVEL>
318  static inline bool is_less_m_grams_id(const uint8_t id_type_len_bytes,
319  const TM_Gram_Id_Value_Ptr & one, const TM_Gram_Id_Value_Ptr & two) {
320  //First compare the types of the id
321  int result = compare(id_type_len_bytes, one, two);
322 
323  //If the id types are equal now we need to compare the keys until the end
324  if (result == 0) {
325  //Extract the id type from the id
326  uint32_t type_one = 0;
327  copy_begin_bytes_to_end(one, id_type_len_bytes, type_one);
328 
329  //Compute the length of the remainder of the m-gram id
330  const uint8_t & id_len_bytes = gram_id_type_2_byte_len<CURR_LEVEL>(type_one);
331 
332  //Compare the remainders of the m-gram ids
333  result = memcmp(one + id_type_len_bytes, two + id_type_len_bytes, id_len_bytes);
334  }
335 
336  //The id types are not equal
337  return (result < 0);
338  }
