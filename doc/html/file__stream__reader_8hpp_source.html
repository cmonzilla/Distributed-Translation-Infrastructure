94  //First read the line from the file

99  LOG_ERROR << "Error while reading the new line in the file!" << END_LOG;
100  //If there was failure during reading the return a failed flag
101  return false;
102  } else {

104  << SSTR(strlen(m_curr_line)) << END_LOG;
105

107  out.set(m_curr_line, strlen(m_curr_line));

110  return true;

114  //This is the end of file or something...
