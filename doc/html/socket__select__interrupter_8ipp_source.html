43 void socket_select_interrupter::open_descriptors()
44 {
45  asio::error_code ec;
46  socket_holder acceptor(socket_ops::socket(
47  AF_INET, SOCK_STREAM, IPPROTO_TCP, ec));
48  if (acceptor.get() == invalid_socket)
49  asio::detail::throw_error(ec, "socket_select_interrupter");
50 
51  int opt = 1;
52  socket_ops::state_type acceptor_state = 0;
53  socket_ops::setsockopt(acceptor.get(), acceptor_state,
54  SOL_SOCKET, SO_REUSEADDR, &opt, sizeof(opt), ec);
55 
56  using namespace std; // For memset.
57  sockaddr_in4_type addr;
58  std::size_t addr_len = sizeof(addr);
59  memset(&addr, 0, sizeof(addr));
60  addr.sin_family = AF_INET;
61  addr.sin_addr.s_addr = socket_ops::host_to_network_long(INADDR_LOOPBACK);
62  addr.sin_port = 0;
63  if (socket_ops::bind(acceptor.get(), (const socket_addr_type*)&addr,
64  addr_len, ec) == socket_error_retval)
65  asio::detail::throw_error(ec, "socket_select_interrupter");
66 
67  if (socket_ops::getsockname(acceptor.get(), (socket_addr_type*)&addr,
68  &addr_len, ec) == socket_error_retval)
69  asio::detail::throw_error(ec, "socket_select_interrupter");
70 
71  // Some broken firewalls on Windows will intermittently cause getsockname to
72  // return 0.0.0.0 when the socket is actually bound to 127.0.0.1. We
73  // explicitly specify the target address here to work around this problem.
74  addr.sin_addr.s_addr = socket_ops::host_to_network_long(INADDR_LOOPBACK);
75 
76  if (socket_ops::listen(acceptor.get(),
77  SOMAXCONN, ec) == socket_error_retval)
78  asio::detail::throw_error(ec, "socket_select_interrupter");
79 
80  socket_holder client(socket_ops::socket(
81  AF_INET, SOCK_STREAM, IPPROTO_TCP, ec));
82  if (client.get() == invalid_socket)
83  asio::detail::throw_error(ec, "socket_select_interrupter");
84 
85  if (socket_ops::connect(client.get(), (const socket_addr_type*)&addr,
86  addr_len, ec) == socket_error_retval)
87  asio::detail::throw_error(ec, "socket_select_interrupter");
88 
89  socket_holder server(socket_ops::accept(acceptor.get(), 0, 0, ec));
90  if (server.get() == invalid_socket)
91  asio::detail::throw_error(ec, "socket_select_interrupter");
92 
93  ioctl_arg_type non_blocking = 1;
94  socket_ops::state_type client_state = 0;
95  if (socket_ops::ioctl(client.get(), client_state,
96  FIONBIO, &non_blocking, ec))
97  asio::detail::throw_error(ec, "socket_select_interrupter");
98 
99  opt = 1;
100  socket_ops::setsockopt(client.get(), client_state,
101  IPPROTO_TCP, TCP_NODELAY, &opt, sizeof(opt), ec);
102 
103  non_blocking = 1;
104  socket_ops::state_type server_state = 0;
105  if (socket_ops::ioctl(server.get(), server_state,
106  FIONBIO, &non_blocking, ec))
107  asio::detail::throw_error(ec, "socket_select_interrupter");
108 
109  opt = 1;
110  socket_ops::setsockopt(server.get(), server_state,
111  IPPROTO_TCP, TCP_NODELAY, &opt, sizeof(opt), ec);
112 
113  read_descriptor_ = server.release();
114  write_descriptor_ = client.release();
115 }

123 {

151 bool socket_select_interrupter::reset()
152 {
153  char data[1024];
154  socket_ops::buf b;
155  socket_ops::init_buf(b, data, sizeof(data));
156  asio::error_code ec;
157  int bytes_read = socket_ops::recv(read_descriptor_, &b, 1, 0, ec);
158  bool was_interrupted = (bytes_read > 0);
159  while (bytes_read == sizeof(data))
160  bytes_read = socket_ops::recv(read_descriptor_, &b, 1, 0, ec);
161  return was_interrupted;
162 }
