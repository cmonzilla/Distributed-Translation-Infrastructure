210  template<const char delim, const uint8_t delim_len = 1 >
211  inline bool get_first(text_piece_reader& out) {
212  //If there is no remaining length or the pointer is NULL then return false
213  if (!m_rest_len || (m_rest_ptr == NULL)) {
214  return false;
215  } else {
216  //The next piece begins where we stopped
217  const char * out_m_begin_ptr = m_rest_ptr;
218 
219  LOG_DEBUG3 << SSTR(static_cast<const void *> (out_m_begin_ptr)) << END_LOG;
220 
221  //Find the sub-sequence
222  const char * char_ptr = find_first_subseq<delim, delim_len>();
223 
224  LOG_DEBUG3 << "Forward searching for the character got: "
225  << SSTR(static_cast<const void *> (char_ptr)) << END_LOG;
226 
227  //The found piece length is first zero
228  size_t out_m_len = 0;
229 
230  //Check if we found something
231  if (char_ptr != NULL) {
232  //Compute the length
233  size_t found_piece_length = char_ptr - m_rest_ptr;
234 
235  LOG_DEBUG3 << "The substring length is " << SSTR(found_piece_length) << END_LOG;
236 
237  //Store the pointer to the remaining text piece
238  m_rest_ptr = char_ptr + 1;
239  //Store the remaining length
240  m_rest_len -= (found_piece_length + 1);
241 
242  LOG_DEBUG3 << "Resetting m_rest_ptr = "
243  << SSTR(static_cast<const void *> (m_rest_ptr))
244  << ", m_rest_len = " << m_rest_len << END_LOG;
245 
246  //Set the resulting length, note that the delimiter can be longer than one character
247  out_m_len = found_piece_length - (delim_len - 1);
248 
249  //If we are looking for end of line, for Windows-format strings, remove the '\r' as well
250  if ((delim == '\n') && found_piece_length && (out_m_begin_ptr[found_piece_length - 1]) == '\r') {
251  out_m_len--;
252  LOG_DEBUG3 << "A \\\\r detected, resetting out_m_len = " << out_m_len << END_LOG;
253  }
254  } else {
255  //If the pointer is not found then the length if the entire remaining length
256  out_m_len = m_rest_len;
257 
258  //If the remaining length is zero then return false as there is nothing to return
259  if (!m_rest_len) {
260  return false;
261  } else {
262  //If there was something left then now we read it all.
263  //Set the remaining length to zero, as there is nothing left.
264  m_rest_len = 0;
265  }
266  }
267 
268  //Set the return data
269  out.set(out_m_begin_ptr, out_m_len);
270 
271  //Return true as we successfully found a delimited substring
272  return true;
273  }
274  }

286  template<const char delim, const uint8_t delim_card = 1 >
287  inline bool get_last(text_piece_reader& out) {
288  //If there is no remaining length or the pointer is NULL then return false
289  if (!m_rest_len || (m_rest_ptr == NULL)) {
290  return false;
291  } else {
292  //Perform some sanity checks for what we are searching
293  ASSERT_SANITY_THROW((delim_card != 1), string("Unsupported delimiter cardinality: ") + to_string(delim_card));
294  ASSERT_SANITY_THROW((delim == '\n') || (delim == '\r'), string("Do not support backwards searching for the end of line!"));
295 
296  LOG_DEBUG4 << "Start searching for a new delimiter, m_rest_ptr: "
297  << SSTR(static_cast<const void *> (m_rest_ptr))
298  << ", m_rest_len: " << m_rest_len << END_LOG;
299 
300  //Search for the next new delimiter from the end
301  const char * out_m_begin_ptr = static_cast<const char *> (memrchr(m_rest_ptr, delim, m_rest_len));
302 
303  LOG_DEBUG4 << "Backward searching for the character got: "
304  << SSTR(static_cast<const void *> (out_m_begin_ptr)) << END_LOG;
305 
306  //Check if we found something
307  if (out_m_begin_ptr != NULL) {
308  //We shall return the substring from the beginning of
309  //the pointer until but excluding the found element
310  const size_t out_m_len = (out_m_begin_ptr - m_rest_ptr);
311  out.set(m_rest_ptr, out_m_len);
312 
313  LOG_DEBUG4 << "The delimiter '" << delim << "' was found, out_m_len = " << out_m_len
314  << ", m_rest_ptr = " << SSTR(static_cast<const void *> (m_rest_ptr)) << END_LOG;
315 
316  //The remaining length will be
317  m_rest_len = m_rest_len - (out_m_len + 1);
318  //The remaining pointer is the found one plus one
319  m_rest_ptr = out_m_begin_ptr + 1;
320 
321  LOG_DEBUG4 << "Setting the original reader, m_rest_len = " << m_rest_len
322  << ", m_rest_ptr = " << SSTR(static_cast<const void *> (m_rest_ptr)) << END_LOG;
323 
324  //Return true as the delimiter was found
325  return true;
326  } else {
327  LOG_DEBUG4 << "The delimiter '" << delim << "' was not found! Returning false!" << END_LOG;
328 
329  //Set the return data to NULL values
330  out.set(NULL, 0);
331 
332  //Return false as the delimiter was not found!
333  return false;
334  }
335  }

511  template<const char delim, const uint8_t delim_len>
512  const char * find_first_subseq() {
513  //Search for the next new delimiter from the front
514  const char * char_ptr = static_cast<const char *> (memchr(m_rest_ptr, delim, m_rest_len));
515 
516  //In case the delimiter length is more than one
517  if (delim_len > 1) {
518  //The pointer to the unread remainder of the file
519  const char * const end_ptr = m_rest_ptr + m_rest_len - 1;
520  //Search for a repeating delimiter substring
521  while (char_ptr != NULL) {
522  //Store the pointer to the last char in the possible delimiter substring
523  const char * const last_ptr = (char_ptr + (delim_len - 1));
524  //If there is enough symbols to check
525  if (last_ptr <= end_ptr) {
526  //Search for the substring of delimiters
527  uint8_t found = 1;
528  do {
529  //Move to the first next symbol and check if it is the right one
530  if (*(++char_ptr) == delim) {
531  ++found;
532  }
533  } while (char_ptr < last_ptr);
534 
535  //Check if we found enough
536  if (found == delim_len) {
537  //If the number of delimiter elements is exactly as we need then we can stop
538  break;
539  } else {
540  //If the number of delimiter elements is less then search on
541  char_ptr = static_cast<const char *> (memchr(char_ptr, delim, (end_ptr - char_ptr + 1)));
542  }
543  } else {
544  //There is not enough symbols, we've failed
545 
546  char_ptr = NULL;
547  }
548  }
549  }
550 
551  return char_ptr;
552  }
