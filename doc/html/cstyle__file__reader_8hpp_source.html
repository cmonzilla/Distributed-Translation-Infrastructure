113  inline bool get_first_line(text_piece_reader& out) {
114  LOG_DEBUG3 << "Searching for a new line, m_file_ptr = " << m_file_ptr << END_LOG;
115 
116  //First read the line from the file
117  ssize_t length = getline(&m_buff_ptr, &m_buff_size, m_file_ptr);
118 
119  if (length != -1) {
120  //Remove the new line symbol, we do not need it!
121  if ((length != 0) && (m_buff_ptr[length - 1] == '\n')) {
122  length = length - 1;
123  }
124 
125  LOG_DEBUG3 << "Read " << length << " symbols: '" << m_buff_ptr << "' !" << END_LOG;
126  //Store the data into the text piece reader
127  out.set(m_buff_ptr, length);
128  LOG_DEBUG3 << "The line has been read!" << END_LOG;
129  return true;
130  } else {
131  LOG_DEBUG3 << "The end of file is reached or an error has occurred!" << END_LOG;
132  //If the end of file is reached or an error occurred -1 is returned
133  return false;
134  }
135  }
