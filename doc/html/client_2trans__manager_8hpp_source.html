215  void write_received_job_result(const uint32_t fis, const uint32_t lis,
216  const trans_job_ptr job, ofstream & target_file) {
217  //The job response is received but it can still be fully or partially canceled or be an error
218  const trans_job_code & code = job->m_response->get_code();
219  switch (code) {
220  case trans_job_code::RESULT_OK:
221  case trans_job_code::RESULT_PARTIAL:
222  //If the result is ok or partial then just put the text into the file
223  target_file << job->m_response->get_text();
224  break;
225  case trans_job_code::RESULT_ERROR:
226  case trans_job_code::RESULT_CANCELED:
227  default:
228  //Report a warning
229  LOG_WARNING << "Sentences from " << fis << " to " << lis << " are not "
230  << "translated, job code: '" << code << "'" << END_LOG;
231 
232  //Write data to file
233  target_file << "<--------- Error: Sentences [" << fis << ":" << lis

242  void write_result_to_file() {
243  //Open the report file
244  ofstream target_file;
245  target_file.open(m_params.m_target_file);
246  //Declare the variable to store the line cursor
247  uint32_t line_cursor = 1;
248 
249  //Go through the translation job data and write it into the files
250  for (jobs_list_iter_type it = m_jobs_list.begin(); (it != m_jobs_list.end()); ++it) {
251  //Get the translation job data
252  const trans_job_ptr job = *it;
253 
254  //Compute the first and last sentence numbers for the case the job is not completed
255  const uint32_t fis = line_cursor;
256  const uint32_t lis = fis + job->m_num_sentences - 1;
257 
258  //Check on the translation job status
259  const trans_job_status & status = job->m_status;
260  switch (status) {
261  case trans_job_status::STATUS_RES_RECEIVED:
262  write_received_job_result(fis, lis, job, target_file);
263  break;
264  case trans_job_status::STATUS_REQ_SENT_GOOD:
265  case trans_job_status::STATUS_REQ_SENT_FAIL:
266  case trans_job_status::STATUS_REQ_INITIALIZED:
267  case trans_job_status::STATUS_UNDEFINED:
268  default:
269  //Report a warning
270  LOG_WARNING << "Sentences from " << fis << " to " << lis << " are not "
271  << "translated, job status: '" << status << "'" << END_LOG;
272 
273  //Write data to file
274  target_file << "<--------- Error: Sentences [" << fis << ":" << lis
275  << "] are not translated, status: '"
276  << status << "'-------->\n";
277  }
278 
279  //Set the line cursor to the next line
280  line_cursor = lis + 1;
281  }
282 
283  //Close the report file
284  target_file.close();

388  void send_translation_jobs() {
389  LOG_DEBUG << "Sending translation job requests ..." << END_LOG;
390 
391  //Send the translation jobs
392  for (jobs_list_iter_type it = m_jobs_list.begin(); (it != m_jobs_list.end()) && !m_is_stopping; ++it) {
393  //Get the pointer to the translation job data
394  trans_job_ptr data = *it;
395  try {
396  //Send the translation job request
397  m_client.send(data->m_request);
398  //Mark the job sending as good in the administration
399  data->m_status = trans_job_status::STATUS_REQ_SENT_GOOD;
400  } catch (uva_exception e) {
401  //Log the error message
402  LOG_ERROR << "Error when sending a translation request "
403  << data->m_request->get_job_id() << ": "
404  << e.get_message() << END_LOG;
405  //Mark the job sending as failed in the administration
406  data->m_status = trans_job_status::STATUS_REQ_SENT_FAIL;
407  //Increments the done jobs count
408  m_num_done_jobs++;
409  }
410  }
411 
412  LOG_DEBUG << "Finished sending translation job requests ..." << END_LOG;
413 
414  //The translation jobs have been sent!
415  notify_jobs_sent();
416 
417  //For the case when all jobs failed, we need to check if the jobs are notified
418  check_jobs_done_and_notify();

479  void create_translation_jobs() {
480  //Declare the variable to store the sentence line
481  text_piece_reader line;
482 
483  LOG_DEBUG << "Reading text from the source file ..." << END_LOG;
484  bool is_done = false;
485  while (!is_done) {
486  //Get the number of sentences to send in the next request
487  const uint64_t num_to_sent = get_num_of_sentences();
488  //Stores the number of read sentences
489  uint64_t num_read = 0;
490  //text to send for translation
491  string source_text;
492 
493  LOG_DEBUG1 << "Planning to send " << num_to_sent << " sentences with the next request" << END_LOG;
494 
495  while ((num_read < num_to_sent) && m_source_file.get_first_line(line)) {
496  //Obtain the read source sentence
497  string source_sent = line.str();
498 
499  LOG_DEBUG2 << "Read line: '" << source_sent << "'" << END_LOG;
500 
501  //If needed, do the source sentence pre-processing
502  if (m_params.is_pre_process) {
503  //Pre-proces the source sentence:
504  //1. Lowercase
505  to_lower(source_sent);
506  //2. Punctuate
507  punctuate(source_sent);
508  //3. Reduce
509  reduce(source_sent);
510  }
511 
512  //Append the new line to the text to be sent
513  source_text += source_sent + UTF8_NEW_LINE_STRING;
514 
515  //Increment the number of read sentences
516  ++num_read;
517 
518  LOG_DEBUG2 << "Read " << num_read << " input sentences!" << END_LOG;
519  }
520 
521  LOG_DEBUG1 << "Read " << num_read << " sentences to be sent with the next request" << END_LOG;
522 
523  //If there were lines read then do translation
524  if (num_read != 0) {
525  //Get the new job id
526  const job_id_type job_id = m_id_mgr.get_next_id();
527 
528  //Remove the last sentence new line symbol
529  source_text.substr(0, source_text.size() - 1);
530 
531  //Create the new job data object
532  trans_job_ptr data = new trans_job();
533 
534  //Create the translation job request
535  data->m_request = new trans_job_request(job_id, m_params.m_source_lang, source_text, m_params.m_target_lang);
536  //Store the number of sentences in the translation request
537  data->m_num_sentences = num_read;
538  //Mark the job sending as good in the administration
539  data->m_status = trans_job_status::STATUS_REQ_INITIALIZED;
540 
541  //Store the translation request
542  m_jobs_list.push_back(data);
543  m_ids_to_jobs_map[job_id] = data;
544  } else {
545  is_done = true;
546  }
547  }
548 
549  LOG_DEBUG << "Finished reading text from the source file!" << END_LOG;
550  }
551  };
552 
553  id_manager<job_id_type> trans_manager::m_id_mgr(job_id::MINIMUM_JOB_ID);
554  }
555  }
556  }
