78  inline bool get_ctx_id(const phrase_length level_idx, const TShortId word_id, TLongId & ctx_id) const {
79  //Compute back the current level pure for debug purposes.
80  const phrase_length curr_level = level_idx + BASE::MGRAM_IDX_OFFSET;
81  LOG_DEBUG3 << "Retrieving context level: " << curr_level << ", word_id: "
82  << word_id << ", ctx_id: " << ctx_id << END_LOG;
83 
84  //Retrieve the context data for the given word
85  StorageContainer* ctx_mapping = m_mgram_mapping[level_idx][word_id];
86 
87  //Check that the context data is available
88  if (ctx_mapping != NULL) {
89  typename StorageContainer::const_iterator result = ctx_mapping->find(ctx_id);
90  if (result == ctx_mapping->end()) {
91  LOG_DEBUG2 << "Can not find ctx_id: " << SSTR(ctx_id) << " for level: "
92  << SSTR(curr_level) << ", word_id: " << SSTR(word_id) << END_LOG;
93  return false;
94  } else {
95  LOG_DEBUG2 << "Found next ctx_id: " << SSTR(result->second)
96  << " for level: " << SSTR(curr_level) << ", word_id: "
97  << SSTR(word_id) << ", ctx_id: " << SSTR(ctx_id) << END_LOG;
98 
99  ctx_id = result->second;
100  return true;
101  }
102  } else {
103  LOG_DEBUG2 << "No context data for: " << SSTR(curr_level)
104  << ", word_id: " << SSTR(word_id) << END_LOG;
105  return false;
106  }

134  template<phrase_length CURR_LEVEL>
135  inline void add_m_gram(const model_m_gram & gram) {
136  const TShortId word_id = gram.get_last_word_id();
137  if (CURR_LEVEL == M_GRAM_LEVEL_1) {
138  //Store the payload
139  m_mgram_data[0][word_id] = gram.m_payload;
140  } else {
141  //Register the m-gram in the hash cache
142  this->register_m_gram_cache(gram);
143 
144  //Define the context id variable
145  TLongId ctx_id = UNKNOWN_WORD_ID;
146  //Obtain the m-gram context id
147  __LayeredTrieBase::get_context_id<w2c_hybrid_trie<WordIndexType, StorageFactory, StorageContainer>, CURR_LEVEL, debug_levels_enum::DEBUG2>(*this, gram, ctx_id);
148 
149  //Store the payload
150  if (CURR_LEVEL == LM_M_GRAM_LEVEL_MAX) {
151  StorageContainer*& ctx_mapping = m_mgram_mapping[BASE::N_GRAM_IDX_IN_M_N_ARR][word_id];
152  if (ctx_mapping == NULL) {
153  ctx_mapping = m_storage_factory->create(LM_M_GRAM_LEVEL_MAX);
154  LOG_DEBUG3 << "Allocating storage for level " << SSTR(LM_M_GRAM_LEVEL_MAX) << ", word_id " << SSTR(word_id) << END_LOG;
155  }
156 
157  LOG_DEBUG3 << "Returning reference to prob., level: " << SSTR(LM_M_GRAM_LEVEL_MAX) << ", word_id "
158  << SSTR(word_id) << ", ctx_id " << SSTR(ctx_id) << END_LOG;
159  //WARNING: We cast to (TLogProbBackOff &) as we misuse the mapping by storing the probability value there!
160  reinterpret_cast<prob_weight&> (ctx_mapping->operator[](ctx_id)) = gram.m_payload.m_prob;
161  } else {
162  const phrase_length idx = (CURR_LEVEL - BASE::MGRAM_IDX_OFFSET);
163 
164  //Get the word mapping first
165  StorageContainer*& ctx_mapping = m_mgram_mapping[idx][word_id];
166 
167  //If the mappings is not there yet for the contexts then create it
168  if (ctx_mapping == NULL) {
169  ctx_mapping = m_storage_factory->create(CURR_LEVEL);
170  LOG_DEBUG3 << "A new ACtxToPBStorage container is allocated for level " << SSTR(CURR_LEVEL) << END_LOG;
171  }
172 
173  //Add the new element to the context mapping
174  TShortId & nextCtxId = ctx_mapping->operator[](ctx_id);
175  nextCtxId = next_ctx_id[idx]++;
176 
177  //Return the reference to it
178  m_mgram_data[CURR_LEVEL - 1][nextCtxId] = gram.m_payload;
179  }
180  }

203  inline void get_m_gram_payload(m_gram_query & query, MGramStatusEnum & status) const {
204  LOG_DEBUG << "Getting the payload for sub-m-gram: " << query << END_LOG;
205 
206  //First ensure the context of the given sub-m-gram
207  LAYERED_BASE_ENSURE_CONTEXT(query, status);
208 
209  //If the context is successfully ensured, then move on to the m-gram and try to obtain its payload
210  if (status == MGramStatusEnum::GOOD_PRESENT_MGS) {
211  //Store the shorthand for the context and end word id
212  TLongId & ctx_id = query.get_curr_ctx_ref();
213  const TShortId & word_id = query.get_curr_end_word_id();
214 
215  //Compute the distance between words
216  const phrase_length & curr_level =query.get_curr_level();
217  LOG_DEBUG << "curr_level: " << SSTR(curr_level) << ", ctx_id: " << ctx_id << ", m_end_word_idx: "
218  << SSTR(query.m_curr_end_word_idx) << ", end word id: " << word_id << END_LOG;
219 
220  //Get the next context id
221  const phrase_length & level_idx = query.get_curr_level_m2();
222  if (get_ctx_id(level_idx, word_id, ctx_id)) {
223  LOG_DEBUG << "ctx_id: " << ctx_id << END_LOG;
224  const phrase_length & idx = query.get_curr_level_m1();
225  //There is data found under this context
226  query.set_curr_payload(&m_mgram_data[idx][ctx_id]);
227  LOG_DEBUG << "The payload is retrieved: " << m_mgram_data[idx][ctx_id] << END_LOG;
228  } else {
229  //The payload could not be found
230  LOG_DEBUG1 << "Unable to find m-gram data for ctx_id: " << SSTR(ctx_id)
231  << ", word_id: " << SSTR(word_id) << END_LOG;
232  status = MGramStatusEnum::BAD_NO_PAYLOAD_MGS;
233  }
234  }

241  inline void get_n_gram_payload(m_gram_query & query, MGramStatusEnum & status) const {
242  //First ensure the context of the given sub-m-gram
243  LAYERED_BASE_ENSURE_CONTEXT(query, status);
244 
245  //If the context is successfully ensured, then move on to the m-gram and try to obtain its payload
246  if (status == MGramStatusEnum::GOOD_PRESENT_MGS) {
247  //Store the shorthand for the context and end word id
248  const TLongId & ctx_id = query.get_curr_ctx_ref();
249  const TShortId & word_id = query.get_curr_end_word_id();
250 
251  //Try to find the word mapping first
252  StorageContainer*& ctx_mapping = m_mgram_mapping[BASE::N_GRAM_IDX_IN_M_N_ARR][word_id];
253 
254  //If the mapping is present the search further, otherwise return false
255  if (ctx_mapping != NULL) {
256  typename StorageContainer::const_iterator result = ctx_mapping->find(ctx_id);
257  if (result == ctx_mapping->end()) {
258  //The data could not be found
259  status = MGramStatusEnum::BAD_NO_PAYLOAD_MGS;
260  } else {
261  //The data could be found
262  query.set_curr_payload(&result->second);
263  LOG_DEBUG << "The payload is retrieved: " << result->second << END_LOG;
264  }
265  } else {
266  //The payload could not be found
267  LOG_DEBUG1 << "Unable to find " << SSTR(LM_M_GRAM_LEVEL_MAX) << "-gram data for ctx_id: "
268  << SSTR(ctx_id) << ", word_id: " << SSTR(word_id) << END_LOG;
269  status = MGramStatusEnum::BAD_NO_PAYLOAD_MGS;
270  }
271  }

289  //M-Gram data for 1 <= M < N. This is a 2D array storing
290  //For each M-Gram level M an array of prob-back_off values
291  // m_mgram_data[M][0] - probability/back-off pair for the given M-gram
292  // m_mgram_data[M][1] --//--
293  // ...
294  // m_mgram_data[M][#M-Grams - 1] --//--
295  // m_mgram_data[M][#M-Grams] --//--
296  m_gram_payload * m_mgram_data[LM_M_GRAM_LEVEL_MAX - 1];
297 
298  //M-Gram data for 1 < M <= N. This is a 2D array storing
299  //For each M-Gram level M an array of #words elements of
300  //pointers to C template parameter type:
301  // m_mgram_mapping[M][0] -> NULL (if there is no M-gram ending with word 0 in the level)
302  // m_mgram_mappin[M]g[1] -> C instance
303  // ...
304  // m_mgram_mapping[M][#words - 1] -> NULL
305  // m_mgram_mapping[M][#words] -> C instance
306  //
307  //For all 1 < M < N instances of C contain mappings from the
308  //context index to the index in the m_mgram_data[M] array -
309  //the array storing probability/back-off values.
310  //For M = N the pair value stored in the instance of C is the
311  //probability itself! Note that that the probabilities are
312  //stored as floats - 4 bytes and m_mgram_data[M] array is also a
313  //4 byte integer, so we minimize memory usage by storing float
314  //probability in place of the index.
315  StorageContainer** m_mgram_mapping[LM_M_GRAM_LEVEL_MAX - 1];
316 
317  //Will store the next context index counters per M-gram level
318  //for 1 < M < N.
319  const static phrase_length NUM_IDX_COUNTERS = LM_M_GRAM_LEVEL_MAX - 2;
320  TShortId next_ctx_id[NUM_IDX_COUNTERS];
321  };
