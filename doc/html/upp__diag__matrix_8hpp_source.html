71  upp_diag_matrix(const size_t dim) : m_max_idx(dim - 1), m_dim(dim) {
72  LOG_DEBUG2 << "Creating the upper diagonal matrix of " << m_dim << " dimensions" << END_LOG;
73 
74  //Compute the number of elements to be used
75  const size_t num_elements = (dim - 1) * dim / 2 + dim;
76 
77  LOG_DEBUG2 << "Allocating " << num_elements << " elements." << END_LOG;
78 
79  //First allocate the number of elements we need
80  m_elems = new element_type[num_elements]();
81 
82  //Allocate the row pointers array
83  m_rows = new element_type_ptr[dim]();
84 
85  //Initialize the row pointers array
86  m_rows[0] = m_elems;
87  for (size_t idx = 1; idx < dim; ++idx) {
88  const size_t offset = dim - idx;
89  LOG_DEBUG2 << "Internal row: " << idx << " offset: " << offset << END_LOG;
90  m_rows[idx] = m_rows[idx - 1] + offset;
91  }
92 
93  LOG_DEBUG2 << "upp_diag_matrix create is finished" << END_LOG;
94  }
